37  package org.atmosphere.jersey.tests;

39  import org.atmosphere.cpr.AtmosphereServlet;
40  import org.slf4j.Logger;
41  import org.slf4j.LoggerFactory;
42  import org.testng.annotations.AfterMethod;
43  import org.testng.annotations.BeforeMethod;
44  
45  import java.io.IOException;
46  import java.net.ServerSocket;
47  
48  public abstract class BaseTest {
49  
50      protected static final Logger logger = LoggerFactory.getLogger(BaseTest.class);
51  
52      protected static final String ROOT = "/*";
53  
54      protected AtmosphereServlet atmoServlet;
55      public String urlTarget;
56      public int port;
57  
58      abstract public void configureCometSupport();
59  
60      abstract public void startServer() throws Exception;
61  
62      abstract public void stopServer() throws Exception;
63  
64      public static class TestHelper {
65  
66          public static int getEnvVariable(final String varName, int defaultValue) {
67              if (null == varName) {
68                  return defaultValue;
69              }
70              String varValue = System.getenv(varName);
71              if (null != varValue) {
72                  try {
73                      return Integer.parseInt(varValue);
74                  } catch (NumberFormatException e) {
75                      // will return default value bellow
76                  }
77              }
78              return defaultValue;
79          }
80      }
81  
82      protected int findFreePort() throws IOException {
83          ServerSocket socket = null;
84  
85          try {
86              socket = new ServerSocket(0);
87  
88              return socket.getLocalPort();
89          } finally {
90              if (socket != null) {
91                  socket.close();
92              }
93          }
94      }
95  
96      @BeforeMethod(alwaysRun = true)
97      public void setUpGlobal() throws Exception {
98          port = TestHelper.getEnvVariable("ATMOSPHERE_HTTP_PORT", findFreePort());
99          urlTarget = getUrlTarget(port);
100         atmoServlet = new AtmosphereServlet();
101         atmoServlet.addInitParameter("com.sun.jersey.config.property.packages", this.getClass().getPackage().getName());
102         atmoServlet.addInitParameter("org.atmosphere.cpr.broadcasterClass", RecyclableBroadcaster.class.getName());
103 
104         configureCometSupport();
105         startServer();
106     }
107 
108     abstract String getUrlTarget(int port);
109 
110     @AfterMethod(alwaysRun = true)
111     public void unsetAtmosphereHandler() throws Exception {
112         if (atmoServlet != null) atmoServlet.destroy();
113         stopServer();
114     }
115 }

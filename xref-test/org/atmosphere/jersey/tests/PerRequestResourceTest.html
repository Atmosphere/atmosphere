47  import static org.testng.Assert.assertNotNull;

51  /**
52   * @author paulsandoz
53   */
54  public class PerRequestResourceTest extends BaseGrizzyTest {
55  
56      String getUrlTarget(int port) {
57          return "http://127.0.0.1:" + port + "/perrequest";
58      }
59  
60      @Test(timeOut = 20000, enabled = true)
61      public void testPerRequestSuspendTimeout() {
62          logger.info("{}: running test:  testPerRequestSuspendTimeout", getClass().getSimpleName());
63  
64          AsyncHttpClient c = new AsyncHttpClient();
65          try {
66              long t1 = System.currentTimeMillis();
67              Response r = c.prepareGet(urlTarget).execute().get(10, TimeUnit.SECONDS);
68              assertNotNull(r);
69              assertEquals(r.getStatusCode(), 200);
70              String resume = r.getResponseBody();
71              assertEquals(resume, "perrequest");
72              long current = System.currentTimeMillis() - t1;
73              assertTrue(current > 5000 && current < 10000);
74          } catch (Exception e) {
75              logger.error("test failed", e);
76              fail(e.getMessage());
77          }
78  
79          c.close();
80      }

16  package org.atmosphere.inject;
17  
18  import org.atmosphere.cpr.AtmosphereConfig;
19  import org.atmosphere.cpr.AtmosphereFramework;
20  import org.atmosphere.cpr.AtmosphereResourceFactory;
21  import org.atmosphere.cpr.AtmosphereResourceSessionFactory;
22  import org.atmosphere.cpr.BroadcasterFactory;
23  import org.atmosphere.cpr.MetaBroadcaster;
24  import org.atmosphere.websocket.WebSocketFactory;
25  
26  import javax.enterprise.context.ApplicationScoped;
27  import javax.enterprise.inject.Produces;
28  
29  /**
30   * Atmosphere Internal Object Injection for DI supporting JSR 330
31   */
32  @ApplicationScoped
33  public class AtmosphereProducers implements CDIProducer {
34  
35      private AtmosphereConfig config;
36  
37      public void configure(AtmosphereConfig config) {
38          this.config = config;
39      }
40  
41      @Produces
42      public BroadcasterFactory getBroadcasterFactory() {
43          return config.getBroadcasterFactory();
44      }
45  
46      @Produces
47      public AtmosphereResourceFactory getAtmosphereResourceFactory() {
48          return config.resourcesFactory();
49      }
50  
51      @Produces
52      public AtmosphereResourceSessionFactory getAtmosphereResourceSessionFactory() {
53          return config.sessionFactory();
54      }
55  
56      @Produces
57      public AtmosphereConfig getAtmosphereConfig() {
58          return config;
59      }
60  
61      @Produces
62      public AtmosphereFramework getAtmosphereFramework() {
63          return config.framework();
64      }
65  
66      @Produces
67      public MetaBroadcaster getMetaBroadcaster() {
68          return config.metaBroadcaster();
69      }
70  
71      @Produces
72      public WebSocketFactory getWebSocketFactory() {
73          return config.websocketFactory();
74      }
75  
76  }

16  package org.atmosphere.inject;
17  
18  import org.atmosphere.cpr.AtmosphereResource;
19  import org.atmosphere.cpr.Broadcaster;
20  import org.atmosphere.inject.annotation.RequestScoped;
21  import org.atmosphere.util.ThreadLocalInvoker;
22  
23  import javax.inject.Named;
24  import java.lang.reflect.Field;
25  import java.lang.reflect.Method;
26  import java.lang.reflect.Proxy;
27  import java.lang.reflect.Type;
28  
29  /**
30   * {@link Broadcaster} and {@link Named} injection support.
31   *
32   * @author Jeanfrancois Arcand
33   */
34  @RequestScoped
35  public class BroadcasterIntrospector extends InjectIntrospectorAdapter<Broadcaster> {
36      private final ThreadLocal<String> nameLocal = new ThreadLocal<String>();
37  
38      @Override
39      public boolean supportedType(Type t) {
40          return (t instanceof Class) && Broadcaster.class.isAssignableFrom((Class) t);
41      }
42  
43      @Override
44      public Broadcaster injectable(AtmosphereResource r) {
45          String named = nameLocal.get();
46  
47          if (named == null) {
48              named = Broadcaster.ROOT_MASTER;
49          }
50  
51          String s = (String) r.getRequest().getAttribute(Named.class.getName());
52          int indx = named.indexOf("{");
53          if (s != null && indx != -1) {
54              named = named.substring(0, indx) + s;
55          }
56  
57          final Broadcaster broadcaster = r.getAtmosphereConfig().getBroadcasterFactory().lookup(named, true);
58  
59          return (Broadcaster) Proxy.newProxyInstance(this.getClass().getClassLoader(),
60                  new Class[]{Broadcaster.class}, new ThreadLocalInvoker() {
61                      {
62                          set(broadcaster);
63                      }
64  
65                      @Override
66                      public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
67                          return method.invoke(broadcaster, args);
68                      }
69                  });
70      }
71  
72      @Override
73      public void introspectField(Class clazz, Field f) {
74          if (f.isAnnotationPresent(Named.class)) {
75              String name = f.getAnnotation(Named.class).value();
76  
77              if (name.isEmpty()) {
78                  name = f.getName();
79              }
80              nameLocal.set(name);
81          }
82      }
83  }

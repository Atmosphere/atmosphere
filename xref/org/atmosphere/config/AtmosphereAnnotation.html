3   import java.lang.annotation.Annotation;
4   import java.lang.annotation.Documented;
5   import java.lang.annotation.ElementType;
6   import java.lang.annotation.Retention;
7   import java.lang.annotation.RetentionPolicy;
8   import java.lang.annotation.Target;
9   
10  /**
11   * This annotation is scanned by Atmosphere at runtime to determine {@link org.atmosphere.annotation.Processor} implementation.
12   * Application that wants to define their own annotation can annotate their class with this annotation.
13   *
14   * @author Jeanfrancois Arcand
15   */
16  @Target({ElementType.TYPE})
17  @Retention(RetentionPolicy.RUNTIME)
18  @Documented
19  public @interface AtmosphereAnnotation {
20      /**
21       * Return the handled annotation associated with the {@link org.atmosphere.annotation.Processor}
22       */
23      Class<? extends Annotation> value();
24  }

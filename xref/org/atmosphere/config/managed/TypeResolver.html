63      /** Cache of type variable/argument pairs */
64      private static final Map<Class<?>, Reference<Map<TypeVariable<?>, Type>>> typeVariableCache = Collections.synchronizedMap(new WeakHashMap<Class<?>, Reference<Map<TypeVariable<?>, Type>>>());
65      private static boolean cacheEnabled = true;

82      /**
83       * Returns the raw class representing the type argument for the {@code targetType} resolved
84       * upwards from the {@code initialType}. If no arguments can be resolved then
85       * {@code Unknown.class} is returned.
86       *
87       * @param initialType to resolve upwards from
88       * @param targetType to resolve arguments for
89       * @return type argument for {@code initialType} else {@code null} if no type arguments are
90       *         declared
91       * @throws IllegalArgumentException if more or less than one type argument is resolved for the
92       *           give types
93       */
94      public static <T, I extends T> Class<?> resolveArgument(Class<I> initialType, Class<T> targetType) {
95          return resolveArgument(resolveGenericType(initialType, targetType), initialType);
96      }

98      /**
99       * Resolves the type argument for the {@code genericType} using type variable information from the
100      * {@code sourceType}. If {@code genericType} is an instance of class, then {@code genericType} is
101      * returned. If no arguments can be resolved then {@code Unknown.class} is returned.
102      *
103      * @param genericType to resolve upwards from
104      * @param targetType to resolve arguments for
105      * @return type argument for {@code initialType} else {@code null} if no type arguments are
106      *         declared
107      * @throws IllegalArgumentException if more or less than one type argument is resolved for the
108      *           give types
109      */
110     public static Class<?> resolveArgument(Type genericType, Class<?> targetType) {
111         Class<?>[] arguments = resolveArguments(genericType, targetType);
112         if (arguments == null)
113             return Unknown.class;
114 
115         if (arguments.length != 1)
116             throw new IllegalArgumentException("Expected 1 type argument on generic type "
117                     + targetType.getName() + " but found " + arguments.length);
118 
119         return arguments[0];
120     }

122     /**
123      * Returns an array of raw classes representing type arguments for the {@code targetType} resolved
124      * upwards from the {@code initialType}. Arguments for {@code targetType} that cannot be resolved
125      * to a Class are returned as {@code Unknown.class}. If no arguments can be resolved then
126      * {@code null} is returned.
127      *
128      * @param initialType to resolve upwards from
129      * @param targetType to resolve arguments for
130      * @return array of raw classes representing type arguments for {@code initialType} else
131      *         {@code null} if no type arguments are declared
132      */
133     public static <T, I extends T> Class<?>[] resolveArguments(Class<I> initialType,
134                                                                Class<T> targetType) {
135         return resolveArguments(resolveGenericType(initialType, targetType), initialType);
136     }

138     /**
139      * Resolves the arguments for the {@code genericType} using the type variable information for the
140      * {@code targetType}. Returns {@code null} if {@code genericType} is not parameterized or if
141      * arguments cannot be resolved.
142      */
143     public static Class<?>[] resolveArguments(Type genericType, Class<?> targetType) {
144         Class<?>[] result = null;
145 
146         if (genericType instanceof ParameterizedType) {
147             ParameterizedType paramType = (ParameterizedType) genericType;
148             Type[] arguments = paramType.getActualTypeArguments();
149             result = new Class[arguments.length];
150             for (int i = 0; i < arguments.length; i++)
151                 result[i] = resolveClass(arguments[i], targetType);
152         } else if (genericType instanceof TypeVariable) {
153             result = new Class[1];
154             result[0] = resolveClass(genericType, targetType);
155         }
156 
157         return result;
158     }

160     /**
161      * Resolves the generic Type for the {@code targetType} by walking the type hierarchy upwards from
162      * the {@code initialType}.
163      */
164     public static Type resolveGenericType(Type initialType, Class<?> targetType) {
165         Class<?> rawType;
166         if (initialType instanceof ParameterizedType)
167             rawType = (Class<?>) ((ParameterizedType) initialType).getRawType();
168         else
169             rawType = (Class<?>) initialType;
170 
171         if (targetType.equals(rawType))
172             return initialType;
173 
174         Type result;
175         if (targetType.isInterface()) {
176             for (Type superInterface : rawType.getGenericInterfaces())
177                 if (superInterface != null && !superInterface.equals(Object.class))
178                     if ((result = resolveGenericType(superInterface, targetType)) != null)
179                         return result;
180         }
181 
182         Type superType = rawType.getGenericSuperclass();
183         if (superType != null && !superType.equals(Object.class))
184             if ((result = resolveGenericType(superType, targetType)) != null)
185                 return result;
186 
187         return null;
188     }

190     /**
191      * Resolves the raw class for the given {@code genericType}, using the type variable information
192      * from the {@code targetType}.
193      */
194     public static Class<?> resolveClass(Type genericType, Class<?> targetType) {
195         if (genericType instanceof Class) {
196             return (Class<?>) genericType;
197         } else if (genericType instanceof ParameterizedType) {
198             return resolveClass(((ParameterizedType) genericType).getRawType(), targetType);
199         } else if (genericType instanceof GenericArrayType) {
200             GenericArrayType arrayType = (GenericArrayType) genericType;
201             Class<?> compoment = resolveClass(arrayType.getGenericComponentType(), targetType);
202             return Array.newInstance(compoment, 0).getClass();
203         } else if (genericType instanceof TypeVariable) {
204             TypeVariable<?> variable = (TypeVariable<?>) genericType;
205             genericType = getTypeVariableMap(targetType).get(variable);
206             genericType = genericType == null ? resolveBound(variable) : resolveClass(genericType,
207                     targetType);
208         }
209 
210         return genericType instanceof Class ? (Class<?>) genericType : Unknown.class;
211     }

213     private static Map<TypeVariable<?>, Type> getTypeVariableMap(final Class<?> targetType) {
214         Reference<Map<TypeVariable<?>, Type>> ref = typeVariableCache.get(targetType);
215         Map<TypeVariable<?>, Type> map = ref != null ? ref.get() : null;
216 
217         if (map == null) {
218             map = new HashMap<TypeVariable<?>, Type>();
219 
220             // Populate interfaces
221             buildTypeVariableMap(targetType.getGenericInterfaces(), map);
222 
223             // Populate super classes and interfaces
224             Type genericType = targetType.getGenericSuperclass();
225             Class<?> type = targetType.getSuperclass();
226             while (type != null && !Object.class.equals(type)) {
227                 if (genericType instanceof ParameterizedType)
228                     buildTypeVariableMap((ParameterizedType) genericType, map);
229                 buildTypeVariableMap(type.getGenericInterfaces(), map);
230 
231                 genericType = type.getGenericSuperclass();
232                 type = type.getSuperclass();
233             }
234 
235             // Populate enclosing classes
236             type = targetType;
237             while (type.isMemberClass()) {
238                 genericType = type.getGenericSuperclass();
239                 if (genericType instanceof ParameterizedType)
240                     buildTypeVariableMap((ParameterizedType) genericType, map);
241 
242                 type = type.getEnclosingClass();
243             }
244 
245             if (cacheEnabled)
246                 typeVariableCache.put(targetType, new WeakReference<Map<TypeVariable<?>, Type>>(map));
247         }
248 
249         return map;
250     }

252     /**
253      * Populates the {@code map} with with variable/argument pairs for the given {@code types}.
254      */
255     static void buildTypeVariableMap(final Type[] types, final Map<TypeVariable<?>, Type> map) {
256         for (Type type : types) {
257             if (type instanceof ParameterizedType) {
258                 ParameterizedType parameterizedType = (ParameterizedType) type;
259                 buildTypeVariableMap(parameterizedType, map);
260                 Type rawType = parameterizedType.getRawType();
261                 if (rawType instanceof Class)
262                     buildTypeVariableMap(((Class<?>) rawType).getGenericInterfaces(), map);
263             } else if (type instanceof Class) {
264                 buildTypeVariableMap(((Class<?>) type).getGenericInterfaces(), map);
265             }
266         }
267     }

269     /**
270      * Populates the {@code typeVariableMap} with type arguments and parameters for the given
271      * {@code type}.
272      */
273     private static void buildTypeVariableMap(ParameterizedType type,
274                                              Map<TypeVariable<?>, Type> typeVariableMap) {
275         if (type.getRawType() instanceof Class) {
276             TypeVariable<?>[] typeVariables = ((Class<?>) type.getRawType()).getTypeParameters();
277             Type[] typeArguments = type.getActualTypeArguments();
278 
279             for (int i = 0; i < typeArguments.length; i++) {
280                 TypeVariable<?> variable = typeVariables[i];
281                 Type typeArgument = typeArguments[i];
282 
283                 if (typeArgument instanceof Class) {
284                     typeVariableMap.put(variable, typeArgument);
285                 } else if (typeArgument instanceof GenericArrayType) {
286                     typeVariableMap.put(variable, typeArgument);
287                 } else if (typeArgument instanceof ParameterizedType) {
288                     typeVariableMap.put(variable, typeArgument);
289                 } else if (typeArgument instanceof TypeVariable) {
290                     TypeVariable<?> typeVariableArgument = (TypeVariable<?>) typeArgument;
291                     Type resolvedType = typeVariableMap.get(typeVariableArgument);
292                     if (resolvedType == null)
293                         resolvedType = resolveBound(typeVariableArgument);
294                     typeVariableMap.put(variable, resolvedType);
295                 }
296             }
297         }
298     }

300     /**
301      * Resolves the first bound for the {@code typeVariable}, returning {@code Unknown.class} if none
302      * can be resolved.
303      */
304     public static Type resolveBound(TypeVariable<?> typeVariable) {
305         Type[] bounds = typeVariable.getBounds();
306         if (bounds.length == 0)
307             return Unknown.class;
308 
309         Type bound = bounds[0];
310         if (bound instanceof TypeVariable)
311             bound = resolveBound((TypeVariable<?>) bound);
312 
313         return bound == Object.class ? Unknown.class : bound;
314     }

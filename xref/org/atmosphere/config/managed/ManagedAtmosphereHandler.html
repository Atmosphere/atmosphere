66      private final static List<Decoder<?, ?>> EMPTY = Collections.<Decoder<?, ?>>emptyList();
67      private Object proxiedInstance;

79      final Map<Method, List<Encoder<?, ?>>> encoders = new HashMap<Method, List<Encoder<?, ?>>>();

85      public ManagedAtmosphereHandler configure(AtmosphereConfig config, Object c) {
86          this.proxiedInstance = c;
87          this.onRuntimeMethod = populateMessage(c, Message.class);
88          this.onDisconnectMethod = populate(c, Disconnect.class);
89          this.onTimeoutMethod = populate(c, Resume.class);
90          this.onGetMethod = populate(c, Get.class);
91          this.onPostMethod = populate(c, Post.class);
92          this.onPutMethod = populate(c, Put.class);
93          this.onDeleteMethod = populate(c, Delete.class);
94          this.onReadyMethod = populate(c, Ready.class);
95          this.onResumeMethod = populate(c, Resume.class);
96          this.config = config;
97  
98          scanForReaderOrInputStream();
99  
100         if (onRuntimeMethod.size() > 0) {
101             populateEncoders();
102             populateDecoders();
103         }
104         return this;
105     }

107     @Override
108     public void onRequest(final AtmosphereResource resource) throws IOException {
109         AtmosphereRequest request = resource.getRequest();
110         String method = request.getMethod();
111         boolean polling = Utils.pollableTransport(resource.transport());
112 
113         if (onReadyMethod != null && !polling) {
114             resource.addEventListener(new OnSuspend() {
115                 @Override
116                 public void onSuspend(AtmosphereResourceEvent event) {
117                     processReady(event.getResource());
118                     event.getResource().removeEventListener(this);
119                 }
120             });
121         }
122 
123         if (onResumeMethod != null && !polling) {
124             resource.addEventListener(new OnResume() {
125                 @Override
126                 public void onResume(AtmosphereResourceEvent event) {
127                     invoke(onResumeMethod, event.getResource());
128                     resource.removeEventListener(this);
129                 }
130             });
131         }
132 
133         if (method.equalsIgnoreCase("get")) {
134             invoke(onGetMethod, resource);
135         } else if (method.equalsIgnoreCase("post")) {
136             String body = null;
137             if (onPostMethod != null) {
138                 body = IOUtils.readEntirely(resource).toString();
139                 resource.getRequest().body(body);
140                 invoke(onPostMethod, resource);
141             }
142 
143             Object o = message(resource, body);
144             if (o != null) {
145                 resource.getBroadcaster().broadcast(new Managed(o));
146             }
147         } else if (method.equalsIgnoreCase("delete")) {
148             invoke(onDeleteMethod, resource);
149         } else if (method.equalsIgnoreCase("put")) {
150             invoke(onPutMethod, resource);
151         }
152     }

154     @Override
155     @SuppressWarnings("unchecked")
156     public void onStateChange(AtmosphereResourceEvent event) throws IOException {
157 
158         // Original Value
159         AtmosphereResourceImpl r = AtmosphereResourceImpl.class.cast(event.getResource());
160         Boolean resumeOnBroadcast = r.resumeOnBroadcast();
161         if (!resumeOnBroadcast) {
162             // For legacy reason, check the attribute as well
163             Object o = r.getRequest(false).getAttribute(ApplicationConfig.RESUME_ON_BROADCAST);
164             if (o != null && Boolean.class.isAssignableFrom(o.getClass())) {
165                 resumeOnBroadcast = Boolean.class.cast(o);
166             }
167         }
168 
169         // Disable resume so cached message can be send in one chunk.
170         if (resumeOnBroadcast) {
171             r.resumeOnBroadcast(false);
172             r.getRequest(false).setAttribute(ApplicationConfig.RESUME_ON_BROADCAST, false);
173         }
174 
175         if (event.isCancelled() || event.isClosedByClient()) {
176             invoke(onDisconnectMethod, event);
177         } else if (event.isResumedOnTimeout() || event.isResuming()) {
178             invoke(onTimeoutMethod, event);
179         } else {
180             Object msg = event.getMessage();
181             Object o;
182             if (msg != null) {
183                 if (Managed.class.isAssignableFrom(msg.getClass())) {
184                     Object newMsg = Managed.class.cast(msg).o;
185                     event.setMessage(newMsg);
186                     // No method matched. Give a last chance by trying to decode the proxiedInstance.
187                     // This makes application development more simpler.
188                     // Chaining of encoder is not supported.
189                     // TODO: This could be problematic with String + method
190                     for (MethodInfo m : onRuntimeMethod) {
191                         o = Invoker.encode(encoders.get(m.method), newMsg);
192                         if (o != null) {
193                             event.setMessage(o);
194                             break;
195                         }
196                     }
197                 } else {
198                     o = message(r, msg);
199                     if (o != null) {
200                         event.setMessage(o);
201                     }
202                 }
203             }
204 
205             super.onStateChange(event);
206         }
207 
208         if (resumeOnBroadcast && r.isSuspended()) {
209             r.resume();
210         }
211     }

213     private Method populate(Object c, Class<? extends Annotation> annotation) {
214         for (Method m : c.getClass().getMethods()) {
215             if (m.isAnnotationPresent(annotation)) {
216                 return m;
217             }
218         }
219         return null;
220     }
221 
222     private List<MethodInfo> populateMessage(Object c, Class<? extends Annotation> annotation) {
223         ArrayList<MethodInfo> methods = new ArrayList<MethodInfo>();
224         for (Method m : c.getClass().getMethods()) {
225             if (m.isAnnotationPresent(annotation)) {
226                 methods.add(new MethodInfo(m));
227             }
228         }
229         return methods;
230     }
231 
232     private void scanForReaderOrInputStream() {
233         for (MethodInfo m : onRuntimeMethod) {
234             Class<?>[] classes = m.method.getParameterTypes();
235             for (Class<?> c : classes) {
236                 if (InputStream.class.isAssignableFrom(c)) {
237                     m.useStream = true;
238                 } else if (Reader.class.isAssignableFrom(c)) {
239                     m.useReader = true;
240                 }
241             }
242         }
243     }
244 
245     private void populateEncoders() {
246         for (MethodInfo m : onRuntimeMethod) {
247             List<Encoder<?, ?>> l = new CopyOnWriteArrayList<Encoder<?, ?>>();
248             for (Class<? extends Encoder> s : m.method.getAnnotation(Message.class).encoders()) {
249                 try {
250                     l.add(config.framework().newClassInstance(Encoder.class, s));
251                 } catch (Exception e) {
252                     logger.error("Unable to load encoder {}", s);
253                 }
254             }
255             encoders.put(m.method, l);
256         }
257 
258         if (onReadyMethod != null) {
259             List<Encoder<?, ?>> l = new CopyOnWriteArrayList<Encoder<?, ?>>();
260             for (Class<? extends Encoder> s : onReadyMethod.getAnnotation(Ready.class).encoders()) {
261                 try {
262                     l.add(config.framework().newClassInstance(Encoder.class, s));
263                 } catch (Exception e) {
264                     logger.error("Unable to load encoder {}", s);
265                 }
266             }
267             encoders.put(onReadyMethod, l);
268         }
269     }
270 
271     private void populateDecoders() {
272         for (MethodInfo m : onRuntimeMethod) {
273             List<Decoder<?, ?>> l = new CopyOnWriteArrayList<Decoder<?, ?>>();
274             for (Class<? extends Decoder> s : m.method.getAnnotation(Message.class).decoders()) {
275                 try {
276                     l.add(config.framework().newClassInstance(Decoder.class, s));
277                 } catch (Exception e) {
278                     logger.error("Unable to load encoder {}", s);
279                 }
280             }
281             decoders.put(m.method, l);
282         }
283     }

285     private Object invoke(Method m, Object o) {
286         if (m != null) {
287             try {
288                 return m.invoke(proxiedInstance, o == null ? new Object[]{} : new Object[]{o});
289             } catch (IllegalAccessException e) {
290                 logger.debug("", e);
291             } catch (InvocationTargetException e) {
292                 logger.debug("", e);
293             }
294         }
295         logger.trace("No Method Mapped for {}", o);
296         return null;
297     }

299     private Object message(AtmosphereResource resource, Object o) {
300         try {
301             for (MethodInfo m : onRuntimeMethod) {
302 
303                 if (m.useReader) {
304                     o = resource.getRequest().getReader();
305                 } else if (m.useStream) {
306                     o = resource.getRequest().getInputStream();
307                 } else if (o == null) {
308                     o = IOUtils.readEntirely(resource).toString();
309                     if (String.class.cast(o).isEmpty()) {
310                         logger.warn("{} received an empty body", ManagedServiceInterceptor.class.getSimpleName());
311                         return null;
312                     }
313                 }
314 
315                 Object decoded = Invoker.decode(decoders.get(m.method), o);
316                 if (decoded == null) {
317                     decoded = o;
318                 }
319                 Object objectToEncode = null;
320 
321                 if (m.method.getParameterTypes().length > 2) {
322                     logger.warn("Injection of more than 2 parameters not supported {}", m);
323                 }
324 
325                 if (m.method.getParameterTypes().length == 2) {
326                     objectToEncode = Invoker.invokeMethod(m.method, proxiedInstance, resource, decoded);
327                 } else {
328                     objectToEncode = Invoker.invokeMethod(m.method, proxiedInstance, decoded);
329                 }
330                 if (objectToEncode != null) {
331                     return Invoker.encode(encoders.get(m.method), objectToEncode);
332                 }
333             }
334         } catch (Throwable t) {
335             logger.error("", t);
336         }
337         return null;
338     }
339 
340     private Object message(Method m, Object o) {
341         if (m != null) {
342             return Invoker.all(encoders.get(m), EMPTY, o, proxiedInstance, m);
343         }
344         return null;
345     }

347     @Override
348     public Object target() {
349         return proxiedInstance;

352     protected void processReady(AtmosphereResource r) {
353         Object o = message(onReadyMethod, r);
354 
355         switch (onReadyMethod.getAnnotation(Ready.class).value()) {
356             case RESOURCE:
357                 if (o != null) {
358                     if (String.class.isAssignableFrom(o.getClass())) {
359                         r.write(o.toString());
360                     } else if (byte[].class.isAssignableFrom(o.getClass())) {
361                         r.write((byte[]) o);
362                     }
363                 }
364                 break;
365             case BROADCASTER:
366                 r.getBroadcaster().broadcast(o);
367                 break;
368             case ALL:
369                 for (Broadcaster b : r.getAtmosphereConfig().getBroadcasterFactory().lookupAll()) {
370                     b.broadcast(o);
371                 }
372                 break;
373 
374         }
375     }

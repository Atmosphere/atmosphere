16  package org.atmosphere.config.managed;
17  
18  import org.atmosphere.config.service.MeteorService;
19  import org.atmosphere.config.service.Singleton;
20  import org.atmosphere.cpr.AtmosphereFramework;
21  import org.atmosphere.cpr.AtmosphereRequest;
22  import org.atmosphere.cpr.AtmosphereResourceImpl;
23  import org.atmosphere.cpr.FrameworkConfig;
24  import org.atmosphere.handler.ReflectorServletProcessor;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import javax.inject.Named;
29  import javax.servlet.Servlet;
30  
31  /**
32   * Handle {@link org.atmosphere.config.service.Singleton},{@link org.atmosphere.config.service.MeteorService}
33   * processing.
34   *
35   * @author Jeanfrancois Arcand
36   */
37  public class MeteorServiceInterceptor extends ServiceInterceptor {
38  
39      private final static Logger logger = LoggerFactory.getLogger(MeteorServiceInterceptor.class);
40  
41      protected void mapAnnotatedService(boolean reMap, String path, AtmosphereRequest request, AtmosphereFramework.AtmosphereHandlerWrapper w) {
42          synchronized (config.handlers()) {
43              if (config.handlers().get(path) == null) {
44                  // MeteorService
45                  if (ReflectorServletProcessor.class.isAssignableFrom(w.atmosphereHandler.getClass())) {
46                      ReflectorServletProcessor r = ReflectorServletProcessor.class.cast(w.atmosphereHandler);
47                      Servlet s = r.getServlet();
48                      MeteorService m = s.getClass().getAnnotation(MeteorService.class);
49                      if (m != null) {
50                          String targetPath = m.path();
51                          if (targetPath.indexOf("{") != -1 && targetPath.indexOf("}") != -1) {
52                              try {
53                                  boolean singleton = s.getClass().getAnnotation(Singleton.class) != null;
54                                  if (!singleton) {
55                                      r = config.framework().newClassInstance(ReflectorServletProcessor.class, ReflectorServletProcessor.class);
56                                      r.setServlet(config.framework().newClassInstance(Servlet.class, s.getClass()));
57                                      r.init(config);
58                                  }
59  
60                                  request.localAttributes().put(Named.class.getName(), path.substring(targetPath.indexOf("{")));
61  
62                                  AtmosphereResourceImpl.class.cast(request.resource()).atmosphereHandler(r);
63  
64                                  config.framework().addAtmosphereHandler(path, r,
65                                          config.getBroadcasterFactory().lookup(w.broadcaster.getClass(), path, true), w.interceptors);
66                                  request.setAttribute(FrameworkConfig.NEW_MAPPING, "true");
67                              } catch (Throwable e) {
68                                  logger.warn("Unable to create AtmosphereHandler", e);
69                              }
70                          }
71                      }
72                  }
73              } else if (reMap) {
74                  request.setAttribute(FrameworkConfig.NEW_MAPPING, "true");
75              }
76          }
77      }
78  
79      @Override
80      public String toString() {
81          return "@MeteorService Interceptor";
82      }
83  }

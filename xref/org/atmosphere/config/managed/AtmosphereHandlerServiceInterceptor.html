16  package org.atmosphere.config.managed;
17  
18  import org.atmosphere.config.service.AtmosphereHandlerService;
19  import org.atmosphere.config.service.Singleton;
20  import org.atmosphere.cpr.AtmosphereFramework.AtmosphereHandlerWrapper;
21  import org.atmosphere.cpr.AtmosphereHandler;
22  import org.atmosphere.cpr.AtmosphereRequest;
23  import org.atmosphere.cpr.AtmosphereResourceImpl;
24  import org.atmosphere.cpr.FrameworkConfig;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import javax.inject.Named;
29  
30  /**
31   * Handle {@link org.atmosphere.config.service.Singleton},{@link org.atmosphere.config.service.MeteorService} and
32   * {@link org.atmosphere.config.service.AtmosphereHandlerService} processing.
33   *
34   * @author Jeanfrancois Arcand
35   */
36  public class AtmosphereHandlerServiceInterceptor extends ServiceInterceptor {
37  
38      private final static Logger logger = LoggerFactory.getLogger(AtmosphereHandlerServiceInterceptor.class);
39  
40      protected void mapAnnotatedService(boolean reMap, String path, AtmosphereRequest request, AtmosphereHandlerWrapper w) {
41          synchronized (config.handlers()) {
42              if (config.handlers().get(path) == null) {
43                  // AtmosphereHandlerService
44                  AtmosphereHandlerService m = w.atmosphereHandler.getClass().getAnnotation(AtmosphereHandlerService.class);
45                  if (m != null) {
46                      try {
47                          String targetPath = m.path();
48                          if (targetPath.indexOf("{") != -1 && targetPath.indexOf("}") != -1) {
49                              boolean singleton = w.atmosphereHandler.getClass().getAnnotation(Singleton.class) != null;
50                              AtmosphereHandler newW = w.atmosphereHandler;
51  
52                              if (!singleton) {
53                                  newW = config.framework().newClassInstance(AtmosphereHandler.class, w.atmosphereHandler.getClass());
54                              }
55  
56                              request.localAttributes().put(Named.class.getName(), path.substring(targetPath.indexOf("{")));
57  
58                              AtmosphereResourceImpl.class.cast(request.resource()).atmosphereHandler(newW);
59  
60                              config.framework().addAtmosphereHandler(path, newW,
61                                      config.getBroadcasterFactory().lookup(w.broadcaster.getClass(), path, true), w.interceptors);
62                              request.setAttribute(FrameworkConfig.NEW_MAPPING, "true");
63                          }
64                      } catch (Throwable e) {
65                          logger.warn("Unable to create AtmosphereHandler", e);
66                      }
67                  }
68  
69              }
70          }
71      }
72  
73      @Override
74      public String toString() {
75          return "@AtmosphereHandlerService Interceptor";
76      }
77  }

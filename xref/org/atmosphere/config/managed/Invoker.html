16  package org.atmosphere.config.managed;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  import java.lang.reflect.InvocationTargetException;
22  import java.lang.reflect.Method;
23  import java.util.List;
24  
25  /**
26   * Invoke a method based on {@link Encoder} and {@link Decoder}
27   *
28   * @author Jeanfrancois Arcand
29   */
30  public class Invoker {
31  
32      private final static Logger logger = LoggerFactory.getLogger(Invoker.class);
33  
34      public static Object decode(
35              List<Decoder<?, ?>> decoders,
36              Object instanceType) {
37  
38          Object decodedObject = matchDecoder(instanceType, decoders);
39          if (instanceType == null) {
40              logger.trace("No Encoder matching {}", instanceType);
41          }
42          return decodedObject;
43      }
44  
45      public static Object invokeMethod(Method method, Object objectToInvoke, Object ... parameters) {
46          Object objectToEncode = null;
47          boolean hasMatch = false;
48          try {
49              objectToEncode = method.invoke(objectToInvoke, method.getParameterTypes().length == 0 ? new Object[]{} : parameters);
50              hasMatch = true;
51          } catch (IllegalAccessException e) {
52              logger.trace("", e);
53          } catch (InvocationTargetException e) {
54              logger.error("", e);
55          } catch (java.lang.IllegalArgumentException e) {
56              logger.trace("", e);
57          } catch (Throwable e) {
58              logger.error("", e);
59          }
60  
61          if (!hasMatch) {
62              logger.trace("No Method's Arguments {} matching {}", method.getName(), objectToInvoke);
63          }
64          return objectToEncode;
65      }
66  
67      public static Object encode(List<Encoder<?, ?>> encoders, Object objectToEncode) {
68          Object encodedObject = matchEncoder(objectToEncode, encoders);
69          if (encodedObject == null) {
70              logger.trace("No Encoder matching {}", objectToEncode);
71          }
72          return encodedObject;
73      }
74  
75      public static Object all(
76              List<Encoder<?, ?>> encoders,
77              List<Decoder<?, ?>> decoders,
78              Object instanceType,
79              Object objectToInvoke,
80              Method method) {
81  
82          Object decodedObject = decode(decoders, instanceType);
83          if (instanceType == null) {
84              logger.trace("No Encoder matching {}", instanceType);
85          }
86          decodedObject = decodedObject == null ? instanceType : decodedObject;
87  
88          logger.trace("{} .on {}", method.getName(), decodedObject);
89          Object objectToEncode = invokeMethod(method, objectToInvoke, decodedObject);
90  
91          Object encodedObject = null;
92          if (objectToEncode != null) {
93              encodedObject = encode(encoders, objectToEncode);
94          }
95          return encodedObject == null ? objectToEncode : encodedObject;
96      }
97  
98      public static Object matchDecoder(Object instanceType, List<Decoder<?, ?>> decoders) {
99          Object decodedObject = decoders.isEmpty() ? instanceType : null;
100         for (Decoder d : decoders) {
101             Class<?>[] typeArguments = TypeResolver.resolveArguments(d.getClass(), Decoder.class);
102             if (instanceType != null && typeArguments.length > 0 && typeArguments[0].isAssignableFrom(instanceType.getClass())) {
103 
104                 logger.trace("{} is trying to decode {}", d, instanceType);
105                 try {
106                     decodedObject = d.decode(instanceType);
107                 } catch (Exception e) {
108                     logger.trace("", e);
109                 }
110             }
111         }
112         return decodedObject;
113     }
114 
115     public static Object matchEncoder(Object instanceType, List<Encoder<?, ?>> encoders) {
116         if (instanceType == null) return null;
117 
118         Object encodedObject = encoders.isEmpty() ? instanceType : null;
119         for (Encoder d : encoders) {
120             Class<?>[] typeArguments = TypeResolver.resolveArguments(d.getClass(), Encoder.class);
121             if (instanceType != null && typeArguments.length > 0 && typeArguments[0].isAssignableFrom(instanceType.getClass())) {
122                 logger.trace("{} is trying to encode {}", d, instanceType);
123                 encodedObject = d.encode(instanceType);
124             }
125         }
126         return encodedObject;
127     }
128 }

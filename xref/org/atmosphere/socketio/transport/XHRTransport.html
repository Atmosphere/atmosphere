93          @Override
94          public void sendMessage(List<SocketIOPacketImpl> messages) throws SocketIOException {
95              if (messages != null) {
96                  for (SocketIOPacketImpl msg : messages) {
97                      switch (msg.getFrameType()) {
98                          case MESSAGE:
99                          case JSON:
100                         case EVENT:
101                         case ACK:
102                         case ERROR:
103                             msg.setPadding(messages.size() > 1);
104                             try {
105                                 sendMessage(msg.toString());
106                             } catch (Exception e) {
107 
108                                 AtmosphereResourceImpl resource = session.getAtmosphereResourceImpl();
109 
110                                 // if BroadcastCache is available, add the message to the cache
111                                 if (resource != null && DefaultBroadcaster.class.isAssignableFrom(resource.getBroadcaster().getClass())) {
112 
113                                     DefaultBroadcaster.class.cast(resource.getBroadcaster()).broadcasterCache.addToCache(resource, msg);
114 
115                                 }
116                             }
117                             break;
118                         default:
119                             logger.error("Unknown SocketIOEvent msg = " + msg);
120                     }
121                 }
122             }
123         }

125         @Override
126         public void sendMessage(String message) throws SocketIOException {
127             logger.trace("Session[" + session.getSessionId() + "]: " + "sendMessage(String): " + message);
128 
129             synchronized (this) {
130                 if (is_open) {
131 
132                     AtmosphereResourceImpl resource = session.getAtmosphereResourceImpl();
133 
134                     logger.trace("Session[" + session.getSessionId() + "]: " + resource.getRequest().getMethod() + "sendMessage");
135 
136                     try {
137                         writeData(resource.getResponse(), message);
138                     } catch (Exception e) {
139                         if (!resource.isCancelled()) {
140                             logger.trace("calling from " + this.getClass().getName() + " : " + "sendMessage ON FORCE UN RESUME");
141                             try {
142                                 finishSend(resource.getResponse());
143                             } catch (IOException ex) {
144                                 //ex.printStackTrace();
145                             }
146 
147                             resource.resume();
148                         }
149                         throw new SocketIOException(e);
150                     }
151                     if (!isStreamingConnection) {
152                         try {
153                             finishSend(resource.getResponse());
154                         } catch (IOException e) {
155                             //e.printStackTrace();
156                         }
157                         resource.resume();
158                     } else {
159                         logger.trace("calling from " + this.getClass().getName() + " : " + "sendMessage");
160                         session.startHeartbeatTimer();
161                     }
162                 } else {
163                     logger.trace("calling from " + this.getClass().getName() + " : " + "SocketIOClosedException sendMessage");
164                     throw new SocketIOClosedException();
165                 }
166             }

170         @Override
171         public Action handle(AtmosphereRequest request, final AtmosphereResponse response, SocketIOSession session) throws IOException {
172             logger.trace("Session id[" + session.getSessionId() + "] method=" + request.getMethod() + "  response HashCode=" + response.hashCode());
173 
174             AtmosphereResourceImpl resource = (AtmosphereResourceImpl) request.getAttribute(ApplicationConfig.ATMOSPHERE_RESOURCE);
175 
176             if ("GET".equals(request.getMethod())) {
177                 synchronized (this) {
178 
179 
180                     if (!is_open) {
181                         response.sendError(HttpServletResponse.SC_NOT_FOUND);
182                     } else {
183                         if (!isStreamingConnection) {
184 
185                             if (resource != null) {
186 
187                                 resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_ID, session.getSessionId());
188 
189                                 // for the Broadcaster
190                                 resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_OUTBOUND, session.getTransportHandler());
191 
192                                 session.setAtmosphereResourceImpl(resource);
193 
194                                 resource.addEventListener(new AtmosphereResourceEventListener() {
195 
196                                     @Override
197                                     public void onThrowable(AtmosphereResourceEvent event) {
198                                     }
199 
200                                     @Override
201                                     public void onSuspend(AtmosphereResourceEvent event) {
202                                     }
203 
204                                     @Override
205                                     public void onResume(AtmosphereResourceEvent event) {
206                                         if (event.isResumedOnTimeout()) {
207                                             try {
208                                                 event.getResource().write(response.getOutputStream(), new SocketIOPacketImpl(PacketType.NOOP).toString());
209                                             } catch (IOException e) {
210                                                 e.printStackTrace();
211                                             }
212                                         }
213                                     }
214 
215                                     @Override
216                                     public void onDisconnect(AtmosphereResourceEvent event) {
217                                     }
218 
219                                     @Override
220                                     public void onBroadcast(AtmosphereResourceEvent event) {
221                                     }
222                                 });
223 
224                                 session.clearTimeoutTimer();
225                                 request.setAttribute(SESSION_KEY, session);
226 
227                                 StringBuilder data = new StringBuilder();
228 
229                                 // if there is a Broadcaster cache, retrieve the messages from the cache, and send them
230                                 if (DefaultBroadcaster.class.isAssignableFrom(resource.getBroadcaster().getClass())) {
231 
232                                     List<Object> cachedMessages = DefaultBroadcaster.class.cast(resource.getBroadcaster()).broadcasterCache.retrieveFromCache(resource);
233 
234                                     if (cachedMessages != null) {
235                                         if (cachedMessages.size() > 1) {
236                                             for (Object object : cachedMessages) {
237                                                 String msg = object.toString();
238                                                 data.append(SocketIOPacketImpl.SOCKETIO_MSG_DELIMITER).append(msg.length()).append(SocketIOPacketImpl.SOCKETIO_MSG_DELIMITER).append(msg);
239                                             }
240                                         } else if (cachedMessages.size() == 1) {
241                                             data.append(cachedMessages.get(0));
242                                         }
243                                     }
244 
245                                     // something to send ?
246                                     if (data.toString().length() > 0) {
247                                         startSend(response);
248                                         writeData(response, data.toString());
249                                         finishSend(response);
250 
251                                         // force a resume, because we sent data
252                                         resource.resume();
253                                     } else {
254                                         resource.suspend(session.getRequestSuspendTime(), false);
255                                     }
256                                 } else {
257                                     // suspend the request
258 
259                                     resource.suspend(session.getRequestSuspendTime(), false);
260                                     resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_ID, session.getSessionId());
261 
262                                     // for the Broadcaster
263                                     resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_OUTBOUND, session.getTransportHandler());
264 
265                                     session.setAtmosphereResourceImpl(resource);
266                                 }
267 
268 
269                             }
270 
271 
272                         } else {
273                             // won't happend, by should be for xhr-streaming transport
274                             response.sendError(HttpServletResponse.SC_NOT_FOUND);
275                         }
276                     }
277                 }
278             } else if ("POST".equals(request.getMethod())) {
279                 if (is_open) {
280                     int size = request.getContentLength();
281                     if (size == 0) {
282                         response.sendError(HttpServletResponse.SC_BAD_REQUEST);
283                     } else {
284                         String data = (String) request.getAttribute(POST_MESSAGE_RECEIVED);
285                         if (data == null) {
286                             data = decodePostData(request.getContentType(), extractString(request.getReader()));
287                         }
288                         if (data != null && data.length() > 0) {
289 
290                             List<SocketIOPacketImpl> list = SocketIOPacketImpl.parse(data);
291 
292                             synchronized (session) {
293                                 for (SocketIOPacketImpl msg : list) {
294 
295                                     if (msg.getFrameType().equals(SocketIOPacketImpl.PacketType.EVENT)) {
296 
297                                         // send message on the suspended request
298                                         session.onMessage(session.getAtmosphereResourceImpl(), session.getTransportHandler(), msg.getData());
299                                         writeData(response, SocketIOPacketImpl.POST_RESPONSE);
300 
301                                     } else {
302                                         writeData(response, SocketIOPacketImpl.POST_RESPONSE);
303                                     }
304 
305                                 }
306                             }
307                         }
308                     }
309 
310                     // force a resume on a POST request
311                     resource.resume();
312                     return Action.CANCELLED;
313                 }
314             } else {
315                 response.sendError(HttpServletResponse.SC_BAD_REQUEST);
316             }
317 
318 
319             return Action.CONTINUE;
320

325         public void connect(AtmosphereResourceImpl resource, AtmosphereHandler atmosphereHandler) throws IOException {
326 
327             HttpServletRequest request = resource.getRequest();
328             HttpServletResponse response = resource.getResponse();
329 
330             request.setAttribute(SESSION_KEY, session);
331             response.setBufferSize(bufferSize);
332 
333             customConnect(request, response);
334             is_open = true;
335             session.onConnect(resource, this);
336             finishSend(response);
337 
338             if (isStreamingConnection) {
339                 resource.suspend();
340             }
341 
342 
343         }

345         @Override
346         public void abort() {
347             logger.error("calling from " + this.getClass().getName() + " : " + "abort");
348             session.clearHeartbeatTimer();
349             session.clearTimeoutTimer();
350             is_open = false;
351             session.onShutdown();
352 
353             // force a resume
354             session.getAtmosphereResourceImpl().resume();
355         }

373     protected SocketIOSession connect(SocketIOSession session, AtmosphereResourceImpl resource, AtmosphereHandler atmosphereHandler, SocketIOSessionFactory sessionFactory) throws IOException {
374 
375         if (session == null) {
376             session = sessionFactory.createSession(resource, atmosphereHandler);
377             resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_ID, session.getSessionId());
378 
379             // for the Broadcaster
380             resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_OUTBOUND, atmosphereHandler);
381         }
382 
383         XHRSessionHelper handler = createHelper(session);
384         handler.connect(resource, atmosphereHandler);
385         return session;
386     }

392     @Override
393     public Action handle(AtmosphereResourceImpl resource, AtmosphereHandler atmosphereHandler, SocketIOSessionFactory sessionFactory) throws IOException {
394 
395         AtmosphereRequest request = resource.getRequest();
396         AtmosphereResponse response = resource.getResponse();
397 
398         Object obj = request.getAttribute(SESSION_KEY);
399         SocketIOSession session = null;
400         String sessionId = null;
401         if (obj != null) {
402             session = (SocketIOSession) obj;
403         } else {
404             sessionId = extractSessionId(request);
405             if (sessionId != null && sessionId.length() > 0) {
406                 session = sessionFactory.getSession(sessionId);
407             }
408         }
409 
410         boolean isDisconnectRequest = isDisconnectRequest(request);
411 
412         if (session != null) {
413             SocketIOSessionOutbound handler = session.getTransportHandler();
414             if (handler != null) {
415                 if (!isDisconnectRequest) {
416                     handler.handle(request, response, session);
417                 } else {
418                     handler.disconnect();
419                     response.setStatus(200);
420                 }
421             } else {
422                 if (!isDisconnectRequest) {
423                     // handle the Connect
424                     session = connect(session, resource, atmosphereHandler, sessionFactory);
425                     if (session == null) {
426                         response.sendError(HttpServletResponse.SC_SERVICE_UNAVAILABLE);
427                     }
428                 } else {
429                     response.setStatus(200);
430                 }
431             }
432         } else if (sessionId != null && sessionId.length() > 0) {
433             logger.trace("Session NULL but not sessionId : wrong session id or the connection was DISCONNECTED");
434             response.sendError(HttpServletResponse.SC_BAD_REQUEST);
435         } else {
436             if ("GET".equals(request.getMethod())) {
437                 session = connect(resource, atmosphereHandler, sessionFactory);
438                 if (session == null) {
439                     response.sendError(HttpServletResponse.SC_SERVICE_UNAVAILABLE);
440                 }
441             } else {
442                 response.sendError(HttpServletResponse.SC_BAD_REQUEST);
443             }
444         }
445 
446         return Action.CONTINUE;
447     }

89          @Override
90          public void sendMessage(List<SocketIOPacketImpl> messages) throws SocketIOException {
91              if (messages != null) {
92                  for (SocketIOPacketImpl msg : messages) {
93                      switch (msg.getFrameType()) {
94                          case MESSAGE:
95                          case JSON:
96                          case EVENT:
97                          case ACK:
98                          case ERROR:
99                              msg.setPadding(messages.size() > 1);
100                             try {
101                                 sendMessage(msg.toString());
102                             } catch (Exception e) {
103 
104                                 AtmosphereResourceImpl resource = session.getAtmosphereResourceImpl();
105                                 // if BroadcastCache is available, add the message to the cache
106                                 if (resource != null && DefaultBroadcaster.class.isAssignableFrom(resource.getBroadcaster().getClass())) {
107                                     DefaultBroadcaster.class.cast(resource.getBroadcaster()).broadcasterCache.addToCache(resource.getBroadcaster().getID(), resource, msg);
108                                 }
109                             }
110                             break;
111                         default:
112                             logger.error("Unknown SocketIOEvent msg = " + msg);
113                     }
114                 }
115             }
116         }

118         @Override
119         public void sendMessage(String message) throws SocketIOException {
120             logger.trace("Session[" + session.getSessionId() + "]: " + "sendMessage(String): " + message);
121 
122             synchronized (this) {
123                 if (is_open) {
124 
125                     AtmosphereResourceImpl resource = session.getAtmosphereResourceImpl();
126 
127                     logger.trace("Session[" + session.getSessionId() + "]: " + resource.getRequest().getMethod() + "sendMessage");
128 
129                     try {
130                         writeData(resource.getResponse(), message);
131                     } catch (Exception e) {
132                         if (!resource.isCancelled()) {
133                             logger.trace("calling from " + this.getClass().getName() + " : " + "sendMessage ON FORCE UN RESUME");
134                             try {
135                                 finishSend(resource.getResponse());
136                             } catch (IOException ex) {
137                                 logger.trace("", ex);
138                             }
139 
140                             resource.resume();
141                         }
142                         throw new SocketIOException(e);
143                     }
144                     if (!isStreamingConnection) {
145                         try {
146                             finishSend(resource.getResponse());
147                         } catch (IOException e) {
148                             logger.trace("", e);
149                         }
150                         resource.resume();
151                     } else {
152                         logger.trace("calling from " + this.getClass().getName() + " : " + "sendMessage");
153                         session.startHeartbeatTimer();
154                     }
155                 } else {
156                     logger.trace("calling from " + this.getClass().getName() + " : " + "SocketIOClosedException sendMessage");
157                     throw new SocketIOClosedException();
158                 }
159             }

163         @Override
164         public Action handle(AtmosphereRequest request, final AtmosphereResponse response, SocketIOSession session) throws IOException {
165             logger.trace("Session id[" + session.getSessionId() + "] method=" + request.getMethod() + "  response HashCode=" + response.hashCode());
166 
167             AtmosphereResourceImpl resource = (AtmosphereResourceImpl) request.getAttribute(ApplicationConfig.ATMOSPHERE_RESOURCE);
168 
169             if ("GET".equals(request.getMethod())) {
170                 synchronized (this) {
171                     if (!is_open) {
172                         response.sendError(AtmosphereResponse.SC_NOT_FOUND);
173                     } else {
174                         if (!isStreamingConnection) {
175                             if (resource != null) {
176                                 resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_ID, session.getSessionId());
177                                 resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_OUTBOUND, session.getTransportHandler());
178                                 session.setAtmosphereResourceImpl(resource);
179 
180                                 resource.addEventListener(new AtmosphereResourceEventListenerAdapter() {
181                                     @Override
182                                     public void onResume(AtmosphereResourceEvent event) {
183                                         if (event.isResumedOnTimeout()) {
184                                             try {
185                                                 event.getResource().write(response.getOutputStream(), new SocketIOPacketImpl(PacketType.NOOP).toString());
186                                             } catch (IOException e) {
187                                                 logger.trace("", e);
188                                             }
189                                         }
190                                     }
191                                 });
192 
193                                 session.clearTimeoutTimer();
194                                 request.setAttribute(SESSION_KEY, session);
195 
196                                 StringBuilder data = new StringBuilder();
197                                 // if there is a Broadcaster cache, retrieve the messages from the cache, and send them
198                                 if (DefaultBroadcaster.class.isAssignableFrom(resource.getBroadcaster().getClass())) {
199 
200                                     List<Object> cachedMessages = DefaultBroadcaster.class.cast(resource.getBroadcaster())
201                                             .broadcasterCache.retrieveFromCache(resource.getBroadcaster().getID(), resource);
202 
203                                     if (cachedMessages != null) {
204                                         if (cachedMessages.size() > 1) {
205                                             for (Object object : cachedMessages) {
206                                                 String msg = object.toString();
207                                                 data.append(SocketIOPacketImpl.SOCKETIO_MSG_DELIMITER)
208                                                         .append(msg.length())
209                                                         .append(SocketIOPacketImpl.SOCKETIO_MSG_DELIMITER).append(msg);
210                                             }
211                                         } else if (cachedMessages.size() == 1) {
212                                             data.append(cachedMessages.get(0));
213                                         }
214                                     }
215 
216                                     // something to send ?
217                                     if (data.toString().length() > 0) {
218                                         startSend(response);
219                                         writeData(response, data.toString());
220                                         finishSend(response);
221 
222                                         // force a resume, because we sent data
223                                         resource.resume();
224                                     }
225                                 }
226 
227                                 resource.disableSuspend(false);
228                                 resource.suspend(session.getRequestSuspendTime(), false);
229                                 resource.disableSuspend(true);
230                             }
231                         } else {
232                             // won't happend, by should be for xhr-streaming transport
233                             response.sendError(AtmosphereResponse.SC_NOT_FOUND);
234                         }
235                     }
236                 }
237             } else if ("POST".equals(request.getMethod())) {
238                 if (is_open) {
239                     int size = request.getContentLength();
240                     if (size == 0) {
241                         response.sendError(AtmosphereResponse.SC_BAD_REQUEST);
242                     } else {
243                         String data = (String) request.getAttribute(POST_MESSAGE_RECEIVED);
244                         if (data == null) {
245                             data = decodePostData(request.getContentType(), extractString(request.getReader()));
246                         }
247                         if (data != null && data.length() > 0) {
248 
249                             List<SocketIOPacketImpl> list = SocketIOPacketImpl.parse(data);
250 
251                             synchronized (session) {
252                                 for (SocketIOPacketImpl msg : list) {
253 
254                                     if (msg.getFrameType().equals(SocketIOPacketImpl.PacketType.EVENT)) {
255 
256                                         // send message on the suspended request
257                                         session.onMessage(session.getAtmosphereResourceImpl(), session.getTransportHandler(), msg.getData());
258                                         writeData(response, SocketIOPacketImpl.POST_RESPONSE);
259 
260                                     } else {
261                                         writeData(response, SocketIOPacketImpl.POST_RESPONSE);
262                                     }
263                                 }
264                             }
265                         }
266                     }
267                     // force a resume on a POST request
268                     resource.resume();
269                 }
270             } else {
271                 response.sendError(AtmosphereResponse.SC_BAD_REQUEST);
272             }
273             return Action.CANCELLED;
274         }

278         public void connect(AtmosphereResourceImpl resource) throws IOException {
279 
280             AtmosphereRequest request = resource.getRequest();
281             AtmosphereResponse response = resource.getResponse();
282 
283             request.setAttribute(SESSION_KEY, session);
284             response.setBufferSize(bufferSize);
285 
286             customConnect(request, response);
287             is_open = true;
288             session.onConnect(resource, this);
289             finishSend(response);
290 
291             if (isStreamingConnection) {
292                 resource.suspend();
293             }
294         }

296         @Override
297         public void abort() {
298             logger.error("calling from " + this.getClass().getName() + " : " + "abort");
299             session.clearHeartbeatTimer();
300             session.clearTimeoutTimer();
301             is_open = false;
302             session.onShutdown();
303 
304             // force a resume
305             session.getAtmosphereResourceImpl().resume();
306         }

324     protected SocketIOSession connect(SocketIOSession session, AtmosphereResourceImpl resource,
325                                       AtmosphereHandler atmosphereHandler, SocketIOSessionFactory sessionFactory) throws IOException {
326 
327         if (session == null) {
328             session = sessionFactory.createSession(resource, atmosphereHandler);
329             resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_ID, session.getSessionId());
330         }
331 
332         XHRSessionHelper handler = createHelper(session);
333         handler.connect(resource);
334         return session;
335     }

341     @Override
342     public Action handle(AtmosphereResourceImpl resource, AtmosphereHandler atmosphereHandler, SocketIOSessionFactory sessionFactory) throws IOException {
343 
344         AtmosphereRequest request = resource.getRequest();
345         AtmosphereResponse response = resource.getResponse();
346 
347         Object obj = request.getAttribute(SESSION_KEY);
348         SocketIOSession session = null;
349         String sessionId = null;
350         if (obj != null) {
351             session = (SocketIOSession) obj;
352         } else {
353             sessionId = extractSessionId(request);
354             if (sessionId != null && sessionId.length() > 0) {
355                 session = sessionFactory.getSession(sessionId);
356             }
357         }
358 
359         boolean isDisconnectRequest = isDisconnectRequest(request);
360         Action action = Action.CONTINUE;
361         if (session != null) {
362             SocketIOSessionOutbound handler = session.getTransportHandler();
363             if (handler != null) {
364                 if (!isDisconnectRequest) {
365                     action = handler.handle(request, response, session);
366                 } else {
367                     handler.disconnect();
368                     response.setStatus(200);
369                 }
370             } else {
371                 if (!isDisconnectRequest) {
372                     // handle the Connect
373                     session = connect(session, resource, atmosphereHandler, sessionFactory);
374                     if (session == null) {
375                         response.sendError(AtmosphereResponse.SC_SERVICE_UNAVAILABLE);
376                     }
377                 } else {
378                     response.setStatus(200);
379                 }
380             }
381         } else if (sessionId != null && sessionId.length() > 0) {
382             logger.trace("Session NULL but not sessionId : wrong session id or the connection was DISCONNECTED");
383             response.sendError(AtmosphereResponse.SC_BAD_REQUEST);
384         } else {
385             if ("GET".equals(request.getMethod())) {
386                 session = connect(resource, atmosphereHandler, sessionFactory);
387                 if (session == null) {
388                     response.sendError(AtmosphereResponse.SC_SERVICE_UNAVAILABLE);
389                 }
390             } else {
391                 response.sendError(AtmosphereResponse.SC_BAD_REQUEST);
392             }
393         }
394 
395         return action;
396     }

44  /**
45   * This class can be used to suspend response programmatically, similar to {@link org.atmosphere.annotation.Suspend}
46   * annotation.
47   * <blockquote><pre>
48   *         SuspendResponse&lt;String&gt; r = new SuspendResponse.SuspendResponseBuilder&lt;String&gt;()
49   *              .broadcaster(broadcaster)
50   *              .outputComments(true)
51   *              .period(5, TimeUnit.SECONDS)
52   *              .entity("foo")
53   *              .build();
54   * </pre></blockquote>
55   *
56   * @param <E> the {@link org.atmosphere.jersey.SuspendResponse#entity type}
57   * @author Jeanfrancois Arcand
58   */
59  public class SuspendResponse<E> extends JResponse {
60      private final TimeSpan suspendTimeout;
61      private final Suspend.SCOPE scope;
62      private final boolean outputComments;
63      private final boolean resumeOnBroadcast;
64      private final Collection<AtmosphereResourceEventListener> listeners;
65      private final Broadcaster broadcaster;
66      private final boolean writeEntity;
67  
68      protected SuspendResponse(SuspendResponseBuilder<E> b) {
69          super(b);
70          this.suspendTimeout = b.suspendTimeout;
71          this.scope = b.scope;
72          this.outputComments = b.outputComments;
73          this.resumeOnBroadcast = b.resumeOnBroadcast;
74          this.listeners = b.listeners;
75          this.broadcaster = b.broadcaster;
76          this.writeEntity = b.writeEntity;
77      }
78

80       * Return the {@link org.atmosphere.annotation.Suspend.SCOPE} value.
81       *
82       * @return the {@link org.atmosphere.annotation.Suspend.SCOPE} value.
83       */
84      public Suspend.SCOPE scope() {
85          return scope;
86      }

145     public static class SuspendResponseBuilder<E> extends AJResponseBuilder<E, SuspendResponseBuilder<E>> {
146 
147         protected TimeSpan suspendTimeout = new TimeSpan(-1, TimeUnit.MILLISECONDS);
148         protected Suspend.SCOPE scope = Suspend.SCOPE.APPLICATION;
149         protected boolean outputComments = true;
150         protected boolean resumeOnBroadcast = false;
151         protected final Collection<AtmosphereResourceEventListener> listeners
152                 = new ArrayList<AtmosphereResourceEventListener>();
153         private Broadcaster broadcaster;
154         private boolean writeEntity = true;
155 
156         /**
157          * Default constructor.
158          */
159         public SuspendResponseBuilder() {
160         }

162         /**
163          * Construct a shallow copy. The metadata map will be copied but not the
164          * key/value references.
165          *
166          * @param that the AJResponseBuilder to copy from.
167          */
168         public SuspendResponseBuilder(SuspendResponseBuilder<E> that) {
169             super(that);
170         }
171 
172         /**
173          * Set the {@link org.atmosphere.annotation.Suspend.SCOPE} value
174          *

176          * @return this
177          */
178         public SuspendResponseBuilder<E> scope(Suspend.SCOPE scope) {
179             this.scope = scope;
180             return this;
181         }
182 
183         /**
184          * Set the timeout period.

187          * @param timeUnit       the {@link java.util.concurrent.TimeUnit}
188          * @return this

190         public SuspendResponseBuilder<E> period(int suspendTimeout, TimeUnit timeUnit) {

236         /**

240          * @return this
241          */
242         public SuspendResponseBuilder<E> addListener(AtmosphereResourceEventListener e) {
243             listeners.add(e);
244             return this;
245         }
246 
247         /**
248          * Build the {@link org.atmosphere.jersey.SuspendResponse}
249          *
250          * @return an instance of {@link org.atmosphere.jersey.SuspendResponse}
251          */
252         public SuspendResponse<E> build() {
253             SuspendResponse<E> r = new SuspendResponse<E>(this);
254             reset();
255             return r;
256         }

16  package org.atmosphere.jersey;
17  
18  import com.sun.jersey.api.JResponse;
19  import org.atmosphere.annotation.Suspend;
20  import org.atmosphere.cpr.AtmosphereResourceEventListener;
21  import org.atmosphere.cpr.Broadcaster;
22  
23  import java.util.ArrayList;
24  import java.util.Collection;
25  import java.util.Collections;
26  import java.util.concurrent.TimeUnit;
27  
28  /**
29   * This class can be used to suspend response programmatically, similar to {@link org.atmosphere.annotation.Suspend}
30   * annotation.
31   * <blockquote><pre>
32   *         SuspendResponse&lt;String&gt; r = new SuspendResponse.SuspendResponseBuilder&lt;String&gt;()
33   *              .broadcaster(broadcaster)
34   *              .outputComments(true)
35   *              .period(5, TimeUnit.SECONDS)
36   *              .entity("foo")
37   *              .build();
38   * </pre></blockquote>
39   *
40   * @param <E> the {@link org.atmosphere.jersey.SuspendResponse#entity type}
41   * @author Jeanfrancois Arcand
42   */
43  public class SuspendResponse<E> extends JResponse {
44      private final TimeSpan suspendTimeout;
45      private final Suspend.SCOPE scope;
46      private final boolean outputComments;
47      private final boolean resumeOnBroadcast;
48      private final Collection<AtmosphereResourceEventListener> listeners;
49      private final Broadcaster broadcaster;
50      private final boolean writeEntity;
51  
52      protected SuspendResponse(SuspendResponseBuilder<E> b) {
53          super(b);
54          this.suspendTimeout = b.suspendTimeout;
55          this.scope = b.scope;
56          this.outputComments = b.outputComments;
57          this.resumeOnBroadcast = b.resumeOnBroadcast;
58          this.listeners = b.listeners;
59          this.broadcaster = b.broadcaster;
60          this.writeEntity = b.writeEntity;
61      }
62  
63      /**
64       * Return the {@link org.atmosphere.annotation.Suspend.SCOPE} value.
65       *
66       * @return the {@link org.atmosphere.annotation.Suspend.SCOPE} value.
67       */
68      public Suspend.SCOPE scope() {
69          return scope;
70      }
71  
72      /**
73       * Return the {@link org.atmosphere.jersey.SuspendResponse.TimeSpan} used to suspend the response.
74       *
75       * @return the {@link org.atmosphere.jersey.SuspendResponse.TimeSpan} used to suspend the response.
76       */
77      public TimeSpan period() {
78          return suspendTimeout;
79      }
80  
81      /**
82       * Tell Atmosphere to write some comments during the connection suspension.
83       *
84       * @return true is comment will be written.
85       */
86      public boolean outputComments() {
87          return outputComments;
88      }
89  
90      /**
91       * Resume the connection on the first {@link org.atmosphere.cpr.Broadcaster#broadcast(Object)} operations.
92       *
93       * @return true if the connection needs to be resumed.
94       */
95      public boolean resumeOnBroadcast() {
96          return resumeOnBroadcast;
97      }
98  
99      /**
100      * Write the returned entity back to the calling connection. Default is false.
101      */
102     public boolean writeEntity() {
103         return writeEntity;
104     }
105 
106     /**
107      * Return the {@link Broadcaster} that will be used to broadcast events.
108      *
109      * @return the {@link Broadcaster} that will be used to broadcast events.
110      */
111     public Broadcaster broadcaster() {
112         return broadcaster;
113     }
114 
115     /**
116      * Return the current list of {@link AtmosphereResourceEventListener} classes.
117      *
118      * @return the current list of {@link AtmosphereResourceEventListener} classes.
119      */
120     public Collection<AtmosphereResourceEventListener> listeners() {
121         return Collections.unmodifiableCollection(listeners);
122     }
123 
124     /**
125      * A Builder for {@link org.atmosphere.jersey.SuspendResponse}
126      *
127      * @param <E>
128      */
129     public static class SuspendResponseBuilder<E> extends AJResponseBuilder<E, SuspendResponseBuilder<E>> {
130 
131         protected TimeSpan suspendTimeout = new TimeSpan(-1, TimeUnit.MILLISECONDS);
132         protected Suspend.SCOPE scope = Suspend.SCOPE.APPLICATION;
133         protected boolean outputComments = true;
134         protected boolean resumeOnBroadcast = false;
135         protected final Collection<AtmosphereResourceEventListener> listeners
136                 = new ArrayList<AtmosphereResourceEventListener>();
137         private Broadcaster broadcaster;
138         private boolean writeEntity = true;
139 
140         /**
141          * Default constructor.
142          */
143         public SuspendResponseBuilder() {
144         }
145 
146         /**
147          * Construct a shallow copy. The metadata map will be copied but not the
148          * key/value references.
149          *
150          * @param that the AJResponseBuilder to copy from.
151          */
152         public SuspendResponseBuilder(SuspendResponseBuilder<E> that) {
153             super(that);
154         }
155 
156         /**
157          * Set the {@link org.atmosphere.annotation.Suspend.SCOPE} value
158          *
159          * @param scope {@link org.atmosphere.annotation.Suspend.SCOPE} value
160          * @return this
161          */
162         public SuspendResponseBuilder<E> scope(Suspend.SCOPE scope) {
163             this.scope = scope;
164             return this;
165         }
166 
167         /**
168          * Set the timeout period.
169          *
170          * @param suspendTimeout the period
171          * @param timeUnit       the {@link java.util.concurrent.TimeUnit}
172          * @return this
173          */
174         public SuspendResponseBuilder<E> period(int suspendTimeout, TimeUnit timeUnit) {
175             this.suspendTimeout = new TimeSpan(suspendTimeout, timeUnit);
176             return this;
177         }
178 
179         /**
180          * Set true to tell Atmosphere to write comments when suspending.
181          *
182          * @param outputComments true to tell Atmosphere to write comments when suspending
183          * @return this
184          */
185         public SuspendResponseBuilder<E> outputComments(boolean outputComments) {
186             this.outputComments = outputComments;
187             return this;
188         }
189 
190         /**
191          * Set to true to resume the connection on the first {@link org.atmosphere.cpr.Broadcaster#broadcast(Object)}
192          *
193          * @param resumeOnBroadcast true to resume the connection on the first {@link org.atmosphere.cpr.Broadcaster#broadcast(Object)}
194          * @return this
195          */
196         public SuspendResponseBuilder<E> resumeOnBroadcast(boolean resumeOnBroadcast) {
197             this.resumeOnBroadcast = resumeOnBroadcast;
198             return this;
199         }
200 
201         /**
202          * Set the {@link Broadcaster}
203          *
204          * @param broadcaster {@link Broadcaster}
205          * @return this
206          */
207         public SuspendResponseBuilder<E> broadcaster(Broadcaster broadcaster) {
208             this.broadcaster = broadcaster;
209             return this;
210         }
211 
212         /**
213          * Write the returned entity back to the calling connection. Default is false.
214          */
215         public SuspendResponseBuilder<E> writeEntity(boolean writeEntity) {
216             this.writeEntity = writeEntity;
217             return this;
218         }
219 
220         /**
221          * Add {@link org.atmosphere.cpr.AtmosphereResourceEventListener}
222          *
223          * @param e {@link org.atmosphere.cpr.AtmosphereResourceEventListener}
224          * @return this
225          */
226         public SuspendResponseBuilder<E> addListener(AtmosphereResourceEventListener e) {
227             listeners.add(e);
228             return this;
229         }
230 
231         /**
232          * Build the {@link org.atmosphere.jersey.SuspendResponse}
233          *
234          * @return an instance of {@link org.atmosphere.jersey.SuspendResponse}
235          */
236         public SuspendResponse<E> build() {
237             SuspendResponse<E> r = new SuspendResponse<E>(this);
238             reset();
239             return r;
240         }
241     }
242 
243     /**
244      * Util class that encapsulate a period and a TimeUnit.
245      */
246     public static class TimeSpan {
247 
248         private final TimeUnit timeUnit;
249         private final int period;
250 
251         public TimeSpan(int period, TimeUnit timeUnit) {
252             this.period = period;
253             this.timeUnit = timeUnit;
254         }
255 
256         public int value() {
257             return period;
258         }
259 
260         public TimeUnit timeUnit() {
261             return timeUnit;
262         }
263     }

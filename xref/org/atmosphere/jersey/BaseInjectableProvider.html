38  package org.atmosphere.jersey;
39  
40  import com.sun.jersey.spi.inject.InjectableProvider;
41  import org.atmosphere.cpr.AtmosphereResource;
42  import org.atmosphere.cpr.FrameworkConfig;
43  
44  import javax.servlet.http.HttpServletRequest;
45  import javax.ws.rs.core.Context;
46  import java.lang.reflect.Type;
47  
48  /**
49   * @author Paul Sandoz
50   */
51  abstract class BaseInjectableProvider implements InjectableProvider<Context, Type> {
52      // The current {@link HttpServletRequest{
53      @Context
54      HttpServletRequest req;
55  
56      protected AtmosphereResource getAtmosphereResource(Class injectType, boolean session) {
57          AtmosphereResource r = null;
58  
59          try {
60              if (session) {
61                  if ((Boolean) req.getAttribute(FrameworkConfig.SUPPORT_SESSION)) {
62                      r = (AtmosphereResource) req.getSession().
63                              getAttribute(AtmosphereFilter.SUSPENDED_RESOURCE);
64                  }
65              }
66  
67              if (r == null) {
68                  r = (AtmosphereResource)
69                          req.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
70              }
71  
72              return r;
73          } catch (IllegalStateException ex) {
74              throw new IllegalStateException("An instance of the class " + injectType.getName() + " could not be injected because there is no HTTP request in scope", ex);
75          }
76      }
77  
78  }

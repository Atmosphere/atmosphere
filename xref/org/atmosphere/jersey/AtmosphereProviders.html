74  /**
75   * Placeholder for injection of Atmosphere object based on
76   * any parameter value (header, cookie, query, matrix or path)
77   *
78   * @author Paul.Sandoz@Sun.Com
79   * @author Jean-Francois Arcand
80   */
81  public class AtmosphereProviders {
82  
83      private static final Logger logger = LoggerFactory.getLogger(AtmosphereProviders.class);
84  
85      public static class BroadcasterProvider implements StringReaderProvider {
86  
87          @Context
88          HttpServletRequest req;
89  
90          @Override
91          public StringReader getStringReader(Class type, Type genericType, Annotation[] annotations) {
92  
93              if (Broadcaster.class.isAssignableFrom(type)) {
94                  return new BroadcasterStringReader();
95              }
96  
97              return null;
98          }
99  
100         @StringReader.ValidateDefaultValue(false)
101         public class BroadcasterStringReader implements StringReader {
102             @Override
103             public Object fromString(String topic) {
104                 Broadcaster broadcaster;
105                 try {
106                     AtmosphereResource r =
107                             (AtmosphereResource)
108                                     req.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
109                     BroadcasterFactory bp = (BroadcasterFactory)
110                             req.getAttribute(ApplicationConfig.BROADCASTER_FACTORY);
111 
112                     Class<? extends Broadcaster> c;
113                     try {
114                         c = (Class<Broadcaster>) Class.forName((String) req.getAttribute(ApplicationConfig.BROADCASTER_CLASS));
115                     } catch (Throwable e) {
116                         throw new IllegalStateException(e.getMessage());
117                     }
118                     broadcaster = bp.lookup(c, topic, true);
119                 } catch (Throwable ex) {
120                     throw new WebApplicationException(ex);
121                 }
122                 logger.trace("Injected Broadcaster {}", broadcaster);
123                 req.setAttribute(AtmosphereFilter.INJECTED_BROADCASTER, broadcaster);
124                 return broadcaster;
125             }
126         }
127     }

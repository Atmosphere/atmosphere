16  package org.atmosphere.jersey;
17  
18  import org.atmosphere.cpr.Trackable;
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  import java.util.concurrent.CountDownLatch;
23  import java.util.concurrent.TimeUnit;
24  
25  /**
26   * Associate any kind of resource to a @Suspend operation. When returning a {@link TrackableResource} from a method
27   * annotated with the suspend annotation, the atmosphere framework will automatically add the X-Atmosphere-tracking-id
28   * header to the response. The header can later be used for injecting {@link TrackableResource}. As simple as
29   * <blockquote><pre>
30   *
31   * @param <T>
32   * @GET
33   * @Suspend public TrackableResource suspend(){
34   * return new TrackableResource(AtmosphereResource.class, "abcdef", Response.OK());
35   * }
36   * @POST public String asyncBroadcast(@HeaderParam("X-Atmosphere-tracking-id") TrackableResource<AtmosphereResource> track) {
37   * AtmosphereResource<?,?> r = track.resource();
38   * ...
39   * }
40   * </blockquote><pre>
41   */
42  public class TrackableResource<T extends Trackable> {
43  
44      private static final Logger logger = LoggerFactory.getLogger(TrackableResource.class);
45      public static final String TRACKING_HEADER = "X-Atmosphere-tracking-id";
46  
47      private final Class<T> type;
48      private T resource;
49      private String trackingID = null;
50      private final Object entity;
51      private final CountDownLatch latch = new CountDownLatch(1);
52  
53      public TrackableResource(Class<T> type, Object entity) {
54          this.type = type;
55          this.entity = entity;
56      }
57  
58      public TrackableResource(Class<T> type, String trackingID, Object entity) {
59          this.type = type;
60          this.trackingID = trackingID;
61          this.entity = entity;
62      }
63  
64      protected void setResource(Trackable resource) {
65          if (!type.isAssignableFrom(resource.getClass())) {
66              throw new IllegalStateException(String.format("Unassignable %s to %s", type.toString(), resource.getClass().toString()));
67          }
68          latch.countDown();
69          this.resource = type.cast(resource);
70      }
71  
72      protected void setTrackingID(String trackingID) {
73          this.trackingID = trackingID;
74      }
75  
76      /**
77       * Return the associated resource of type T
78       *
79       * @return the associated resource of type T
80       */
81      public T resource() {
82          try {
83              latch.await(5, TimeUnit.SECONDS);
84          } catch (InterruptedException e) {
85              logger.trace("", e);
86          }
87          return resource;
88      }
89  
90      /**
91       * Retunr the class's type.
92       *
93       * @return
94       */
95      public Class<T> type() {
96          return type;
97      }
98  
99      /**
100      * Return the trackingID token associated with the resource.
101      *
102      * @return the trackingID token associated with the resource.
103      */
104     public String trackingID() {
105         return trackingID;
106     }
107 
108     /**
109      * Return the Entity associated with the resource.
110      *
111      * @return the Entity associated with the resource.
112      */
113     public Object entity() {
114         return entity;
115     }
116 }

190         boolean resumeOnBroadcast(boolean resumeOnBroadcast) {
191             String transport = servletReq.getHeader(X_ATMOSPHERE_TRANSPORT);
192             if (transport != null && (transport.equals(JSONP_TRANSPORT) || transport.equals(LONG_POLLING_TRANSPORT))) {
193                 return true;

195             return resumeOnBroadcast;

198         /**
199          * Configure the {@link AtmosphereResourceEvent} state (suspend, resume, broadcast)
200          * based on the annotation the web application has used.
201          *
202          * @param request  the {@link ContainerRequest}
203          * @param response the {@link ContainerResponse}
204          * @return the {@link ContainerResponse}
205          */
206         public ContainerResponse filter(final ContainerRequest request, final ContainerResponse response) {
207             if (response.getMappedThrowable() != null) {
208                 logger.debug("Unexpected exception", response.getMappedThrowable());
209                 return response;
210             }
211 
212             if (action == Action.NONE) return response;
213 
214             // Check first if something was defined in web.xml
215             AtmosphereConfig config = (AtmosphereConfig) servletReq.getAttribute(ATMOSPHERE_CONFIG);
216             if (config == null) {
217                 logger.error(INSTALLATION_ERROR);
218                 throw new WebApplicationException(new IllegalStateException(INSTALLATION_ERROR));
219             }
220 
221             AtmosphereFramework atmosphereFramework = config.framework();
222 
223             String p = config.getInitParameter(JERSEY_CONTAINER_RESPONSE_WRITER_CLASS);
224             ContainerResponseWriter w;
225             if (p != null) {
226                 try {
227                     w = (ContainerResponseWriter) Thread.currentThread().getContextClassLoader().loadClass(p).newInstance();
228                     logger.trace("Installing ContainerResponseWriter {}", p);
229                 } catch (Throwable e) {
230                     logger.error("Error loading ContainerResponseWriter {}", p, e);
231                 }
232             }
233 
234             // Now check if it was defined as an attribute
235             w = (ContainerResponseWriter) servletReq.getAttribute(FrameworkConfig.JERSEY_CONTAINER_RESPONSE_WRITER_INSTANCE);
236             if (w != null) {
237                 response.setContainerResponseWriter(w);
238             }
239 
240             AtmosphereResource r =
241                     (AtmosphereResource) servletReq
242                             .getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
243 
244             if (Boolean.parseBoolean(config.getInitParameter(SUPPORT_LOCATION_HEADER))) {
245                 useResumeAnnotation = true;
246             }
247 
248             // Force the status code to 200 events independently of the value of the entity (null or not)
249             if (response.getStatus() == 204) {
250                 response.setStatus(200);
251             }
252 
253             switch (action) {
254                 case ASYNCHRONOUS:
255                     String transport = getHeaderOrQueryValue(X_ATMOSPHERE_TRANSPORT);
256                     String broadcasterName = uuid(r);
257 
258                     if (!topic.equalsIgnoreCase(HeaderConfig.X_ATMOSPHERE_TRACKING_ID)) {
259                         broadcasterName = getHeaderOrQueryValue(topic);
260                     }
261 
262                     if (transport == null) {
263                         transport = HeaderConfig.LONG_POLLING_TRANSPORT;
264                     }
265 
266                     if (broadcasterName == null) {
267                         StringBuilder s = new StringBuilder();
268                         Enumeration<String> e = servletReq.getHeaderNames();
269                         String t;
270                         while (e.hasMoreElements()) {
271                             t = e.nextElement();
272                             s.append(t).append("=").append(servletReq.getHeader(t)).append("\n");
273                         }
274 
275                         logger.error("\nQueryString:\n{}\n\nHeaders:\n{}", servletReq.getQueryString(), s.toString());
276 
277                         throw new WebApplicationException(new IllegalStateException("Must specify transport using header value "
278                                 + transport
279                                 + " and uuid " + broadcasterName));
280                     }
281                     String subProtocol = (String) servletReq.getAttribute(FrameworkConfig.WEBSOCKET_SUBPROTOCOL);
282 
283                     final boolean waitForResource = waitFor == -1 ? true : false;
284                     Broadcaster newBroadcaster;
285                     // See issue https://github.com/Atmosphere/atmosphere/issues/676
286                     synchronized (broadcasterName.intern()) {
287                         newBroadcaster = config.getBroadcasterFactory().lookup(broadcasterName, true);
288                         newBroadcaster.setBroadcasterLifeCyclePolicy(BroadcasterLifeCyclePolicy.EMPTY_DESTROY);
289                     }
290                     final Broadcaster bcaster = newBroadcaster;
291 
292                     if (!waitForResource || (!transport.startsWith(POLLING_TRANSPORT) && subProtocol == null)) {
293                         final boolean resumeOnBroadcast = transport.equals(JSONP_TRANSPORT) || transport.equals(LONG_POLLING_TRANSPORT);
294 
295                         if (listeners != null) {
296                             for (Class<? extends AtmosphereResourceEventListener> listener : listeners) {
297                                 try {
298                                     AtmosphereResourceEventListener el = atmosphereFramework.newClassInstance(AtmosphereResourceEventListener.class, listener);
299                                     r.addEventListener(el);
300                                 } catch (Throwable t) {
301                                     throw new WebApplicationException(
302                                             new IllegalStateException("Invalid AtmosphereResourceEventListener " + listener));
303                                 }
304                             }
305                         }
306                         final Object entity = response.getEntity();
307 
308                         r.addEventListener(new OnSuspend() {
309                             @Override
310                             public void onSuspend(AtmosphereResourceEvent event) {
311                                 try {
312                                     if (entity != null) {
313                                         if (waitForResource) {
314                                             bcaster.awaitAndBroadcast(entity, 30, TimeUnit.SECONDS);
315                                         } else {
316                                             bcaster.broadcastOnResume(entity);
317                                             event.getResource().resume();
318                                         }
319                                     }
320                                 } finally {
321                                     event.getResource().removeEventListener(this);
322                                 }
323                             }
324                         });
325 
326                         if (resumeOnBroadcast) {
327                             servletReq.setAttribute(RESUME_ON_BROADCAST, new Boolean(true));
328                         }
329 
330                         r.setBroadcaster(bcaster);
331                         executeSuspend(r, timeout, resumeOnBroadcast, null, request, response, writeEntity);
332                     } else {
333                         Object entity = response.getEntity();
334                         if (waitForResource) {
335                             bcaster.awaitAndBroadcast(entity, 30, TimeUnit.SECONDS);
336                         } else {
337                             bcaster.broadcast(entity);
338                         }
339 
340                         if (subProtocol == null && writeEntity) {
341                             try {
342                                 if (Callable.class.isAssignableFrom(entity.getClass())) {
343                                     entity = Callable.class.cast(entity).call();
344                                 }
345                                 synchronized (response) {
346                                     response.setEntity(entity);
347                                     response.write();
348                                 }
349                             } catch (Throwable t) {
350                                 logger.debug("Error running Callable", t);
351                                 response.setEntity(null);
352                             }
353                         } else {
354                             response.setEntity(null);
355                         }
356                     }
357                     break;
358                 case SUSPEND_RESPONSE:
359                     SuspendResponse<?> s = SuspendResponse.class.cast(JResponseAsResponse.class.cast(response.getResponse()).getJResponse());
360                     boolean resumeOnBroadcast = resumeOnBroadcast(s.resumeOnBroadcast());
361 
362                     for (AtmosphereResourceEventListener el : s.listeners()) {
363                         r.addEventListener(el);
364                     }
365 
366                     if (s.getEntity() == null) {
367                         //https://github.com/Atmosphere/atmosphere/issues/423
368                         response.setEntity("");
369                     }
370 
371                     Broadcaster bc = s.broadcaster();
372                     if (bc == null && s.scope() != Suspend.SCOPE.REQUEST) {
373                         bc = (Broadcaster) servletReq.getAttribute(INJECTED_BROADCASTER);
374                     }
375 
376                     suspend(resumeOnBroadcast,
377                             translateTimeUnit(s.period().value(), s.period().timeUnit()), request, response, bc, r, s.scope(), s.writeEntity());
378 
379                     break;
380                 case SUBSCRIBE:
381                 case SUSPEND:
382                 case SUSPEND_RESUME:
383                     resumeOnBroadcast = resumeOnBroadcast((action == Action.SUSPEND_RESUME));
384 
385                     if (listeners != null) {
386                         for (Class<? extends AtmosphereResourceEventListener> listener : listeners) {
387                             try {
388                                 AtmosphereResourceEventListener el = atmosphereFramework.newClassInstance(AtmosphereResourceEventListener.class, listener);
389                                 r.addEventListener(el);
390                             } catch (Throwable t) {
391                                 throw new WebApplicationException(
392                                         new IllegalStateException("Invalid AtmosphereResourceEventListener " + listener, t));
393                             }
394                         }
395                     }
396 
397                     Broadcaster broadcaster = (Broadcaster) servletReq.getAttribute(INJECTED_BROADCASTER);
398                     // @Subscribe
399                     // TODO: Optimize me
400                     if (action == Action.SUBSCRIBE) {
401                         Class<Broadcaster> c = null;
402                         try {
403                             c = (Class<Broadcaster>) Class.forName((String) servletReq.getAttribute(BROADCASTER_CLASS));
404                         } catch (Throwable e) {
405                             throw new IllegalStateException(e.getMessage());
406                         }
407                         broadcaster = config.getBroadcasterFactory().lookup(c, topic, true);
408                     }
409 
410                     suspend(resumeOnBroadcast, timeout, request, response,
411                             broadcaster, r, scope, writeEntity);
412 
413                     break;
414                 case RESUME:
415                     if (response.getEntity() != null) {
416                         try {
417                             synchronized (response) {
418                                 response.write();
419                             }
420                         } catch (IOException ex) {
421                             throw new WebApplicationException(ex);
422                         }
423                     }
424 
425                     String path = response.getContainerRequest().getPath();
426                     r = resumeCandidates.remove(path.substring(path.lastIndexOf("/") + 1));
427 
428                     if (r != null) {
429                         resume(r);
430                     } else {
431                         throw new WebApplicationException(
432                                 new IllegalStateException("Unable to retrieve suspended Response. " +
433                                         "Either session-support is not enabled in atmosphere.xml or the" +
434                                         "path used to resume is invalid."));
435 
436                     }
437                     break;
438                 case BROADCAST:
439                 case PUBLISH:
440                 case RESUME_ON_BROADCAST:
441                     AtmosphereResource ar = (AtmosphereResource) servletReq.getAttribute(SUSPENDED_RESOURCE);
442                     if (ar != null) {
443                         r = ar;
444                     }
445 
446                     if (action == Action.PUBLISH) {
447                         Class<Broadcaster> c = null;
448                         try {
449                             c = (Class<Broadcaster>) Class.forName((String) servletReq.getAttribute(BROADCASTER_CLASS));
450                         } catch (Throwable e) {
451                             throw new IllegalStateException(e.getMessage());
452                         }
453                         r.setBroadcaster(config.getBroadcasterFactory().lookup(c, topic, true));
454                     }
455 
456                     broadcast(response, r, timeout);
457                     if (!writeEntity) {
458                         synchronized (response) {
459                             response.setEntity(null);
460                         }
461                     }
462                     break;
463                 case SCHEDULE:
464                 case SCHEDULE_RESUME:
465                     Object o = response.getEntity();
466                     Broadcaster b = r.getBroadcaster();
467                     if (response.getEntity() instanceof Broadcastable) {
468                         b = ((Broadcastable) response.getEntity()).getBroadcaster();
469                         o = ((Broadcastable) response.getEntity()).getMessage();
470                         response.setEntity(((Broadcastable) response.getEntity()).getResponseMessage());
471                     }
472 
473                     if (response.getEntity() != null) {
474                         try {
475                             synchronized (response) {
476                                 response.write();
477                             }
478                         } catch (IOException ex) {
479                             throw new WebApplicationException(ex);
480                         }
481                     }
482 
483                     if (action == Action.SCHEDULE_RESUME) {
484                         configureResumeOnBroadcast(b);
485                     }
486 
487                     b.scheduleFixedBroadcast(o, waitFor, timeout, TimeUnit.SECONDS);
488                     break;
489             }
490 
491             return response;
492         }

494         String uuid(AtmosphereResource r) {
495             String s = (String) r.getRequest().getAttribute(SUSPENDED_ATMOSPHERE_RESOURCE_UUID);
496             if (s != null) {
497                 return s;
498             }
499 
500             s = r.getRequest().getHeader(HeaderConfig.X_ATMOSPHERE_TRACKING_ID);
501             if (s != null && s != "0") {
502                 return s;
503             } else {
504                 return r.uuid();
505             }
506         }

508         String getHeaderOrQueryValue(String name) {
509             String value = servletReq.getHeader(name);
510             if (value == null) {
511                 value = servletReq.getParameter(name);
512                 // https://github.com/Atmosphere/atmosphere/issues/166
513                 if (value == null) {
514                     value = servletReq.getParameter(name.toLowerCase());
515                     // Last Chance
516                     if (value == null) {
517                         String qs = servletReq.getQueryString();
518                         if (qs != null && qs.indexOf(name) != -1) {
519                             String[] s = qs.split("&");
520                             String[] query;
521                             for (String a : s) {
522                                 if (a.startsWith(name) || a.startsWith(name.toLowerCase())) {
523                                     query = a.split("=");
524                                     if (query.length == 2) {
525                                         return query[1];
526                                     }
527                                 }
528                             }
529                         }
530                     }
531                 }
532             }
533             return value;
534         }

536         Response.ResponseBuilder configureHeaders(Response.ResponseBuilder b) throws IOException {
537             boolean webSocketSupported = servletReq.getAttribute(WebSocket.WEBSOCKET_SUSPEND) != null;
538 
539             if (servletReq.getHeaders("Connection") != null && servletReq.getHeaders("Connection").hasMoreElements()) {
540                 String[] e = ((Enumeration<String>) servletReq.getHeaders("Connection")).nextElement().toString().split(",");
541                 for (String upgrade : e) {
542                     if (upgrade != null && upgrade.equalsIgnoreCase(WEBSOCKET_UPGRADE)) {
543                         if (!webSocketSupported) {
544                             b = b.header(X_ATMOSPHERE_ERROR, "Websocket protocol not supported");
545                         }
546                     }
547                 }
548             }
549             return b;
550         }

552         void configureResumeOnBroadcast(Broadcaster b) {
553             Iterator<AtmosphereResource> i = b.getAtmosphereResources().iterator();
554             while (i.hasNext()) {
555                 HttpServletRequest r = i.next().getRequest();
556                 r.setAttribute(RESUME_ON_BROADCAST, true);
557             }
558         }

560         void configureFilter(Broadcaster bc) {
561             if (bc == null) throw new WebApplicationException(new IllegalStateException("Broadcaster cannot be null"));
562 
563             /**
564              * Here we can't predict if it's the same set of filter shared across all Broadcaster as
565              * Broadcaster can have their own BroadcasterConfig instance.
566              */
567             BroadcasterConfig c = bc.getBroadcasterConfig();
568             // Already configured
569             if (c.hasFilters()) {
570                 return;
571             }
572 
573             // Always the first one, before any transformation/filtering
574             for (ClusterBroadcastFilter cbf : clusters) {
575                 cbf.setBroadcaster(bc);
576                 c.addFilter(cbf);
577             }
578 
579             BroadcastFilter f = null;
580             if (filters != null) {
581                 for (Class<BroadcastFilter> filter : filters) {
582                     try {
583                         f = filter.newInstance();
584                     } catch (Throwable t) {
585                         logger.warn("Invalid @BroadcastFilter: " + filter, t);
586                     }
587                     c.addFilter(f);
588                 }
589             }
590         }

596         void broadcast(ContainerResponse r, AtmosphereResource ar, long delay) {
597             Object o = r.getEntity();
598 
599             Broadcaster b = ar.getBroadcaster();
600             Object msg = o;
601             Object returnMsg = null;
602             // Something went wrong if null.
603             if (o instanceof Broadcastable) {
604                 if (((Broadcastable) o).getBroadcaster() != null) {
605                     b = ((Broadcastable) o).getBroadcaster();
606                 }
607                 msg = ((Broadcastable) o).getMessage();
608                 returnMsg = ((Broadcastable) o).getResponseMessage();
609             }
610 
611             if (action == Action.RESUME_ON_BROADCAST) {
612                 configureResumeOnBroadcast(b);
613             }
614 
615             if (o != null) {
616                 addFilter(b);
617                 r.setEntity(msg);
618                 if (msg == null) return;
619 
620                 if (delay == -1) {
621                     b.broadcast(msg);
622                     if (o instanceof Broadcastable) {
623                         r.setEntity(returnMsg);
624                     }
625                 } else if (delay == 0) {
626                     b.delayBroadcast(msg);
627                 } else {
628                     b.delayBroadcast(msg, delay, TimeUnit.SECONDS);
629                 }
630             }
631         }

645         void suspend(boolean resumeOnBroadcast,
646                      long timeout,
647                      ContainerRequest request,
648                      ContainerResponse response,
649                      Broadcaster bc,
650                      AtmosphereResource r,
651                      Suspend.SCOPE localScope,
652                      boolean flushEntity) {
653 
654             // Force the status code to 200 events independently of the value of the entity (null or not)
655             if (response.getStatus() == 204) {
656                 response.setStatus(200);
657             }
658 
659             BroadcasterFactory broadcasterFactory = r.getAtmosphereConfig().getBroadcasterFactory();
660 
661             boolean sessionSupported = (Boolean) servletReq.getAttribute(FrameworkConfig.SUPPORT_SESSION);
662             URI location = null;
663             // Do not add location header if already there.
664             if (useResumeAnnotation && !sessionSupported && !resumeOnBroadcast && response.getHttpHeaders().getFirst("Location") == null) {
665                 String uuid = UUID.randomUUID().toString();
666 
667                 location = uriInfo.getAbsolutePathBuilder().path(uuid).build("");
668 
669                 resumeCandidates.put(uuid, r);
670                 servletReq.setAttribute(RESUME_UUID, uuid);
671                 servletReq.setAttribute(RESUME_CANDIDATES, resumeCandidates);
672             }
673 
674             if (bc == null && localScope != Suspend.SCOPE.REQUEST) {
675                 bc = r.getBroadcaster();
676             }
677 
678             if (response.getEntity() == null) {
679                 //https://github.com/Atmosphere/atmosphere/issues/423
680                 response.setEntity("");
681             }
682 
683             if (response.getEntity() instanceof Broadcastable) {
684                 Broadcastable b = (Broadcastable) response.getEntity();
685                 bc = b.getBroadcaster();
686                 response.setEntity(b.getResponseMessage());
687             }
688 
689             if ((localScope == Suspend.SCOPE.REQUEST) && bc == null) {
690                 if (bc == null) {
691                     try {
692                         String id = servletReq.getHeader(X_ATMOSPHERE_TRACKING_ID);
693                         if (id == null) {
694                             id = UUID.randomUUID().toString();
695                         }
696 
697                         bc = broadcasterFactory.get(id);
698                         bc.setScope(Broadcaster.SCOPE.REQUEST);
699                     } catch (Exception ex) {
700                         logger.error("failed to instantiate broadcaster with factory: " + broadcasterFactory, ex);
701                     }
702                 } else {
703                     bc.setScope(Broadcaster.SCOPE.REQUEST);
704                 }
705             }
706             r.setBroadcaster(bc);
707 
708             if (resumeOnBroadcast) {
709                 servletReq.setAttribute(RESUME_ON_BROADCAST, new Boolean(true));
710             }
711 
712             executeSuspend(r, timeout, resumeOnBroadcast, location, request, response, flushEntity);
713 
714         }

716         void executeSuspend(AtmosphereResource r,
717                             long timeout,
718                             boolean resumeOnBroadcast,
719                             URI location,
720                             ContainerRequest request,
721                             ContainerResponse response,
722                             boolean flushEntity) {
723 
724             servletReq.setAttribute(FrameworkConfig.CONTAINER_RESPONSE, response);
725             configureFilter(r.getBroadcaster());
726             servletReq.setAttribute(SUSPENDED_RESOURCE, r);
727 
728             // Set the content-type based on the returned entity.
729             try {
730                 MediaType contentType = response.getMediaType();
731                 if (contentType == null && response.getEntity() != null) {
732                     LinkedList<MediaType> l = new LinkedList<MediaType>();
733                     // Will retrun the first
734                     l.add(request.getAcceptableMediaType(new LinkedList<MediaType>()));
735                     contentType = response.getMessageBodyWorkers().getMessageBodyWriterMediaType(
736                             response.getEntity().getClass(),
737                             response.getEntityType(),
738                             response.getAnnotations(),
739                             l);
740 
741                     if (contentType == null ||
742                             contentType.isWildcardType() || contentType.isWildcardSubtype())
743                         contentType = MediaType.APPLICATION_OCTET_STREAM_TYPE;
744                 }
745 
746                 Object entity = response.getEntity();
747 
748                 Response.ResponseBuilder b = Response.ok();
749                 b = configureHeaders(b);
750 
751                 AtmosphereConfig config = r.getAtmosphereConfig();
752 
753                 String defaultCT = config.getInitParameter(DEFAULT_CONTENT_TYPE);
754                 if (defaultCT == null) {
755                     defaultCT = "text/plain; charset=ISO-8859-1";
756                 }
757 
758                 String ct = contentType == null ? defaultCT : contentType.toString();
759 
760                 if (defaultContentType != null) {
761                     ct = defaultContentType;
762                 }
763 
764                 if (entity != null) {
765                     b = b.header("Content-Type", ct);
766                 }
767                 servletReq.setAttribute(FrameworkConfig.EXPECTED_CONTENT_TYPE, ct);
768 
769                 if (entity != null && flushEntity) {
770                     try {
771                         if (Callable.class.isAssignableFrom(entity.getClass())) {
772                             entity = Callable.class.cast(entity).call();
773                         }
774                     } catch (Throwable t) {
775                         logger.error("Error executing callable {}", entity);
776                         entity = null;
777                     }
778 
779                     if (location != null) {
780                         b = b.header(HttpHeaders.LOCATION, location);
781                     }
782 
783                     synchronized (response) {
784                         AtmosphereResourceEventListenerAdapter a =
785                                 (AtmosphereResourceEventListenerAdapter) servletReq.getAttribute(CALLBACK_JAVASCRIPT_PROTOCOL);
786                         if (a != null) {
787                             try {
788                                 a.onSuspend(r.getAtmosphereResourceEvent());
789                             } catch (Exception ex) {
790                                 logger.debug("AtmosphereFilter error", ex);
791                             } finally {
792                                 servletReq.removeAttribute(CALLBACK_JAVASCRIPT_PROTOCOL);
793                             }
794                         }
795                         r.removeEventListener(a);
796                         response.setResponse(b.entity(entity).build());
797                         response.write();
798                     }
799                 }
800 
801                 response.setEntity(null);
802                 r.suspend(timeout);
803             } catch (IOException ex) {
804                 throw new WebApplicationException(ex);
805             }
806         }
807     }

809     /**
810      * Create a {@link ResourceFilter} which contains the information about the
811      * annotation being processed.
812      * <p/>
813      * XXX Need to filter invalid mix of annotation.
814      *
815      * @param am an {@link AbstractMethod}
816      * @return a List of {@link ResourceFilter} to invoke.
817      */
818     @Override
819     public List<ResourceFilter> create(AbstractMethod am) {
820         LinkedList<ResourceFilter> list = new LinkedList<ResourceFilter>();
821         Filter f;
822 
823         for (Annotation annotation : am.getAnnotations()) {
824             logger.trace("AtmosphereFilter processing annotation: {}", annotation);
825         }
826 
827         if (am.getMethod() == null) {
828             return null;
829         }
830 
831         if (SuspendResponse.class.isAssignableFrom(am.getMethod().getReturnType())) {
832             list.addLast(new Filter(Action.SUSPEND_RESPONSE));
833             return list;
834         }
835 
836         if (am.isAnnotationPresent(Broadcast.class)) {
837             int delay = am.getAnnotation(Broadcast.class).delay();
838             Class[] broadcastFilter = am.getAnnotation(Broadcast.class).filters();
839 
840             if (am.getAnnotation(Broadcast.class).resumeOnBroadcast()) {
841                 f = new Filter(Action.RESUME_ON_BROADCAST, delay, 0, Suspend.SCOPE.APPLICATION, broadcastFilter, null,
842                         am.getAnnotation(Broadcast.class).writeEntity());
843             } else {
844                 f = new Filter(Action.BROADCAST, delay, 0, Suspend.SCOPE.APPLICATION, broadcastFilter, null,
845                         am.getAnnotation(Broadcast.class).writeEntity());
846             }
847 
848             list.addLast(f);
849 
850             if (am.isAnnotationPresent(Cluster.class)) {
851                 broadcastFilter = am.getAnnotation(Cluster.class).value();
852                 for (Class<ClusterBroadcastFilter> c : broadcastFilter) {
853                     try {
854                         ClusterBroadcastFilter cbf = c.newInstance();
855                         cbf.setUri(am.getAnnotation(Cluster.class).name());
856                         f.addCluster(cbf);
857                     } catch (Throwable t) {
858                         logger.warn("Invalid ClusterBroadcastFilter", t);
859                     }
860                 }
861             }
862         }
863 
864         if (am.isAnnotationPresent(Asynchronous.class)) {
865             int suspendTimeout = am.getAnnotation(Asynchronous.class).period();
866             Class[] broadcastFilter = am.getAnnotation(Asynchronous.class).broadcastFilter();
867 
868             boolean wait = am.getAnnotation(Asynchronous.class).waitForResource();
869             f = new Filter(Action.ASYNCHRONOUS,
870                     suspendTimeout,
871                     wait ? -1 : 0,
872                     null,
873                     broadcastFilter,
874                     am.getAnnotation(Asynchronous.class).header(),
875                     am.getAnnotation(Asynchronous.class).writeEntity(),
876                     am.getAnnotation(Asynchronous.class).contentType());
877             f.setListeners(am.getAnnotation(Asynchronous.class).eventListeners());
878             list.addFirst(f);
879         }
880 
881         if (am.isAnnotationPresent(Suspend.class)) {
882 
883             long suspendTimeout = am.getAnnotation(Suspend.class).period();
884             TimeUnit tu = am.getAnnotation(Suspend.class).timeUnit();
885             suspendTimeout = translateTimeUnit(suspendTimeout, tu);
886 
887             Suspend.SCOPE scope = am.getAnnotation(Suspend.class).scope();
888 
889             if (am.getAnnotation(Suspend.class).resumeOnBroadcast()) {
890                 f = new Filter(Action.SUSPEND_RESUME,
891                         suspendTimeout,
892                         0,
893                         scope,
894                         null,
895                         null,
896                         am.getAnnotation(Suspend.class).writeEntity(),
897                         am.getAnnotation(Suspend.class).contentType());
898             } else {
899                 f = new Filter(Action.SUSPEND,
900                         suspendTimeout,
901                         0,
902                         scope,
903                         null,
904                         null,
905                         am.getAnnotation(Suspend.class).writeEntity(),
906                         am.getAnnotation(Suspend.class).contentType());
907             }
908             f.setListeners(am.getAnnotation(Suspend.class).listeners());
909 
910             list.addFirst(f);
911         }
912 
913         if (am.isAnnotationPresent(Subscribe.class)) {
914             int timeout = am.getAnnotation(Subscribe.class).timeout();
915             f = new Filter(Action.SUBSCRIBE, timeout, -1, Suspend.SCOPE.APPLICATION,
916                      null, am.getAnnotation(Subscribe.class).value(), am.getAnnotation(Subscribe.class).writeEntity());
917 
918             f.setListeners(am.getAnnotation(Subscribe.class).listeners());
919 
920             list.addFirst(f);
921         }
922 
923         if (am.isAnnotationPresent(Publish.class)) {
924             f = new Filter(Action.PUBLISH, -1, -1, Suspend.SCOPE.APPLICATION,
925                      null, am.getAnnotation(Publish.class).value(), true);
926             list.addFirst(f);
927         }
928 
929         if (am.isAnnotationPresent(Resume.class)) {
930             useResumeAnnotation = true;
931             int suspendTimeout = am.getAnnotation(Resume.class).value();
932             list.addFirst(new Filter(Action.RESUME, suspendTimeout));
933         }
934 
935         if (am.isAnnotationPresent(Schedule.class)) {
936             int period = am.getAnnotation(Schedule.class).period();
937             int waitFor = am.getAnnotation(Schedule.class).waitFor();
938 
939             if (am.getAnnotation(Schedule.class).resumeOnBroadcast()) {
940                 list.addFirst(new Filter(Action.SCHEDULE_RESUME, period, waitFor));
941             } else {
942                 list.addFirst(new Filter(Action.SCHEDULE, period, waitFor));
943             }
944         }
945 
946         if (!list.isEmpty()) {
947             f = new Filter(Action.NONE);
948             list.addFirst(f);
949         }
950 
951         return list;
952     }

954     protected long translateTimeUnit(long period, TimeUnit tu) {
955         if (period == -1) return period;
956 
957         switch (tu) {
958             case SECONDS:
959                 return TimeUnit.MILLISECONDS.convert(period, TimeUnit.SECONDS);
960             case MINUTES:
961                 return TimeUnit.MILLISECONDS.convert(period, TimeUnit.MINUTES);
962             case HOURS:
963                 return TimeUnit.MILLISECONDS.convert(period, TimeUnit.HOURS);
964             case DAYS:
965                 return TimeUnit.MILLISECONDS.convert(period, TimeUnit.DAYS);
966             case MILLISECONDS:
967                 return period;
968             case MICROSECONDS:
969                 return TimeUnit.MILLISECONDS.convert(period, TimeUnit.MICROSECONDS);
970             case NANOSECONDS:
971                 return TimeUnit.MILLISECONDS.convert(period, TimeUnit.NANOSECONDS);
972         }
973         return period;
974     }

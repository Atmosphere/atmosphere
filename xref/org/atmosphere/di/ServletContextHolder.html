16  package org.atmosphere.di;
17  
18  import javax.servlet.ServletContext;
19  import java.lang.ref.WeakReference;
20  import java.util.concurrent.atomic.AtomicReference;
21  
22  /**
23   * Holds the reference of an object being able to return the current ServletContext of the web application
24   *
25   * @author Mathieu Carbou
26   * @since 0.7
27   */
28  public final class ServletContextHolder {
29  
30      private static AtomicReference<WeakReference<ServletContextProvider>> PROVIDER = new AtomicReference<WeakReference<ServletContextProvider>>();
31  
32      private ServletContextHolder() {
33      }
34  
35      public static void register(ServletContextProvider provider) {
36          PROVIDER.set(new WeakReference<ServletContextProvider>(provider));
37      }
38  
39      public static ServletContext getServletContext() {
40          WeakReference<ServletContextProvider> ref = PROVIDER.get();
41          if (ref != null) {
42              ServletContextProvider provider = ref.get();
43              if (provider != null) {
44                  return provider.getServletContext();
45              }
46          }
47          throw new IllegalStateException("No " + ServletContextProvider.class.getSimpleName() + " found.");
48      }
49  }

16  package org.atmosphere.handler;
17  
18  import org.atmosphere.cpr.AtmosphereHandler;
19  import org.atmosphere.cpr.AtmosphereResource;
20  import org.atmosphere.cpr.AtmosphereResourceEvent;
21  import org.atmosphere.cpr.AtmosphereResourceImpl;
22  import org.atmosphere.cpr.AtmosphereResponse;
23  import org.atmosphere.cpr.AtmosphereResponseImpl;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import java.io.IOException;
28  import java.util.List;
29  
30  /**
31   * Simple {@link AtmosphereHandler} that must be used with the {@link org.atmosphere.interceptor.AtmosphereResourceLifecycleInterceptor}
32   * and {@link org.atmosphere.interceptor.BroadcastOnPostAtmosphereInterceptor} to reduce the handling of the suspend/resume/disconnect and
33   * broadcast operation.
34   * <p/>
35   * You can also safely used this class with {@link org.atmosphere.cpr.BroadcasterCache}.
36   * method
37   *
38   * @author Jeanfrancois Arcand
39   */
40  public abstract class OnMessage<T> extends AbstractReflectorAtmosphereHandler {
41  
42      public final static String MESSAGE_DELIMITER = "|";
43      private final Logger logger = LoggerFactory.getLogger(OnMessage.class);
44  
45      @Override
46      public final void onRequest(AtmosphereResource resource) throws IOException {
47          if (resource.getRequest().getMethod().equalsIgnoreCase("GET")) {
48              onOpen(resource);
49          }
50      }
51  
52      @Override
53      public final void onStateChange(AtmosphereResourceEvent event) throws IOException {
54          AtmosphereResponse response = ((AtmosphereResourceImpl) event.getResource()).getResponse(false);
55  
56          logger.trace("{} with event {}", event.getResource().uuid(), event);
57          if (event.isCancelled() || event.isClosedByApplication() || event.isClosedByClient()) {
58              onDisconnect(response);
59          } else if (event.getMessage() != null && List.class.isAssignableFrom(event.getMessage().getClass())) {
60              List<T> messages = List.class.cast(event.getMessage());
61              for (T t : messages) {
62                  onMessage(response, t);
63              }
64          } else if (event.isResuming()) {
65              onResume(response);
66          } else if (event.isResumedOnTimeout()) {
67              onTimeout(response);
68          } else if (event.isSuspended()) {
69              onMessage(response, (T) event.getMessage());
70          }
71          postStateChange(event);
72      }
73  
74      @Override
75      public final void destroy() {
76      }
77  
78      /**
79       * This method will be invoked when an connection has been received and not haven't yet be suspended. Note that
80       * the connection will be suspended AFTER the method has been invoked when used with {@link org.atmosphere.interceptor.AtmosphereResourceLifecycleInterceptor}
81       *
82       * @param resource an {@link AtmosphereResource}
83       * @throws IOException
84       */
85      public void onOpen(AtmosphereResource resource) throws IOException {
86      }
87  
88      /**
89       * Implement this method to get invoked every time a new {@link org.atmosphere.cpr.Broadcaster#broadcast(Object)}
90       * occurs.
91       *
92       * @param response an {@link AtmosphereResponse}
93       * @param message  a message of type T
94       */
95      abstract public void onMessage(AtmosphereResponse response, T message) throws IOException;
96  
97      /**
98       * This method will be invoked during the process of resuming a connection. By default this method does nothing.
99       *
100      * @param response an {@link AtmosphereResponse}.
101      * @throws IOException
102      */
103     public void onResume(AtmosphereResponse response) throws IOException {
104     }
105 
106     /**
107      * This method will be invoked when a suspended connection times out, e.g no activity has occurred for the
108      * specified time used when suspending. By default this method does nothing.
109      *
110      * @param response an {@link AtmosphereResponse}.
111      * @throws IOException
112      */
113     public void onTimeout(AtmosphereResponse response) throws IOException {
114     }
115 
116     /**
117      * This method will be invoked when the underlying WebServer detects a connection has been closed. Please
118      * note that not all WebServer supports that features (see Atmosphere's WIKI for help). By default this method does nothing.
119      *
120      * @param response an {@link AtmosphereResponse}.
121      * @throws IOException
122      */
123     public void onDisconnect(AtmosphereResponse response) throws IOException {
124     }
125 }

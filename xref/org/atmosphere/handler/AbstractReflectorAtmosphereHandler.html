38  package org.atmosphere.handler;
39  
40  import org.atmosphere.cpr.ApplicationConfig;
41  import org.atmosphere.cpr.AtmosphereHandler;
42  import org.atmosphere.cpr.AtmosphereResourceEvent;
43  import org.atmosphere.cpr.Broadcaster;
44  import org.slf4j.Logger;
45  import org.slf4j.LoggerFactory;
46  
47  import javax.servlet.http.HttpServletRequest;
48  import javax.servlet.http.HttpServletResponse;
49  import java.io.IOException;
50  import java.util.List;
51  
52  import static org.atmosphere.cpr.ApplicationConfig.PROPERTY_USE_STREAM;
53  
54  /**
55   * Simple {@link AtmosphereHandler} that reflect every call to
56   * {@link Broadcaster#broadcast}, e.g sent the broadcasted event back to the remote client. All broadcast will be by default returned
57   * as it is to the suspended {@link HttpServletResponse#getOutputStream}
58   * or {@link HttpServletResponse#getWriter()}
59   *
60   * @author Jean-francois Arcand
61   */
62  public abstract class AbstractReflectorAtmosphereHandler
63          implements AtmosphereHandler<HttpServletRequest, HttpServletResponse> {
64  
65      private static final Logger logger = LoggerFactory.getLogger(AbstractReflectorAtmosphereHandler.class);
66  
67      /**
68       * Write the {@link AtmosphereResourceEvent#getMessage()} back to the client using
69       * the {@link HttpServletResponse#getOutputStream()} or {@link HttpServletResponse#getWriter()}.
70       * If a {@link org.atmosphere.cpr.Serializer} is defined, it will be invoked and the writ operation
71       * will be delegated to to it.
72       * <p/>
73       * By default, this method will try to use {@link HttpServletResponse#getWriter()}.
74       *
75       * @param event the {@link AtmosphereResourceEvent#getMessage()}
76       * @throws java.io.IOException
77       */
78      public void onStateChange(AtmosphereResourceEvent<HttpServletRequest, HttpServletResponse> event)
79              throws IOException {
80  
81          Object message = event.getMessage();
82          if (message == null || event.isCancelled()) return;
83  
84          if (event.getResource().getSerializer() != null) {
85              try {
86                  event.getResource().getSerializer().write(event.getResource().getResponse().getOutputStream(), message);
87              } catch (Throwable ex) {
88                  logger.warn("Serializer exception: message: " + message, ex);
89                  throw new IOException(ex);
90              }
91          } else {
92              boolean isUsingStream = (Boolean) event.getResource().getRequest().getAttribute(PROPERTY_USE_STREAM);
93  
94              if (!isUsingStream) {
95                  try {
96                      event.getResource().getResponse().getWriter();
97                  } catch (IllegalStateException e) {
98                      isUsingStream = true;
99                  }
100             }
101 
102             if (message instanceof List) {
103                 for (String s : (List<String>) message) {
104                     if (isUsingStream) {
105                         event.getResource().getResponse().getOutputStream().write(s.getBytes());
106                         event.getResource().getResponse().getOutputStream().flush();
107                     } else {
108                         event.getResource().getResponse().getWriter().write(s);
109                         event.getResource().getResponse().getWriter().flush();
110                     }
111                 }
112             } else {
113                 if (isUsingStream) {
114                     event.getResource().getResponse().getOutputStream().write(message.toString().getBytes());
115                     event.getResource().getResponse().getOutputStream().flush();
116                 } else {
117                     event.getResource().getResponse().getWriter().write(message.toString());
118                     event.getResource().getResponse().getWriter().flush();
119                 }
120             }
121 
122             Boolean resumeOnBroadcast = false;
123             Object o = event.getResource().getRequest().getAttribute(ApplicationConfig.RESUME_ON_BROADCAST);
124             if (o != null && Boolean.class.isAssignableFrom(o.getClass())) {
125                 resumeOnBroadcast = Boolean.class.cast(o);
126             }
127 
128             if (resumeOnBroadcast != null && resumeOnBroadcast) {
129                 event.getResource().resume();
130             }
131         }
132     }

16  package org.atmosphere.interceptor;
17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.AsyncIOWriter;
20  import org.atmosphere.cpr.AsyncIOWriterAdapter;
21  import org.atmosphere.cpr.AtmosphereInterceptor;
22  import org.atmosphere.cpr.AtmosphereResource;
23  import org.atmosphere.cpr.AtmosphereResponse;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import java.io.IOException;
28  import java.io.PrintWriter;
29  
30  /**
31   * HTML 5 Server Side Events implementation.
32   *
33   * @author Jeanfrancois Arcand
34   */
35  public class SSEAtmosphereInterceptor implements AtmosphereInterceptor {
36  
37      private static final Logger logger = LoggerFactory.getLogger(SSEAtmosphereInterceptor.class);
38  
39      private static final StringBuffer whitespace = new StringBuffer();
40  
41      static {
42          for (int i = 0; i < 2000; i++) {
43              whitespace.append(" ");
44          }
45          whitespace.append("\n");
46      }
47  
48      @Override
49      public Action inspect(AtmosphereResource r) {
50          final AtmosphereResponse response = r.getResponse();
51  
52          if (r.transport().equals(AtmosphereResource.TRANSPORT.SSE)) {
53  
54              String contentType = response.getContentType();
55              response.setContentType("text/event-stream");
56              response.setCharacterEncoding("utf-8");
57              PrintWriter writer = null;
58              try {
59                  writer = response.getWriter();
60              } catch (IOException e) {
61                  logger.trace("", e);
62              }
63  
64              writer.print(whitespace);
65              writer.flush();
66              response.setContentType(contentType);
67  
68              response.asyncIOWriter(new AsyncIOWriterAdapter() {
69                  @Override
70                  public AsyncIOWriter redirect(String location) throws IOException {
71                      response.sendRedirect(location);
72                      return this;
73                  }
74  
75                  @Override
76                  public AsyncIOWriter writeError(int errorCode, String message) throws IOException {
77                      response.sendError(errorCode);
78                      return this;
79                  }
80  
81                  @Override
82                  public AsyncIOWriter write(String data) throws IOException {
83                      response.write("data:" + data + "\n\n");
84                      return this;
85                  }
86  
87                  // TODO: Performance: execute a single write
88                  @Override
89                  public AsyncIOWriter write(byte[] data) throws IOException {
90                      response.write("data:").write(data).write("\n\n");
91                      return this;
92                  }
93  
94                  @Override
95                  public AsyncIOWriter write(byte[] data, int offset, int length) throws IOException {
96                      response.write("data:").write(data, offset, length).write("\n\n");
97                      return this;
98                  }
99  
100                 @Override
101                 public void close() throws IOException {
102                     response.closeStreamOrWriter();
103                 }
104 
105                 @Override
106                 public AsyncIOWriter flush() throws IOException {
107                     response.flushBuffer();
108                     return this;
109                 }
110             });
111         }
112         return Action.CONTINUE;
113     }
114 
115     @Override
116     public String toString() {
117         return "SSE Interceptor Support";
118     }
119 }

43  /**
44   * HTML 5 Server-Sent Events implementation.
45   *
46   * @author Jeanfrancois Arcand
47   */
48  public class SSEAtmosphereInterceptor extends AtmosphereInterceptorAdapter {
49  
50      private static final Logger logger = LoggerFactory.getLogger(SSEAtmosphereInterceptor.class);
51  
52      private static final byte[] padding;
53      private static final String paddingText;
54      private static final byte[] DATA = "data:".getBytes();
55      private static final byte[] NEWLINE = "\r\n".getBytes();
56      private static final byte[] END = "\r\n\r\n".getBytes();
57      private String contentType = "text/event-stream";
58  
59      static {
60          StringBuffer whitespace = new StringBuffer();
61          for (int i = 0; i < 2000; i++) {
62              whitespace.append(" ");
63          }
64          whitespace.append("\n");
65          paddingText = whitespace.toString();
66          padding = paddingText.getBytes();
67      }
68  
69      @Override
70      public void configure(AtmosphereConfig config) {
71          String s = config.getInitParameter(ApplicationConfig.SSE_DEFAULT_CONTENTTYPE);
72          if (s != null) {
73              contentType = s;
74          }
75      }
76  
77      private boolean writePadding(AtmosphereResponse response) {
78          if (response.request() != null && response.request().getAttribute("paddingWritten") != null) return false;
79  
80          response.setContentType(contentType);
81          response.setCharacterEncoding("utf-8");
82          boolean isUsingStream = (Boolean) response.request().getAttribute(PROPERTY_USE_STREAM);
83          if (isUsingStream) {
84              try {
85                  OutputStream stream = response.getResponse().getOutputStream();
86                  try {
87                      stream.write(padding);
88                      stream.flush();
89                  } catch (IOException ex) {
90                      logger.warn("SSE may not work", ex);
91                  }
92              } catch (IOException e) {
93                  logger.trace("", e);
94              }
95          } else {
96              try {
97                  PrintWriter w = response.getResponse().getWriter();
98                  w.println(paddingText);
99                  w.flush();
100             } catch (IOException e) {
101                 logger.trace("", e);
102             }
103         }
104         response.resource().getRequest().setAttribute("paddingWritten", "true");
105         return true;
106     }
107 
108     private final class P extends OnPreSuspend implements AllowInterceptor {
109 
110         private final AtmosphereResponse response;
111 
112         private P(AtmosphereResponse response) {
113             this.response = response;
114         }
115 
116         @Override
117         public void onPreSuspend(AtmosphereResourceEvent event) {
118             writePadding(response);
119         }
120     }
121 
122     @Override
123     public Action inspect(final AtmosphereResource r) {
124 
125         if (Utils.webSocketMessage(r)) return Action.CONTINUE;
126 
127         final AtmosphereResponse response = r.getResponse();
128         final AtmosphereRequest request = r.getRequest();
129         String accept = request.getHeader("Accept") == null ? "text/plain" : request.getHeader("Accept").trim();
130 
131         if (r.transport().equals(AtmosphereResource.TRANSPORT.SSE) || contentType.equalsIgnoreCase(accept)) {
132             super.inspect(r);
133 
134             r.addEventListener(new P(response));
135 
136             AsyncIOWriter writer = response.getAsyncIOWriter();
137             if (AtmosphereInterceptorWriter.class.isAssignableFrom(writer.getClass())) {
138                 AtmosphereInterceptorWriter.class.cast(writer).interceptor(new AsyncIOInterceptorAdapter() {
139                     private boolean padding() {
140                         if (!r.isSuspended()) {
141                             return writePadding(response);
142                         }
143                         return false;
144                     }
145 
146                     @Override
147                     public void prePayload(AtmosphereResponse response, byte[] data, int offset, int length) {
148                         boolean noPadding = padding();
149                         // The CALLBACK_JAVASCRIPT_PROTOCOL may be called by a framework running on top of Atmosphere
150                         // In that case, we must pad/protocol indenendently of the state of the AtmosphereResource
151                         if (!noPadding || r.getRequest().getAttribute(CALLBACK_JAVASCRIPT_PROTOCOL) != null) {
152                             // write other meta info such as (id, event, etc)?
153                             response.write(DATA, true);
154                         }
155                     }
156 
157                     @Override
158                     public byte[] transformPayload(AtmosphereResponse response, byte[] responseDraft,
159                                                    byte[] data) throws IOException {
160                         boolean noPadding = padding();
161                         // The CALLBACK_JAVASCRIPT_PROTOCOL may be called by a framework running on top of Atmosphere
162                         // In that case, we must pad/protocol indenendently of the state of the AtmosphereResource
163                         if (!noPadding || r.getRequest().getAttribute(CALLBACK_JAVASCRIPT_PROTOCOL) != null) {
164                             if (isMultilineData(responseDraft)) {
165                                 // return a padded multiline-data
166                                 return encodeMultilineData(responseDraft);
167                             }
168                         }
169                         return responseDraft;
170                     }
171 
172                     @Override
173                     public void postPayload(AtmosphereResponse response, byte[] data, int offset, int length) {
174                         // The CALLBACK_JAVASCRIPT_PROTOCOL may be called by a framework running on top of Atmosphere
175                         // In that case, we must pad/protocol indenendently of the state of the AtmosphereResource
176                         if (r.isSuspended() || r.getRequest().getAttribute(CALLBACK_JAVASCRIPT_PROTOCOL) != null
177                                 || r.getRequest().getAttribute(CONTAINER_RESPONSE) != null) {
178                             response.write(END, true);
179                         }
180 
181                         /**
182                          * When used with https://github.com/remy/polyfills/blob/master/EventSource.js , we
183                          * resume after every message.
184                          */
185                         String ua = r.getRequest().getHeader("User-Agent");
186                         if (ua != null && ua.contains("MSIE")) {
187                             try {
188                                 response.flushBuffer();
189                             } catch (IOException e) {
190                                 logger.trace("", e);
191                             }
192                             r.resume();
193                         }
194                     }
195                 });
196             } else {
197                 logger.warn("Unable to apply {}. Your AsyncIOWriter must implement {}", getClass().getName(), AtmosphereInterceptorWriter.class.getName());
198             }
199         }
200         return Action.CONTINUE;
201     }
202 
203     @Override
204     public String toString() {
205         return "SSE Interceptor Support";
206     }

208     // utilities
209     private static byte[] encodeMultilineData(byte[] data) {
210         // add "data" field for each line
211         ByteArrayOutputStream baos = new ByteArrayOutputStream();
212         int bp = 0;
213         int ep = 0;
214         try {
215             while (ep < data.length) {
216                 int c = data[ep];
217                 if (c == '\r') {
218                     if (baos.size() > 0) {
219                         baos.write(NEWLINE);
220                         baos.write(DATA);
221                     }
222                     baos.write(data, bp, ep - bp);
223                     if (ep + 1 < data.length && data[ep + 1] == '\n') {
224                         ep++;
225                     }
226                     bp = ep + 1;
227                 } else if (c == '\n') {
228                     if (baos.size() > 0) {
229                         baos.write(NEWLINE);
230                         baos.write(DATA);
231                     }
232                     baos.write(data, bp, ep - bp);
233                     bp = ep + 1;
234                 }
235                 ep++;
236             }
237             if (baos.size() > 0) {
238                 baos.write(NEWLINE);
239                 baos.write(DATA);
240             }
241             baos.write(data, bp, ep - bp);
242         } catch (IOException e) {
243             //ignore
244         }
245         return baos.toByteArray();
246     }
247 
248     private static boolean isMultilineData(byte[] data) {
249         for (byte b : data) {
250             if (b == '\r' || b == '\n') {
251                 return true;
252             }
253         }
254         return false;
255     }

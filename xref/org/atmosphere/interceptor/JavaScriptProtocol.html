19  import org.atmosphere.cpr.Action;
20  import org.atmosphere.cpr.ApplicationConfig;
21  import org.atmosphere.cpr.AtmosphereConfig;
22  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;
23  import org.atmosphere.cpr.AtmosphereResource;
24  import org.atmosphere.cpr.AtmosphereResourceEvent;
25  import org.atmosphere.cpr.BroadcastFilter;
26  import org.atmosphere.cpr.HeaderConfig;
27  import org.atmosphere.util.Utils;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import java.io.IOException;
32  import java.util.concurrent.atomic.AtomicReference;
33  
34  import static org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter.OnSuspend;
35  import static org.atmosphere.cpr.FrameworkConfig.CALLBACK_JAVASCRIPT_PROTOCOL;
36  
37  /**
38   * An Interceptor that send back to a websocket and http client the value of {@link HeaderConfig#X_ATMOSPHERE_TRACKING_ID}
39   * and {@link HeaderConfig#X_CACHE_DATE}
40   *
41   * @author Jeanfrancois Arcand
42   */
43  public class JavaScriptProtocol extends AtmosphereInterceptorAdapter {
44  
45      private final static Logger logger = LoggerFactory.getLogger(JavaScriptProtocol.class);
46      private String wsDelimiter = "|";
47      private final TrackMessageSizeFilter f = new TrackMessageSizeFilter();
48  
49      @Override
50      public void configure(AtmosphereConfig config) {
51          String s = config.getInitParameter(ApplicationConfig.MESSAGE_DELIMITER);
52          if (s != null) {
53              wsDelimiter = s;
54          }
55      }
56  
57      @Override
58      public Action inspect(final AtmosphereResource r) {
59  
60          String uuid = r.getRequest().getHeader(HeaderConfig.X_ATMOSPHERE_TRACKING_ID);
61          String handshakeUUID = r.getRequest().getHeader(HeaderConfig.X_ATMO_PROTOCOL);
62          if (uuid != null && uuid.equals("0") && handshakeUUID != null) {
63              r.getRequest().header(HeaderConfig.X_ATMO_PROTOCOL, null);
64              // Since 1.0.10
65  
66              final StringBuffer message = new StringBuffer(r.uuid()).append(wsDelimiter).append(System.currentTimeMillis());
67  
68              // https://github.com/Atmosphere/atmosphere/issues/993
69              boolean track = false;
70              if (r.getBroadcaster().getBroadcasterConfig().hasFilters()) {
71                  for (BroadcastFilter bf : r.getBroadcaster().getBroadcasterConfig().filters()) {
72                      if (TrackMessageSizeFilter.class.isAssignableFrom(bf.getClass())) {
73                          track = true;
74                          break;
75                      }
76                  }
77              }
78  
79              final AtomicReference<String> protocolMessage = new AtomicReference<String>(message.toString());
80              if (track) {
81                  protocolMessage.set((String) f.filter(r, protocolMessage.get(), protocolMessage.get()).message());
82              }
83  
84              if (!Utils.resumableTransport(r.transport())) {
85                  OnSuspend a = new OnSuspend() {
86                      @Override
87                      public void onSuspend(AtmosphereResourceEvent event) {
88                          r.getResponse().write(protocolMessage.get());
89                          try {
90                              r.getResponse().flushBuffer();
91                          } catch (IOException e) {
92                              logger.trace("", e);
93                          }
94                      }
95                  };
96                  // Pass the information to Servlet Based Framework
97                  r.getRequest().setAttribute(CALLBACK_JAVASCRIPT_PROTOCOL, a);
98                  r.addEventListener(a);
99              } else {
100                 r.getResponse().write(protocolMessage.get());
101             }
102 
103             // We don't need to reconnect here
104             if (r.transport() == AtmosphereResource.TRANSPORT.WEBSOCKET
105                     || r.transport() == AtmosphereResource.TRANSPORT.STREAMING
106                     || r.transport() == AtmosphereResource.TRANSPORT.SSE) {
107                 return Action.CONTINUE;
108             } else {
109                 return Action.SKIP_ATMOSPHEREHANDLER;
110             }
111         }
112         return Action.CONTINUE;
113     }
114 
115     @Override
116     public String toString() {
117         return "Atmosphere JavaScript Protocol";
118     }
119 }

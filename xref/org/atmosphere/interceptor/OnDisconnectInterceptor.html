16  package org.atmosphere.interceptor;
17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.AsynchronousProcessor;
20  import org.atmosphere.cpr.AtmosphereConfig;
21  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;
22  import org.atmosphere.cpr.AtmosphereRequest;
23  import org.atmosphere.cpr.AtmosphereResource;
24  import org.atmosphere.cpr.AtmosphereResourceEventImpl;
25  import org.atmosphere.cpr.AtmosphereResourceImpl;
26  import org.atmosphere.util.Utils;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import static org.atmosphere.util.Utils.closeMessage;
31  
32  
33  /**
34   * When the browser close the connection, the atmosphere.js will send an unsubscribe message to tell
35   * framework the browser is disconnecting.
36   *
37   * @author Jeanfrancois Arcand
38   */
39  public class OnDisconnectInterceptor extends AtmosphereInterceptorAdapter {
40  
41      private final Logger logger = LoggerFactory.getLogger(OnDisconnectInterceptor.class);
42      private AsynchronousProcessor p;
43      private AtmosphereConfig config;
44  
45      @Override
46      public void configure(AtmosphereConfig config) {
47          if (AsynchronousProcessor.class.isAssignableFrom(config.framework().getAsyncSupport().getClass())) {
48              p = AsynchronousProcessor.class.cast(config.framework().getAsyncSupport());
49          }
50          this.config = config;
51      }
52  
53      @Override
54      public Action inspect(final AtmosphereResource r) {
55  
56          if (Utils.webSocketMessage(r)) return Action.CONTINUE;
57  
58          AtmosphereRequest request = AtmosphereResourceImpl.class.cast(r).getRequest(false);
59          String uuid = r.uuid();
60          if (closeMessage(request)) {
61              AtmosphereResource ss = config.resourcesFactory().find(uuid);
62  
63              if (ss == null) {
64                  logger.debug("No Suspended Connection found for {}. Using the AtmosphereResource associated with the close message", uuid);
65                  ss = r;
66              }
67  
68              if (ss == null) {
69                  logger.debug("Was unable to execute onDisconnect on {}", r.uuid());
70                  return Action.CONTINUE;
71              }
72              logger.debug("AtmosphereResource {} disconnected", uuid);
73  
74              // Block websocket closing detection
75              AtmosphereResourceEventImpl.class.cast(ss.getAtmosphereResourceEvent()).isClosedByClient(true);
76  
77              p.completeLifecycle(ss, false);
78              return Action.CANCELLED;
79          }
80          return Action.CONTINUE;
81      }
82  
83      public String toString() {
84          return "Browser disconnection detection";
85      }
86  }

16  package org.atmosphere.interceptor;
17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.AsyncIOInterceptorAdapter;
20  import org.atmosphere.cpr.AsyncIOWriter;
21  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;
22  import org.atmosphere.cpr.AtmosphereInterceptorWriter;
23  import org.atmosphere.cpr.AtmosphereResource;
24  import org.atmosphere.cpr.AtmosphereResource.TRANSPORT;
25  import org.atmosphere.cpr.AtmosphereResourceImpl;
26  import org.atmosphere.cpr.AtmosphereResponse;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  /**
31   * Fix for the Android 2.2.x bogus HTTP implementation.
32   *
33   * @author Jeanfrancois Arcand
34   */
35  public class AndroidAtmosphereInterceptor extends AtmosphereInterceptorAdapter {
36  
37      private static final Logger logger = LoggerFactory.getLogger(AndroidAtmosphereInterceptor.class);
38  
39      private static final byte[] padding;
40      private static final String paddingText;
41  
42      static {
43          StringBuffer whitespace = new StringBuffer();
44          for (int i = 0; i < 4096; i++) {
45              whitespace.append(" ");
46          }
47          whitespace.append("\n");
48          paddingText = whitespace.toString();
49          padding = paddingText.getBytes();
50      }
51  
52      @Override
53      public Action inspect(final AtmosphereResource r) {
54  
55          if (!r.transport().equals(TRANSPORT.STREAMING)) return Action.CONTINUE;
56  
57          final AtmosphereResponse response = AtmosphereResourceImpl.class.cast(r).getResponse(false);
58          String userAgent = AtmosphereResourceImpl.class.cast(r).getRequest(false).getHeader("User-Agent");
59  
60          if (userAgent != null &&
61                  (userAgent.indexOf("Android 2.") != -1 || userAgent.indexOf("Android 3.") != -1)) {
62              super.inspect(r);
63  
64              AsyncIOWriter writer = response.getAsyncIOWriter();
65              if (AtmosphereInterceptorWriter.class.isAssignableFrom(writer.getClass())) {
66                  AtmosphereInterceptorWriter.class.cast(writer).interceptor(new AsyncIOInterceptorAdapter() {
67  
68                      @Override
69                      public void prePayload(AtmosphereResponse response, byte[] data, int offset, int length) {
70                          response.write(padding, true);
71                      }
72  
73                      @Override
74                      public void postPayload(AtmosphereResponse response, byte[] data, int offset, int length) {
75                          response.write(padding, true);
76                      }
77  
78                  });
79              } else {
80                  logger.warn("Unable to apply {}. Your AsyncIOWriter must implement {}", getClass().getName(), AtmosphereInterceptorWriter.class.getName());
81              }
82          }
83          return Action.CONTINUE;
84      }
85  
86      @Override
87      public String toString() {
88          return "Android Interceptor Support";
89      }
90  }

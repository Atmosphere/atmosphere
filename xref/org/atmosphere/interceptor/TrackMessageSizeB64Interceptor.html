39  import static org.atmosphere.cpr.ApplicationConfig.EXCLUDED_CONTENT_TYPES;

41  /**
42   * An {@link org.atmosphere.cpr.AtmosphereInterceptor} that adds message size and delimiter, and encodes the message in Base64.
43   * This allows for broadcasting of messages containing the delimiter character.
44   * <p/>
45   * You can configure this class to exclude some response's content-type by using the {@link ApplicationConfig#EXCLUDED_CONTENT_TYPES}
46   *
47   * @author Jeanfrancois Arcand
48   * @author Martin Maƒçura
49   */
50  public class TrackMessageSizeB64Interceptor extends AtmosphereInterceptorAdapter {
51  
52      private static final Logger logger = LoggerFactory.getLogger(TrackMessageSizeB64Interceptor.class);
53      private static final String DELIMITER = "|";
54      private final static String OUT_ENCODING = "UTF-8";
55      public final static String SKIP_INTERCEPTOR = TrackMessageSizeB64Interceptor.class.getName() + ".skip";
56  
57      private final HashSet<String> excludedContentTypes = new HashSet<String>();
58  
59      private final Interceptor interceptor = new Interceptor();
60  
61      @Override
62      public void configure(AtmosphereConfig config) {
63          String s = config.getInitParameter(EXCLUDED_CONTENT_TYPES);
64          if (s != null) {
65              excludedContentTypes.addAll(Arrays.asList(s.split(",")));
66          }
67      }
68  
69      /**
70       * Excluse response's content-type from being processed by this class.
71       *
72       * @param excludedContentType the value of {@link AtmosphereResponseImpl#getContentType()}
73       * @return this
74       */
75      public TrackMessageSizeB64Interceptor excludedContentType(String excludedContentType) {
76          excludedContentTypes.add(excludedContentType.toLowerCase());
77          return this;
78      }
79  
80      @Override
81      public Action inspect(final AtmosphereResource r) {
82  
83          if (Utils.webSocketMessage(r)) return Action.CONTINUE;
84  
85          final AtmosphereResponse response = r.getResponse();
86  
87          super.inspect(r);
88  
89          AsyncIOWriter writer = response.getAsyncIOWriter();
90          if (AtmosphereInterceptorWriter.class.isAssignableFrom(writer.getClass())) {
91              AtmosphereInterceptorWriter.class.cast(writer).interceptor(interceptor);
92          } else {
93              logger.warn("Unable to apply {}. Your AsyncIOWriter must implement {}", getClass().getName(), AtmosphereInterceptorWriter.class.getName());
94          }
95          return Action.CONTINUE;
96      }
97  
98      @Override
99      public String toString() {
100         return " Track Message Size Base64 Interceptor using " + DELIMITER;
101     }
102 
103     private final class Interceptor extends AsyncIOInterceptorAdapter {
104         @Override
105         public byte[] transformPayload(AtmosphereResponse response, byte[] responseDraft, byte[] data) throws IOException {
106 
107             if (response.request().getAttribute(SKIP_INTERCEPTOR) == null
108                     && Boolean.valueOf(response.request().getHeader(HeaderConfig.X_ATMOSPHERE_TRACKMESSAGESIZE))
109                     && (response.getContentType() == null
110                     || !excludedContentTypes.contains(response.getContentType().toLowerCase()))) {
111                 response.setCharacterEncoding(OUT_ENCODING);
112                 String s = DatatypeConverter.printBase64Binary(responseDraft);
113                 StringBuilder sb = new StringBuilder();
114                 sb.append(s.length()).append(DELIMITER).append(s);
115                 return sb.toString().getBytes(OUT_ENCODING);
116             } else {
117                 return responseDraft;
118             }
119 
120         }
121     }
122 }

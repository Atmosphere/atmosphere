48  /**
49   * This interceptor associates a {@link AtmosphereResource} to all {@link Broadcaster} the resource was added before
50   * the underlying connection got closed and resume. This allow an application to restore the state of the client before the
51   * disconnection occurred, and for the long-polling transport to return to it's previous state.
52   *
53   * @author Jeanfrancois Arcand
54   */
55  public class AtmosphereResourceStateRecovery implements AtmosphereInterceptor {
56  
57      private final static Logger logger = LoggerFactory.getLogger(AtmosphereResourceStateRecovery.class);
58      private final ConcurrentHashMap<String, BroadcasterTracker> states = new ConcurrentHashMap<String, BroadcasterTracker>();
59      private BroadcasterFactory factory;
60      private ScheduledExecutorService stateTracker;
61      private long timeout = 5 * 1000 * 60;
62      private Future<?> trackerFuture;
63  
64      @Override
65      public void configure(AtmosphereConfig config) {
66          factory = config.getBroadcasterFactory();
67          factory.addBroadcasterListener(new B());
68  
69          stateTracker = ExecutorsFactory.getScheduler(config);
70          String s = config.getInitParameter(STATE_RECOVERY_TIMEOUT);
71          if (s != null) {
72              timeout = Long.parseLong(s);
73          }
74          startStateTracker();
75          logger.trace("{} started.", AtmosphereResourceStateRecovery.class.getName());
76      }
77  
78      public AtmosphereResourceStateRecovery timeout(long timeout){
79          this.timeout = timeout;
80          trackerFuture.cancel(false);
81          startStateTracker();
82          return this;
83      }
84  
85      public long timeout(){
86          return timeout;
87      }
88  
89      protected void startStateTracker() {
90          trackerFuture = stateTracker.scheduleAtFixedRate(new Runnable() {
91              @Override
92              public void run() {
93                  long now = System.currentTimeMillis();
94                  for (Map.Entry<String, BroadcasterTracker> t : states.entrySet()) {
95                      // The resource may still be suspended but we don't want to keep a reference to it, so we swap
96                      // the state and will recover.
97                      if (now - t.getValue().lastTick() > timeout) {
98                          logger.trace("AtmosphereResource {} state destroyed.", t.getKey());
99                          states.remove(t.getKey());
100                     }
101                 }
102             }
103         }, timeout, timeout, TimeUnit.MILLISECONDS);
104     }

106     @Override
107     public Action inspect(final AtmosphereResource r) {
108 
109         if (!Utils.pollableTransport(r.transport()) && !Utils.webSocketMessage(r)){
110 
111             final BroadcasterTracker tracker = track(r).tick();
112 
113             List<Object> cachedMessages = retrieveCache(r, tracker, false);
114             if (!cachedMessages.isEmpty()) {
115                 logger.trace("cached messages");
116                 writeCache(r, cachedMessages);
117                 return Action.CANCELLED;
118             } else {
119                 r.addEventListener(new OnAlwaysSuspend() {
120                     public void onSuspend(AtmosphereResourceEvent event) {
121                         r.removeEventListener(this);
122 
123                         logger.trace("onSuspend first");
124                         final AtomicBoolean doNotSuspend = new AtomicBoolean(false);
125                         /**
126                          * If a message gets broadcasted during the execution of the code below, we don't need to
127                          * suspend the connection. This code is needed to prevent the connection being suspended
128                          * with messages already written.
129                          */
130                         r.addEventListener(new OnBroadcast() {
131                             @Override
132                             public void onBroadcast(AtmosphereResourceEvent event) {
133                                 r.removeEventListener(this);
134                                 doNotSuspend.set(true);
135                                 logger.trace("onBroadcast");
136                             }
137                         });
138 
139                         for (String broadcasterID : tracker.ids()) {
140                             Broadcaster b = factory.lookup(broadcasterID, false);
141                             logger.trace("About to associate resource {} with Broadcaster {}", r.uuid(), broadcasterID);
142                             if (b != null && !b.getID().equalsIgnoreCase(r.getBroadcaster().getID())) {
143                                 logger.trace("Associate AtmosphereResource {} with Broadcaster {}", r.uuid(), broadcasterID);
144                                 b.addAtmosphereResource(r);
145                             } else if (b == null) {
146                                 logger.trace("Broadcaster {} is no longer available for {}", broadcasterID, r);
147                             } else {
148                                 logger.trace("AtmosphereResource {} already associated with {}", r.uuid(), broadcasterID);
149                             }
150                         }
151 
152                         /**
153                          * Check the cache to see if messages has been added directly by using
154                          * {@link BroadcasterCache#addToCache(String, org.atmosphere.cpr.AtmosphereResource, org.atmosphere.cache.BroadcastMessage)}
155                          * after {@link Broadcaster#addAtmosphereResource(org.atmosphere.cpr.AtmosphereResource)} has been
156                          * invoked.
157                          */
158                         final List<Object> cachedMessages = retrieveCache(r, tracker, true);
159                         if (logger.isTraceEnabled()) {
160                             logger.trace("message size {}", cachedMessages.size());
161                         }
162                         if (!cachedMessages.isEmpty()) {
163                             logger.trace("About to write to the cache {}", r.uuid());
164                             writeCache(r, cachedMessages);
165                             doNotSuspend.set(true);
166                         }
167 
168                         // Force doNotSuspend.
169                         if (doNotSuspend.get()) {
170                             AtmosphereResourceImpl.class.cast(r).action().type(Action.TYPE.CONTINUE);
171                         }
172                         if (logger.isTraceEnabled()) {
173                             logger.trace("doNotSuspend {}", doNotSuspend.get());
174                         }
175                     }
176                 });
177             }
178         }
179         return Action.CONTINUE;
180     }
181 
182     private BroadcasterTracker track(AtmosphereResource r) {
183         BroadcasterTracker tracker = states.get(r.uuid());
184         if (tracker == null) {
185             tracker = new BroadcasterTracker();
186             states.put(r.uuid(), tracker);
187             logger.trace("AtmosphereResource {} state now tracked", r.uuid());
188         }
189         return tracker;
190     }

193     public void postInspect(AtmosphereResource r) {
194     }

196     @Override
197     public void destroy() {
198         trackerFuture.cancel(true);
199     }

201     public final class B extends BroadcasterListenerAdapter {
202         @Override
203         public void onAddAtmosphereResource(Broadcaster b, AtmosphereResource r) {
204 
205             BroadcasterTracker t = states.get(r.uuid());
206             if (t == null) {
207                 t = track(r);
208             }
209             logger.trace("Starting tracking the state of {} with broadcaster {}", r.uuid(), b.getID());
210             t.add(b);
211         }
212 
213         @Override
214         public void onRemoveAtmosphereResource(Broadcaster b, AtmosphereResource r) {
215             // We track cancelled and resumed connection only.
216             BroadcasterTracker t = states.get(r.uuid());
217             AtmosphereResourceEvent e = r.getAtmosphereResourceEvent();
218             if (e.isClosedByClient() || !r.isResumed() && !e.isResumedOnTimeout()) {
219                 logger.trace("Deleting the state of {} with broadcaster {}", r.uuid(), b.getID());
220                 if (t != null) {
221                     t.remove(b);
222                 }
223             } else {
224                 // The BroadcasterTracker was swapped
225                 onAddAtmosphereResource(b, r);
226                 logger.trace("Keeping the state of {} with broadcaster {}", r.uuid(), b.getID());
227                 logger.trace("State for {} with broadcaster {}", r.uuid(), t != null ? t.ids() : "null");
228             }
229         }
230     }

232     public final static class BroadcasterTracker {
233 
234         private final ConcurrentLinkedQueue<String> broadcasterIds;
235         private long tick;
236 
237         public BroadcasterTracker() {
238             this.broadcasterIds = new ConcurrentLinkedQueue<String>();
239             tick = System.currentTimeMillis();
240         }
241 
242         public BroadcasterTracker add(Broadcaster b) {
243             logger.trace("Adding {}", b.getID());
244             if (!broadcasterIds.contains(b.getID())) {
245                 broadcasterIds.add(b.getID());
246             }
247             return this;
248         }
249 
250         public BroadcasterTracker remove(Broadcaster b) {
251             logger.trace("Removing {}", b.getID());
252             broadcasterIds.remove(b.getID());
253             return this;
254         }
255 
256         public ConcurrentLinkedQueue<String> ids() {
257             return broadcasterIds;
258         }
259 
260         public BroadcasterTracker tick() {
261             tick = System.currentTimeMillis();
262             return this;
263         }
264 
265         public long lastTick() {
266             return tick;
267         }
268     }
269 
270     public ConcurrentHashMap<String, BroadcasterTracker> states() {
271         return states;
272     }

274     @Override
275     public String toString() {
276         return "AtmosphereResource state recovery";
277     }

279     public List<Object> retrieveCache(AtmosphereResource r, BroadcasterTracker tracker, boolean force) {
280         List<Object> cachedMessages = new LinkedList<Object>();
281         for (String broadcasterID : tracker.ids()) {
282             Broadcaster b = factory.lookup(broadcasterID, false);
283             BroadcasterCache cache;
284             logger.trace("About to retrieve cached messages for resource {} with Broadcaster {}, tracked by " + b, r.uuid(), r.getBroadcaster());
285             if (b != null && (force || !b.getID().equalsIgnoreCase(r.getBroadcaster().getID()))) {
286                 // We cannot add the resource now. we need to first make sure there is no cached message.
287                 cache = b.getBroadcasterConfig().getBroadcasterCache();
288                 List<Object> t = cache.retrieveFromCache(b.getID(), r.uuid());
289 
290                 t = b.getBroadcasterConfig().applyFilters(r, t);
291                 if (!t.isEmpty()) {
292                     logger.trace("Found Cached Messages For AtmosphereResource {} with Broadcaster {}", r.uuid(), broadcasterID);
293                     cachedMessages.addAll(t);
294                 }
295             } else {
296                 logger.trace("Broadcaster {} is no longer available for {}", broadcasterID, r);
297             }
298         }
299         return cachedMessages;
300     }
301 
302     private void writeCache(AtmosphereResource r, List<Object> cachedMessages) {
303         try {
304             logger.trace("Writing cached messages {} for {}", cachedMessages, r.uuid());
305             r.getAtmosphereHandler().onStateChange(
306                     new AtmosphereResourceEventImpl(AtmosphereResourceImpl.class.cast(r), false, false, null)
307                             .setMessage(cachedMessages));
308         } catch (IOException e) {
309             logger.warn("Unable to recover from state recovery {}", r.uuid(), e);
310         }
311     }

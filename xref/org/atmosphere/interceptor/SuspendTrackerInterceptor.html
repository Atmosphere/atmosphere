19  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;

21  import org.atmosphere.cpr.AtmosphereResource;
22  import org.atmosphere.cpr.AtmosphereResourceEvent;
23  import org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter;
24  import org.atmosphere.cpr.AtmosphereResourceImpl;
25  import org.atmosphere.util.Utils;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import java.util.Collections;
30  import java.util.HashSet;
31  import java.util.Set;
32  
33  import static org.atmosphere.cpr.HeaderConfig.X_ATMOSPHERE_TRACKING_ID;
34  
35  /**
36   * An interceptor that keep track of {@link AtmosphereResource#uuid()} and disable invocation of {@link org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter#onSuspend(org.atmosphere.cpr.AtmosphereResourceEvent)}
37   * and {@link org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter#onPreSuspend(org.atmosphere.cpr.AtmosphereResourceEvent)}
38   * </p>
39   * When used, the onSuspend will be only called ONCE for every transport, when the first request is made.
40   *
41   * @author Jeanfrancois Arcand
42   */
43  public class SuspendTrackerInterceptor extends AtmosphereInterceptorAdapter {
44  
45      private final Set<String> trackedUUID = Collections.synchronizedSet(new HashSet<String>());
46      private final Logger logger = LoggerFactory.getLogger(SuspendTrackerInterceptor.class);
47  
48      @Override
49      public Action inspect(final AtmosphereResource r) {
50          if (Utils.webSocketMessage(r)) return Action.CONTINUE;
51  
52          final AtmosphereRequest request = AtmosphereResourceImpl.class.cast(r).getRequest(false);
53          boolean connecting = request.getHeader(X_ATMOSPHERE_TRACKING_ID) != null && request.getHeader(X_ATMOSPHERE_TRACKING_ID).equals("0");
54  
55          if (!connecting && !Utils.pollableTransport(r.transport())) {
56              if (!trackedUUID.add(r.uuid())) {
57                  logger.trace("Blocking {} from suspend", r.uuid());
58                  AtmosphereResourceImpl.class.cast(r).disableSuspendEvent(true);
59              }
60  
61              r.addEventListener(new AtmosphereResourceEventListenerAdapter() {
62                  @Override
63                  public void onDisconnect(AtmosphereResourceEvent event) {
64                      logger.trace("Untracking {}", r.uuid());
65                      trackedUUID.remove(r.uuid());
66                  }
67  
68                  @Override
69                  public void onClose(AtmosphereResourceEvent event) {
70                      onDisconnect(event);
71                  }
72              });
73          }
74          return Action.CONTINUE;
75      }
76  
77      @Override
78      public String toString() {
79          return "UUID Tracking Interceptor";
80      }
81  
82      @Override
83      public PRIORITY priority() {
84          return InvokationOrder.BEFORE_DEFAULT;
85      }
86  }

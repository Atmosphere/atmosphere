16  package org.atmosphere.interceptor;
17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;
20  import org.atmosphere.cpr.AtmosphereRequest;
21  import org.atmosphere.cpr.AtmosphereResource;
22  import org.atmosphere.cpr.AtmosphereResourceEvent;
23  import org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter;
24  import org.atmosphere.cpr.AtmosphereResourceImpl;
25  import org.atmosphere.cpr.HeaderConfig;
26  import org.atmosphere.util.Utils;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import java.util.Collections;
31  import java.util.HashSet;
32  import java.util.Set;
33  
34  /**
35   * An interceptor that keep track of {@link AtmosphereResource#uuid()} and disable invocation of {@link org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter#onSuspend(org.atmosphere.cpr.AtmosphereResourceEvent)}
36   * and {@link org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter#onPreSuspend(org.atmosphere.cpr.AtmosphereResourceEvent)}
37   * </p>
38   * When used, the onSuspend will be only called ONCE for every transport, when the first request is made.
39   *
40   * @author Jeanfrancois Arcand
41   */
42  public class SuspendTrackerInterceptor extends AtmosphereInterceptorAdapter {
43  
44      private final Set<String> trackedUUID = Collections.synchronizedSet(new HashSet<String>());
45      private final Logger logger = LoggerFactory.getLogger(SuspendTrackerInterceptor.class);
46  
47      @Override
48      public Action inspect(final AtmosphereResource r) {
49  
50          if (Utils.webSocketMessage(r)) return Action.CONTINUE;
51  
52          final AtmosphereRequest request = AtmosphereResourceImpl.class.cast(r).getRequest(false);
53          boolean connecting = request.getHeader(HeaderConfig.X_ATMOSPHERE_TRACKING_ID) != null && request.getHeader(HeaderConfig.X_ATMOSPHERE_TRACKING_ID).equals("0");
54  
55          if (!connecting && !Utils.pollableTransport(r.transport())) {
56              if (!trackedUUID.add(r.uuid())) {
57                  logger.trace("Blocking {} from suspend", r.uuid());
58                  AtmosphereResourceImpl.class.cast(r).disableSuspendEvent(true);
59              }
60  
61              r.addEventListener(new AtmosphereResourceEventListenerAdapter() {
62                  @Override
63                  public void onDisconnect(AtmosphereResourceEvent event) {
64                      logger.trace("Untracking {}", r.uuid());
65                      trackedUUID.remove(r.uuid());
66                  }
67  
68                  @Override
69                  public void onClose(AtmosphereResourceEvent event) {
70                      onDisconnect(event);
71                  }
72              });
73          }
74          return Action.CONTINUE;
75      }
76  
77      public Set<String> trackedUUID(){
78          return trackedUUID;
79      }
80  
81      @Override
82      public String toString() {
83          return "UUID Tracking Interceptor";
84      }
85  
86      @Override
87      public PRIORITY priority() {
88          return InvokationOrder.BEFORE_DEFAULT;
89      }
90  }

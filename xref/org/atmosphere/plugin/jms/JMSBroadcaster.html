58  /**
59   * Simple {@link org.atmosphere.cpr.Broadcaster} implementation based on JMS
60   * <p/>
61   * The {@link ConnectionFactory} name's is jms/atmosphereFactory The
62   * {@link Topic} by constructing "BroadcasterId =
63   * {@link org.atmosphere.cpr.Broadcaster#getID}
64   *
65   * @author Jeanfrancois Arcand
66   */
67  public class JMSBroadcaster extends AbstractBroadcasterProxy {
68      private static final String JMS_TOPIC = JMSBroadcaster.class.getName() + ".topic";
69      private static final String JNDI_NAMESPACE = JMSBroadcaster.class.getName() + ".JNDINamespace";
70      private static final String JNDI_FACTORY_NAME = JMSBroadcaster.class.getName() + ".JNDIConnectionFactoryName";
71      private static final String JNDI_TOPIC = JMSBroadcaster.class.getName() + ".JNDITopic";
72      private static final Logger logger = LoggerFactory.getLogger(JMSBroadcaster.class);
73  
74      private Connection connection;
75      private Session consumerSession;
76      private Session publisherSession;
77      private Topic topic;
78      private MessageConsumer consumer;
79      private MessageProducer publisher;
80  
81      private String topicId = "atmosphere";
82      private String factoryName = "atmosphereFactory";
83      private String namespace = "jms/";
84  
85      public JMSBroadcaster(String id, AtmosphereConfig config) {
86          super(id, null, config);
87          setUp();
88      }
89  
90      private void setUp() {
91          try {
92              // For backward compatibility.
93              if (config.getInitParameter(JMS_TOPIC) != null) {
94                  topicId = config.getInitParameter(JMS_TOPIC);
95              }
96  
97              if (config.getInitParameter(JNDI_NAMESPACE) != null) {
98                  namespace = config.getInitParameter(JNDI_NAMESPACE);
99              }
100 
101             if (config.getInitParameter(JNDI_FACTORY_NAME) != null) {
102                 factoryName = config.getInitParameter(JNDI_FACTORY_NAME);
103             }
104 
105             if (config.getInitParameter(JNDI_TOPIC) != null) {
106                 topicId = config.getInitParameter(JNDI_TOPIC);
107             }
108 
109             logger.info("Looking up Connection Factory {}", namespace + factoryName);
110             Context ctx = new InitialContext();
111             ConnectionFactory connectionFactory = (ConnectionFactory) ctx.lookup(namespace + factoryName);
112 
113             logger.info("Looking up topic: {}", topicId);
114             topic = (Topic) ctx.lookup(namespace + topicId);
115 
116             connection = connectionFactory.createConnection();
117             consumerSession = connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
118             publisherSession = connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
119 
120             publisher = publisherSession.createProducer(topic);
121             connection.start();
122             logger.info("JMS created for topic {}", topicId);
123             // Unfortunately we need the getID() to complete the configuration
124             // But setID() is called after configure(), therefore we do the
125             // rest of the configuration in incomingBroadcast() (which is called
126             // once during configuration). We cannot do all the configuration in
127             // incomingBroadcast() though, as using bc.getAtmosphereConfig() would
128             // introduce a race condition (the configuration is loaded in a different
129             // thread).
130         } catch (Exception e) {
131             String msg = "Unable to configure JMSBroadcaster";
132             logger.error(msg, e);
133             throw new RuntimeException(msg, e);
134         }
135     }
136 
137     /**
138      * {@inheritDoc}
139      */
140     @Override
141     public void incomingBroadcast() {
142         // we setup the consumer in the setID method. No need to do it here too
143     }
144 
145     @Override
146     public void setID(String id) {
147         super.setID(id);
148         restartConsumer();
149     }
150 
151     void restartConsumer() {
152         try {
153             String id = getID();
154             if (id.startsWith("/*")) {
155                 id = "atmosphere";
156             }
157             if (consumer != null) {
158                 consumer.close();
159                 consumer = null;
160             }
161 
162             logger.info("Create JMS consumer: {}", id);
163             String selector = String.format("BroadcasterId = '%s'", id);
164 
165             consumer = consumerSession.createConsumer(topic, selector);
166             consumer.setMessageListener(new MessageListener() {
167                 @Override
168                 public void onMessage(Message msg) {
169                     try {
170                         TextMessage textMessage = (TextMessage) msg;
171                         String message = textMessage.getText();
172 
173                         if (message != null && bc != null) {
174                             broadcastReceivedMessage(message);
175                         }
176                     } catch (JMSException ex) {
177                         logger.warn("Failed to broadcast message", ex);
178                     }
179                 }
180             });
181             logger.info("Consumer created for topic {}, with filter {}",
182                     topicId, selector);
183         } catch (Throwable ex) {
184             String msg = "Unable to initialize JMSBroadcaster";
185             logger.error(msg, ex);
186             throw new IllegalStateException(msg, ex);
187         }
188     }
189 
190     /**
191      * {@inheritDoc}
192      */
193     @Override
194     public void outgoingBroadcast(Object message) {
195         try {
196             String id = getID();
197             if (id.startsWith("/*")) {
198                 id = "atmosphere";
199             }
200 
201             if (publisherSession == null) {
202                 throw new IllegalStateException("JMS Session is null");
203             }
204 
205             TextMessage textMessage = publisherSession.createTextMessage(message
206                     .toString());
207             textMessage.setStringProperty("BroadcasterId", id);
208             publisher.send(textMessage);
209         } catch (JMSException ex) {
210             logger.warn("Failed to send message over JMS", ex);
211         }
212     }
213 
214     /**
215      * Close all related JMS factory, connection, etc.
216      */
217     @Override
218     public synchronized void releaseExternalResources() {
219         try {
220             connection.close();
221             consumerSession.close();
222             publisherSession.close();
223             consumer.close();
224             publisher.close();
225         } catch (Throwable ex) {
226             logger.warn("releaseExternalResources", ex);
227         }
228     }

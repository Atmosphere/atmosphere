55  package org.atmosphere.util;
56  
57  import java.lang.reflect.InvocationHandler;
58  import java.lang.reflect.InvocationTargetException;
59  import java.lang.reflect.Method;
60  
61  /**
62   * This class has been highly inspired by the Jersey's 1.x.
63   * @param <T>
64   */
65  public class ThreadLocalInvoker<T> implements InvocationHandler {
66  
67      private ThreadLocal<T> threadLocalInstance = new ThreadLocal<T>();
68  
69      public void set(T threadLocalInstance) {
70          this.threadLocalInstance.set(threadLocalInstance);
71      }
72  
73      public T get() {
74          return this.threadLocalInstance.get();
75      }
76  
77      public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
78          if (threadLocalInstance.get() == null) {
79              throw new IllegalStateException("No thread local" + proxy.getClass());
80          }
81  
82          try {
83              return method.invoke(threadLocalInstance.get(), args);
84          } catch (IllegalAccessException ex) {
85              throw new IllegalStateException(ex);
86          } catch (InvocationTargetException ex) {
87              throw ex.getTargetException();
88          }
89      }
90  }

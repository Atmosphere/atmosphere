27  /**
28   * A package-local class holding common representation and mechanics
29   * for classes supporting dynamic striping on 64bit values. The class
30   * extends Number so that concrete subclasses must publicly do so.
31   */
32  @SuppressWarnings("all")
33  abstract class Striped64 extends Number {
34      /*
35       * This class maintains a lazily-initialized table of atomically
36       * updated variables, plus an extra "base" field. The table size
37       * is a power of two. Indexing uses masked per-thread hash codes.
38       * Nearly all declarations in this class are package-private,
39       * accessed directly by subclasses.
40       *
41       * Table entries are of class Cell; a variant of AtomicLong padded
42       * to reduce cache contention on most processors. Padding is
43       * overkill for most Atomics because they are usually irregularly
44       * scattered in memory and thus don't interfere much with each
45       * other. But Atomic objects residing in arrays will tend to be
46       * placed adjacent to each other, and so will most often share
47       * cache lines (with a huge negative performance impact) without
48       * this precaution.
49       *
50       * In part because Cells are relatively large, we avoid creating
51       * them until they are needed.  When there is no contention, all
52       * updates are made to the base field.  Upon first contention (a
53       * failed CAS on base update), the table is initialized to size 2.
54       * The table size is doubled upon further contention until
55       * reaching the nearest power of two greater than or equal to the
56       * number of CPUS. Table slots remain empty (null) until they are
57       * needed.
58       *
59       * A single spinlock ("busy") is used for initializing and
60       * resizing the table, as well as populating slots with new Cells.
61       * There is no need for a blocking lock; when the lock is not
62       * available, threads try other slots (or the base).  During these
63       * retries, there is increased contention and reduced locality,
64       * which is still better than alternatives.
65       *
66       * Per-thread hash codes are initialized to random values.
67       * Contention and/or table collisions are indicated by failed
68       * CASes when performing an update operation (see method
69       * retryUpdate). Upon a collision, if the table size is less than
70       * the capacity, it is doubled in size unless some other thread
71       * holds the lock. If a hashed slot is empty, and lock is
72       * available, a new Cell is created. Otherwise, if the slot
73       * exists, a CAS is tried.  Retries proceed by "double hashing",
74       * using a secondary hash (Marsaglia XorShift) to try to find a
75       * free slot.
76       *
77       * The table size is capped because, when there are more threads
78       * than CPUs, supposing that each thread were bound to a CPU,
79       * there would exist a perfect hash function mapping threads to
80       * slots that eliminates collisions. When we reach capacity, we
81       * search for this mapping by randomly varying the hash codes of
82       * colliding threads.  Because search is random, and collisions
83       * only become known via CAS failures, convergence can be slow,
84       * and because threads are typically not bound to CPUS forever,
85       * may not occur at all. However, despite these limitations,
86       * observed contention rates are typically low in these cases.
87       *
88       * It is possible for a Cell to become unused when threads that
89       * once hashed to it terminate, as well as in the case where
90       * doubling the table causes no thread to hash to it under
91       * expanded mask.  We do not try to detect or remove such cells,
92       * under the assumption that for long-running instances, observed
93       * contention levels will recur, so the cells will eventually be
94       * needed again; and for short-lived ones, it does not matter.
95       */
96  
97      /**
98       * Padded variant of AtomicLong supporting only raw accesses plus CAS.
99       * The value field is placed between pads, hoping that the JVM doesn't
100      * reorder them.
101      *
102      * JVM intrinsics note: It would be possible to use a release-only
103      * form of CAS here, if it were provided.
104      */
105     static final class Cell {
106         volatile long p0, p1, p2, p3, p4, p5, p6;
107         volatile long value;
108         volatile long q0, q1, q2, q3, q4, q5, q6;
109         Cell(long x) { value = x; }
110 
111         final boolean cas(long cmp, long val) {
112             return UNSAFE.compareAndSwapLong(this, valueOffset, cmp, val);
113         }
114 
115         // Unsafe mechanics
116         private static final sun.misc.Unsafe UNSAFE;
117         private static final long valueOffset;
118         static {
119             try {
120                 UNSAFE = getUnsafe();
121                 Class<?> ak = Cell.class;
122                 valueOffset = UNSAFE.objectFieldOffset
123                         (ak.getDeclaredField("value"));
124             } catch (Exception e) {
125                 throw new Error(e);
126             }
127         }
128 
129     }
130 
131     /**
132      * Holder for the thread-local hash code. The code is initially
133      * random, but may be set to a different value upon collisions.
134      */
135     static final class HashCode {
136         static final Random rng = new Random();
137         int code;
138         HashCode() {
139             int h = rng.nextInt(); // Avoid zero to allow xorShift rehash
140             code = (h == 0) ? 1 : h;
141         }
142     }
143 
144     /**
145      * The corresponding ThreadLocal class
146      */
147     static final class ThreadHashCode extends ThreadLocal<HashCode> {
148         public HashCode initialValue() { return new HashCode(); }
149     }
150 
151     /**
152      * Static per-thread hash codes. Shared across all instances to
153      * reduce ThreadLocal pollution and because adjustments due to
154      * collisions in one table are likely to be appropriate for
155      * others.
156      */
157     static final ThreadHashCode threadHashCode = new ThreadHashCode();
158 
159     /** Number of CPUS, to place bound on table size */
160     static final int NCPU = Runtime.getRuntime().availableProcessors();
161 
162     /**
163      * Table of cells. When non-null, size is a power of 2.
164      */
165     transient volatile Cell[] cells;
166 
167     /**
168      * Base value, used mainly when there is no contention, but also as
169      * a fallback during table initialization races. Updated via CAS.
170      */
171     transient volatile long base;
172 
173     /**
174      * Spinlock (locked via CAS) used when resizing and/or creating Cells.
175      */
176     transient volatile int busy;
177 
178     /**
179      * Package-private default constructor
180      */
181     Striped64() {
182     }
183 
184     /**
185      * CASes the base field.
186      */
187     final boolean casBase(long cmp, long val) {
188         return UNSAFE.compareAndSwapLong(this, baseOffset, cmp, val);
189     }
190 
191     /**
192      * CASes the busy field from 0 to 1 to acquire lock.
193      */
194     final boolean casBusy() {
195         return UNSAFE.compareAndSwapInt(this, busyOffset, 0, 1);
196     }
197 
198     /**
199      * Computes the function of current and new value. Subclasses
200      * should open-code this update function for most uses, but the
201      * virtualized form is needed within retryUpdate.
202      *
203      * @param currentValue the current value (of either base or a cell)
204      * @param newValue the argument from a user update call
205      * @return result of the update function
206      */
207     abstract long fn(long currentValue, long newValue);
208 
209     /**
210      * Handles cases of updates involving initialization, resizing,
211      * creating new Cells, and/or contention. See above for
212      * explanation. This method suffers the usual non-modularity
213      * problems of optimistic retry code, relying on rechecked sets of
214      * reads.
215      *
216      * @param x the value
217      * @param hc the hash code holder
218      * @param wasUncontended false if CAS failed before call
219      */
220     final void retryUpdate(long x, HashCode hc, boolean wasUncontended) {
221         int h = hc.code;
222         boolean collide = false;                // True if last slot nonempty
223         for (;;) {
224             Cell[] as; Cell a; int n; long v;
225             if ((as = cells) != null && (n = as.length) > 0) {
226                 if ((a = as[(n - 1) & h]) == null) {
227                     if (busy == 0) {            // Try to attach new Cell
228                         Cell r = new Cell(x);   // Optimistically create
229                         if (busy == 0 && casBusy()) {
230                             boolean created = false;
231                             try {               // Recheck under lock
232                                 Cell[] rs; int m, j;
233                                 if ((rs = cells) != null &&
234                                         (m = rs.length) > 0 &&
235                                         rs[j = (m - 1) & h] == null) {
236                                     rs[j] = r;
237                                     created = true;
238                                 }
239                             } finally {
240                                 busy = 0;
241                             }
242                             if (created)
243                                 break;
244                             continue;           // Slot is now non-empty
245                         }
246                     }
247                     collide = false;
248                 }
249                 else if (!wasUncontended)       // CAS already known to fail
250                     wasUncontended = true;      // Continue after rehash
251                 else if (a.cas(v = a.value, fn(v, x)))
252                     break;
253                 else if (n >= NCPU || cells != as)
254                     collide = false;            // At max size or stale
255                 else if (!collide)
256                     collide = true;
257                 else if (busy == 0 && casBusy()) {
258                     try {
259                         if (cells == as) {      // Expand table unless stale
260                             Cell[] rs = new Cell[n << 1];
261                             for (int i = 0; i < n; ++i)
262                                 rs[i] = as[i];
263                             cells = rs;
264                         }
265                     } finally {
266                         busy = 0;
267                     }
268                     collide = false;
269                     continue;                   // Retry with expanded table
270                 }
271                 h ^= h << 13;                   // Rehash
272                 h ^= h >>> 17;
273                 h ^= h << 5;
274             }
275             else if (busy == 0 && cells == as && casBusy()) {
276                 boolean init = false;
277                 try {                           // Initialize table
278                     if (cells == as) {
279                         Cell[] rs = new Cell[2];
280                         rs[h & 1] = new Cell(x);
281                         cells = rs;
282                         init = true;
283                     }
284                 } finally {
285                     busy = 0;
286                 }
287                 if (init)
288                     break;
289             }
290             else if (casBase(v = base, fn(v, x)))
291                 break;                          // Fall back on using base
292         }
293         hc.code = h;                            // Record index for next time
294     }

297     /**
298      * Sets base and all cells to the given value.
299      */
300     final void internalReset(long initialValue) {
301         Cell[] as = cells;
302         base = initialValue;
303         if (as != null) {
304             int n = as.length;
305             for (int i = 0; i < n; ++i) {
306                 Cell a = as[i];
307                 if (a != null)
308                     a.value = initialValue;
309             }
310         }
311     }

313     // Unsafe mechanics
314     private static final sun.misc.Unsafe UNSAFE;
315     private static final long baseOffset;
316     private static final long busyOffset;
317     static {
318         try {
319             UNSAFE = getUnsafe();
320             Class<?> sk = Striped64.class;
321             baseOffset = UNSAFE.objectFieldOffset
322                     (sk.getDeclaredField("base"));
323             busyOffset = UNSAFE.objectFieldOffset
324                     (sk.getDeclaredField("busy"));
325         } catch (Exception e) {
326             throw new Error(e);
327         }
328     }

330     /**
331      * Returns a sun.misc.Unsafe.  Suitable for use in a 3rd party package.
332      * Replace with a simple call to Unsafe.getUnsafe when integrating
333      * into a jdk.
334      *
335      * @return a sun.misc.Unsafe
336      */
337     private static sun.misc.Unsafe getUnsafe() {
338         try {
339             return sun.misc.Unsafe.getUnsafe();
340         } catch (SecurityException tryReflectionInstead) {}
341         try {
342             return java.security.AccessController.doPrivileged
343                     (new java.security.PrivilegedExceptionAction<sun.misc.Unsafe>() {
344                         public sun.misc.Unsafe run() throws Exception {
345                             Class<sun.misc.Unsafe> k = sun.misc.Unsafe.class;
346                             for (java.lang.reflect.Field f : k.getDeclaredFields()) {
347                                 f.setAccessible(true);
348                                 Object x = f.get(null);
349                                 if (k.isInstance(x))
350                                     return k.cast(x);
351                             }
352                             throw new NoSuchFieldError("the Unsafe");
353                         }});
354         } catch (java.security.PrivilegedActionException e) {
355             throw new RuntimeException("Could not initialize intrinsics",
356                     e.getCause());
357         }
358     }

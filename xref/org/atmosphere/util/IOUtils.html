73      /**
74       * <p>
75       * Delivers the given message according to the specified {@link org.atmosphere.config.service.DeliverTo configuration).
76       * </p>
77       *
78       * @param o              the message
79       * @param deliverConfig  the annotation state
80       * @param defaultDeliver the strategy applied if deliverConfig is {@code null}
81       * @param r              the resource
82       */
83      public static void deliver(final Object o,
84                                 final DeliverTo deliverConfig,
85                                 final DeliverTo.DELIVER_TO defaultDeliver,
86                                 final AtmosphereResource r) {
87          final DeliverTo.DELIVER_TO deliverTo = deliverConfig == null ? defaultDeliver : deliverConfig.value();
88          switch (deliverTo) {
89              case RESOURCE:
90                  r.getBroadcaster().broadcast(o, r);
91                  break;
92              case BROADCASTER:
93                  r.getBroadcaster().broadcast(o);
94                  break;
95              case ALL:
96                  for (Broadcaster b : r.getAtmosphereConfig().getBroadcasterFactory().lookupAll()) {
97                      b.broadcast(o);
98                  }
99                  break;
100 
101         }
102     }

104     public static Object readEntirely(AtmosphereResource r) throws IOException {
105         AtmosphereRequest request = r.getRequest();
106         return isBodyBinary(request) ? readEntirelyAsByte(r) : readEntirelyAsString(r).toString();
107     }

109     public final static boolean isBodyBinary(AtmosphereRequest request) {
110         if (request.getContentType() != null
111                 && request.getContentType().equalsIgnoreCase(FORCE_BINARY) || request.getHeader(X_ATMO_BINARY) != null)
112             return true;
113         return false;
114     }

116     public final static boolean isBodyEmpty(Object o) {
117         if (o != null && (String.class.isAssignableFrom(o.getClass()) && String.class.cast(o).isEmpty())
118                 || (Byte[].class.isAssignableFrom(o.getClass()) && Byte[].class.cast(o).length == 0)) {
119             return true;
120         } else {
121             return false;
122         }
123     }

125     public static StringBuilder readEntirelyAsString(AtmosphereResource r) throws IOException {
126         final StringBuilder stringBuilder = new StringBuilder();
127 
128         boolean readGetBody = r.getAtmosphereConfig().getInitParameter(ApplicationConfig.READ_GET_BODY, false);
129         if (!readGetBody && AtmosphereResourceImpl.class.cast(r).getRequest(false).getMethod().equalsIgnoreCase("GET")) {
130             logger.debug("Blocking an I/O read operation from a GET request. To enable GET + body, set {} to true", ApplicationConfig.READ_GET_BODY);
131             return stringBuilder;
132         }
133 
134         AtmosphereRequest request = r.getRequest();
135         if (request.body().isEmpty()) {
136             BufferedReader bufferedReader = null;
137             try {
138                 try {
139                     InputStream inputStream = request.getInputStream();
140                     if (inputStream != null) {
141                         bufferedReader = new BufferedReader(new InputStreamReader(inputStream));
142                     }
143                 } catch (IllegalStateException ex) {
144                     logger.trace("", ex);
145                     Reader reader = request.getReader();
146                     if (reader != null) {
147                         bufferedReader = new BufferedReader(reader);
148                     }
149                 }
150 
151                 if (bufferedReader != null) {
152                     char[] charBuffer = new char[8192];
153                     int bytesRead = -1;
154                     try {
155                         while ((bytesRead = bufferedReader.read(charBuffer)) > 0) {
156                             stringBuilder.append(charBuffer, 0, bytesRead);
157                         }
158                     } catch (NullPointerException ex) {
159                         // https://java.net/jira/browse/GRIZZLY-1676
160                     }
161                 } else {
162                     stringBuilder.append("");
163                 }
164             } finally {
165                 if (bufferedReader != null) {
166                     try {
167                         bufferedReader.close();
168                     } catch (IOException ex) {
169                         logger.warn("", ex);
170                     }
171                 }
172             }
173         } else {
174             AtmosphereRequestImpl.Body body = request.body();
175             try {
176                 stringBuilder.append(body.hasString() ? body.asString() : new String(body.asBytes(), body.byteOffset(), body.byteLength(), request.getCharacterEncoding()));
177             } catch (UnsupportedEncodingException e) {
178                 logger.error("", e);
179             }
180         }
181         return stringBuilder;
182     }

184     public static byte[] readEntirelyAsByte(AtmosphereResource r) throws IOException {
185         AtmosphereRequest request = r.getRequest();
186 
187         boolean readGetBody = r.getAtmosphereConfig().getInitParameter(ApplicationConfig.READ_GET_BODY, false);
188         if (!readGetBody && AtmosphereResourceImpl.class.cast(r).getRequest(false).getMethod().equalsIgnoreCase("GET")) {
189             logger.debug("Blocking an I/O read operation from a GET request. To enable GET + body, set {} to true", ApplicationConfig.READ_GET_BODY);
190             return new byte[0];
191         }
192 
193         AtmosphereRequestImpl.Body body = request.body();
194         if (request.body().isEmpty()) {
195             BufferedInputStream bufferedStream = null;
196             ByteArrayOutputStream bbIS = new ByteArrayOutputStream();
197             try {
198                 try {
199                     InputStream inputStream = request.getInputStream();
200                     if (inputStream != null) {
201                         bufferedStream = new BufferedInputStream(inputStream);
202                     }
203                 } catch (IllegalStateException ex) {
204                     logger.trace("", ex);
205                     Reader reader = request.getReader();
206                     if (reader != null) {
207                         bufferedStream = new BufferedInputStream(new ReaderInputStream(reader));
208                     }
209                 }
210 
211                 if (bufferedStream != null) {
212                     byte[] bytes = new byte[8192];
213                     int bytesRead = 0;
214                     while (bytesRead != -1) {
215                         bytesRead = bufferedStream.read(bytes);
216                         if (bytesRead > 0)
217                             bbIS.write(bytes, 0, bytesRead);
218                     }
219 
220                 } else {
221                     bbIS.write("".getBytes());
222                 }
223             } finally {
224                 if (bufferedStream != null) {
225                     try {
226                         bufferedStream.close();
227                     } catch (IOException ex) {
228                         logger.warn("", ex);
229                     }
230                 }
231             }
232             return bbIS.toByteArray();
233         } else if (body.hasString()) {
234             try {
235                 return readEntirelyAsString(r).toString().getBytes(request.getCharacterEncoding());
236             } catch (UnsupportedEncodingException e) {
237                 logger.error("", e);
238             }
239         } else if (body.hasBytes()) {
240             return Arrays.copyOfRange(body.asBytes(), body.byteOffset(), body.byteOffset() + body.byteLength());
241         }
242         throw new IllegalStateException("No body " + r);
243     }

245     public static String guestServletPath(AtmosphereConfig config) {
246         String servletPath = "";
247         if (config.getServletConfig() != null) {
248             servletPath = getCleanedServletPath(guestRawServletPath(config));
249         } else {
250             throw new IllegalStateException("Unable to configure jsr356 at that stage");
251         }
252         return servletPath;
253     }

255     public static String guestRawServletPath(AtmosphereConfig config) {
256         String servletPath = "";
257         try {
258             if (config.getServletConfig() != null) {
259                 ServletRegistration s = config.getServletContext().getServletRegistration(config.getServletConfig().getServletName());
260 
261                 if (s == null) {
262                     s = config.getServletContext().getServletRegistration(VoidServletConfig.ATMOSPHERE_SERVLET);
263                 }
264 
265                 if ( s == null) {
266                     for (Map.Entry<String, ? extends ServletRegistration> servlet : config.getServletContext().getServletRegistrations().entrySet()) {
267                         if (knownClasses.contains(servlet.getValue().getClassName())) {
268                             s = servlet.getValue();
269                             break;
270                         }
271                     }
272 
273                     if (s == null) {
274                         throw new IllegalStateException("Unable to configure jsr356 at that stage. No Servlet associated with "
275                                 + config.getServletConfig().getServletName());
276                     }
277                 }
278 
279                 if (s.getMappings().size() > 1) {
280                     logger.warn("More than one Servlet Mapping defined. WebSocket may not work {}", s);
281                 }
282 
283                 for (String m : s.getMappings()) {
284                     servletPath = m;
285                 }
286             } else {
287                 throw new IllegalStateException("Unable to configure jsr356 at that stage");
288             }
289             return servletPath;
290         } catch (Exception ex) {
291             logger.error("", ex);
292             throw new IllegalStateException("Unable to configure jsr356 at that stage");
293         }
294     }

297     /**
298      * Used to remove trailing slash and wildcard from a servlet path.<br/><br/>
299      * Examples :<br/>
300      * - "/foo/" becomes "/foo"<br/>
301      * - "foo/bar" becomes "/foo/bar"<br/>
302      *
303      * @param fullServletPath : Servlet mapping
304      * @return Servlet mapping without trailing slash and wildcard
305      */
306     public static String getCleanedServletPath(String fullServletPath) {
307 
308         if (fullServletPath.equalsIgnoreCase("/*")) return "";
309 
310         Matcher matcher = SERVLET_PATH_PATTERN.matcher(fullServletPath);
311 
312         // It should not happen if the servlet path is valid
313         if (!matcher.find()) return fullServletPath;
314 
315         String servletPath = matcher.group(0);
316         if (!servletPath.startsWith("/")) {
317             servletPath = "/" + servletPath;
318         }
319 
320         return servletPath;
321     }

323     private static boolean scanForAtmosphereFramework(Class<?> classToScan) {
324         if (classToScan == null) return false;
325 
326         logger.trace("Scanning {}", classToScan.getName());
327 
328         // Before doing the Siberian traversal, look locally
329         if (knownClasses.contains(classToScan.getName())) {
330             return true;
331         }
332 
333         try {
334             Field[] fields = classToScan.getDeclaredFields();
335             for (Field f : fields) {
336                 f.setAccessible(true);
337                 if (AtmosphereFramework.class.isAssignableFrom(f.getType())) {
338                     return true;
339                 }
340             }
341         } catch (Exception ex) {
342             logger.trace("", ex);
343         }
344 
345         // Now try with parent
346         if (scanForAtmosphereFramework(classToScan.getSuperclass())) return true;
347         return false;
348     }

351         try {
352             return Thread.currentThread().getContextClassLoader().loadClass(className);
353         } catch (Throwable t) {
354             return thisClass.getClassLoader().loadClass(className);
355         }
356     }

358     public static boolean isAtmosphere(String className) {
359         Class<? extends AtmosphereServlet> clazz;
360         try {
361             clazz = (Class<? extends AtmosphereServlet>) Thread.currentThread().getContextClassLoader().loadClass(className);
362         } catch (Throwable t) {
363             try {
364                 clazz = (Class<? extends AtmosphereServlet>) IOUtils.class.getClassLoader().loadClass(className);
365             } catch (Exception ex) {
366                 return false;
367             }
368         }
369         return AtmosphereServlet.class.isAssignableFrom(clazz);
370     }

372     /**
373      * <p>
374      * This method reads the given file stored under "META-INF/services" and accessed through the framework's class loader
375      * to specify a list of {@link org.atmosphere.cpr.AtmosphereFramework.MetaServiceAction actions} to be done on different
376      * service classes ({@link org.atmosphere.cpr.AtmosphereInterceptor}, {@link org.atmosphere.cpr.BroadcastFilter}, etc).
377      * </p>
378      * <p/>
379      * <p>
380      * The file content should follows the following format:
381      * <pre>
382      * INSTALL
383      * com.mycompany.MyInterceptor
384      * com.mycompany.MyFilter
385      * EXCLUDE
386      * org.atmosphere.interceptor.HeartbeatInterceptor
387      * </pre>
388      * </p>
389      * <p/>
390      * <p>
391      * If you don't specify any {@link org.atmosphere.cpr.AtmosphereFramework.MetaServiceAction} before a class, then
392      * default action will be {@link org.atmosphere.cpr.AtmosphereFramework.MetaServiceAction#INSTALL}.
393      * </p>
394      * <p/>
395      * <p>
396      * Important note: you must specify a class declared inside a package. Since creating classes in the source root is
397      * a bad practice, the method does not deal with it to improve its performances.
398      * </p>
399      *
400      * @param path the service file to read
401      * @return the map associating class to action
402      */
403     public static Map<String, AtmosphereFramework.MetaServiceAction> readServiceFile(final String path) {
404         final Map<String, AtmosphereFramework.MetaServiceAction> b = new LinkedHashMap<String, AtmosphereFramework.MetaServiceAction>();
405 
406         String line;
407         InputStream is = null;
408         BufferedReader reader = null;
409         AtmosphereFramework.MetaServiceAction action = AtmosphereFramework.MetaServiceAction.INSTALL;
410 
411         try {
412             is = AtmosphereFramework.class.getClassLoader().getResourceAsStream(path);
413 
414             if (is == null) {
415                 logger.trace("META-INF/services/{} not found in class loader", path);
416                 return b;
417             }
418 
419             reader = new BufferedReader(new InputStreamReader(is));
420 
421             while (true) {
422                 line = reader.readLine();
423 
424                 if (line == null) {
425                     break;
426                 } else if (line.isEmpty()) {
427                     continue;
428                 } else if (line.indexOf('.') == -1) {
429                     action = AtmosphereFramework.MetaServiceAction.valueOf(line);
430                 } else {
431                     b.put(line, action);
432                 }
433             }
434             logger.info("Successfully loaded and installed {}", path);
435         } catch (IOException e) {
436             logger.trace("Unable to read META-INF/services/{} from class loader", path, e);
437         } finally {
438             close(is, reader);
439         }
440 
441         return b;
442     }

444     /**
445      * <p>
446      * Tries to close the given objects and log the {@link IOException} at INFO level
447      * to make the code more readable when we assume that the {@link IOException} won't be managed.
448      * </p>
449      * <p/>
450      * <p>
451      * Also ignore {@code null} parameters.
452      * </p>
453      *
454      * @param closeableArray the objects to close
455      */
456     public static void close(final Closeable... closeableArray) {
457         for (Closeable closeable : closeableArray) {
458             try {
459                 if (closeable != null) {
460                     closeable.close();
461                 }
462             } catch (IOException ioe) {
463                 logger.info("Can't close the object", ioe);
464             }
465         }
466     }

468     public static String realPath(ServletContext servletContext, String targetPath) throws MalformedURLException {
469         String realPath = servletContext.getRealPath(targetPath);
470         if (realPath == null) {
471             URL u = servletContext.getResource(targetPath);
472             if (u != null) {
473                 realPath = u.getPath();
474             } else {
475                 return "";
476             }
477         }
478         return realPath;
479     }

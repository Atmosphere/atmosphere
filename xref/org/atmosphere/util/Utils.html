21  import javax.servlet.http.HttpServletRequest;
22  import java.util.Enumeration;

24  import static org.atmosphere.cpr.HeaderConfig.WEBSOCKET_UPGRADE;
25  
26  /**
27   * Utils class.
28   *
29   * @author Jeanfrancois Arcand
30   */
31  public final class Utils {
32  
33      public final static boolean webSocketEnabled(HttpServletRequest request) {
34  
35          boolean allowWebSocketWithoutHeaders = request.getHeader(HeaderConfig.X_ATMO_WEBSOCKET_PROXY) != null ? true : false;
36          if (allowWebSocketWithoutHeaders) return true;
37  
38          boolean webSocketEnabled = false;
39          Enumeration<String> connection = request.getHeaders("Connection");
40          if (connection == null || !connection.hasMoreElements()) {
41              connection = request.getHeaders("connection");
42          }
43  
44          if (connection != null && connection.hasMoreElements()) {
45              String[] e = connection.nextElement().toString().split(",");
46              for (String upgrade : e) {
47                  if (upgrade.trim().equalsIgnoreCase(WEBSOCKET_UPGRADE)) {
48                      webSocketEnabled = true;
49                      break;
50                  }
51              }
52          }
53          return webSocketEnabled;
54      }
55  
56      public final static boolean resumableTransport(AtmosphereResource.TRANSPORT t) {
57          switch (t) {
58              case JSONP:
59              case LONG_POLLING:
60                  return true;
61              default:
62                  return false;
63          }
64      }
65  
66      public final static boolean pollableTransport(AtmosphereResource.TRANSPORT t) {
67          switch (t) {
68              case POLLING:
69              case UNDEFINED:
70              case AJAX:
71                  return true;
72              default:
73                  return false;
74          }
75      }
76  }

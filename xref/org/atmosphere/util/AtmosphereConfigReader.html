66  public class AtmosphereConfigReader {

68      private static final Logger logger = LoggerFactory.getLogger(AtmosphereConfigReader.class);

70      private final Map<String, String> tuples = new HashMap<String, String>();
71      private final Map<String, ArrayList<Property>> atmosphereHandlerProperties = new HashMap<String, ArrayList<Property>>();
72      private final Map<String, String> broadcasters = new HashMap<String, String>();
73      private final Map<String, String> broadcasterCache = new HashMap<String, String>();

75      private String cometSupportClass = null;
76      private String supportSession = "";
77      private String[] broadcastFilterClasses;

79      /**
80       * Create a {@link DocumentBuilderFactory} element from META-INF/atmosphere.xml
81       *
82       * @param stream
83       */
84      public AtmosphereConfigReader(InputStream stream) {
85          DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
86          try {
87              parse(factory.newDocumentBuilder().parse(stream));
88          } catch (SAXException e) {
89              logger.error(e.getMessage(), e);
90              throw new RuntimeException(e);
91          } catch (IOException e) {
92              logger.error(e.getMessage(), e);
93              throw new RuntimeException(e);
94          } catch (ParserConfigurationException e) {
95              logger.error(e.getMessage(), e);
96              throw new RuntimeException(e);
97          }
98      }

100     /**
101      * Parse the META-INF/atmosphere.xml element.
102      *
103      * @param document
104      */
105     private void parse(Document document) {
106         Element element = document.getDocumentElement();
107         NodeList atmosphereHandlers = element.getElementsByTagName("atmosphere-handler");
108         for (int i = 0; i < atmosphereHandlers.getLength(); i++) {
109             Node atmosphereHandler = atmosphereHandlers.item(i);
110             NamedNodeMap attrs = atmosphereHandler.getAttributes();
111             NodeList properties = atmosphereHandler.getChildNodes();
112             ArrayList<Property> list = new ArrayList<Property>();
113 
114             // Read the properties to be set on a AtmosphereHandler
115             for (int j = 0; j < properties.getLength(); j++) {
116                 Node property = properties.item(j);
117                 NamedNodeMap values = property.getAttributes();
118                 if (values != null) {
119                     list.add(new Property(values.getNamedItem("name").getNodeValue(),
120                             values.getNamedItem("value").getNodeValue()));
121                 }
122             }
123 
124             if (attrs != null) {
125                 atmosphereHandlerProperties.put(attrs.getNamedItem("context-root").getNodeValue(), list);
126 
127                 addAtmosphereHandler(attrs.getNamedItem("context-root").getNodeValue(),
128                         attrs.getNamedItem("class-name").getNodeValue());
129 
130                 if (attrs.getNamedItem("broadcaster") != null) {
131                     String broadcasterClass = attrs.getNamedItem("broadcaster").getNodeValue();
132                     if (broadcasterClass != null) {
133                         broadcasters.put(attrs.getNamedItem("context-root").getNodeValue(), broadcasterClass);
134                     }
135                 }
136 
137                 if (attrs.getNamedItem("broadcastFilterClasses") != null) {
138                     broadcastFilterClasses = attrs.getNamedItem("broadcastFilter").getNodeValue().split(",");
139                 }
140 
141                 if (attrs.getNamedItem("broadcasterCache") != null) {
142                     String bc = attrs.getNamedItem("broadcasterCache").getNodeValue();
143                     if (bc != null) {
144                         broadcasterCache.put(attrs.getNamedItem("context-root").getNodeValue(), bc);
145                     }
146                 }
147                 if (attrs.getNamedItem("comet-support") != null) {
148                     cometSupportClass = attrs.getNamedItem("comet-support").getNodeValue();
149                 }
150 
151                 if (attrs.getNamedItem("support-session") != null) {
152                     supportSession = attrs.getNamedItem("support-session").getNodeValue();
153                 }
154             }
155         }
156     }

158     /**

163     public String supportSession() {

167     /**
168      * Add a {@link AtmosphereHandler} to the list of discovered element.
169      *
170      * @param contextPath The context-path which will map the {@link AtmosphereHandler}
171      * @param className   The associated {@lin AtmosphereHandler} class name.
172      */
173     void addAtmosphereHandler(String contextPath, String className) {
174         if (tuples.containsKey(contextPath)) {
175             throw new RuntimeException("duplicate context root in configuration :" + contextPath);
176         }
177         tuples.put(contextPath, className);
178     }

185      */
186     public Map<String, String> getAtmosphereHandlers() {
187         return tuples;
188     }

18  import javax.servlet.http.Cookie;
19  import java.text.DateFormat;
20  import java.text.FieldPosition;
21  import java.text.SimpleDateFormat;
22  import java.util.Date;
23  import java.util.Locale;
24  import java.util.TimeZone;

26  /**
27   * This class is a cut & paste from Tomcat's Cookie creation.
28   */
29  public class CookieUtil {
30      private static final String OLD_COOKIE_PATTERN =
31              "EEE, dd-MMM-yyyy HH:mm:ss z";
32      private static final ThreadLocal<DateFormat> OLD_COOKIE_FORMAT =
33              new ThreadLocal<DateFormat>() {
34                  @Override
35                  protected DateFormat initialValue() {
36                      DateFormat df =
37                              new SimpleDateFormat(OLD_COOKIE_PATTERN, Locale.US);
38                      df.setTimeZone(TimeZone.getTimeZone("GMT"));
39                      return df;
40                  }
41              };
42      private static final String tspecials = ",; ";
43      private static final String ancientDate;
44  
45      static {
46          ancientDate = OLD_COOKIE_FORMAT.get().format(new Date(10000));
47      }
48  
49      public static boolean isToken(String value) {
50          if (value == null) return true;
51          int len = value.length();
52  
53          for (int i = 0; i < len; i++) {
54              char c = value.charAt(i);
55  
56              if (c < 0x20 || c >= 0x7f || tspecials.indexOf(c) != -1)
57                  return false;
58          }
59          return true;
60      }
61  
62      private static String escapeDoubleQuotes(String s) {
63  
64          if (s == null || s.length() == 0 || s.indexOf('"') == -1) {
65              return s;
66          }
67  
68          StringBuffer b = new StringBuffer();
69          for (int i = 0; i < s.length(); i++) {
70              char c = s.charAt(i);
71              if (c == '"')
72                  b.append('\\').append('"');
73              else
74                  b.append(c);
75          }
76  
77          return b.toString();
78      }
79  
80      public static String toString(Cookie c) {
81          StringBuffer buf = new StringBuffer();
82          // Servlet implementation checks name
83          buf.append(c.getName());
84          buf.append("=");
85          // Servlet implementation does not check anything else
86  
87          /*
88           * The spec allows some latitude on when to send the version attribute
89           * with a Set-Cookie header. To be nice to clients, we'll make sure the
90           * version attribute is first. That means checking the various things
91           * that can cause us to switch to a v1 cookie first.
92           *
93           * Note that by checking for tokens we will also throw an exception if a
94           * control character is encountered.
95           */
96          // Start by using the version we were asked for
97          int newVersion = c.getVersion();
98  
99          // Now build the cookie header
100         // Value
101         maybeQuote(buf, c.getValue());
102         // Add version 1 specific information
103         if (newVersion == 1) {
104             // Version=1 ... required
105             buf.append("; Version=1");
106 
107             // Comment=comment
108             if (c.getComment() != null) {
109                 buf.append("; Comment=");
110                 maybeQuote(buf, c.getComment());
111             }
112         }
113 
114         // Add domain information, if present
115         if (c.getDomain() != null) {
116             buf.append("; Domain=");
117             maybeQuote(buf, c.getDomain());
118         }
119 
120         // Max-Age=secs ... or use old "Expires" format
121         if (c.getMaxAge() >= 0) {
122             if (newVersion > 0) {
123                 buf.append("; Max-Age=");
124                 buf.append(c.getMaxAge());
125             }
126 
127             if (newVersion == 0) {
128                 // Wdy, DD-Mon-YY HH:MM:SS GMT ( Expires Netscape format )
129                 buf.append("; Expires=");
130                 // To expire immediately we need to set the time in past
131                 if (c.getMaxAge() == 0) {
132                     buf.append(ancientDate);
133                 } else {
134                     OLD_COOKIE_FORMAT.get().format(
135                             new Date(System.currentTimeMillis() +
136                                     c.getMaxAge() * 1000L),
137                             buf, new FieldPosition(0));
138                 }
139             }
140         }
141 
142         // Path=path
143         if (c.getPath() != null) {
144             buf.append("; Path=");
145             maybeQuote(buf, c.getPath());
146         }
147 
148         // Secure
149         if (c.getSecure()) {
150             buf.append("; Secure");
151         }
152 
153         // HttpOnly
154         if (c.isHttpOnly()) {
155             buf.append("; HttpOnly");
156         }
157         return buf.toString();
158     }
159 
160     /**
161      * Quotes values if required.
162      *
163      * @param buf
164      * @param value
165      */
166     private static void maybeQuote(StringBuffer buf, String value) {
167         if (value == null || value.length() == 0) {
168             buf.append("\"\"");
169         } else if (alreadyQuoted(value)) {
170             buf.append('"');
171             buf.append(escapeDoubleQuotes(value, 1, value.length() - 1));
172             buf.append('"');
173         } else {
174             buf.append(value);
175         }
176     }
177 
178     public static boolean alreadyQuoted(String value) {
179         if (value == null || value.length() < 2) {
180             return false;
181         }
182         return (value.charAt(0) == '\"' && value.charAt(value.length() - 1) == '\"');
183     }
184 
185     /**
186      * Escapes any double quotes in the given string.
187      *
188      * @param s          the input string
189      * @param beginIndex start index inclusive
190      * @param endIndex   exclusive
191      * @return The (possibly) escaped string
192      */
193     private static String escapeDoubleQuotes(String s, int beginIndex, int endIndex) {
194 
195         if (s == null || s.length() == 0 || s.indexOf('"') == -1) {
196             return s;
197         }
198 
199         StringBuffer b = new StringBuffer();
200         for (int i = beginIndex; i < endIndex; i++) {
201             char c = s.charAt(i);
202             if (c == '\\') {
203                 b.append(c);
204                 //ignore the character after an escape, just append it
205                 if (++i >= endIndex) {
206                     throw new IllegalArgumentException("Invalid escape character in cookie value.");
207                 }
208                 b.append(s.charAt(i));
209             } else if (c == '"') {
210                 b.append('\\').append('"');
211             } else {
212                 b.append(c);
213             }
214         }
215 
216         return b.toString();
217     }

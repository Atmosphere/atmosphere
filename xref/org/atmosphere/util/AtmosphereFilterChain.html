69  package org.atmosphere.util;
70  
71  import javax.servlet.Filter;
72  import javax.servlet.FilterChain;
73  import javax.servlet.RequestDispatcher;
74  import javax.servlet.Servlet;
75  import javax.servlet.ServletConfig;
76  import javax.servlet.ServletException;
77  import javax.servlet.ServletRequest;
78  import javax.servlet.ServletResponse;
79  import java.io.IOException;
80  import java.util.concurrent.atomic.AtomicInteger;
81  
82  /**
83   * Implementation of <code>javax.servlet.FilterChain</code> used to manage
84   * the execution of a set of filters for a particular request.  When the
85   * set of defined filters has all been executed, the next call to
86   * <code>doFilter()</code> will execute the servlet's <code>service()</code>
87   * method itself.
88   *
89   * @author Craig R. McClanahan
90   */
91  public final class AtmosphereFilterChain implements FilterChain {
92  
93      public static final int INCREMENT = 20;
94  
95      /**
96       * Filters.
97       */
98      private FilterConfigImpl[] filters = new FilterConfigImpl[20];
99  
100     /**
101      * The int which gives the current number of filters in the chain.
102      */
103     private int n = 0;
104     /**
105      * The servlet instance to be executed by this chain.
106      */
107     private Servlet servlet = null;
108     private ServletConfig configImpl;
109 
110     public AtmosphereFilterChain() {
111     }
112 
113     /**
114      * Initialize the {@link Filter}
115      *
116      * @throws javax.servlet.ServletException
117      */
118     public void init() throws ServletException {
119         for (FilterConfigImpl f : filters) {
120             if (f != null) {
121                 f.getFilter().init(f);
122             }
123         }
124         if (servlet != null) {
125             servlet.init(configImpl);
126         }
127     }
128 
129     public void invokeFilterChain(ServletRequest request, ServletResponse response)
130             throws IOException, ServletException {
131         request.setAttribute("pos", new AtomicInteger(0));
132         doFilter(request, response);
133     }
134 
135     /**
136      * Invoke the next filter in this chain, passing the specified request
137      * and response.  If there are no more filters in this chain, invoke
138      * the <code>service()</code> method of the servlet itself.
139      *
140      * @param request  The servlet request we are processing
141      * @param response The servlet response we are creating
142      * @throws IOException      if an input/output error occurs
143      * @throws ServletException if a servlet exception occurs
144      */
145     public void doFilter(ServletRequest request, ServletResponse response)
146             throws IOException, ServletException {
147 
148         // Call the next filter if there is one
149         AtomicInteger pos = ((AtomicInteger) request.getAttribute("pos"));
150         if (pos.get() < n) {
151             FilterConfigImpl filterConfig = filters[pos.getAndIncrement()];
152             Filter filter = null;
153             try {
154                 filter = filterConfig.getFilter();
155                 filter.doFilter(request, response, this);
156             } catch (IOException e) {
157                 throw e;
158             } catch (ServletException e) {
159                 throw e;
160             } catch (RuntimeException e) {
161                 throw e;
162             } catch (Throwable e) {
163                 throw new ServletException("Throwable", e);
164             }
165 
166             return;
167         }
168 
169         try {
170             if (servlet != null) {
171                 servlet.service(request, response);
172             } else {
173                 RequestDispatcher rd = configImpl.getServletContext().getNamedDispatcher("default");
174                 if (rd == null) {
175                     throw new ServletException("No Servlet Found");
176                 }
177                 rd.forward(request,  response);
178             }
179 
180         } catch (IOException e) {
181             throw e;
182         } catch (ServletException e) {
183             throw e;
184         } catch (RuntimeException e) {
185             throw e;
186         } catch (Throwable e) {
187             throw new ServletException("Throwable", e);
188         }
189 
190     }
191 
192     // -------------------------------------------------------- Package Methods
193 
194     /**
195      * Add a filter to the set of filters that will be executed in this chain.
196      *
197      * @param filterConfig The FilterConfig for the servlet to be executed
198      */
199     public void addFilter(FilterConfigImpl filterConfig) {
200 
201         if (filterConfig.getFilter() == null) {
202             throw new NullPointerException("Filter is null");
203         }
204 
205         if (n == filters.length) {
206             FilterConfigImpl[] newFilters =
207                     new FilterConfigImpl[n + INCREMENT];
208             System.arraycopy(filters, 0, newFilters, 0, n);
209             filters = newFilters;
210         }
211 
212         filters[n++] = filterConfig;
213     }
214 
215     /**
216      * Set the servlet that will be executed at the end of this chain.
217      * Set by the mapper filter
218      */
219     public void setServlet(ServletConfig configImpl, Servlet servlet) {
220         this.configImpl = configImpl;
221         this.servlet = servlet;
222     }
223 
224     public FilterConfigImpl getFilter(int i) {
225         return filters[i];
226     }
227 
228     public Servlet getServlet() {
229         return servlet;
230     }
231 
232     public ServletConfig getServletConfig() {
233         return configImpl;
234     }
235 
236     public void destroy() {
237         if (n > 0 && filters != null) {
238             for (int i = 0; i < filters.length; i++) {
239                 if (filters[i] != null) {
240                     filters[i].recycle();
241                 }
242             }
243             filters = null;
244         }
245 
246         if (servlet != null) {
247             servlet.destroy();
248         }
249     }

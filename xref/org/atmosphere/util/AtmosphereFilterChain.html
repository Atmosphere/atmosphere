33  package org.atmosphere.util;
34  
35  import javax.servlet.Filter;
36  import javax.servlet.FilterChain;
37  import javax.servlet.RequestDispatcher;
38  import javax.servlet.Servlet;
39  import javax.servlet.ServletConfig;
40  import javax.servlet.ServletException;
41  import javax.servlet.ServletRequest;
42  import javax.servlet.ServletResponse;
43  import java.io.IOException;
44  import java.util.concurrent.atomic.AtomicInteger;
45  
46  /**
47   * Implementation of <code>javax.servlet.FilterChain</code> used to manage
48   * the execution of a set of filters for a particular request.  When the
49   * set of defined filters has all been executed, the next call to
50   * <code>doFilter()</code> will execute the servlet's <code>service()</code>
51   * method itself.
52   *
53   * @author Craig R. McClanahan
54   */
55  public final class AtmosphereFilterChain implements FilterChain {
56  
57      public static final int INCREMENT = 20;
58  
59      /**
60       * Filters.
61       */
62      private FilterConfigImpl[] filters = new FilterConfigImpl[20];
63  
64      /**
65       * The int which gives the current number of filters in the chain.
66       */
67      private int n = 0;
68      /**
69       * The servlet instance to be executed by this chain.
70       */
71      private Servlet servlet = null;
72      private ServletConfig configImpl;
73  
74      public AtmosphereFilterChain() {
75      }
76  
77      /**
78       * Initialize the {@link Filter}
79       *
80       * @throws javax.servlet.ServletException
81       */
82      public void init() throws ServletException {
83          for (FilterConfigImpl f : filters) {
84              if (f != null) {
85                  f.getFilter().init(f);
86              }
87          }
88          if (servlet != null) {
89              servlet.init(configImpl);
90          }
91      }
92  
93      public void invokeFilterChain(ServletRequest request, ServletResponse response)
94              throws IOException, ServletException {
95          request.setAttribute("pos", new AtomicInteger(0));
96          doFilter(request, response);
97      }
98  
99      /**
100      * Invoke the next filter in this chain, passing the specified request
101      * and response.  If there are no more filters in this chain, invoke
102      * the <code>service()</code> method of the servlet itself.
103      *
104      * @param request  The servlet request we are processing
105      * @param response The servlet response we are creating
106      * @throws IOException      if an input/output error occurs
107      * @throws ServletException if a servlet exception occurs
108      */
109     public void doFilter(ServletRequest request, ServletResponse response)
110             throws IOException, ServletException {
111 
112         // Call the next filter if there is one
113         AtomicInteger pos = ((AtomicInteger) request.getAttribute("pos"));
114         if (pos.get() < n) {
115             FilterConfigImpl filterConfig = filters[pos.getAndIncrement()];
116             Filter filter = null;
117             try {
118                 filter = filterConfig.getFilter();
119                 filter.doFilter(request, response, this);
120             } catch (IOException e) {
121                 throw e;
122             } catch (ServletException e) {
123                 throw e;
124             } catch (RuntimeException e) {
125                 throw e;
126             } catch (Throwable e) {
127                 throw new ServletException("Throwable", e);
128             }
129 
130             return;
131         }
132 
133         try {
134             if (servlet != null) {
135                 servlet.service(request, response);
136             } else {
137                 RequestDispatcher rd = configImpl.getServletContext().getNamedDispatcher("default");
138                 if (rd == null) {
139                     throw new ServletException("No Servlet Found");
140                 }
141                 rd.forward(request, response);
142             }
143 
144         } catch (IOException e) {
145             throw e;
146         } catch (ServletException e) {
147             throw e;
148         } catch (RuntimeException e) {
149             throw e;
150         } catch (Throwable e) {
151             throw new ServletException("Throwable", e);
152         }
153 
154     }
155 
156     // -------------------------------------------------------- Package Methods
157 
158     /**
159      * Add a filter to the set of filters that will be executed in this chain.
160      *
161      * @param filterConfig The FilterConfig for the servlet to be executed
162      */
163     public void addFilter(FilterConfigImpl filterConfig) {
164 
165         if (filterConfig.getFilter() == null) {
166             throw new NullPointerException("Filter is null");
167         }
168 
169         if (n == filters.length) {
170             FilterConfigImpl[] newFilters =
171                     new FilterConfigImpl[n + INCREMENT];
172             System.arraycopy(filters, 0, newFilters, 0, n);
173             filters = newFilters;
174         }
175 
176         filters[n++] = filterConfig;
177     }
178 
179     /**
180      * Set the servlet that will be executed at the end of this chain.
181      * Set by the mapper filter
182      */
183     public void setServlet(ServletConfig configImpl, Servlet servlet) {
184         this.configImpl = configImpl;
185         this.servlet = servlet;
186     }
187 
188     public FilterConfigImpl getFilter(int i) {
189         return filters[i];
190     }
191 
192     public Servlet getServlet() {
193         return servlet;
194     }
195 
196     public ServletConfig getServletConfig() {
197         return configImpl;
198     }
199 
200     public void destroy() {
201         if (n > 0 && filters != null) {
202             for (int i = 0; i < filters.length; i++) {
203                 if (filters[i] != null) {
204                     filters[i].recycle();
205                 }
206             }
207             filters = null;
208         }
209 
210         if (servlet != null) {
211             servlet.destroy();
212         }
213     }

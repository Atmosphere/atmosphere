16  package org.atmosphere.util;
17  
18  import org.atmosphere.cpr.AtmosphereConfig;
19  import org.atmosphere.cpr.AtmosphereRequest;
20  import org.atmosphere.cpr.FrameworkConfig;
21  import org.atmosphere.util.uri.UriTemplate;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import java.util.HashMap;
26  import java.util.Map;
27  
28  /**
29   * Default implementation of the {@link EndpointMapper} used by the {@link org.atmosphere.cpr.AsynchronousProcessor}
30   * and {@link org.atmosphere.websocket.DefaultWebSocketProcessor}
31   *
32   * @author Jeanfrancois Arcand
33   */
34  public class DefaultEndpointMapper<U> implements EndpointMapper<U> {
35  
36      private static final Logger logger = LoggerFactory.getLogger(DefaultEndpointMapper.class);
37  
38      public DefaultEndpointMapper() {
39      }
40  
41      protected U match(String path, Map<String, U> handlers) {
42          U handler = handlers.get(path);
43  
44          if (handler == null) {
45              final Map<String, String> m = new HashMap<String, String>();
46              for (Map.Entry<String, U> e : handlers.entrySet()) {
47                  UriTemplate t = null;
48                  try {
49                      t = new UriTemplate(e.getKey());
50                      logger.trace("Trying to map {} to {}", t, path);
51                      if (t.match(path, m)) {
52                          handler = e.getValue();
53                          logger.trace("Mapped {} to {}", t, e.getValue());
54                          break;
55                      }
56                  } finally {
57                      if (t != null) t.destroy();
58                  }
59              }
60          }
61          return handler;
62      }
63  
64      public String computePath(AtmosphereRequest req) {
65          return Utils.pathInfo(req);
66      }
67  
68      @Override
69      public void configure(AtmosphereConfig config) {
70      }
71  
72      @Override
73      public U map(AtmosphereRequest req, Map<String, U> handlers) {
74          String path = computePath(req);
75  
76          U handler = map(path, handlers);
77          if (handler == null) {
78              // (2) First, try exact match
79              handler = map(path + (path.endsWith("/") ? "all" : "/all"), handlers);
80  
81              if (handler == null) {
82                  // (3) Wildcard
83                  handler = map(path + "*", handlers);
84  
85                  // (4) try without a path
86                  if (handler == null) {
87                      String p = path.lastIndexOf("/") <= 0 ? "/" : path.substring(0, path.lastIndexOf("/"));
88                      while (p.length() > 0) {
89                          handler = map(p, handlers);
90  
91                          // (3.1) Try path wildcard
92                          if (handler != null) {
93                              break;
94                          }
95                          p = p.substring(0, p.lastIndexOf("/"));
96                      }
97                  }
98  
99                  // Glassfish 3.1.2 issue .. BEUUUURRRRKKKKKK!!
100                 if (handler == null && req.getContextPath().length() < path.length()) {
101                     path = path.substring(req.getContextPath().length());
102                     handler = map(path, handlers);
103                 }
104             }
105         }
106         req.setAttribute(FrameworkConfig.MAPPED_PATH, path);
107         return handler;
108     }
109 
110     @Override
111     public U map(String path, Map<String, U> handlers) {
112 
113         if (path == null || path.isEmpty()) {
114             path = "/";
115         }
116 
117         U handler = match(path, handlers);
118         if (handler == null) {
119             // (2) First, try exact match
120             handler = match(path + (path.endsWith("/") ? "all" : "/all"), handlers);
121 
122             if (handler == null) {
123                 // (3) Wildcard
124                 handler = match(path + "*", handlers);
125 
126                 // (4) try without a path
127                 if (handler == null) {
128                     String p = path.lastIndexOf("/") <= 0 ? "/" : path.substring(0, path.lastIndexOf("/"));
129                     while (p.length() > 0 && p.indexOf("/") != -1) {
130                         handler = match(p, handlers);
131 
132                         // (3.1) Try path wildcard
133                         if (handler != null) {
134                             break;
135                         }
136                         p = p.substring(0, p.lastIndexOf("/"));
137                     }
138                 }
139             }
140         }
141         return handler;
142     }
143 }

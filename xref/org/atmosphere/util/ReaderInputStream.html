32  package org.atmosphere.util;
33  
34  
35  import java.io.IOException;
36  import java.io.InputStream;
37  import java.io.Reader;
38  import java.nio.ByteBuffer;
39  import java.nio.CharBuffer;
40  import java.nio.charset.Charset;
41  import java.nio.charset.CharsetEncoder;
42  import java.nio.charset.CoderResult;
43  import java.nio.charset.CodingErrorAction;
44  
45  /**
46   * {@link java.io.InputStream} implementation that reads a character stream from a {@link java.io.Reader}
47   * and transforms it to a byte stream using a specified charset encoding. The stream
48   * is transformed using a {@link java.nio.charset.CharsetEncoder} object, guaranteeing that all charset
49   * encodings supported by the JRE are handled correctly. In particular for charsets such as
50   * UTF-16, the implementation ensures that one and only one byte order marker
51   * is produced.
52   * <p/>
53   * Since in general it is not possible to predict the number of characters to be read from the
54   * {@link java.io.Reader} to satisfy a read request on the {@link ReaderInputStream}, all reads from
55   * the {@link java.io.Reader} are buffered. There is therefore no well defined correlation
56   * between the current position of the {@link java.io.Reader} and that of the {@link ReaderInputStream}.
57   * This also implies that in general there is no need to wrap the underlying {@link java.io.Reader}
58   * in a {@link java.io.BufferedReader}.
59   * <p/>
60   * {@link ReaderInputStream} implements the inverse transformation of {@link java.io.InputStreamReader};
61   * in the following example, reading from <tt>in2</tt> would return the same byte
62   * sequence as reading from <tt>in</tt> (provided that the initial byte sequence is legal
63   * with respect to the charset encoding):
64   * <pre>
65   * InputStream in = ...
66   * Charset cs = ...
67   * InputStreamReader reader = new InputStreamReader(in, cs);
68   * ReaderInputStream in2 = new ReaderInputStream(reader, cs);</pre>
69   * {@link ReaderInputStream} implements the same transformation as {@link java.io.OutputStreamWriter},
70   * except that the control flow is reversed: both classes transform a character stream
71   * into a byte stream, but {@link java.io.OutputStreamWriter} pushes data to the underlying stream,
72   * while {@link ReaderInputStream} pulls it from the underlying stream.
73   *
74   * Note that while there are use cases where there is no alternative to using
75   * this class, very often the need to use this class is an indication of a flaw
76   * in the design of the code. This class is typically used in situations where an existing
77   * API only accepts an {@link java.io.InputStream}, but where the most natural way to produce the data
78   * is as a character stream, i.e. by providing a {@link java.io.Reader} instance. An example of a situation
79   * where this problem may appear is when implementing the {@link javax.activation.DataSource}
80   * interface from the Java Activation Framework.
81   *
82   * Given the fact that the {@link java.io.Reader} class doesn't provide any way to predict whether the next
83   * read operation will block or not, it is not possible to provide a meaningful
84   * implementation of the {@link java.io.InputStream#available()} method. A call to this method
85   * will always return 0. Also, this class doesn't support {@link java.io.InputStream#mark(int)}.
86   *
87   * Instances of {@link ReaderInputStream} are not thread safe.
88   *
89   */
90  public class ReaderInputStream extends InputStream {
91      private static final int DEFAULT_BUFFER_SIZE = 1024;
92  
93      private final Reader reader;
94      private final CharsetEncoder encoder;
95  
96      /**
97       * CharBuffer used as input for the decoder. It should be reasonably
98       * large as we read data from the underlying Reader into this buffer.
99       */
100     private final CharBuffer encoderIn;
101 
102     /**
103      * ByteBuffer used as output for the decoder. This buffer can be small
104      * as it is only used to transfer data from the decoder to the
105      * buffer provided by the caller.
106      */
107     private final ByteBuffer encoderOut;
108 
109     private CoderResult lastCoderResult;
110     private boolean endOfInput;
111 
112     /**
113      * Construct a new {@link ReaderInputStream}.
114      *
115      * @param reader  the target {@link java.io.Reader}
116      * @param encoder the charset encoder
117      * @since 2.1
118      */
119     public ReaderInputStream(Reader reader, CharsetEncoder encoder) {
120         this(reader, encoder, DEFAULT_BUFFER_SIZE);
121     }
122 
123     /**
124      * Construct a new {@link ReaderInputStream}.
125      *
126      * @param reader     the target {@link java.io.Reader}
127      * @param encoder    the charset encoder
128      * @param bufferSize the size of the input buffer in number of characters
129      * @since 2.1
130      */
131     public ReaderInputStream(Reader reader, CharsetEncoder encoder, int bufferSize) {
132         this.reader = reader;
133         this.encoder = encoder;
134         this.encoderIn = CharBuffer.allocate(bufferSize);
135         this.encoderIn.flip();
136         this.encoderOut = ByteBuffer.allocate(128);
137         this.encoderOut.flip();
138     }
139 
140     /**
141      * Construct a new {@link ReaderInputStream}.
142      *
143      * @param reader     the target {@link java.io.Reader}
144      * @param charset    the charset encoding
145      * @param bufferSize the size of the input buffer in number of characters
146      */
147     public ReaderInputStream(Reader reader, Charset charset, int bufferSize) {
148         this(reader,
149                 charset.newEncoder()
150                         .onMalformedInput(CodingErrorAction.REPLACE)
151                         .onUnmappableCharacter(CodingErrorAction.REPLACE),
152                 bufferSize);
153     }
154 
155     /**
156      * Construct a new {@link ReaderInputStream} with a default input buffer size of
157      * 1024 characters.
158      *
159      * @param reader  the target {@link java.io.Reader}
160      * @param charset the charset encoding
161      */
162     public ReaderInputStream(Reader reader, Charset charset) {
163         this(reader, charset, DEFAULT_BUFFER_SIZE);
164     }
165 
166     /**
167      * Construct a new {@link ReaderInputStream}.
168      *
169      * @param reader      the target {@link java.io.Reader}
170      * @param charsetName the name of the charset encoding
171      * @param bufferSize  the size of the input buffer in number of characters
172      */
173     public ReaderInputStream(Reader reader, String charsetName, int bufferSize) {
174         this(reader, Charset.forName(charsetName), bufferSize);
175     }
176 
177     /**
178      * Construct a new {@link ReaderInputStream} with a default input buffer size of
179      * 1024 characters.
180      *
181      * @param reader      the target {@link java.io.Reader}
182      * @param charsetName the name of the charset encoding
183      */
184     public ReaderInputStream(Reader reader, String charsetName) {
185         this(reader, charsetName, DEFAULT_BUFFER_SIZE);
186     }
187 
188     /**
189      * Construct a new {@link ReaderInputStream} that uses the default character encoding
190      * with a default input buffer size of 1024 characters.
191      *
192      * @param reader the target {@link java.io.Reader}
193      */
194     public ReaderInputStream(Reader reader) {
195         this(reader, Charset.defaultCharset());
196     }
197 
198     /**
199      * Fills the internal char buffer from the reader.
200      *
201      * @throws java.io.IOException If an I/O error occurs
202      */
203     private void fillBuffer() throws IOException {
204         if (!endOfInput && (lastCoderResult == null || lastCoderResult.isUnderflow())) {
205             encoderIn.compact();
206             int position = encoderIn.position();
207             // We don't use Reader#read(CharBuffer) here because it is more efficient
208             // to write directly to the underlying char array (the default implementation
209             // copies data to a temporary char array).
210             int c = reader.read(encoderIn.array(), position, encoderIn.remaining());
211             if (c == -1) {
212                 endOfInput = true;
213             } else {
214                 encoderIn.position(position + c);
215             }
216             encoderIn.flip();
217         }
218         encoderOut.compact();
219         lastCoderResult = encoder.encode(encoderIn, encoderOut, endOfInput);
220         encoderOut.flip();
221     }
222 
223     /**
224      * Read the specified number of bytes into an array.
225      *
226      * @param b   the byte array to read into
227      * @param off the offset to start reading bytes into
228      * @param len the number of bytes to read
229      * @return the number of bytes read or <code>-1</code>
230      *         if the end of the stream has been reached
231      * @throws java.io.IOException if an I/O error occurs
232      */
233     @Override
234     public int read(byte[] b, int off, int len) throws IOException {
235         if (b == null) {
236             throw new NullPointerException("Byte array must not be null");
237         }
238         if (len < 0 || off < 0 || (off + len) > b.length) {
239             throw new IndexOutOfBoundsException("Array Size=" + b.length +
240                     ", offset=" + off + ", length=" + len);
241         }
242         int read = 0;
243         if (len == 0) {
244             return 0; // Always return 0 if len == 0
245         }
246         while (len > 0) {
247             if (encoderOut.hasRemaining()) {
248                 int c = Math.min(encoderOut.remaining(), len);
249                 encoderOut.get(b, off, c);
250                 off += c;
251                 len -= c;
252                 read += c;
253             } else {
254                 fillBuffer();
255                 if (endOfInput && !encoderOut.hasRemaining()) {
256                     break;
257                 }
258             }
259         }
260         return read == 0 && endOfInput ? -1 : read;
261     }
262 
263     /**
264      * Read the specified number of bytes into an array.
265      *
266      * @param b the byte array to read into
267      * @return the number of bytes read or <code>-1</code>
268      *         if the end of the stream has been reached
269      * @throws java.io.IOException if an I/O error occurs
270      */
271     @Override
272     public int read(byte[] b) throws IOException {
273         return read(b, 0, b.length);
274     }
275 
276     /**
277      * Read a single byte.
278      *
279      * @return either the byte read or <code>-1</code> if the end of the stream
280      *         has been reached
281      * @throws java.io.IOException if an I/O error occurs
282      */
283     @Override
284     public int read() throws IOException {
285         for (; ; ) {
286             if (encoderOut.hasRemaining()) {
287                 return encoderOut.get() & 0xFF;
288             } else {
289                 fillBuffer();
290                 if (endOfInput && !encoderOut.hasRemaining()) {
291                     return -1;
292                 }
293             }
294         }
295     }
296 
297     /**
298      * Close the stream. This method will cause the underlying {@link java.io.Reader}
299      * to be closed.
300      *
301      * @throws java.io.IOException if an I/O error occurs
302      */
303     @Override
304     public void close() throws IOException {
305         reader.close();
306     }
307 }

17  package org.atmosphere.util;
18  
19  import org.atmosphere.cpr.FrameworkConfig;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  import org.w3c.dom.Document;
23  import org.w3c.dom.Element;
24  import org.w3c.dom.Node;
25  import org.w3c.dom.NodeList;
26  import org.xml.sax.SAXException;
27  
28  import javax.xml.parsers.DocumentBuilderFactory;
29  import javax.xml.parsers.ParserConfigurationException;
30  import java.io.IOException;
31  import java.io.InputStream;
32  import java.util.ArrayList;
33  
34  /**
35   * web.xml reader.
36   *
37   * @author Jeanfrancois Arcand
38   */
39  public class WebDotXmlReader {
40  
41      private static final Logger logger = LoggerFactory.getLogger(WebDotXmlReader.class);
42  
43      private final ArrayList<String> mappings = new ArrayList<String>();
44  
45      /**
46       * Create a {@link DocumentBuilderFactory} element from WEB-INF/web.xml
47       *
48       * @param stream
49       */
50      public WebDotXmlReader(InputStream stream) {
51          DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
52          try {
53              parse(factory.newDocumentBuilder().parse(stream));
54          } catch (SAXException e) {
55              logger.error(e.getMessage(), e);
56              throw new RuntimeException(e);
57          } catch (IOException e) {
58              logger.error(e.getMessage(), e);
59              throw new RuntimeException(e);
60          } catch (ParserConfigurationException e) {
61              logger.error(e.getMessage(), e);
62              throw new RuntimeException(e);
63          }
64      }
65  
66      /**
67       * Parse the web.xml element.
68       *
69       * @param document
70       */
71      private void parse(Document document) {
72          Element element = document.getDocumentElement();
73          NodeList servlets = element.getElementsByTagName("servlet");
74          String atmosphereServletName = null;
75          for (int i = 0; i < servlets.getLength(); i++) {
76              Node m = servlets.item(i);
77              NodeList list = m.getChildNodes();
78              for (int j = 0; j < list.getLength(); j++) {
79                  Node n = list.item(j);
80                  if (n.getNodeName().equals("servlet-name")) {
81                      atmosphereServletName = n.getFirstChild().getNodeValue();
82                  }
83  
84                  if (n.getNodeName().equals("servlet-class")) {
85                      if (n.getFirstChild().getNodeValue().equals(FrameworkConfig.ATMOSPHERE_SERVLET)) {
86                          break;
87                      }
88                  }
89              }
90          }
91  
92          NodeList servletMappings = element.getElementsByTagName("servlet-mapping");
93          for (int i = 0; i < servletMappings.getLength(); i++) {
94              Node m = servletMappings.item(i);
95              NodeList list = m.getChildNodes();
96  
97              String urlMapping = null;
98              String servletName = null;
99              for (int j = 0; j < list.getLength(); j++) {
100                 Node n = list.item(j);
101                 if (n.getNodeName().equals("servlet-name")) {
102                     servletName = n.getFirstChild().getNodeValue();
103                     if (!servletName.equals(atmosphereServletName)) {
104                         servletName = null;
105                     }
106                 }
107 
108                 if (n.getNodeName().equals("url-pattern")) {
109                     urlMapping = n.getFirstChild().getNodeValue();
110                 }
111 
112                 if (servletName != null && urlMapping != null) {
113                     mappings.add(urlMapping);
114                     servletName = null;
115                     urlMapping = null;
116                 }
117             }
118         }
119     }
120 
121     /**
122      * Return an {@link ArrayList} which maps to the {@link org.atmosphere.cpr.AtmosphereServlet}
123      */
124     public ArrayList<String> getMappings() {
125         return mappings;
126     }
127 }

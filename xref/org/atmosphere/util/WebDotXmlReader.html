38  package org.atmosphere.util;

40  import org.atmosphere.cpr.AtmosphereServlet;
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  import org.w3c.dom.Document;
44  import org.w3c.dom.Element;
45  import org.w3c.dom.Node;
46  import org.w3c.dom.NodeList;
47  import org.xml.sax.SAXException;
48  
49  import javax.xml.parsers.DocumentBuilderFactory;
50  import javax.xml.parsers.ParserConfigurationException;
51  import java.io.IOException;
52  import java.io.InputStream;
53  import java.util.ArrayList;
54  
55  /**
56   * web.xml reader.
57   *
58   * @author Jeanfrancois Arcand
59   */
60  public class WebDotXmlReader {
61  
62      private static final Logger logger = LoggerFactory.getLogger(WebDotXmlReader.class);
63  
64      private static final String ATMOSPHERE_SERVLET = AtmosphereServlet.class.getName();
65  
66      private final ArrayList<String> mappings = new ArrayList<String>();
67  
68      /**
69       * Create a {@link DocumentBuilderFactory} element from WEB-INF/web.xml
70       *
71       * @param stream
72       */
73      public WebDotXmlReader(InputStream stream) {
74          DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
75          try {
76              parse(factory.newDocumentBuilder().parse(stream));
77          } catch (SAXException e) {
78              logger.error(e.getMessage(), e);
79              throw new RuntimeException(e);
80          } catch (IOException e) {
81              logger.error(e.getMessage(), e);
82              throw new RuntimeException(e);
83          } catch (ParserConfigurationException e) {
84              logger.error(e.getMessage(), e);
85              throw new RuntimeException(e);
86          }
87      }
88  
89      /**
90       * Parse the web.xml element.
91       *
92       * @param document
93       */
94      private void parse(Document document) {
95          Element element = document.getDocumentElement();
96          NodeList servlets = element.getElementsByTagName("servlet");
97          String atmosphereServletName = null;
98          for (int i = 0; i < servlets.getLength(); i++) {
99              Node m = servlets.item(i);
100             NodeList list = m.getChildNodes();
101             for (int j = 0; j < list.getLength(); j++) {
102                 Node n = list.item(j);
103                 if (n.getNodeName().equals("servlet-name")) {
104                     atmosphereServletName = n.getFirstChild().getNodeValue();
105                 }
106 
107                 if (n.getNodeName().equals("servlet-class")) {
108                     if (n.getFirstChild().getNodeValue().equals(ATMOSPHERE_SERVLET)) {
109                         break;
110                     }
111                 }
112             }
113         }
114 
115         NodeList servletMappings = element.getElementsByTagName("servlet-mapping");
116         for (int i = 0; i < servletMappings.getLength(); i++) {
117             Node m = servletMappings.item(i);
118             NodeList list = m.getChildNodes();
119 
120             String urlMapping = null;
121             String servletName = null;
122             for (int j = 0; j < list.getLength(); j++) {
123                 Node n = list.item(j);
124                 if (n.getNodeName().equals("servlet-name")) {
125                     servletName = n.getFirstChild().getNodeValue();
126                     if (!servletName.equals(atmosphereServletName)) {
127                         servletName = null;
128                     }
129                 }
130 
131                 if (n.getNodeName().equals("url-pattern")) {
132                     urlMapping = n.getFirstChild().getNodeValue();
133                 }
134 
135                 if (servletName != null && urlMapping != null) {
136                     mappings.add(urlMapping);
137                     servletName = null;
138                     urlMapping = null;
139                 }
140             }
141         }
142     }
143 
144     /**
145      * Return an {@link ArrayList} which maps to the {@link AtmosphereServlet}
146      */
147     public ArrayList<String> getMappings() {
148         return mappings;
149     }

16  package org.atmosphere.util.analytics;
17  
18  import java.util.HashMap;
19  import java.util.Map;
20  
21  public class ModuleDetection {
22  
23      private final static HashMap<String, String> modules = new HashMap<String, String>();
24  
25      static {
26          modules.put("gwt2", "org.atmosphere.gwt20.server.GwtRpcInterceptor");
27          modules.put("jersey", "org.atmosphere.jersey.AtmosphereFilter");
28          modules.put("gwt", "org.atmosphere.gwt.serve.AtmosphereGwtHandler");
29          modules.put("cometd", "org.atmosphere.cometd.CometdServlet");
30          modules.put("socketio", "org.atmosphere.socketio.cpr.SocketIOAtmosphereHandler");
31          modules.put("weblogic", "org.atmosphere.weblogic.AtmosphereWebLogicServlet");
32          modules.put("portal", "com.github.flowersinthesand.portal.atmosphere.AtmosphereModule");
33          modules.put("sockjs", "org.atmosphere.sockjs.SockJsAtmosphereInterceptor");
34          modules.put("primefaces", "org.primefaces.push.PushServlet");
35          modules.put("vaadin", "com.vaadin.server.VaadinServlet");
36          modules.put("wicket", "org.apache.wicket.atmosphere.EventBus");
37          modules.put("scalatra", "org.scalatra.ScalatraServlet");
38          modules.put("eventPush", "org.grails.plugin.platform.events.push.EventsPushHandler");
39          modules.put("nettosphere", "org.atmosphere.nettosphere.BridgeRuntime");
40          modules.put("jersey2", "org.glassfish.jersey.servlet.ServletContainer");
41          modules.put("swaggersocket", "com.wordnik.swaggersocket.server.SwaggerSocketProtocolInterceptor");
42          modules.put("richfaces", "org.ajax4jsf.Filter");
43          modules.put("grails", "org.grails.plugins.atmosphere_meteor.MeteorHandlerArtefactHandler");
44  
45      }
46  
47      public final static String detect() {
48          for (Map.Entry<String, String> e : modules.entrySet()) {
49              if (check(e.getValue())) {
50                  return e.getKey();
51              }
52          }
53          return "runtime";
54      }
55  
56      static boolean check(String clazz) {
57          try {
58              Class.forName(clazz);
59              return true;
60          } catch (ClassNotFoundException e) {
61              return false;
62          }
63      }
64  
65  }

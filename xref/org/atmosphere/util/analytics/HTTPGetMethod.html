16  package org.atmosphere.util.analytics;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  import java.io.IOException;
22  import java.net.HttpURLConnection;
23  import java.net.URL;
24  
25  /**
26   * Fork of https://code.google.com/p/jgoogleanalytics/
27   * Simple class peforming HTTP Get method on the requested url
28   *
29   * @author : Siddique Hameed
30   * @version : 0.1
31   */
32  public class HTTPGetMethod {
33      private static final String GET_METHOD_NAME = "GET";
34      private static final String SUCCESS_MESSAGE = "JGoogleAnalytics: Tracking Successful!";
35      private static String uaName; // User Agent name
36      private static String osString = "Unknown";
37      private final Logger logger = LoggerFactory.getLogger(HTTPGetMethod.class);
38  
39      HTTPGetMethod() {
40          // Initialise the static parameters if we need to.
41          if (uaName == null) {
42              uaName = "Java/" + System.getProperty("java.version"); // java version info appended
43              // os string is architecture+osname+version concatenated with _
44              osString = System.getProperty("os.arch");
45              if (osString == null || osString.length() < 1) {
46                  osString = "";
47              } else {
48                  osString += "; ";
49                  osString += System.getProperty("os.name") + " "
50                          + System.getProperty("os.version");
51              }
52          }
53      }
54  
55      public void request(String urlString) {
56          HttpURLConnection urlConnection = null;
57          try {
58              URL url = new URL(urlString);
59              urlConnection = openURLConnection(url);
60              urlConnection.setInstanceFollowRedirects(true);
61              urlConnection.setRequestMethod(GET_METHOD_NAME);
62              urlConnection.setRequestProperty("User-agent", uaName + " (" + osString + ")");
63  
64              logger.debug("Sending Server's information to Atmosphere's Google Analytics {} {}", urlString, uaName + " (" + osString + ")");
65              urlConnection.connect();
66              int responseCode = getResponseCode(urlConnection);
67              if (responseCode != HttpURLConnection.HTTP_OK) {
68                  logError("JGoogleAnalytics: Error tracking, url=" + urlString);
69              } else {
70                  logMessage(SUCCESS_MESSAGE);
71              }
72          } catch (Exception e) {
73              logError(e.getMessage());
74          } finally {
75              if (urlConnection != null) {
76                  urlConnection.disconnect();
77              }
78          }
79      }
80  
81      protected int getResponseCode(HttpURLConnection urlConnection)
82              throws IOException {
83          return urlConnection.getResponseCode();
84      }
85  
86      private HttpURLConnection openURLConnection(URL url) throws IOException {
87          return (HttpURLConnection) url.openConnection();
88      }
89  
90      private void logMessage(String message) {
91          logger.trace("{}", message);
92      }
93  
94      private void logError(String errorMesssage) {
95          logger.trace("{}", errorMesssage);
96      }
97  }

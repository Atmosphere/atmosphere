74      /**
75       * Order the templates according according to JAX-RS.
76       * <p>
77       * Sort the set of matching resource classes using the number of 
78       * characters in the regular expression not resulting from template 
79       * variables as the primary key, the number of matching groups
80       * as a secondary key, and the number of explicit regular expression
81       * declarations as the tertiary key.
82       */
83      static public final Comparator<UriTemplate> COMPARATOR = new Comparator<UriTemplate>() {
84          public int compare(UriTemplate o1, UriTemplate o2) {
85              if (o1 == null && o2 == null)
86                  return 0;
87              if (o1 == null)
88                  return 1;
89              if (o2 == null)
90                  return -1;
91              
92              if (o1 == EMPTY && o2 == EMPTY)
93                  return 0;
94              if (o1 == EMPTY)
95                  return 1;
96              if (o2 == EMPTY)
97                  return -1;
98  
99              // Compare the number of explicit characters
100             // Note that it is important that o2 is compared against o1
101             // so that a regular expression with say 10 explicit characters
102             // is less than a regular expression with say 5 explicit characters.
103             int i = o2.getNumberOfExplicitCharacters() - o1.getNumberOfExplicitCharacters();
104             if (i != 0) return i;
105 
106             // If the number of explicit characters is equal
107             // compare the number of template variables
108             // Note that it is important that o2 is compared against o1
109             // so that a regular expression with say 10 template variables
110             // is less than a regular expression with say 5 template variables.
111             i = o2.getNumberOfTemplateVariables() - o1.getNumberOfTemplateVariables();
112             if (i != 0) return i;
113 
114             // If the number of template variables is equal
115             // compare the number of explicit regexes
116             i = o2.getNumberOfExplicitRegexes() - o1.getNumberOfExplicitRegexes();
117             if (i != 0) return i;
118             
119             // If the number of explicit characters and template variables
120             // are equal then comapre the regexes
121             // The order does not matter as long as templates with different
122             // explicit characters are distinguishable
123             return o2.pattern.getRegex().compareTo(o1.pattern.getRegex());
124         }
125     };

212      * to match URIs according to the template and map template variables to
213      * template values.
214      * <p>

218      * @throws IllegalArgumentException if the template is null or
219      *         an empty string.

221     protected UriTemplate(UriTemplateParser templateParser) throws 
222             PatternSyntaxException, IllegalArgumentException {
223         this.template = templateParser.getTemplate();
224         
225         this.normalizedTemplate = templateParser.getNormalizedTemplate();
226 
227         this.pattern = createUriPattern(templateParser);
228 
229         this.numOfExplicitRegexes = templateParser.getNumberOfExplicitRegexes();
230         
231         this.numOfCharacters = templateParser.getNumberOfLiteralCharacters();
232 
233         this.endsWithSlash = template.charAt(template.length() - 1) == '/';
234 
235         this.templateVariables = Collections.unmodifiableList(templateParser.getNames());
236     }

371     /**
372      * Create a URI by substituting any template variables
373      * for corresponding template values.
374      * <p>
375      * A URI template variable without a value will be substituted by the 
376      * empty string.
377      *
378      * @param values the map of template variables to template values.
379      * @return the URI.
380      */
381     public final String createURI(Map<String, String> values) {
382         StringBuilder b = new StringBuilder();
383         // Find all template variables
384         Matcher m = TEMPLATE_NAMES_PATTERN.matcher(normalizedTemplate);
385         int i = 0;
386         while(m.find()) {
387             b.append(normalizedTemplate, i, m.start());
388             String tValue = values.get(m.group(1));
389             if (tValue != null) b.append(tValue);
390             i = m.end();
391         }
392         b.append(normalizedTemplate, i, normalizedTemplate.length());
393         return b.toString();
394     }

411     /**
412      * Create a URI by substituting any template variables
413      * for corresponding template values.
414      * <p>
415      * A URI template variable without a value will be substituted by the 
416      * empty string.
417      *
418      * @param values the array of template values. The values will be 
419      *        substituted in order of occurence of unique template variables.
420      * @param offset the offset into the array
421      * @param length the length of the array
422      * @return the URI.
423      */
424     public final String createURI(String[] values, int offset, int length) {
425         Map<String, String> mapValues = new HashMap<String, String>();
426         StringBuilder b = new StringBuilder();
427         // Find all template variables
428         Matcher m = TEMPLATE_NAMES_PATTERN.matcher(normalizedTemplate);
429         int v = offset;
430         length += offset;
431         int i = 0;
432         while(m.find()) {
433             b.append(normalizedTemplate, i, m.start());
434             String tVariable = m.group(1);
435             // Check if a template variable has already occurred
436             // If so use the value to ensure that two or more declarations of 
437             // a template variable have the same value
438             String tValue = mapValues.get(tVariable);
439             if (tValue != null) {
440                 b.append(tValue);
441             } else {
442                 if (v < length) {
443                     tValue = values[v++];
444                     if (tValue != null) {
445                         mapValues.put(tVariable, tValue);
446                         b.append(tValue);
447                     }
448                 }
449             }
450             i = m.end();
451         }
452         b.append(normalizedTemplate, i, normalizedTemplate.length());
453         return b.toString();
454     }

581     /**
582      * Construct a URI from the component parts each of which may contain
583      * template variables.
584      * <p>
585      * A template value is an Object instance that MUST support the toString()
586      * method to convert the template value to a String instance.
587      *
588      * @param scheme the URI scheme component
589      * @param authority the URI authority info component
590      * @param userInfo the URI user info component
591      * @param host the URI host component
592      * @param port the URI port component
593      * @param path the URI path component
594      * @param query the URI query componnet
595      * @param fragment the URI fragment component
596      * @param values the template variable to value map
597      * @param encode if true encode a template value according to the correspond
598      *        component type of the associated template variable, otherwise
599      *        contextually encode the template value
600      * @return a URI
601      */
602     public final static String createURIWithStringValues(
603             final String scheme, final String authority,
604             final String userInfo, final String host, final String port,
605             final String path, final String query, final String fragment,
606             final Map<String, ? extends Object> values, final boolean encode) {
607 
608         StringBuilder sb = new StringBuilder();
609 
610         if (scheme != null)
611             createURIComponent(UriComponent.Type.SCHEME, scheme, values, false, sb).
612                     append(':');
613 
614         if (userInfo != null || host != null || port != null) {
615             sb.append("//");
616 
617             if (userInfo != null && userInfo.length() > 0)
618                 createURIComponent(UriComponent.Type.USER_INFO, userInfo, values, encode, sb).
619                     append('@');
620 
621             if (host != null) {
622                 // TODO check IPv6 address
623                 createURIComponent(UriComponent.Type.HOST, host, values, encode, sb);
624             }
625 
626             if (port != null && port.length() > 0) {
627                 sb.append(':');
628                 createURIComponent(UriComponent.Type.PORT, port, values, false, sb);
629             }
630         } else if (authority != null) {
631             sb.append("//");
632             
633             createURIComponent(UriComponent.Type.AUTHORITY, authority, values, encode, sb);
634         }
635 
636         if (path != null)
637             createURIComponent(UriComponent.Type.PATH, path, values, encode, sb);
638 
639         if (query != null && query.length() > 0) {
640             sb.append('?');
641             createURIComponent(UriComponent.Type.QUERY_PARAM, query, values, encode, sb);
642         }
643 
644         if (fragment != null && fragment.length() > 0) {
645             sb.append('#');
646             createURIComponent(UriComponent.Type.FRAGMENT, fragment, values, encode, sb);
647         }
648         return sb.toString();
649     }

651     private static StringBuilder createURIComponent(final UriComponent.Type t, 
652             String template,
653             final Map<String, ? extends Object> values, 
654             final boolean encode, 
655             final StringBuilder b) {
656         if (template.indexOf('{') == -1) {
657             b.append(template);
658             return b;
659         }
660         
661         // Find all template variables
662         template = new UriTemplateParser(template).getNormalizedTemplate();
663         final Matcher m = TEMPLATE_NAMES_PATTERN.matcher(template);
664 
665         int i = 0;
666         while(m.find()) {
667             b.append(template, i, m.start());
668             Object tValue = values.get(m.group(1));
669             if (tValue != null) {
670                 if (encode)
671                     tValue = UriComponent.encode(tValue.toString(), t);
672                 else
673                     tValue =UriComponent.contextualEncode(tValue.toString(), t);
674                 b.append(tValue);
675             } else {
676                 throw templateVariableHasNoValue(m.group(1));
677             }
678             i = m.end();
679         }
680         b.append(template, i, template.length());
681         return b;
682     }

860     private static int createURIComponent(final UriComponent.Type t, 
861             String template,
862             final String[] values, final int offset,
863             final boolean encode, 
864             final Map<String, String> mapValues,
865             final StringBuilder b) {
866         if (template.indexOf('{') == -1) {
867             b.append(template);
868             return offset;
869         }
870 
871         // Find all template variables
872         template = new UriTemplateParser(template).getNormalizedTemplate();
873         final Matcher m = TEMPLATE_NAMES_PATTERN.matcher(template);
874         int v = offset;
875         int i = 0;
876         while(m.find()) {
877             b.append(template, i, m.start());
878             final String tVariable = m.group(1);
879             // Check if a template variable has already occurred
880             // If so use the value to ensure that two or more declarations of 
881             // a template variable have the same value
882             String tValue = mapValues.get(tVariable);
883             if (tValue != null) {
884                 b.append(tValue);
885             } else if (v < values.length) {
886                 tValue = values[v++];
887                 if (tValue != null) {
888                     if (encode)
889                         tValue = UriComponent.encode(tValue, t);
890                     else
891                         tValue = UriComponent.contextualEncode(tValue, t);
892                     mapValues.put(tVariable, tValue);
893                     b.append(tValue);
894                 } else {
895                     throw templateVariableHasNoValue(tVariable);
896                 }
897             } else {
898                 throw templateVariableHasNoValue(tVariable);
899             }
900             i = m.end();
901         }
902         b.append(template, i, template.length());
903         return v;
904     }

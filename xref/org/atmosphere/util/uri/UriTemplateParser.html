76      private static Set<Character> RESERVED_REGEX_CHARACTERS = createReserved();

78      private static Set<Character> createReserved() {
79          /* TODO need to escape all regex characters present */
80          char[] reserved = {
81              '.',
82              '?',
83              '(', 
84              ')'};
85  
86          Set<Character> s = new HashSet<Character>(reserved.length);
87          for (char c : reserved) s.add(c);
88          return s;
89      }

91      private static final Pattern TEMPLATE_VALUE_PATTERN = Pattern.compile("[^/]+?");

93      private interface CharacterIterator {
94          boolean hasNext();
95          char next();
96          char peek();
97          int pos();
98      }
99  
100     private static final class StringCharacterIterator implements CharacterIterator {
101         int pos;
102         String s;
103 
104         public StringCharacterIterator(String s) {
105             this.s = s;
106         }
107 
108         public boolean hasNext() {
109             return pos < s.length();
110         }
111 
112         public char next() {
113             if (!hasNext())
114                 throw new NoSuchElementException();
115             return s.charAt(pos++);
116         }
117 
118         public char peek() {
119             if (!hasNext())
120                 throw new NoSuchElementException();
121 
122             return s.charAt(pos++);
123         }
124 
125         public int pos() {
126             if (pos == 0) return 0;
127             return pos - 1;
128         }
129 
130     }

152     /**
153      * Parse a template.
154      * 
155      * @param template the template.
156      * @throws IllegalArgumentException if the template is null, an empty string
157      *         or does not conform to a JAX-RS URI template.
158      */
159     public UriTemplateParser(String template) {
160         if (template == null || template.length() == 0)
161             throw new IllegalArgumentException();
162 
163         this.template = template;
164         parse(new StringCharacterIterator(template));
165         try {
166             pattern = Pattern.compile(regex.toString());
167         } catch (PatternSyntaxException ex) {
168             throw new IllegalArgumentException("Invalid syntax for the template expression '" + 
169                     regex + "'", 
170                     ex);            
171         }
172     }

174     public UriTemplateParser destroy() {

176         groupCounts.clear();
177         nameToPattern.clear();
178         regex.setLength(0);
179         normalizedTemplate.setLength(0);
180         literalCharactersBuffer.setLength(0);
181         return this;
182     }

242     /**
243      * Get the group indexes to capturing groups.
244      * <p>
245      * Any nested capturing groups will be ignored and the
246      * the group index will refer to the top-level capturing
247      * groups associated with the templates variables.
248      *
249      * @return the group indexes to capturing groups.
250      */
251     public final int[] getGroupIndexes() {
252         if (names.isEmpty()) return EMPTY_INT_ARRAY;
253 
254         int[] indexes = new int[names.size() + 1];        
255         indexes[0] = 1;
256         for (int i = 1; i < indexes.length; i++) {
257             indexes[i] = indexes[i - 1] + groupCounts.get(i - 1);
258         }
259         for (int i = 0; i < indexes.length; i++) {
260             if (indexes[i] != i + 1)
261                 return indexes;
262         }
263         return EMPTY_INT_ARRAY;
264     }

294     private void parse(CharacterIterator ci) {
295         try {
296             while (ci.hasNext()) {
297                 char c = ci.next();
298                 if (c == '{') {
299                     processLiteralCharacters();
300                     parseName(ci);
301                 } else {
302                     literalCharactersBuffer.append(c);
303                 }
304             }
305             processLiteralCharacters();
306         } catch (NoSuchElementException ex) {
307             throw new IllegalArgumentException(
308                     "Invalid syntax for the template, \"" + template +
309                     "\". Check if a path parameter is terminated with a '}'.",
310                     ex);
311         }
312     }
313 
314     private void processLiteralCharacters() {
315         if (literalCharactersBuffer.length() > 0) {
316             literalCharacters += literalCharactersBuffer.length();
317 
318             String s = encodeLiteralCharacters(literalCharactersBuffer.toString());
319 
320             normalizedTemplate.append(s);
321 
322             // Escape if reserved regex character
323             for (int i = 0; i < s.length(); i++) {
324                 char c = s.charAt(i);
325                 if (RESERVED_REGEX_CHARACTERS.contains(c))
326                     regex.append("\\");
327                 regex.append(c);
328             }
329 
330             literalCharactersBuffer.setLength(0);
331         }
332     }
333 
334     private void parseName(CharacterIterator ci) {
335         char c = consumeWhiteSpace(ci);
336 
337         StringBuffer nameBuffer = new StringBuffer();        
338         if (Character.isLetterOrDigit(c) || c == '_') {
339             // Template name character
340             nameBuffer.append(c);
341         } else {
342             throw new IllegalArgumentException("Illegal character '" + c + 
343                     "' at position " + ci.pos() + " is not as the start of a name");
344         }
345 
346         String nameRegexString = "";
347         while(true) {
348             c = ci.next();
349             // "\\{(\\w[-\\w\\.]*)
350             if (Character.isLetterOrDigit(c) || c == '_' || c == '-' || c == '.') {
351                 // Template name character             
352                 nameBuffer.append(c);
353             } else if (c == ':') {
354                 nameRegexString = parseRegex(ci);
355                 break;
356             } else if (c == '}') {
357                 break;
358             } else if (c == ' ') {
359                 c = consumeWhiteSpace(ci);
360 
361                 if (c == ':') {
362                     nameRegexString = parseRegex(ci);
363                     break;
364                 } else if (c == '}') {
365                     break;
366                 } else {
367                     // Error
368                     throw new IllegalArgumentException("Illegal character '" + c + 
369                             "' at position " + ci.pos() + " is not allowed after a name");
370                 }
371             } else {
372                 throw new IllegalArgumentException("Illegal character '" + c + 
373                         "' at position " + ci.pos() + " is not allowed as part of a name");
374             }
375         }        
376         String name = nameBuffer.toString();
377         names.add(name);
378 
379         try {
380             if (nameRegexString.length() > 0)
381                 numOfExplicitRegexes++;
382             Pattern namePattern = (nameRegexString.length() == 0) 
383                     ? TEMPLATE_VALUE_PATTERN : Pattern.compile(nameRegexString);
384             if (nameToPattern.containsKey(name)) {
385                 if (!nameToPattern.get(name).equals(namePattern)) {
386                     throw new IllegalArgumentException("The name '" + name + 
387                             "' is declared " +
388                             "more than once with different regular expressions");
389                 }
390             } else {
391                 nameToPattern.put(name, namePattern);            
392             }
393 
394             // Determine group count of pattern
395             Matcher m = namePattern.matcher("");
396             int g = m.groupCount();
397             groupCounts.add(g + 1);
398 
399             regex.append('(').
400                     append(namePattern).
401                     append(')');
402             normalizedTemplate.append('{').
403                     append(name).
404                     append('}');
405         } catch (PatternSyntaxException ex) {
406             throw new IllegalArgumentException("Invalid syntax for the expression '" + nameRegexString + 
407                     "' associated with the name '" + name + "'", 
408                     ex);
409         }
410     }
411 
412     private String parseRegex(CharacterIterator ci) {
413         StringBuffer regexBuffer = new StringBuffer();
414 
415         int braceCount = 1;
416         while (true) {
417             char c = ci.next();
418             if (c == '{') {
419                 braceCount++;
420             } else if (c == '}') {
421                 braceCount--;
422                 if (braceCount == 0)
423                     break;
424             }            
425             regexBuffer.append(c);
426         }
427 
428         return regexBuffer.toString().trim();
429     }

431     private char consumeWhiteSpace(CharacterIterator ci) {
432         char c = ci.next();
433         // Consume white space;
434         // TODO use correct c
435         while (c == ' ') c = ci.next();
436 
437         return c;
438     }

17  package org.atmosphere.util;
18  
19  import org.atmosphere.cpr.BroadcastFilter;
20  
21  /**
22   * Simple {@link BroadcastFilter} which automatically filter
23   * HTML/HTTP character into proper value, like \n replace by &lt;br&gt;. Using
24   * this BroadcastFilter prevent XSS attack.
25   *
26   * @author Jeanfrancois Arcand
27   */
28  public class XSSHtmlFilter implements BroadcastFilter {
29  
30      /**
31       * Transform a message into a well formed HTML message.
32       *
33       * @param o The object to introspect.
34       * @return a well formed
35       */
36      public BroadcastAction filter(String broadcasterId, Object originalMessage, Object o) {
37          if (o instanceof String) {
38              String message = (String) o;
39  
40              StringBuffer buffer = new StringBuffer(message.length());
41  
42              for (int i = 0; i < message.length(); i++) {
43                  char c = message.charAt(i);
44                  switch (c) {
45                      case '\b':
46                          buffer.append("\\b");
47                          break;
48                      case '\f':
49                          buffer.append("\\f");
50                          break;
51                      case '\n':
52                          buffer.append("<br />");
53                          break;
54                      case '\r':
55                          // ignore
56                          break;
57                      case '\t':
58                          buffer.append("\\t");
59                          break;
60                      case '\'':
61                          buffer.append("\\'");
62                          break;
63                      case '\"':
64                          buffer.append("\\\"");
65                          break;
66                      case '\\':
67                          buffer.append("\\\\");
68                          break;
69                      case '<':
70                          buffer.append("&lt;");
71                          break;
72                      case '>':
73                          buffer.append("&gt;");
74                          break;
75                      case '&':
76                          buffer.append("&amp;");
77                          break;
78                      default:
79                          buffer.append(c);
80                  }
81              }
82              return new BroadcastAction(buffer.toString());
83          } else {
84              return new BroadcastAction(o);
85          }
86      }
87  }

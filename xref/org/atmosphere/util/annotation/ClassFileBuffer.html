38  package org.atmosphere.util.annotation;
39  
40  import java.io.DataInput;
41  import java.io.DataInputStream;
42  import java.io.EOFException;
43  import java.io.IOException;
44  import java.io.InputStream;
45  
46  /**
47   * {@code ClassFileBuffer} is used by {@link eu.infomas.annotation.AnnotationDetector} to efficiently
48   * read Java ClassFile files from an {@link java.io.InputStream} and parse the content
49   * via the {@link java.io.DataInput} interface.
50   * <br/>
51   * Note that Java ClassFile files can grow really big,
52   * {@code com.sun.corba.se.impl.logging.ORBUtilSystemException} is 128.2 kb!
53   *
54   * @author <a href="mailto:rmuller@xiam.nl">Ronald K. Muller</a>
55   * @since annotation-detector 3.0.0
56   */
57  final class ClassFileBuffer implements DataInput {
58  
59      private byte[] buffer;
60      private int size; // the number of significant bytes read
61      private int pointer; // the "read pointer"
62  
63      /**
64       * Create a new, empty {@code ClassFileBuffer} with the default initial capacity (8 kb).
65       */
66      ClassFileBuffer() {
67          this(8 * 1024);
68      }
69  
70      /**
71       * Create a new, empty {@code ClassFileBuffer} with the specified initial capacity.
72       * The initial capacity must be greater than zero. The internal buffer will grow
73       * automatically when a higher capacity is required. However, buffer resizing occurs
74       * extra overhead. So in good initial capacity is important in performance critical
75       * situations.
76       */
77      ClassFileBuffer(final int initialCapacity) {
78          if (initialCapacity < 1) {
79              throw new IllegalArgumentException("initialCapacity < 1: " + initialCapacity);
80          }
81          this.buffer = new byte[initialCapacity];
82      }
83  
84      /**
85       * Clear and fill the buffer of this {@code ClassFileBuffer} with the
86       * supplied byte stream.
87       * The read pointer is reset to the start of the byte array.
88       */
89      public void readFrom(final InputStream in) throws IOException {
90          pointer = 0;
91          size = 0;
92          int n;
93          do {
94              n = in.read(buffer, size, buffer.length - size);
95              if (n > 0) {
96                  size += n;
97              }
98              resizeIfNeeded();
99          } while (n >= 0);
100     }
101 
102     /**
103      * Sets the file-pointer offset, measured from the beginning of this file,
104      * at which the next read or write occurs.
105      */
106     public void seek(final int position) throws IOException {
107         if (position < 0) {
108             throw new IllegalArgumentException("position < 0: " + position);
109         }
110         if (position > size) {
111             throw new EOFException();
112         }
113         this.pointer = position;
114     }
115 
116     /**
117      * Return the size (in bytes) of this Java ClassFile file.
118      */
119     public int size() {
120         return size;
121     }
122 
123     // DataInput
124 
125     @Override
126     public void readFully(final byte bytes[]) throws IOException {
127         readFully(bytes, 0, bytes.length);
128     }
129 
130     @Override
131     public void readFully(final byte bytes[], final int offset, final int length) throws IOException {
132         if (length < 0 || offset < 0 || offset + length > bytes.length) {
133             throw new IndexOutOfBoundsException();
134         }
135         if (pointer + length > size) {
136             throw new EOFException();
137         }
138         System.arraycopy(buffer, pointer, bytes, offset, length);
139         pointer += length;
140     }
141 
142     @Override
143     public int skipBytes(final int n) throws IOException {
144         seek(pointer + n);
145         return n;
146     }
147 
148     @Override
149     public byte readByte() throws IOException {
150         if (pointer >= size) {
151             throw new EOFException();
152         }
153         return buffer[pointer++];
154     }
155 
156     @Override
157     public boolean readBoolean() throws IOException {
158         return readByte() != 0;
159     }
160 
161     @Override
162     public int readUnsignedByte() throws IOException {
163         if (pointer >= size) {
164             throw new EOFException();
165         }
166         return read();
167     }
168 
169     @Override
170     public int readUnsignedShort() throws IOException {
171         if (pointer + 2 > size) {
172             throw new EOFException();
173         }
174         return (read() << 8) + read();
175     }
176 
177     @Override
178     public short readShort() throws IOException {
179         return (short) readUnsignedShort();
180     }
181 
182     @Override
183     public char readChar() throws IOException {
184         return (char) readUnsignedShort();
185     }
186 
187     @Override
188     public int readInt() throws IOException {
189         if (pointer + 4 > size) {
190             throw new EOFException();
191         }
192         return (read() << 24) +
193                 (read() << 16) +
194                 (read() << 8) +
195                 read();
196     }
197 
198     @Override
199     public long readLong() throws IOException {
200         if (pointer + 8 > size) {
201             throw new EOFException();
202         }
203         return (read() << 56) +
204                 (read() << 48) +
205                 (read() << 40) +
206                 (read() << 32) +
207                 (read() << 24) +
208                 (read() << 16) +
209                 (read() << 8) +
210                 read();
211     }
212 
213     @Override
214     public float readFloat() throws IOException {
215         return Float.intBitsToFloat(readInt());
216     }
217 
218     @Override
219     public double readDouble() throws IOException {
220         return Double.longBitsToDouble(readLong());
221     }
222 
223     /**
224      * This methods throws an {@link UnsupportedOperationException} because the method
225      * is deprecated and not used in the context of this implementation.
226      */
227     @Override
228     @Deprecated
229     public String readLine() throws IOException {
230         throw new UnsupportedOperationException("readLine() is deprecated and not supported");
231     }
232 
233     @Override
234     public String readUTF() throws IOException {
235         return DataInputStream.readUTF(this);
236     }
237 
238     // private
239 
240     private int read() {
241         return buffer[pointer++] & 0xff;
242     }
243 
244     private void resizeIfNeeded() {
245         if (size >= buffer.length) {
246             final byte[] newBuffer = new byte[buffer.length * 2];
247             System.arraycopy(buffer, 0, newBuffer, 0, buffer.length);
248             buffer = newBuffer;
249         }
250     }
251 
252     public void destroy() {
253         buffer = null;
254     }
255 }

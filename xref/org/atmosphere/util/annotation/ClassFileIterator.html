39  package org.atmosphere.util.annotation;
40  
41  import org.atmosphere.util.FileIterator;
42  
43  import java.io.File;
44  import java.io.FileInputStream;
45  import java.io.IOException;
46  import java.io.InputStream;
47  
48  
49  /**
50   * {@code ClassFileIterator} is used to iterate over all Java ClassFile files
51   * available within a specific context. For every Java ClassFile ({@code .class})
52   * an {@link InputStream} is returned.
53   *
54   * @author <a href="mailto:rmuller@xiam.nl">Ronald K. Muller</a>
55   * @since annotation-detector 3.0.0
56   */
57  final class ClassFileIterator {
58  
59      private final FileIterator fileIterator;
60      private ZipFileIterator zipIterator;
61      private boolean isFile;
62  
63      /**
64       * Create a new {@code ClassFileIterator} returning all Java ClassFile files
65       * available from the class path ({@code System.getProperty("java.class.path")}).
66       */
67      ClassFileIterator() throws IOException {
68          this(classPath());
69      }
70  
71      /**
72       * Create a new {@code ClassFileIterator} returning all Java ClassFile files
73       * available from the specified files and/or directories, including sub
74       * directories.
75       */
76      public ClassFileIterator(final File... filesOrDirectories) throws IOException {
77          fileIterator = new FileIterator(filesOrDirectories);
78      }
79  
80      /**
81       * Return the name of the Java ClassFile returned from the last call to
82       * {@link #next()}. The name is either the path name of a file or the name of
83       * an ZIP/JAR file entry.
84       */
85      public String getName() {
86          // Both getPath() and getName() are very light weight method calls
87          return zipIterator == null ?
88                  fileIterator.getFile().getPath() :
89                  zipIterator.getEntry().getName();
90      }
91  
92      /**
93       * Return {@code true} if the current {@link InputStream} is reading from a
94       * plain {@link File}. Return {@code false} if the current {@link InputStream}
95       * is reading from a ZIP File Entry.
96       */
97      public boolean isFile() {
98          return isFile;
99      }
100 
101     /**
102      * Return the next Java ClassFile as an {@code InputStream}.
103      * <br/>
104      * NOTICE: Client code MUST close the returned {@code InputStream}!
105      */
106     public InputStream next() throws IOException {
107         while (true) {
108             if (zipIterator == null) {
109                 final File file = fileIterator.next();
110                 if (file == null) {
111                     return null;
112                 } else {
113                     final String name = file.getName();
114                     if (name.endsWith(".class")) {
115                         isFile = true;
116                         return new FileInputStream(file);
117                     } else if (fileIterator.isRootFile() && endsWithIgnoreCase(name, ".jar")) {
118                         zipIterator = new ZipFileIterator(file);
119                     } // else just ignore
120                     continue;
121                 }
122             } else {
123                 final InputStream is = zipIterator.next();
124                 if (is == null) {
125                     zipIterator = null;
126                     continue;
127                 } else {
128                     isFile = false;
129                     return is;
130                 }
131             }
132         }
133     }
134 
135     // private
136 
137     /**
138      * Returns the class path of the current JVM instance as an array of {@link File} objects.
139      */
140     private static File[] classPath() {
141         final String[] fileNames = System.getProperty("java.class.path").split(File.pathSeparator);
142         final File[] files = new File[fileNames.length];
143         for (int i = 0; i < files.length; ++i) {
144             files[i] = new File(fileNames[i]);
145         }
146         return files;
147     }
148 
149     private static boolean endsWithIgnoreCase(final String value, final String suffix) {
150         final int n = suffix.length();
151         return value.regionMatches(true, value.length() - n, suffix, 0, n);
152     }

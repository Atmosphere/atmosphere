17  import java.io.IOException;
18  import java.io.InputStream;
19  import java.util.Deque;
20  import java.util.LinkedList;
21  
22  /**
23   * {@code FileIterator} enables iteration over all files in a directory and all
24   * its sub directories.
25   * <br/>
26   * Usage:
27   * <pre>
28   * FileIterator iter = new FileIterator(new File("./src"));
29   * File f;
30   * while ((f = iter.next()) != null) {
31   *     // do something with f
32   *     assert f == iter.getCurrent();
33   * }
34   * </pre>
35   *
36   * @author <a href="mailto:rmuller@xiam.nl">Ronald K. Muller</a>
37   * @since annotation-detector 3.0.0
38   */
39  public final class InputStreamIterator {
40  
41      private final Deque<InputStream> stack = new LinkedList<InputStream>();
42      private int rootCount;
43      private InputStream current;
44  
45      /**
46       * Create a new {@code FileIterator} using the specified 'filesOrDirectories' as root.
47       * <br/>
48       * If 'filesOrDirectories' contains a file, the iterator just returns that single file.
49       * If 'filesOrDirectories' contains a directory, all files in that directory
50       * and its sub directories are returned (depth first).
51       *
52       * @param filesOrDirectories Zero or more {@link java.io.File} objects, which are iterated
53       * in the specified order (depth first)
54       */
55      public InputStreamIterator(final InputStream... filesOrDirectories) {
56          addReverse(filesOrDirectories);
57          rootCount = stack.size();
58      }
59  
60      /**
61       * Return the next {@link java.io.File} object or {@code null} if no more files are
62       * available.
63       *
64       */
65      public InputStream next() throws IOException {
66          if (stack.isEmpty()) {
67              current = null;
68              return null;
69          } else {
70              current = stack.removeLast();
71              return current;
72          }
73      }
74  
75      /**
76       * Add the specified files in reverse order.
77       */
78      private void addReverse(final InputStream[] files) {
79          for (int i = files.length - 1; i >=0; --i) {
80              stack.add(files[i]);
81          }
82      }
83  
84  }

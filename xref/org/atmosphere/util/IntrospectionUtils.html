80       * Call execute() - any ant-like task should work
81       */
82      public static void execute(Object proxy, String method) throws Exception {
83          Method executeM = null;
84          Class<?> c = proxy.getClass();
85          Class<?> params[] = new Class[0];
86          //	params[0]=args.getClass();
87          executeM = findMethod(c, method, params);
88          if (executeM == null) {
89              throw new RuntimeException("No execute in " + proxy.getClass());
90          }
91          executeM.invoke(proxy, (Object[]) null);//new Object[] { args });
92      }

94      /**
95       * Call void setAttribute( String ,Object )
96       */
97      public static void setAttribute(Object proxy, String name, Object value)
98              throws Exception {
99          if (proxy instanceof AttributeHolder) {
100             ((AttributeHolder) proxy).setAttribute(name, value);
101             return;
102         }
103 
104         Method executeM = null;
105         Class<?> c = proxy.getClass();
106         Class<?> params[] = new Class[2];
107         params[0] = String.class;
108         params[1] = Object.class;
109         executeM = findMethod(c, "setAttribute", params);
110         if (executeM == null) {
111             logger.debug("No setAttribute in {}", proxy.getClass());
112             return;
113         }
114 
115         logger.debug("Setting {}={} in proxy: {}", new Object[]{name, value, proxy});
116         executeM.invoke(proxy, new Object[]{name, value});
117         return;
118     }

135     /**
136      * Construct a URLClassLoader. Will compile and work in JDK1.1 too.
137      */
138     public static ClassLoader getURLClassLoader(URL urls[], ClassLoader parent) {
139         try {
140             Class<?> urlCL = Class.forName("java.net.URLClassLoader");
141             Class<?> paramT[] = new Class[2];
142             paramT[0] = urls.getClass();
143             paramT[1] = ClassLoader.class;
144             Method m = findMethod(urlCL, "newInstance", paramT);
145             if (m == null) {
146                 return null;
147             }
148 
149             ClassLoader cl = (ClassLoader) m.invoke(urlCL, new Object[]{urls, parent});
150             return cl;
151         } catch (ClassNotFoundException ex) {
152             // jdk1.1
153             return null;
154         } catch (Exception ex) {
155             logger.error("failed getting URLClassLoader", ex);
156             return null;
157         }
158     }

176     final static public boolean invokeProperty(Object object, String setter, String name, String value) {
177         try {
178             Method methods[] = findMethods(object.getClass());
179             Method setPropertyMethodVoid = null;
180             Method setPropertyMethodBool = null;
181 
182             // First, the ideal case - a setFoo( String ) method
183             for (int i = 0; i < methods.length; i++) {
184                 Class<?> paramT[] = methods[i].getParameterTypes();
185                 if (setter.equals(methods[i].getName()) && paramT.length == 1
186                         && "java.lang.String".equals(paramT[0].getName())) {
187 
188                     methods[i].invoke(object, new Object[]{value});
189                     return true;
190                 }
191             }
192 
193             // Try a setFoo ( int ) or ( boolean )
194             for (int i = 0; i < methods.length; i++) {
195                 boolean ok = true;
196                 if (setter.equals(methods[i].getName())
197                         && methods[i].getParameterTypes().length == 1) {
198 
199                     // match - find the type and invoke it
200                     Class<?> paramType = methods[i].getParameterTypes()[0];
201                     Object params[] = new Object[1];
202 
203                     // Try a setFoo ( int )
204                     if ("java.lang.Integer".equals(paramType.getName())
205                             || "int".equals(paramType.getName())) {
206                         try {
207                             params[0] = new Integer(value);
208                         } catch (NumberFormatException ex) {
209                             ok = false;
210                         }
211                         // Try a setFoo ( long )
212                     } else if ("java.lang.Long".equals(paramType.getName())
213                             || "long".equals(paramType.getName())) {
214                         try {
215                             params[0] = new Long(value);
216                         } catch (NumberFormatException ex) {
217                             ok = false;
218                         }
219 
220                         // Try a setFoo ( boolean )
221                     } else if ("java.lang.Boolean".equals(paramType.getName())
222                             || "boolean".equals(paramType.getName())) {
223                         params[0] = new Boolean(value);
224 
225                         // Try a setFoo ( InetAddress )
226                     } else if ("java.net.InetAddress".equals(paramType
227                             .getName())) {
228                         try {
229                             params[0] = InetAddress.getByName(value);
230                         } catch (UnknownHostException exc) {
231                             debug("Unable to resolve host name:" + value);
232                             ok = false;
233                         }
234 
235                         // Unknown type
236                     } else {
237                         debug("Unknown type " + paramType.getName());
238                     }
239 
240                     if (ok) {
241                         methods[i].invoke(object, params);
242                         return true;
243                     }
244                 }
245 
246                 // save "setProperty" for later
247                 if ("setProperty".equals(methods[i].getName())) {
248                     if (methods[i].getReturnType() == Boolean.TYPE) {
249                         setPropertyMethodBool = methods[i];
250                     } else {
251                         setPropertyMethodVoid = methods[i];
252                     }
253 
254                 }
255             }
256 
257             // Ok, no setXXX found, try a setProperty("name", "value")
258             if (setPropertyMethodBool != null || setPropertyMethodVoid != null) {
259                 Object params[] = new Object[2];
260                 params[0] = name;
261                 params[1] = value;
262                 if (setPropertyMethodBool != null) {
263                     try {
264                         return (Boolean) setPropertyMethodBool.invoke(object, params);
265                     } catch (IllegalArgumentException biae) {
266                         //the boolean method had the wrong
267                         //parameter types. lets try the other
268                         if (setPropertyMethodVoid != null) {
269                             setPropertyMethodVoid.invoke(object, params);
270                             return true;
271                         } else {
272                             throw biae;
273                         }
274                     }
275                 } else {
276                     setPropertyMethodVoid.invoke(object, params);
277                     return true;
278                 }
279             }
280 
281         } catch (IllegalArgumentException e) {
282             logger.info("failed, object: " + object + ", setter: " + setter + ", value: " + value, e);
283         } catch (Exception e) {
284             if (dbg > 0) {
285                 debug(e.getClass().getSimpleName() + " for " + object.getClass() + " " + setter + "=" + value + ")");
286             }
287             if (dbg > 1) {
288                 logger.debug("", e);
289             }
290         }
291 
292         return false;
293     }

295     public static Object getProperty(Object object, String name) {
296         String getter = "get" + capitalize(name);
297         String isGetter = "is" + capitalize(name);
298 
299         try {
300             Method methods[] = findMethods(object.getClass());
301             Method getPropertyMethod = null;
302 
303             // First, the ideal case - a getFoo() method
304             for (int i = 0; i < methods.length; i++) {
305                 Class<?> paramT[] = methods[i].getParameterTypes();
306                 if (getter.equals(methods[i].getName()) && paramT.length == 0) {
307                     return methods[i].invoke(object, (Object[]) null);
308                 }
309                 if (isGetter.equals(methods[i].getName()) && paramT.length == 0) {
310                     return methods[i].invoke(object, (Object[]) null);
311                 }
312 
313                 if ("getProperty".equals(methods[i].getName())) {
314                     getPropertyMethod = methods[i];
315                 }
316             }
317 
318             // Ok, no setXXX found, try a getProperty("name")
319             if (getPropertyMethod != null) {
320                 Object params[] = new Object[1];
321                 params[0] = name;
322                 return getPropertyMethod.invoke(object, params);
323             }
324 
325         } catch (IllegalArgumentException e) {
326             logger.info("failed, object: " + object + ", name: " + name, e);
327         } catch (Exception e) {
328             if (dbg > 0) {
329                 debug(e.getClass().getSimpleName() + " for " + object.getClass() + " " + name + ")");
330             }
331             if (dbg > 1) {
332                 logger.debug("", e);
333             }
334         }
335 
336         return null;
337     }

340      */
341     public static void setProperty(Object object, String name) {
342         String setter = "set" + capitalize(name);
343         try {
344             Method methods[] = findMethods(object.getClass());
345             Method setPropertyMethod = null;
346             // find setFoo() method
347             for (int i = 0; i < methods.length; i++) {
348                 Class<?> paramT[] = methods[i].getParameterTypes();
349                 if (setter.equals(methods[i].getName()) && paramT.length == 0) {
350                     methods[i].invoke(object, new Object[]{});
351                     return;
352                 }
353             }
354         } catch (Exception e) {
355             if (dbg > 0) {
356                 debug("Exception for " + object.getClass() + " " + name);
357             }
358             if (dbg > 1) {
359                 logger.debug("", e);
360             }
361         }
362     }

364     /**
365      * Replace ${NAME} with the property value
366      */
367     public static String replaceProperties(String value,
368                                            Hashtable<String, String> staticProp, PropertySource dynamicProp[]) {
369         if (value.indexOf("$") < 0) {
370             return value;
371         }
372         StringBuilder sb = new StringBuilder();
373         int prev = 0;
374         // assert value!=nil
375         int pos;
376         while ((pos = value.indexOf("$", prev)) >= 0) {
377             if (pos > 0) {
378                 sb.append(value.substring(prev, pos));
379             }
380             if (pos == (value.length() - 1)) {
381                 sb.append('$');
382                 prev = pos + 1;
383             } else if (value.charAt(pos + 1) != '{') {
384                 sb.append('$');
385                 prev = pos + 1; // XXX
386             } else {
387                 int endName = value.indexOf('}', pos);
388                 if (endName < 0) {
389                     sb.append(value.substring(pos));
390                     prev = value.length();
391                     continue;
392                 }
393                 String n = value.substring(pos + 2, endName);
394                 String v = null;
395                 if (staticProp != null) {
396                     v = staticProp.get(n);
397                 }
398                 if (v == null && dynamicProp != null) {
399                     for (int i = 0; i < dynamicProp.length; i++) {
400                         v = dynamicProp[i].getProperty(n);
401                         if (v != null) {
402                             break;
403                         }
404                     }
405                 }
406                 if (v == null)
407                     v = "${" + n + "}";
408 
409                 sb.append(v);
410                 prev = endName + 1;
411             }
412         }
413         if (prev < value.length())
414             sb.append(value.substring(prev));
415         return sb.toString();
416     }

418     /**
419      * Reverse of Introspector.decapitalize
420      */
421     public static String capitalize(String name) {
422         if (name == null || name.length() == 0) {
423             return name;
424         }
425         char chars[] = name.toCharArray();
426         chars[0] = Character.toUpperCase(chars[0]);
427         return new String(chars);
428     }

430     public static String unCapitalize(String name) {
431         if (name == null || name.length() == 0) {
432             return name;
433         }
434         char chars[] = name.toCharArray();
435         chars[0] = Character.toLowerCase(chars[0]);
436         return new String(chars);
437     }

441     /**
442      * Add all the jar files in a dir to the classpath, represented as a Vector
443      * of URLs.
444      */
445     @SuppressWarnings("unchecked")
446     public static void addToClassPath(Vector<URL> cpV, String dir) {
447         try {
448             String cpComp[] = getFilesByExt(dir, ".jar");
449             if (cpComp != null) {
450                 int jarCount = cpComp.length;
451                 for (int i = 0; i < jarCount; i++) {
452                     URL url = getURL(dir, cpComp[i]);
453                     if (url != null)
454                         cpV.addElement(url);
455                 }
456             }
457         } catch (Exception ex) {
458             logger.debug("failed to add urls to classpath", ex);
459         }
460     }

462     @SuppressWarnings("unchecked")
463     public static void addToolsJar(Vector<URL> v) {
464         try {
465             // Add tools.jar in any case
466             File f = new File(System.getProperty("java.home")
467                     + "/../lib/tools.jar");
468 
469             if (!f.exists()) {
470                 // On some systems java.home gets set to the root of jdk.
471                 // That's a bug, but we can work around and be nice.
472                 f = new File(System.getProperty("java.home") + "/lib/tools.jar");
473                 if (f.exists()) {
474                     logger.debug("Detected strange java.home value {}, it should point to jre",
475                             System.getProperty("java.home"));
476                 }
477             }
478             URL url = new URL("file", "", f.getAbsolutePath());
479 
480             v.addElement(url);
481         } catch (MalformedURLException ex) {
482             logger.debug("failed to add tools jar url to vector", ex);
483         }
484     }

486     /**
487      * Return all files with a given extension in a dir
488      */
489     public static String[] getFilesByExt(String ld, String ext) {
490         File dir = new File(ld);
491         String[] names = null;
492         final String lext = ext;
493         if (dir.isDirectory()) {
494             names = dir.list(new FilenameFilter() {
495                 public boolean accept(File d, String name) {
496                     if (name.endsWith(lext)) {
497                         return true;
498                     }
499                     return false;
500                 }
501             });
502         }
503         return names;
504     }

506     /**
507      * Construct a file url from a file, using a base dir
508      */
509     public static URL getURL(String base, String file) {
510         try {
511             File baseF = new File(base);
512             File f = new File(baseF, file);
513             String path = f.getCanonicalPath();
514             if (f.isDirectory()) {
515                 path += "/";
516             }
517             if (!f.exists()) {
518                 return null;
519             }
520             return new URL("file", "", path);
521         } catch (Exception ex) {
522             logger.debug("failed to get url, base: " + base + ", file: " + file, ex);
523             return null;
524         }
525     }

527     /**
528      * Add elements from the classpath <i>cp </i> to a Vector <i>jars </i> as
529      * file URLs (We use Vector for JDK 1.1 compat).
530      * <p/>
531      *
532      * @param jars The jar list
533      * @param cp   a String classpath of directory or jar file elements
534      *             separated by path.separator delimiters.
535      * @throws IOException           If an I/O error occurs
536      * @throws MalformedURLException Doh ;)
537      */
538     @SuppressWarnings("unchecked")
539     public static void addJarsFromClassPath(Vector<URL> jars, String cp)
540             throws IOException, MalformedURLException {
541         String sep = System.getProperty("path.separator");
542         String token;
543         StringTokenizer st;
544         if (cp != null) {
545             st = new StringTokenizer(cp, sep);
546             while (st.hasMoreTokens()) {
547                 File f = new File(st.nextToken());
548                 String path = f.getCanonicalPath();
549                 if (f.isDirectory()) {
550                     path += "/";
551                 }
552                 URL url = new URL("file", "", path);
553                 if (!jars.contains(url)) {
554                     jars.addElement(url);
555                 }
556             }
557         }
558     }

571     /**
572      * Construct a URL classpath from files in a directory, a cpath property,
573      * and tools.jar.
574      */
575     @SuppressWarnings("unchecked")
576     public static URL[] getClassPath(String dir, String cpath,
577                                      String cpathProp, boolean addTools) throws IOException,
578             MalformedURLException {
579         Vector<URL> jarsV = new Vector<URL>();
580         if (dir != null) {
581             // Add dir/classes first, if it exists
582             URL url = getURL(dir, "classes");
583             if (url != null)
584                 jarsV.addElement(url);
585             addToClassPath(jarsV, dir);
586         }
587 
588         if (cpath != null)
589             addJarsFromClassPath(jarsV, cpath);
590 
591         if (cpathProp != null) {
592             String cpath1 = System.getProperty(cpathProp);
593             addJarsFromClassPath(jarsV, cpath1);
594         }
595 
596         if (addTools)
597             addToolsJar(jarsV);
598 
599         return getClassPath(jarsV);
600     }

604     public static boolean processArgs(Object proxy, String args[])
605             throws Exception {
606         String args0[] = null;
607         if (null != findMethod(proxy.getClass(), "getOptions1", new Class[]{})) {
608             args0 = (String[]) callMethod0(proxy, "getOptions1");
609         }
610 
611         if (args0 == null) {
612             //args0=findVoidSetters(proxy.getClass());
613             args0 = findBooleanSetters(proxy.getClass());
614         }
615         Hashtable<String, String> h = null;
616         if (null != findMethod(proxy.getClass(), "getOptionAliases",
617                 new Class[]{})) {
618             h = (Hashtable<String, String>) callMethod0(proxy, "getOptionAliases");
619         }
620         return processArgs(proxy, args, args0, null, h);
621     }

623     public static boolean processArgs(Object proxy, String args[],
624                                       String args0[], String args1[],
625                                       Hashtable<String, String> aliases) throws Exception {
626         for (int i = 0; i < args.length; i++) {
627             String arg = args[i];
628             if (arg.startsWith("-"))
629                 arg = arg.substring(1);
630             if (aliases != null && aliases.get(arg) != null)
631                 arg = aliases.get(arg);
632 
633             if (args0 != null) {
634                 boolean set = false;
635                 for (int j = 0; j < args0.length; j++) {
636                     if (args0[j].equalsIgnoreCase(arg)) {
637                         setProperty(proxy, args0[j], "true");
638                         set = true;
639                         break;
640                     }
641                 }
642                 if (set)
643                     continue;
644             }
645             if (args1 != null) {
646                 for (int j = 0; j < args1.length; j++) {
647                     if (args1[j].equalsIgnoreCase(arg)) {
648                         i++;
649                         if (i >= args.length)
650                             return false;
651                         setProperty(proxy, arg, args[i]);
652                         break;
653                     }
654                 }
655             } else {
656                 // if args1 is not specified,assume all other options have param
657                 i++;
658                 if (i >= args.length)
659                     return false;
660                 setProperty(proxy, arg, args[i]);
661             }
662 
663         }
664         return true;
665     }

673     @SuppressWarnings("unchecked")
674     public static String[] findVoidSetters(Class<?> c) {
675         Method m[] = findMethods(c);
676         if (m == null)
677             return null;
678         Vector<String> v = new Vector<String>();
679         for (int i = 0; i < m.length; i++) {
680             if (m[i].getName().startsWith("set")
681                     && m[i].getParameterTypes().length == 0) {
682                 String arg = m[i].getName().substring(3);
683                 v.addElement(unCapitalize(arg));
684             }
685         }
686         String s[] = new String[v.size()];
687         for (int i = 0; i < s.length; i++) {
688             s[i] = (String) v.elementAt(i);
689         }
690         return s;
691     }

693     @SuppressWarnings("unchecked")
694     public static String[] findBooleanSetters(Class<?> c) {
695         Method m[] = findMethods(c);
696         if (m == null)
697             return null;
698         Vector<String> v = new Vector<String>();
699         for (int i = 0; i < m.length; i++) {
700             if (m[i].getName().startsWith("set")
701                     && m[i].getParameterTypes().length == 1
702                     && "boolean".equalsIgnoreCase(m[i].getParameterTypes()[0]
703                     .getName())) {
704                 String arg = m[i].getName().substring(3);
705                 v.addElement(unCapitalize(arg));
706             }
707         }
708         String s[] = new String[v.size()];
709         for (int i = 0; i < s.length; i++) {
710             s[i] = v.elementAt(i);
711         }
712         return s;

729     public static Method findMethod(Class<?> c, String name,
730                                     Class<?> params[]) {
731         Method methods[] = findMethods(c);
732         if (methods == null)
733             return null;
734         for (int i = 0; i < methods.length; i++) {
735             if (methods[i].getName().equals(name)) {
736                 Class<?> methodParams[] = methods[i].getParameterTypes();
737                 if (methodParams == null)
738                     if (params == null || params.length == 0)
739                         return methods[i];
740                 if (params == null)
741                     if (methodParams == null || methodParams.length == 0)
742                         return methods[i];
743                 if (params.length != methodParams.length)
744                     continue;
745                 boolean found = true;
746                 for (int j = 0; j < params.length; j++) {
747                     if (params[j] != methodParams[j]) {
748                         found = false;
749                         break;
750                     }
751                 }
752                 if (found)
753                     return methods[i];
754             }
755         }
756         return null;
757     }

863     public static Object convert(String object, Class<?> paramType) {
864         Object result = null;
865         if ("java.lang.String".equals(paramType.getName())) {
866             result = object;
867         } else if ("java.lang.Integer".equals(paramType.getName())
868                 || "int".equals(paramType.getName())) {
869             try {
870                 result = new Integer(object);
871             } catch (NumberFormatException ex) {
872             }
873             // Try a setFoo ( boolean )
874         } else if ("java.lang.Boolean".equals(paramType.getName())
875                 || "boolean".equals(paramType.getName())) {
876             result = new Boolean(object);
877 
878             // Try a setFoo ( InetAddress )
879         } else if ("java.net.InetAddress".equals(paramType
880                 .getName())) {
881             try {
882                 result = InetAddress.getByName(object);
883             } catch (UnknownHostException exc) {
884                 debug("Unable to resolve host name:" + object);
885             }
886 
887             // Unknown type
888         } else {
889             debug("Unknown type " + paramType.getName());
890         }
891         if (result == null) {
892             throw new IllegalArgumentException("Can't convert argument: " + object);
893         }
894         return result;
895     }

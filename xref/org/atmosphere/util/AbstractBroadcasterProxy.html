36  /**
37   * Abstract {@link org.atmosphere.cpr.Broadcaster} that delegates the internal processing to a proxy.
38   *
39   * @author Jeanfrancois Arcand
40   */
41  public abstract class AbstractBroadcasterProxy extends DefaultBroadcaster {
42      private static final Logger logger = LoggerFactory.getLogger(AbstractBroadcasterProxy.class);
43  
44      private Method jerseyBroadcast;
45  
46      public AbstractBroadcasterProxy(String id, URI uri, AtmosphereServlet.AtmosphereConfig config) {
47          super(id, uri, config);
48      }
49  
50      /**
51       * Implement this method to broadcast message received from an external source like JGroups, Redis, etc.
52       */
53      abstract public void incomingBroadcast();
54  
55      /**
56       * Implement this method to broadcast message to external source like JGroups, Redis, etc.
57       *
58       * @param message outgoing message
59       */
60      abstract public void outgoingBroadcast(Object message);
61  
62      /**
63       * {@inheritDoc}
64       */
65      @Override
66      protected Runnable getBroadcastHandler() {
67          return new Runnable() {
68              public void run() {
69                  try {
70                      incomingBroadcast();
71                  } catch (Throwable t) {
72                      logger.trace("incomingBroadcast Exception. Broadcaster will be broken unless reconfigured", t);
73                      destroy();
74                      return;
75                  }
76              }
77          };
78      }
79  
80      protected void reconfigure() {
81          if (!started.get()) {
82              return;
83          }
84  
85          logger.debug("Reconfiguring Broadcaster {}", getID());
86          if (notifierFuture != null) {
87              notifierFuture.cancel(true);
88          }
89  
90          if (asyncWriteFuture != null) {
91              asyncWriteFuture.cancel(true);
92          }
93  
94          notifierFuture = bc.getExecutorService().submit(getBroadcastHandler());
95          asyncWriteFuture = bc.getAsyncWriteService().submit(getAsyncWriteHandler());
96      }
97  
98      /**
99       * {@inheritDoc}
100      */
101     @Override
102     protected void broadcast(final AtmosphereResource<?, ?> r, final AtmosphereResourceEvent e) {
103         if (r.getRequest() instanceof HttpServletRequest) {
104             if (((HttpServletRequest) r.getRequest()).getAttribute(FrameworkConfig.CONTAINER_RESPONSE) != null) {
105                 try {
106                     if (jerseyBroadcast == null) {
107                         Class jerseyBroadcasterUtil = Class.forName("org.atmosphere.jersey.util.JerseyBroadcasterUtil");
108                         jerseyBroadcast = jerseyBroadcasterUtil.getMethod("broadcast", new Class[]{AtmosphereResource.class, AtmosphereResourceEvent.class});
109                     }
110                     jerseyBroadcast.invoke(null, new Object[]{r, e});
111                 } catch (Throwable t) {
112                     super.broadcast(r, e);
113                 }
114             } else {
115                 super.broadcast(r, e);
116             }
117         }
118     }
119 
120     protected void broadcastReceivedMessage(Object message) {
121         try {
122             Object newMsg = filter(message);
123             push(new Entry(newMsg, null, new BroadcasterFuture<Object>(newMsg), message));
124         } catch (Throwable t) {
125             logger.error("failed to push message: " + message, t);
126         }
127     }
128 
129     /**
130      * {@inheritDoc}
131      */
132     @Override
133     public <T> Future<T> broadcast(T msg) {
134         if (destroyed.get()) {
135             logger.warn("This Broadcaster has been destroyed and cannot be used {}", getID());
136             return null;
137         }
138 
139         start();
140 
141         Object newMsg = filter(msg);
142         if (newMsg == null) return null;
143         BroadcasterFuture<Object> f = new BroadcasterFuture<Object>(newMsg);
144         try {
145             outgoingBroadcast(msg);
146             push(new Entry(newMsg, null, f, false));
147         } finally {
148             f.done();
149         }
150         return f;
151     }

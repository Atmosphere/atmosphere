16  package org.atmosphere.util;
17  
18  import org.atmosphere.cpr.AtmosphereConfig;
19  import org.atmosphere.cpr.AtmosphereResource;
20  import org.atmosphere.cpr.AtmosphereResourceEvent;
21  import org.atmosphere.cpr.Broadcaster;
22  import org.atmosphere.cpr.BroadcasterFuture;
23  import org.atmosphere.cpr.DefaultBroadcaster;
24  import org.atmosphere.cpr.Deliver;
25  import org.atmosphere.cpr.FrameworkConfig;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import java.lang.reflect.Method;
30  import java.net.URI;
31  import java.util.Set;
32  import java.util.concurrent.Future;
33  
34  
35  /**
36   * Abstract {@link org.atmosphere.cpr.Broadcaster} that delegates the internal processing to a proxy.
37   *
38   * @author Jeanfrancois Arcand
39   */
40  public abstract class AbstractBroadcasterProxy extends DefaultBroadcaster {
41  
42      private static final Logger logger = LoggerFactory.getLogger(AbstractBroadcasterProxy.class);
43      private Method jerseyBroadcast;
44  
45      public AbstractBroadcasterProxy() {
46          try {
47              Class jerseyBroadcasterUtil = Class.forName("org.atmosphere.jersey.util.JerseyBroadcasterUtil");
48              jerseyBroadcast = jerseyBroadcasterUtil.getMethod("broadcast", new Class[]{AtmosphereResource.class, AtmosphereResourceEvent.class, Broadcaster.class});
49          } catch (Exception e) {
50              logger.trace("", e);
51          }
52      }
53  
54      public Broadcaster initialize(String id, URI uri, AtmosphereConfig config) {
55          return super.initialize(id, uri, config);
56      }
57  
58      /**
59       * Implement this method to broadcast message received from an external source like JGroups, Redis, etc.
60       */
61      abstract public void incomingBroadcast();
62  
63      /**
64       * Implement this method to broadcast message to external source like JGroups, Redis, etc.
65       *
66       * @param message outgoing message
67       */
68      abstract public void outgoingBroadcast(Object message);
69  
70      @Override
71      protected Runnable getBroadcastHandler() {
72          return new Runnable() {
73              public void run() {
74                  try {
75                      incomingBroadcast();
76                  } catch (Throwable t) {
77                      logger.warn("incomingBroadcast Exception. Broadcaster will be broken unless reconfigured", t);
78                      destroy();
79                      return;
80                  }
81              }
82          };
83      }
84  
85      protected void reconfigure() {
86          if (!started.get()) {
87              return;
88          }
89  
90          logger.debug("Reconfiguring Broadcaster {}", getID());
91          spawnReactor();
92      }
93  
94      @Override
95      protected void invokeOnStateChange(final AtmosphereResource r, final AtmosphereResourceEvent e) {
96          if (r.getRequest().getAttribute(FrameworkConfig.CONTAINER_RESPONSE) != null) {
97              try {
98                  jerseyBroadcast.invoke(null, new Object[]{r, e, r.getBroadcaster()});
99              } catch (Throwable t) {
100                 super.invokeOnStateChange(r, e);
101             }
102         } else {
103             super.invokeOnStateChange(r, e);
104         }
105     }
106 
107     protected void broadcastReceivedMessage(Object message) {
108         try {
109             Object newMsg = filter(message);
110             // if newSgw == null, that means the message has been filtered.
111             if (newMsg != null) {
112                 push(new Deliver(newMsg, new BroadcasterFuture<Object>(newMsg), message));
113             }
114         } catch (Throwable t) {
115             logger.error("failed to push message: " + message, t);
116         }
117     }
118 
119     @Override
120     public Future<Object> broadcast(Object msg) {
121         return b(msg);
122     }
123 
124     protected Future<Object> b(Object msg) {
125         if (destroyed.get()) {
126             logger.warn("This Broadcaster has been destroyed and cannot be used {}", getID());
127             return null;
128         }
129 
130         start();
131 
132         BroadcasterFuture<Object> f = new BroadcasterFuture<Object>(msg);
133         try {
134             outgoingBroadcast(msg);
135         } finally {
136             futureDone(f);
137         }
138         return f;
139     }
140 
141     @Override
142     public Future<Object> broadcast(Object msg, AtmosphereResource r) {
143         logger.warn("This feature is not supported with {}", r.getBroadcaster().getClass().getName());
144         return b(msg);
145     }
146 
147     @Override
148     public Future<Object> broadcast(Object msg, Set<AtmosphereResource> subset) {
149         logger.warn("This feature is not supported with {}", subset.iterator().next().getBroadcaster().getClass().getName());
150         return b(msg);
151     }
152 }

101     @Override
102     public void handle(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
103         // Is this a resumed connect?
104         LongPollScheduler scheduler = (LongPollScheduler) request.getAttribute(LongPollScheduler.ATTRIBUTE);
105         if (scheduler == null) {
106             // No - process messages
107 
108             // Remember if we start a batch
109             boolean batch = false;
110 
111             // Don't know the session until first message or handshake response.
112             ServerSessionImpl session = null;
113             boolean connect = false;
114 
115             try {
116                 ServerMessage.Mutable[] messages = parseMessages(request);
117                 if (messages == null)
118                     return;
119 
120                 PrintWriter writer = null;
121                 for (ServerMessage.Mutable message : messages) {
122                     // Is this a connect?
123                     connect = Channel.META_CONNECT.equals(message.getChannel());
124 
125                     // Get the session from the message
126                     String client_id = message.getClientId();
127                     if (session == null || client_id != null && !client_id.equals(session.getId())) {
128                         session = (ServerSessionImpl) getBayeux().getSession(client_id);
129                         if (_autoBatch && !batch && session != null && !connect && !message.isMeta()) {
130                             // start a batch to group all resulting messages into a single response.
131                             batch = true;
132                             session.startBatch();
133                         }
134                     } else if (!session.isHandshook()) {
135                         batch = false;
136                         session = null;
137                     }
138 
139                     if (connect && session != null) {
140                         // cancel previous scheduler to cancel any prior waiting long poll
141                         // this should also dec the browser ID
142                         session.setScheduler(null);
143                     }
144 
145                     boolean wasConnected = session != null && session.isConnected();
146 
147                     // Forward handling of the message.
148                     // The actual reply is return from the call, but other messages may
149                     // also be queued on the session.
150                     ServerMessage.Mutable reply = bayeuxServerHandle(session, message);
151 
152                     // Do we have a reply ?
153                     if (reply != null) {
154                         if (session == null) {
155                             // This must be a handshake, extract a session from the reply
156                             session = (ServerSessionImpl) getBayeux().getSession(reply.getClientId());
157 
158                             // Get the user agent while we are at it, and add the browser ID cookie
159                             if (session != null) {
160                                 String userAgent = request.getHeader("User-Agent");
161                                 session.setUserAgent(userAgent);
162 
163                                 String browserId = findBrowserId(request);
164                                 if (browserId == null)
165                                     setBrowserId(request, response);
166                             }
167                         } else {
168                             // Special handling for connect
169                             if (connect) {
170                                 try {
171                                     writer = sendQueue(request, response, session, writer);
172 
173                                     // If the writer is non null, we have already started sending a response, so we should not suspend
174                                     if (writer == null && reply.isSuccessful() && session.isQueueEmpty()) {
175                                         // Detect if we have multiple sessions from the same browser
176                                         // Note that CORS requests do not send cookies, so we need to handle them specially
177                                         // CORS requests always have the Origin header
178 
179                                         String browserId = findBrowserId(request);
180                                         boolean allowSuspendConnect;
181                                         if (browserId != null)
182                                             allowSuspendConnect = incBrowserId(browserId);
183                                         else
184                                             allowSuspendConnect = _allowMultiSessionsNoBrowser;
185 
186                                         if (allowSuspendConnect) {
187                                             long timeout = session.calculateTimeout(getTimeout());
188 
189                                             // Support old clients that do not send advice:{timeout:0} on the first connect
190                                             if (timeout > 0 && wasConnected && session.isConnected()) {
191                                                 // Suspend and wait for messages
192                                                 Continuation continuation = ContinuationSupport.getContinuation(request);
193                                                 continuation.setTimeout(timeout);
194                                                 continuation.suspend(response);
195                                                 scheduler = new LongPollScheduler(session, continuation, reply, browserId);
196                                                 session.setScheduler(scheduler);
197                                                 request.setAttribute(LongPollScheduler.ATTRIBUTE, scheduler);
198                                                 reply = null;
199                                                 metaConnectSuspended(request, session, timeout);
200                                             } else {
201                                                 decBrowserId(browserId);
202                                             }
203                                         } else {
204                                             // There are multiple sessions from the same browser
205                                             Map<String, Object> advice = reply.getAdvice(true);
206 
207                                             if (browserId != null)
208                                                 advice.put("multiple-clients", true);
209 
210                                             if (_multiSessionInterval > 0) {
211                                                 advice.put(Message.RECONNECT_FIELD, Message.RECONNECT_RETRY_VALUE);
212                                                 advice.put(Message.INTERVAL_FIELD, _multiSessionInterval);
213                                             } else {
214                                                 advice.put(Message.RECONNECT_FIELD, Message.RECONNECT_NONE_VALUE);
215                                                 reply.setSuccessful(false);
216                                             }
217                                             session.reAdvise();
218                                         }
219                                     }
220                                 } finally {
221                                     if (reply != null && session.isConnected())
222                                         session.startIntervalTimeout(getInterval());
223                                 }
224                             } else {
225                                 if (!isMetaConnectDeliveryOnly() && !session.isMetaConnectDeliveryOnly()) {
226                                     writer = sendQueue(request, response, session, writer);
227                                 }
228                             }
229                         }
230 
231                         // If the reply has not been otherwise handled, send it
232                         if (reply != null) {
233                             if (connect && session != null && !session.isConnected())
234                                 reply.getAdvice(true).put(Message.RECONNECT_FIELD, Message.RECONNECT_NONE_VALUE);
235 
236                             reply = getBayeux().extendReply(session, session, reply);
237 
238                             if (reply != null) {
239                                 getBayeux().freeze(reply);
240                                 writer = send(request, response, writer, reply);
241                             }
242                         }
243                     }
244 
245                     // Disassociate the reply
246                     message.setAssociated(null);
247                 }
248                 if (writer != null)
249                     complete(writer);
250             } catch (ParseException x) {
251                 handleJSONParseException(request, response, x.getMessage(), x.getCause());
252             } finally {
253                 // If we started a batch, end it now
254                 if (batch) {
255                     boolean ended = session.endBatch();
256 
257                     // Flush session if not done by the batch, since some browser order <script> requests
258                     if (!ended && isAlwaysFlushingAfterHandle())
259                         session.flush();
260                 } else if (session != null && !connect && isAlwaysFlushingAfterHandle()) {
261                     session.flush();
262                 }
263             }
264         } else {
265             // Get the resumed session
266             ServerSessionImpl session = scheduler.getSession();
267             metaConnectResumed(request, session);
268 
269             PrintWriter writer;
270             try {
271                 // Send the message queue
272                 writer = sendQueue(request, response, session, null);
273             } finally {
274                 // We need to start the interval timeout before the connect reply
275                 // otherwise we open up a race condition where the client receives
276                 // the connect reply and sends a new connect request before we start
277                 // the interval timeout, which will be wrong.
278                 // We need to put this into a finally block in case sending the queue
279                 // throws an exception (for example because the client is gone), so that
280                 // we start the interval timeout that is important to sweep the session
281                 if (session.isConnected())
282                     session.startIntervalTimeout(getInterval());
283             }
284 
285             // Send the connect reply
286             ServerMessage.Mutable reply = scheduler.getReply();
287 
288             if (!session.isConnected())
289                 reply.getAdvice(true).put(Message.RECONNECT_FIELD, Message.RECONNECT_NONE_VALUE);
290 
291             reply = getBayeux().extendReply(session, session, reply);
292 
293             if (reply != null) {
294                 getBayeux().freeze(reply);
295                 writer = send(request, response, writer, reply);
296             }
297 
298             complete(writer);
299         }
300     }

302     private PrintWriter sendQueue(HttpServletRequest request, HttpServletResponse response, ServerSessionImpl session, PrintWriter writer)
303             throws IOException {
304         final List<ServerMessage> queue = session.takeQueue();
305         for (ServerMessage m : queue)
306             writer = send(request, response, writer, m);
307         return writer;
308     }

315     @Override
316     protected PrintWriter send(HttpServletRequest request, HttpServletResponse response, PrintWriter writer, ServerMessage message) throws IOException {
317         StringBuilder builder = new StringBuilder(message.size() * 32);
318         if (writer == null) {
319             response.setContentType(_mimeType);
320             writer = response.getWriter();
321         }
322         builder.append("[").append(message.getJSON()).append("]");
323         writer.append(builder.toString());
324         return writer;
325     }

331     private class LongPollScheduler implements AbstractServerTransport.OneTimeScheduler, ContinuationListener {
332         private static final String ATTRIBUTE = "org.cometd.scheduler";
333 
334         private final ServerSessionImpl _session;
335         private final Continuation _continuation;
336         private final ServerMessage.Mutable _reply;
337         private String _browserId;
338 
339         public LongPollScheduler(ServerSessionImpl session, Continuation continuation, ServerMessage.Mutable reply, String browserId) {
340             _session = session;
341             _continuation = continuation;
342             _continuation.addContinuationListener(this);
343             _reply = reply;
344             _browserId = browserId;
345         }
346 
347         public void cancel() {
348             if (_continuation != null && _continuation.isSuspended() && !_continuation.isExpired()) {
349                 try {
350                     decBrowserId();
351                     ((HttpServletResponse) _continuation.getServletResponse()).sendError(HttpServletResponse.SC_REQUEST_TIMEOUT);
352                 } catch (IOException x) {
353                     logger.trace("", x);
354                 }
355 
356                 try {
357                     _continuation.complete();
358                 } catch (Exception x) {
359                     logger.trace("", x);
360                 }
361             }
362         }
363 
364         public void schedule() {
365             decBrowserId();
366             _continuation.resume();
367         }
368 
369         public ServerSessionImpl getSession() {
370             return _session;
371         }
372 
373         public ServerMessage.Mutable getReply() {
374             Map<String, Object> advice = _session.takeAdvice();
375             if (advice != null)
376                 _reply.put(Message.ADVICE_FIELD, advice);
377             return _reply;
378         }
379 
380         public void onComplete(Continuation continuation) {
381             decBrowserId();
382         }
383 
384         public void onTimeout(Continuation continuation) {
385             _session.setScheduler(null);
386         }
387 
388         private void decBrowserId() {
389             WebSocketTransport.this.decBrowserId(_browserId);
390             _browserId = null;
391         }
392     }

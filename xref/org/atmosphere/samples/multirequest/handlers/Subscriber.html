16  package org.atmosphere.samples.multirequest.handlers;
17  
18  import org.apache.log4j.Logger;
19  import org.atmosphere.annotation.Broadcast;
20  import org.atmosphere.cpr.Broadcaster;
21  import org.atmosphere.jersey.Broadcastable;
22  import org.atmosphere.jersey.SuspendResponse;
23  
24  import javax.ws.rs.FormParam;
25  import javax.ws.rs.GET;
26  import javax.ws.rs.POST;
27  import javax.ws.rs.Path;
28  import javax.ws.rs.PathParam;
29  import javax.ws.rs.Produces;
30  
31  @Path("/subscribe/{topic}")
32  @Produces("text/html;charset=ISO-8859-1")
33  public class Subscriber {
34  
35  	private static final Logger LOG = Logger.getLogger(Subscriber.class);
36  
37  	@PathParam("topic")
38  	private Broadcaster topic;
39  
40  	@GET
41  	public SuspendResponse<String> subscribe() {
42  		LOG.debug("OnSubscribe to topic");
43  		SuspendResponse<String> sr = new SuspendResponse.SuspendResponseBuilder<String>().broadcaster(topic).outputComments(true)
44  				.addListener(new EventsLogger()).build();
45  		return sr;
46  	}
47  
48  	@POST
49  	@Broadcast
50  	public Broadcastable publish(@FormParam("message") String message) {
51  		LOG.debug("Receive message <" + message + ">, dispatch to other connected");
52  		return new Broadcastable(message, "", topic);
53  	}
54  }

16  package org.atmosphere.samples.twitter;
17  
18  import com.ning.http.client.AsyncCompletionHandler;
19  import com.ning.http.client.AsyncHttpClient;
20  import com.ning.http.client.Response;
21  import com.sun.jersey.spi.resource.Singleton;
22  import org.atmosphere.cpr.AtmosphereResourceEvent;
23  import org.atmosphere.cpr.Broadcaster;
24  import org.atmosphere.jersey.SuspendResponse;
25  import org.codehaus.jettison.json.JSONObject;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import javax.ws.rs.GET;
30  import javax.ws.rs.Path;
31  import javax.ws.rs.PathParam;
32  import javax.ws.rs.Produces;
33  import javax.ws.rs.WebApplicationException;
34  import java.util.concurrent.Callable;
35  import java.util.concurrent.ConcurrentHashMap;
36  import java.util.concurrent.CountDownLatch;
37  import java.util.concurrent.Future;
38  import java.util.concurrent.TimeUnit;
39  import java.util.concurrent.atomic.AtomicReference;
40  
41  @Path("/search/{tagid}")
42  @Produces("text/html;charset=ISO-8859-1")
43  @Singleton
44  public class TwitterFeed {
45  
46      private static final Logger logger = LoggerFactory.getLogger(TwitterFeed.class);
47  
48      private final AsyncHttpClient asyncClient = new AsyncHttpClient();
49      private final ConcurrentHashMap<String, Future<?>> futures = new ConcurrentHashMap<String, Future<?>>();
50      private final CountDownLatch suspendLatch = new CountDownLatch(1);
51  
52      @GET
53      public SuspendResponse<String> search(final @PathParam("tagid") Broadcaster feed,
54                                            final @PathParam("tagid") String tagid) {
55  
56          if (tagid.isEmpty()) {
57              throw new WebApplicationException();
58          }
59  
60          if (feed.getAtmosphereResources().size() == 0) {
61  
62              final Future<?> future = feed.scheduleFixedBroadcast(new Callable<String>() {
63  
64                  private final AtomicReference<String> refreshUrl = new AtomicReference<String>("");
65  
66                  public String call() throws Exception {
67                      String query = null;
68                      if (!refreshUrl.get().isEmpty()) {
69                          query = refreshUrl.get();
70                      } else {
71                          query = "?q=" + tagid;
72                      }
73  
74                      // Wait for the connection to be suspended.
75                      suspendLatch.await();
76                      asyncClient.prepareGet("http://search.twitter.com/search.json" + query).execute(
77                              new AsyncCompletionHandler<Object>() {
78  
79                                  @Override
80                                  public Object onCompleted(Response response) throws Exception {
81                                      String s = response.getResponseBody();
82  
83                                      if (response.getStatusCode() != 200) {
84                                          feed.resumeAll();
85                                          feed.destroy();
86                                          logger.info("Twitter Search API unavaileble\n{}", s);
87                                          return null;
88                                      }
89  
90                                      JSONObject json = new JSONObject(s);
91                                      refreshUrl.set(json.getString("refresh_url"));
92                                      if (json.getJSONArray("results").length() > 1) {
93                                          feed.broadcast(s).get();
94                                      }
95                                      return null;
96                                  }
97  
98                              });
99                      return null;
100                 }
101 
102             }, 5, TimeUnit.SECONDS);
103 
104             futures.put(tagid, future);
105         }
106 
107         return new SuspendResponse.SuspendResponseBuilder<String>().broadcaster(feed).outputComments(true)
108                 .addListener(new EventsLogger() {
109 
110                     @Override
111                     public void onSuspend(
112                             final AtmosphereResourceEvent event) {
113                         super.onSuspend(event);
114 
115                         // OK, we can start polling Twitter!
116                         suspendLatch.countDown();
117                     }
118                 }).build();
119     }
120 
121     @GET
122     @Path("stop")
123     public String stopSearch(final @PathParam("tagid") Broadcaster feed,
124                              final @PathParam("tagid") String tagid) {
125         feed.resumeAll();
126         if (futures.get(tagid) != null) {
127             futures.get(tagid).cancel(true);
128         }
129         logger.info("Stopping real time update for {}", tagid);
130         return "DONE";
131     }
132 }

38  package org.atmosphere.samples.twitter;
39  
40  import org.atmosphere.samples.twitter.UsersState.UserStateData;
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  
44  import javax.ws.rs.Consumes;
45  import javax.ws.rs.DELETE;
46  import javax.ws.rs.FormParam;
47  import javax.ws.rs.GET;
48  import javax.ws.rs.POST;
49  import javax.ws.rs.Path;
50  import javax.ws.rs.WebApplicationException;
51  import javax.ws.rs.core.MediaType;
52  
53  public class UserResource {
54  
55      private static final Logger logger = LoggerFactory.getLogger(UserResource.class);
56  
57      private final UsersState us;
58  
59      private String user;
60  
61      private UserStateData usd;
62  
63      public UserResource(UsersState us, String name, UserStateData usd) {
64          this.us = us;
65          this.user = name;
66          this.usd = usd;
67      }
68  
69      @GET
70      public String get() {
71          return "";
72      }
73  
74      @DELETE
75      public void delete() {
76          us.remove(user);
77      }
78  
79      @Path("messages")
80      @POST
81      public void post(String message) {
82          logger.info("MESSAGE: {}", message);
83      }
84  
85      @Path("follows")
86      @POST
87      @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
88      public void follow(@FormParam("follower") String follower) {
89          UserStateData followerState = us.get(follower);
90          // User does not exist
91          if (followerState == null) {
92              throw new WebApplicationException(404);
93          }
94  
95          followerState.bc.addAtmosphereResource(us.get(user).bc.getUserAtmosphereEvent().getResource());
96          followerState.bc.broadcast(user + " is now follow you ",
97                  followerState.bc.getUserAtmosphereEvent().getResource());
98  
99          logger.info("{} is following {}", user, follower);
100     }
101 }

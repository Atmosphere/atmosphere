53  package org.atmosphere.samples.tictactoe;
54  
55  public class TTTGame {
56  
57      final public static int X = 10;
58      final public static int O = 1;
59      int[] board = {0, 0, 0, 0, 0, 0, 0, 0, 0};
60      int turnNum = 0;
61      int[][] wins = {{0, 1, 2}, {3, 4, 5,}, {6, 7, 8}, {0, 3, 6},
62              {1, 4, 7}, {2, 5, 8}, {0, 4, 8}, {2, 4, 6}};
63      int winner = -1;
64  
65      //return false if cell is an invalid move
66      public boolean turn(int cell) {
67          if (cell < 0 || cell > 8) {
68              return false; // invalid move
69          }
70          if (winner != -1) {
71              return false;
72          }
73          if (board[cell] != 0) {
74              return false; // invalid move
75          }
76          turnNum++;
77          if (turnNum % 2 == 1) { //then X
78              board[cell] = X;
79          } else { // else O
80              board[cell] = O;
81          }
82          return true;
83      }
84  
85      // return 
86      private int whoseTurn() {
87          if (turnNum == 0 || turnNum % 2 == 0) {
88              return X;
89          } else {
90              return O;
91          }
92      }
93  
94      private boolean done() {
95          return (turnNum > 8);
96      }
97  
98      // return -1 for no win, 0 for tie, 1 for x win, 2 for o win
99      public int win() {
100         if (winner != -1) {
101             return winner;
102         }
103         for (int i = 0; i < 8; i++) {
104             int winSum = board[wins[i][0]] + board[wins[i][1]] + board[wins[i][2]];
105             if (winSum == 3) {
106                 winner = 2;
107             } else if (winSum == 30) {
108                 winner = 1;
109             }
110         }
111         if (winner == -1 && turnNum > 8) {
112             winner = 0;
113         }
114         return winner;
115     }
116 
117     private int[] getBoard() {
118         return board;
119     }
120 
121     /**
122      * Create a JSON representation of the game state.  It will look something like this:
123      * <code>
124      * { "win": "-1", "board": ["0","0","0","0","0","0","0","0","0"] }
125      * </code>
126      *
127      * @return json
128      */
129     public String getJSON() {
130         String response = "{" +
131                 "\"win\": \"" + win() + "\", \"board\": [";
132         for (int i = 0; i < 9; i++) {
133             response = response + "\"" + board[i] + "\"";
134             if (i < 8) {  // no trailing comma
135                 response = response + ",";
136             }
137         }
138         response = response + "]," + "\"turn\": " + whoseTurn() + " }";
139         return response;
140     }

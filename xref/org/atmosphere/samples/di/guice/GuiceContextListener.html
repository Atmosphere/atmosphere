16  package org.atmosphere.samples.di.guice;
17  
18  import com.google.inject.Guice;
19  import com.google.inject.Injector;
20  import com.google.inject.servlet.GuiceServletContextListener;
21  import com.google.inject.servlet.ServletModule;
22  import org.atmosphere.guice.GuiceContainer;
23  import org.atmosphere.guice.GuiceManagedAtmosphereServlet;
24  
25  import java.util.HashMap;
26  
27  /**
28   * @author Mathieu Carbou
29   * @since 0.7
30   */
31  public final class GuiceContextListener extends GuiceServletContextListener {
32      @Override
33      protected Injector getInjector() {
34          return Guice.createInjector(new ServletModule() {
35              @Override
36              protected void configureServlets() {
37                  bind(MessageResource.class);
38                  serve("/async/*").with(GuiceManagedAtmosphereServlet.class, new HashMap<String, String>() {
39                      {
40                          put("org.atmosphere.useWebSocket", "true");
41                          put("org.atmosphere.useNative", "true");
42                      }
43                  });
44                  serve("/rest/*").with(GuiceContainer.class);
45              }
46          });
47      }
48  }

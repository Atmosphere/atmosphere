16  package org.atmosphere.samples.di.guice;
17  
18  import com.google.inject.Inject;
19  import com.google.inject.Singleton;
20  import org.atmosphere.annotation.Broadcast;
21  import org.atmosphere.annotation.Suspend;
22  import org.atmosphere.cpr.Broadcaster;
23  import org.atmosphere.cpr.BroadcasterFactory;
24  import org.atmosphere.cpr.DefaultBroadcaster;
25  import org.atmosphere.jersey.Broadcastable;
26  import org.codehaus.jettison.json.JSONException;
27  import org.codehaus.jettison.json.JSONObject;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import javax.ws.rs.FormParam;
32  import javax.ws.rs.GET;
33  import javax.ws.rs.POST;
34  import javax.ws.rs.Path;
35  import javax.ws.rs.PathParam;
36  import javax.ws.rs.Produces;
37  
38  /**
39   * @author Mathieu Carbou
40   * @since 0.7
41   */
42  @Path("/topic")
43  @Singleton
44  @Produces("application/json")
45  public class MessageResource {
46  
47      private static final Logger logger = LoggerFactory.getLogger(MessageResource.class);
48  
49      @Inject
50      Service service;
51  
52      @GET
53      @Path("{name}")
54      @Suspend(outputComments = true, resumeOnBroadcast = false, listeners = EventsLogger.class)
55      public Broadcastable listen(@PathParam("name") String topic) throws JSONException {
56          Broadcaster broadcaster = BroadcasterFactory.getDefault().lookup(DefaultBroadcaster.class, topic, true);
57          logger.info("thread: {} LISTENING to '{}'", Thread.currentThread().getName(), broadcaster.getID());
58          if (service == null) {
59              throw new AssertionError();
60          }
61          return new Broadcastable(new JSONObject().put("from", "system").put("msg", "Connected !"), broadcaster);
62      }
63  
64      @POST
65      @Path("{name}")
66      @Broadcast
67      public Broadcastable publish(@PathParam("name") String topic, @FormParam("from") String from,
68                                   @FormParam("msg") String message) throws JSONException {
69          Broadcaster broadcaster = BroadcasterFactory.getDefault().lookup(DefaultBroadcaster.class, topic, true);
70          logger.info("thread: {} PUBLISH to '{}' from {}: {}",
71                  new Object[]{Thread.currentThread().getName(), broadcaster.getID(), from, message});
72          if (service == null) {
73              throw new AssertionError();
74          }
75          return new Broadcastable(new JSONObject().put("from", from).put("msg", message), "", broadcaster);
76      }
77  
78  }

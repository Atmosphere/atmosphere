38  package org.atmosphere.samples.chat;
39  
40  import org.atmosphere.commons.jersey.JsonpFilter;
41  import org.atmosphere.commons.util.EventsLogger;
42  import org.atmosphere.cpr.BroadcastFilter;
43  import org.atmosphere.cpr.Meteor;
44  import org.atmosphere.util.XSSHtmlFilter;
45  
46  import javax.servlet.http.HttpServlet;
47  import javax.servlet.http.HttpServletRequest;
48  import javax.servlet.http.HttpServletResponse;
49  import java.io.IOException;
50  import java.util.LinkedList;
51  import java.util.List;
52  
53  /**
54   * Simple Servlet that implement the logic to build a Chat application using
55   * a {@link Meteor} to suspend and broadcast chat message.
56   *
57   * @author Jeanfrancois Arcand
58   * @autor TAKAI Naoto (Orginial author for the Comet based Chat).
59   */
60  public class MeteorChat extends HttpServlet {
61  
62      private final static long serialVersionUID = -2919167206889576860L;
63  
64      /**
65       * List of {@link BroadcastFilter}
66       */
67      private final List<BroadcastFilter> list;
68  
69      public MeteorChat() {
70          list = new LinkedList<BroadcastFilter>();
71          list.add(new XSSHtmlFilter());
72          list.add(new JsonpFilter());
73      }
74  
75      /**
76       * Create a {@link Meteor} and use it to suspend the response.
77       *
78       * @param req An {@link HttpServletRequest}
79       * @param res An {@link HttpServletResponse}
80       */
81      @Override
82      public void doGet(HttpServletRequest req, HttpServletResponse res) throws IOException {
83          Meteor m = Meteor.build(req, list, null);
84  
85          // Log all events on the concole.
86          m.addListener(new EventsLogger());
87  
88          req.getSession().setAttribute("meteor", m);
89  
90          res.setContentType("text/html;charset=ISO-8859-1");
91  
92          m.suspend(-1);
93          m.broadcast(req.getServerName() + "__has suspended a connection from " + req.getRemoteAddr());
94      }
95  
96      /**
97       * Re-use the {@link Meteor} created onthe first GET for broadcasting message.
98       *
99       * @param req An {@link HttpServletRequest}
100      * @param res An {@link HttpServletResponse}
101      */
102     @Override
103     public void doPost(HttpServletRequest req, HttpServletResponse res) throws IOException {
104         Meteor m = (Meteor) req.getSession().getAttribute("meteor");
105         res.setCharacterEncoding("UTF-8");
106         String action = req.getParameterValues("action")[0];
107         String name = req.getParameterValues("name")[0];
108 
109         if ("login".equals(action)) {
110             req.getSession().setAttribute("name", name);
111             m.broadcast("System Message from " + req.getServerName() + "__" + name + " has joined.");
112             res.getWriter().write("success");
113             res.getWriter().flush();
114         } else if ("post".equals(action)) {
115             String message = req.getParameterValues("message")[0];
116             m.broadcast(name + "__" + message);
117             res.getWriter().write("success");
118             res.getWriter().flush();
119         } else {
120             res.setStatus(422);
121 
122             res.getWriter().write("success");
123             res.getWriter().flush();
124         }
125     }
126 
127 }

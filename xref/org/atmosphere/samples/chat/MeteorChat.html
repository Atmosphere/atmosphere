16  package org.atmosphere.samples.chat;
17  
18  import org.atmosphere.config.service.MeteorService;
19  import org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter;
20  import org.atmosphere.cpr.BroadcasterFactory;
21  import org.atmosphere.cpr.DefaultBroadcaster;
22  import org.atmosphere.cpr.Meteor;
23  
24  import javax.servlet.http.HttpServlet;
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpServletResponse;
27  import java.io.IOException;
28  import java.util.Date;
29  
30  import static org.atmosphere.cpr.AtmosphereResource.TRANSPORT.LONG_POLLING;
31  
32  /**
33   * Simple Servlet that implement the logic to build a Chat application using
34   * a {@link Meteor} to suspend and broadcast chat message.  The Meteor is annotated using the {@link MeteorService}
35   * but can also be defined in web.xml using
36   * <blockquote>
37   * <init-param>
38   *    <param-name>org.atmosphere.servlet</param-name>
39   *    <param-value>org.atmosphere.samples.chat.MeteorChat</param-value>
40   * </init-param>
41   * <p/>
42   * </blockquote>
43   *
44   * @author Jeanfrancois Arcand
45   */
46  @MeteorService("/*")
47  public class MeteorChat extends HttpServlet {
48  
49      /**
50       * Create a {@link Meteor} and use it to suspend the response.
51       *
52       * @param req An {@link HttpServletRequest}
53       * @param res An {@link HttpServletResponse}
54       */
55      @Override
56      public void doGet(HttpServletRequest req, HttpServletResponse res) throws IOException {
57          // Set the logger level to TRACE to see what's happening.
58          Meteor m = Meteor.build(req).addListener(new AtmosphereResourceEventListenerAdapter());
59  
60          m.resumeOnBroadcast(m.transport() == LONG_POLLING ? true : false).suspend(-1);
61      }
62  
63      /**
64       * Re-use the {@link Meteor} created on the first GET for broadcasting message.
65       *
66       * @param req An {@link HttpServletRequest}
67       * @param res An {@link HttpServletResponse}
68       */
69      @Override
70      public void doPost(HttpServletRequest req, HttpServletResponse res) throws IOException {
71          String body = req.getReader().readLine().trim();
72          // Simple JSON -- Use Jackson for more complex structure
73          // Message looks like { "author" : "foo", "message" : "bar" }
74          String author = body.substring(body.indexOf(":") + 2, body.indexOf(",") - 1);
75          String message = body.substring(body.lastIndexOf(":") + 2, body.length() - 2);
76          BroadcasterFactory.getDefault().lookup(DefaultBroadcaster.class, "/*").broadcast(new Data(author, message).toString());
77      }
78  
79      private final static class Data {
80  
81          private final String text;
82          private final String author;
83  
84          public Data(String author, String text) {
85              this.author = author;
86              this.text = text;
87          }
88  
89          public String toString() {
90              return "{ \"text\" : \"" + text + "\", \"author\" : \"" + author + "\" , \"time\" : " + new Date().getTime() + "}";
91          }
92      }
93  }

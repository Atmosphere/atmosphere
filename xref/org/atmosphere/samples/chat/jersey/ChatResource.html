16  package org.atmosphere.samples.chat.jersey;
17  
18  import org.atmosphere.annotation.Broadcast;
19  import org.atmosphere.annotation.Suspend;
20  import org.atmosphere.cpr.AtmosphereResource;
21  import org.atmosphere.cpr.Broadcaster;
22  
23  import javax.ws.rs.Consumes;
24  import javax.ws.rs.GET;
25  import javax.ws.rs.POST;
26  import javax.ws.rs.Path;
27  import javax.ws.rs.Produces;
28  import javax.ws.rs.core.Context;
29  
30  /**
31   * Simple chat resource demonstrating the power of Atmosphere. This resource supports transport like WebSocket, Streaming, JSONP and Long-Polling.
32   *
33   * @author Jeanfrancois Arcand
34   */
35  @Path("/")
36  public class ChatResource {
37  
38      /**
39       * Suspend the response without writing anything back to the client.
40       * @return a white space
41       */
42      @Suspend(contentType = "application/json")
43      @GET
44      public String suspend() {
45          return "";
46      }
47  
48      /**
49       * Broadcast the received message object to all suspended response. Do not write back the message to the calling connection.
50       * @param message a {@link Message}
51       * @return a {@link Response}
52       */
53      @Broadcast(writeEntity = false)
54      @POST
55      @Produces("application/json")
56      public Response broadcast(Message message, @Context AtmosphereResource r, @Context Broadcaster b) {
57          b.addAtmosphereResource(r);
58          return new Response(message.author, message.message);
59      }
60  
61  }

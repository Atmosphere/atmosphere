16  package org.atmosphere.samples.chat.jersey;
17  
18  import org.atmosphere.client.TrackMessageSizeInterceptor;
19  import org.atmosphere.config.service.AtmosphereService;
20  import org.atmosphere.cpr.BroadcasterFactory;
21  import org.atmosphere.interceptor.AtmosphereResourceLifecycleInterceptor;
22  
23  import javax.ws.rs.POST;
24  import javax.ws.rs.Path;
25  
26  /**
27   * Extremely simple chat application supportiong WebSocket, Server Side-Events, Long-Polling and Streaming.
28   *
29   * @author Jeanfrancois Arcand
30   */
31  @Path("/chat")
32  @AtmosphereService(
33          dispatch = false,
34          interceptors = {AtmosphereResourceLifecycleInterceptor.class, TrackMessageSizeInterceptor.class},
35          path = "/chat",
36          servlet = "org.glassfish.jersey.servlet.ServletContainer")
37  public class Jersey2Resource {
38  
39      /**
40       * Echo the chat message. Jackson can clearly be used here, but for simplicity we just echo what we receive.
41       * @param message
42       */
43      @POST
44      public void broadcast(String message) {
45          BroadcasterFactory.getDefault().lookup("/chat").broadcast(message);
46      }
47  
48  }

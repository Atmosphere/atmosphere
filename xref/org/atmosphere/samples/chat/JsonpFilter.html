37  package org.atmosphere.samples.chat;
38  
39  import org.atmosphere.cpr.BroadcastFilter;
40  
41  /**
42   * Simple {@link org.atmosphere.cpr.BroadcastFilter} that produce jsonp String.
43   *
44   * @author Jeanfrancois Arcand
45   */
46  public class JsonpFilter implements BroadcastFilter {
47  
48      private static final String BEGIN_SCRIPT_TAG = "<script type='text/javascript'>\n";
49      private static final String END_SCRIPT_TAG = "</script>\n";
50  
51      public BroadcastAction filter(Object originalMessage, Object o) {
52          if (o instanceof String) {
53              String m = (String) o;
54              String name = m;
55              String message = "";
56  
57              if (m.indexOf("__") > 0) {
58                  name = m.substring(0, m.indexOf("__"));
59                  message = m.substring(m.indexOf("__") + 2);
60              }
61  
62              return new BroadcastAction(BEGIN_SCRIPT_TAG + "window.parent.app.update({ name: \""
63                      + name + "\", message: \"" + message + "\" });\n" + END_SCRIPT_TAG);
64          } else {
65              return new BroadcastAction(o);
66          }
67      }
68  }

27  /**
28   * Simple Asynchronous sample that demonstrate how task can be asynchronously executed. This class support all protocol
29   * and use the {@link org.atmosphere.cpr.HeaderConfig#X_ATMOSPHERE_TRACKING_ID} to create a single {@link org.atmosphere.cpr.Broadcaster}
30   * associated with the remote client.
31   *
32   * @author Jeanfrancois Arcand
33   */
34  @Path("/async")
35  public class AsynchronousExecution {
36      @POST
37      @Asynchronous(waitForResource = false, contentType = "application/json")
38      public Callable<Response> publish(final Message message) {
39          return new Callable<Response>() {
40  
41              public Response call() throws Exception {
42                  return new Response("Asynchronous Execution", message.message);
43              }
44          };
45      }
46  }

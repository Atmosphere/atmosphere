16  package org.atmosphere.samples.pubsub;

18  import org.atmosphere.client.TrackMessageSizeInterceptor;
19  import org.atmosphere.config.service.AtmosphereHandlerService;
20  import org.atmosphere.config.service.Singleton;
21  import org.atmosphere.cpr.AtmosphereResourceEvent;
22  import org.atmosphere.handler.AtmosphereHandlerAdapter;
23  import org.atmosphere.interceptor.AtmosphereResourceLifecycleInterceptor;
24  import org.atmosphere.interceptor.BroadcastOnPostAtmosphereInterceptor;
25  import org.atmosphere.interceptor.SuspendTrackerInterceptor;
26  import org.atmosphere.util.SimpleBroadcaster;
27  
28  import java.io.IOException;
29  
30  /**
31   * Simple PubSub resource that demonstrate many functionality supported by
32   * Atmosphere JQuery Plugin and AtmosphereHandler extension.  You can compare that implementation
33   * with the MeteorPubSub and the JQueryPubsub sample
34   * <p/>
35   * This sample support out of the box WebSocket, Long-Polling, SSE and Streaming. You can also use the
36   * @ManagedService annotation for more out of the box supported feature.
37   *
38   * @author Jeanfrancois Arcand
39   */
40  @Singleton
41  @AtmosphereHandlerService(path = "/{chat}",
42          interceptors = {
43              AtmosphereResourceLifecycleInterceptor.class,
44              TrackMessageSizeInterceptor.class,
45              BroadcastOnPostAtmosphereInterceptor.class,
46              SuspendTrackerInterceptor.class},
47          broadcaster = SimpleBroadcaster.class)
48  public class AtmosphereHandlerPubSub extends AtmosphereHandlerAdapter {
49  
50      @Override
51      public void onStateChange(AtmosphereResourceEvent event) throws IOException {
52          if (event.isSuspended()) {
53              String message = event.getMessage() == null ? null : event.getMessage().toString();
54              if (message != null && message.indexOf("message") != -1) {
55                  event.getResource().write(message.substring("message=".length()));
56              } else {
57                  event.getResource().write("=error=");
58              }
59          }
60      }
61  
62  }

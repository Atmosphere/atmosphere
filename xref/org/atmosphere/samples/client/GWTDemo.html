47      PollAsync polling = GWT.create(Poll.class);

52      @Override
53      public void onModuleLoad() {
54  
55          Button button = new Button("Broadcast");
56          button.addClickHandler(new ClickHandler() {
57              @Override
58              public void onClick(ClickEvent event) {
59                  sendMessage();
60              }
61          });
62  
63          Button post = new Button("Post");
64          post.addClickHandler(new ClickHandler() {
65              @Override
66              public void onClick(ClickEvent event) {
67                  client.post(new Event(count++, "This was send using the post mechanism"));
68              }
69          });
70  
71          Button pollButton = new Button("Poll");
72          pollButton.addClickHandler(new ClickHandler() {
73              @Override
74              public void onClick(ClickEvent event) {
75                  polling.pollDelayed(3000, new AsyncCallback<Event>() {
76  
77                      @Override
78                      public void onFailure(Throwable caught) {
79                          GWT.log("Failed to poll", caught);
80                      }
81  
82                      @Override
83                      public void onSuccess(Event result) {
84                          Info.display("Polling message received: " + result.getCode(), result.getData());
85                      }
86                  });
87              }
88          });
89  
90          Button wnd = new Button("Open Window");
91          wnd.addClickHandler(new ClickHandler() {
92              @Override
93              public void onClick(ClickEvent event) {
94                  Scheduler.get().scheduleDeferred(new ScheduledCommand() {
95                      @Override
96                      public void execute() {
97                          screen = Window.current().open(Document.get().getURL(), "child", new WindowFeatures().setStatus(true).setResizable(true));
98                      }
99                  });
100             }
101         });
102 
103         Button sendWindow = new Button("Send to window");
104         sendWindow.addClickHandler(new ClickHandler() {
105             @Override
106             public void onClick(ClickEvent event) {
107                 if (screen != null) {
108                     WindowSocket.post(screen, "wsock", "Hello Child!");
109                 }
110             }
111         });
112 
113         WindowSocket socket = new WindowSocket();
114         socket.addHandler(new WindowSocket.MessageHandler() {
115             @Override
116             public void onMessage(String message) {
117                 Info.display("Received through window socket", message);
118             }
119         });
120         socket.bind("wsock");
121 
122         RootPanel.get("buttons").add(button);
123         RootPanel.get("buttons").add(post);
124         RootPanel.get("buttons").add(pollButton);
125         RootPanel.get("buttons").add(wnd);
126         RootPanel.get("buttons").add(sendWindow);
127 
128         initialize();
129 
130         Button killbutton = new Button("Stop");
131         killbutton.addClickHandler(new ClickHandler() {
132             @Override
133             public void onClick(ClickEvent event) {
134                 client.stop();
135             }
136         });
137 
138         RootPanel.get("buttons").add(killbutton);

142     public static void displayCookies() {
143         StringBuilder cookies = new StringBuilder();
144         for (String name : Cookies.getCookieNames()) {
145             String value = Cookies.getCookie(name);
146             cookies.append(name).append(" = ").append(value).append("<br/>");
147         }
148         Info.display("Cookies", cookies.toString());        
149     }

151     private class MyCometListener implements AtmosphereListener {
152 
153         @Override
154         public void onConnected(int heartbeat, int connectionID) {
155             GWT.log("comet.connected [" + heartbeat + ", " + connectionID + "]");
156             displayCookies();
157         }
158 
159         @Override
160         public void onBeforeDisconnected() {
161             logger.log(Level.INFO, "comet.beforeDisconnected");
162         }
163 
164         @Override
165         public void onDisconnected() {
166             GWT.log("comet.disconnected");
167         }
168 
169         @Override
170         public void onError(Throwable exception, boolean connected) {
171             int statuscode = -1;
172             if (exception instanceof StatusCodeException) {
173                 statuscode = ((StatusCodeException) exception).getStatusCode();
174             }
175             GWT.log("comet.error [connected=" + connected + "] (" + statuscode + ")", exception);
176         }
177 
178         @Override
179         public void onHeartbeat() {
180             GWT.log("comet.heartbeat [" + client.getConnectionID() + "]");
181         }
182 
183         @Override
184         public void onRefresh() {
185             GWT.log("comet.refresh [" + client.getConnectionID() + "]");
186         }
187 
188         @Override
189         public void onMessage(List<? extends Serializable> messages) {
190             StringBuilder result = new StringBuilder();
191             for (Serializable obj : messages) {
192                 result.append(obj.toString()).append("<br/>");
193             }
194             logger.log(Level.INFO, "comet.message [" + client.getConnectionID() + "] " + result.toString());
195             Info.display("[" + client.getConnectionID() + "] Received " + messages.size() + " messages", result.toString());
196         }
197     }

201     public void initialize() {
202 
203         MyCometListener cometListener = new MyCometListener();
204 
205         AtmosphereGWTSerializer serializer = GWT.create(EventSerializer.class);
206         // set a small length parameter to force refreshes
207         // normally you should remove the length parameter
208         client = new AtmosphereClient(GWT.getModuleBaseURL() + "gwtComet", serializer, cometListener);
209         client.start();
210     }

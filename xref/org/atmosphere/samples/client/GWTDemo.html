55      static final Logger logger = Logger.getLogger(GWTDemo.class.getName());
56      static final String LABEL_ENTER_ROOM = "Type your name to enter the room";
57      static final String LABEL_TYPE_MESSAGE = "Type a message to send to the room";
58      static final String MESSAGE_JOINED_ROOM = "&lt;joined the room&gt;";
59      static final String MESSAGE_LEFT_ROOM = "&lt;left the room&gt;";
60      static final String MESSAGE_ROOM_CONNECTED = "[connected to room]";
61      static final String MESSAGE_ROOM_DISCONNECTED = "[disconnected from room]";
62      static final String MESSAGE_ROOM_ERROR = "Error: ";
63      static final String COLOR_SYSTEM_MESSAGE = "grey";
64      static final String COLOR_MESSAGE_SELF = "green";
65      static final String COLOR_MESSAGE_OTHERS = "red";

67      int count = 0;

71      AtmosphereGWTSerializer serializer = GWT.create(EventSerializer.class);

79      @Override
80      public void onModuleLoad() {
81          
82          final ListBox roomSelect = new ListBox();
83          roomSelect.addItem("Room 1", "room1");
84          roomSelect.addItem("Room 2", "room2");
85          roomSelect.addItem("Room 3", "room3");
86          roomSelect.addItem("Room 4", "room4");
87          roomSelect.setSelectedIndex(0);
88          roomSelect.addChangeHandler(new ChangeHandler() {
89              @Override
90              public void onChange(ChangeEvent event) {
91                  String room = roomSelect.getValue(roomSelect.getSelectedIndex());
92                  changeRoom(room);
93              }
94          });
95          RootPanel.get("room").add(roomSelect);
96          
97          chat = Document.get().getElementById("chat");
98          
99          label = new Label(LABEL_ENTER_ROOM);
100         RootPanel.get("label").add(label);
101         
102         input = new TextBox();
103         input.addKeyDownHandler(new KeyDownHandler() {
104             @Override
105             public void onKeyDown(KeyDownEvent event) {
106                 if (event.getNativeKeyCode() == KeyCodes.KEY_ENTER) {
107                     sendMessage(input.getValue());
108                     input.setText("");
109                 }
110             }
111         });
112         RootPanel.get("input").add(input);
113 
114         Button send = new Button("Send");
115         send.addClickHandler(new ClickHandler() {
116             @Override
117             public void onClick(ClickEvent event) {
118                 sendMessage(input.getValue());
119                 input.setText("");
120             }
121         });
122         RootPanel.get("send").add(send);
123         
124         HTMLPanel logPanel = new HTMLPanel("") {
125             @Override
126             public void add(Widget widget) {
127                 super.add(widget);
128                 widget.getElement().scrollIntoView();
129             }
130         };
131         RootPanel.get("logger").add(logPanel);
132         Logger.getLogger("").addHandler(new HasWidgetsLogHandler(logPanel));
133     
134         changeRoom(room);
135     }

137     void sendMessage(String message) {
138         if (author == null) {
139             author = message;
140             client.broadcast(new Event(author, MESSAGE_JOINED_ROOM));
141             label.setText(LABEL_TYPE_MESSAGE);
142         } else {
143             client.broadcast(new Event(author, message));
144         }
145     }
146     
147     String getUrl() {
148         return GWT.getModuleBaseURL() + "gwtComet/" + room;
149     }
150     
151     void changeRoom(final String newRoom) {
152         if (client != null) {
153             if (author != null) {
154                 client.broadcast(new Event(author, MESSAGE_LEFT_ROOM));
155             }
156             client.stop();
157             client = null;
158         }
159         author = null;
160         Scheduler.get().scheduleDeferred(new Scheduler.ScheduledCommand() {
161             @Override
162             public void execute() {
163                 room = newRoom;
164                 client = new AtmosphereClient(getUrl(), serializer, cometListener);
165                 clearChat();
166                 label.setText(LABEL_ENTER_ROOM);
167                 client.start();
168             }
169         });
170     }
171     
172     void clearChat() {
173         chat.setInnerHTML("");
174     }
175     
176     void addChatLine(String line, String color) {
177         HTML newLine = new HTML(line);
178         newLine.getElement().getStyle().setColor(color);
179         chat.appendChild(newLine.getElement());
180         newLine.getElement().scrollIntoView();
181     }

183     private class MyCometListener implements AtmosphereListener {
184         
185         DateTimeFormat timeFormat = DateTimeFormat.getFormat(DateTimeFormat.PredefinedFormat.TIME_MEDIUM);
186 
187         @Override
188         public void onConnected(int heartbeat, int connectionID) {
189             logger.info("comet.connected [" + heartbeat + ", " + connectionID + "]");
190             addChatLine(MESSAGE_ROOM_CONNECTED, COLOR_SYSTEM_MESSAGE);
191         }
192 
193         @Override
194         public void onBeforeDisconnected() {
195             logger.log(Level.INFO, "comet.beforeDisconnected");
196             if (author != null) {
197                 client.broadcast(new Event(author, MESSAGE_LEFT_ROOM));
198             }
199         }
200 
201         @Override
202         public void onDisconnected() {
203             logger.info("comet.disconnected");
204             addChatLine(MESSAGE_ROOM_DISCONNECTED, COLOR_SYSTEM_MESSAGE);
205         }
206 
207         @Override
208         public void onError(Throwable exception, boolean connected) {
209             int statuscode = -1;
210             if (exception instanceof StatusCodeException) {
211                 statuscode = ((StatusCodeException) exception).getStatusCode();
212             }
213             logger.log(Level.SEVERE, "comet.error [connected=" + connected + "] (" + statuscode + ")", exception);
214             addChatLine(MESSAGE_ROOM_ERROR + exception.getMessage(), COLOR_SYSTEM_MESSAGE);
215             client.stop();
216         }
217 
218         @Override
219         public void onHeartbeat() {
220             logger.info("comet.heartbeat [" + client.getConnectionID() + "]");
221         }
222 
223         @Override
224         public void onRefresh() {
225             logger.info("comet.refresh [" + client.getConnectionID() + "]");
226         }
227 
228         @Override
229         public void onAfterRefresh() {
230             logger.info("comet.afterRefresh [" + client.getConnectionID() + "]");
231         }
232 
233         @Override
234         public void onMessage(List<?> messages) {
235             for (Object obj : messages) {
236                 if (obj instanceof Event) {
237                     Event e = (Event)obj;
238                     String line = timeFormat.format(e.getTime())
239                             + " <b>" + e.getAuthor() + "</b> " + e.getMessage();
240                     if (e.getAuthor().equals(author)) {
241                         addChatLine(line, COLOR_MESSAGE_SELF);
242                     } else {
243                         addChatLine(line, COLOR_MESSAGE_OTHERS);
244                     }
245                 }
246             }
247         }
248     }

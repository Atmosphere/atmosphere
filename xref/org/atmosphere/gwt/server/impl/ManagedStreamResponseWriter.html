17  package org.atmosphere.gwt.server.impl;
18  
19  
20  import java.io.IOException;
21  import java.io.OutputStream;
22  import java.io.Serializable;
23  import java.util.List;
24  import org.atmosphere.gwt.server.SerializationException;
25  
26  /**
27   * @author p.havelaar
28   */
29  abstract public class ManagedStreamResponseWriter extends GwtResponseWriterImpl {
30  
31      private CountOutputStream countOutputStream;
32      private boolean refresh;
33  
34      protected Integer length;
35  
36      protected final boolean chrome;
37  
38      public ManagedStreamResponseWriter(GwtAtmosphereResourceImpl resource) {
39          super(resource);
40  
41          String userAgent = resource.getAtmosphereResource().getRequest().getHeader("User-Agent");
42          chrome = userAgent != null && userAgent.contains("Chrome");
43      }
44  
45      @Override
46      protected OutputStream getOutputStream(OutputStream outputStream) {
47          countOutputStream = new CountOutputStream(outputStream);
48          return countOutputStream;
49      }
50  
51      @Override
52      protected void doSuspend() throws IOException {
53          int paddingRequired;
54          String paddingParameter = getRequest().getParameter("padding");
55          if (paddingParameter != null) {
56              paddingRequired = Integer.parseInt(paddingParameter);
57          } else {
58              paddingRequired = getPaddingRequired();
59          }
60  
61          String lengthParameter = getRequest().getParameter("length");
62          if (lengthParameter != null) {
63              length = Integer.parseInt(lengthParameter);
64          }
65  
66          if (paddingRequired > 0) {
67              countOutputStream.setIgnoreFlush(true);
68              writer.flush();
69  
70              int written = countOutputStream.getCount();
71  
72              if (paddingRequired > written) {
73                  CharSequence padding = getPadding(paddingRequired - written);
74                  if (padding != null) {
75                      writer.append(padding);
76                  }
77              }
78  
79              countOutputStream.setIgnoreFlush(false);
80          }
81      }
82  
83      @Override
84      public synchronized void write(List<? extends Serializable> messages, boolean flush) 
85                      throws IOException, SerializationException {
86          super.write(messages, flush);
87          checkLength();
88      }
89  
90      @Override
91      public synchronized void heartbeat() throws IOException {
92          super.heartbeat();
93          checkLength();
94      }
95  
96      private void checkLength() throws IOException {
97          int count = countOutputStream.getCount();
98          // Chrome seems to have a problem with lots of small messages consuming lots of memory.
99          // I'm guessing for each readyState = 3 event it copies the responseText from its IO system to its
100         // JavaScript
101         // engine and does not clean up all the events until the HTTP request is finished.
102         if (chrome) {
103             count = 2 * count;
104         }
105         if (!refresh && isOverRefreshLength(count)) {
106             refresh = true;
107             doRefresh();
108         }
109     }
110 
111     protected abstract void doRefresh() throws IOException;
112 
113     protected abstract int getPaddingRequired();
114 
115     protected abstract CharSequence getPadding(int padding);
116 
117     protected abstract boolean isOverRefreshLength(int written);
118 }

20  import com.google.gwt.user.server.rpc.SerializationPolicy;

22  import java.io.IOException;
23  import java.io.Serializable;

29  abstract public class StreamingProtocolResponseWriter extends ManagedStreamResponseWriter {
30  
31      private static final int MAX_PADDING_REQUIRED = 2048;
32      private static final String PADDING_STRING;
33  
34      static {
35          char[] padding = new char[MAX_PADDING_REQUIRED + 1];
36          for (int i = 0; i < padding.length - 1; i++) {
37              padding[i] = '*';
38          }
39          padding[padding.length - 1] = '\n';
40          PADDING_STRING = new String(padding);
41      }
42  
43      public StreamingProtocolResponseWriter(GwtAtmosphereResourceImpl resource, SerializationPolicy serializationPolicy, ClientOracle clientOracle) {
44          super(resource, serializationPolicy, clientOracle);
45  
46      }
47  
48      abstract String getContentType();
49  
50      @Override
51      public void initiate() throws IOException {
52          getResponse().setContentType(getContentType());
53  
54          String origin = getRequest().getHeader("Origin");
55          if (origin != null) {
56              if (logger.isDebugEnabled()) {
57                  logger.debug("Origin: " + origin);
58              }
59              getResponse().setHeader("Access-Control-Allow-Origin", origin);
60          }
61  
62          super.initiate();
63  
64          // send connection parameters to client
65          writer.append('!').append(String.valueOf(resource.getHeartBeatInterval())).append(':');
66          writer.append(String.valueOf(connectionID)).append('\n');
67      }
68  
69  
70      static CharSequence escape(CharSequence string) {
71          int length = (string != null) ? string.length() : 0;
72          int i = 0;
73          loop:
74          while (i < length) {
75              char ch = string.charAt(i);
76              switch (ch) {
77                  case '\\':
78                  case '\n':
79                      break loop;
80              }
81              i++;
82          }
83  
84          if (i == length)
85              return string;
86  
87          StringBuilder str = new StringBuilder(string.length() * 2);
88          str.append(string, 0, i);
89          while (i < length) {
90              char ch = string.charAt(i);
91              switch (ch) {
92                  case '\\':
93                      str.append("\\\\");
94                      break;
95                  case '\n':
96                      str.append("\\n");
97                      break;
98                  default:
99                      str.append(ch);
100             }
101             i++;
102         }
103         return str;
104     }

106     @Override
107     protected CharSequence getPadding(int padding) {
108         if (padding > PADDING_STRING.length() - 1) {
109             StringBuilder result = new StringBuilder(padding);
110             while (padding > 0) {
111                 if (padding > PADDING_STRING.length() - 1) {
112                     result.append(PADDING_STRING);
113                     padding -= PADDING_STRING.length() - 1;
114                 } else {
115                     result.append(PADDING_STRING.substring(PADDING_STRING.length() - 1 - padding));
116                     padding = 0;
117                 }
118             }
119             return result.toString();
120         } else {
121             return PADDING_STRING.substring(PADDING_STRING.length() - padding - 1);
122         }
123     }
124 
125     @Override
126     protected void doSendError(int statusCode, String message) throws IOException {
127         getResponse().setStatus(statusCode);
128         if (message != null) {
129             writer.append(message);
130         }
131     }
132 
133     @Override
134     protected void doWrite(List<? extends Serializable> messages) throws IOException {
135         for (Serializable message : messages) {
136             CharSequence string;
137             if (message instanceof CharSequence) {
138                 string = escape((CharSequence) message);
139                 if (string == message) {
140                     writer.append('|');
141                 } else {
142                     writer.append(']');
143                 }
144             } else {
145                 string = serialize(message);
146             }
147 
148             writer.append(string).append('\n');
149         }
150     }
151 
152     @Override
153     protected boolean isOverRefreshLength(int written) {
154         if (length != null) {
155             return written > length;
156         } else {
157             return written > 5 * 1024 * 1024;
158         }
159     }
160 
161     @Override
162     protected void doHeartbeat() throws IOException {
163         writer.append("#\n");
164     }
165 
166     @Override
167     protected void doTerminate() throws IOException {
168         writer.append("?\n");
169     }
170 
171     @Override
172     protected void doRefresh() throws IOException {
173         writer.append("@\n");
174     }

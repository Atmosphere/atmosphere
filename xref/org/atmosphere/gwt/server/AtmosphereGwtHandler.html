71      public int doComet(GwtAtmosphereResource resource) throws ServletException, IOException {
72          Broadcaster broadcaster = BroadcasterFactory.getDefault().lookup(Broadcaster.class, GWT_BROADCASTER_ID);
73          if (broadcaster == null) {
74              try {
75                  broadcaster = BroadcasterFactory.getDefault().get(DefaultBroadcaster.class, GWT_BROADCASTER_ID);
76              } catch (IllegalAccessException ex) {
77                  logger.error("Failed to get broadcaster", ex);
78              } catch (InstantiationException ex) {
79                  logger.error("Failed to get broadcaster", ex);
80              }
81          }
82          resource.getAtmosphereResource().setBroadcaster(broadcaster);
83          return NO_TIMEOUT;
84      }

117     protected GwtAtmosphereResource lookupResource(int connectionId) {
118         GwtAtmosphereResource r = resources.get(connectionId);

169     @Override
170     public void onRequest(AtmosphereResource<HttpServletRequest, HttpServletResponse> resource) throws IOException {
171 
172         HttpServletRequest request = resource.getRequest();
173 
174         String servertransport = request.getParameter("servertransport");
175         if ("rpcprotocol".equals(servertransport)) {
176             Integer connectionID = Integer.parseInt(request.getParameter("connectionID"));
177             doServerMessage(request.getReader(), connectionID);
178             return;
179         }
180 
181         try {
182             int requestHeartbeat = heartbeat;
183             String requestedHeartbeat = request.getParameter("heartbeat");
184             if (requestedHeartbeat != null) {
185                 try {
186                     requestHeartbeat = Integer.parseInt(requestedHeartbeat);
187                     if (requestHeartbeat <= 0) {
188                         throw new IOException("invalid heartbeat parameter");
189                     }
190                     requestHeartbeat = computeHeartbeat(requestHeartbeat);
191                 } catch (NumberFormatException e) {
192                     throw new IOException("invalid heartbeat parameter");
193                 }
194             }
195 
196             GwtAtmosphereResourceImpl resourceWrapper = new GwtAtmosphereResourceImpl(resource, this, requestHeartbeat);
197             doCometImpl(resourceWrapper);
198         } catch (IOException e) {
199 //            GwtAtmosphereResourceImpl resource = new GwtAtmosphereResourceImpl(atm, this, -1);
200             logger.error("Unable to initiated comet" + e.getMessage(), e);
201 //			resource.getResponseWriter().sendError(HttpServletResponse.SC_BAD_REQUEST, e.getMessage());
202         }

207     protected void doServerMessage(BufferedReader data, int connectionID) {
208         List<Serializable> postMessages = new ArrayList<Serializable>();
209         GwtAtmosphereResource resource = lookupResource(connectionID);
210         if (resource == null) {
211             return;
212         }
213         try {
214             while (true) {
215                 String event = data.readLine();
216                 if (event == null) {
217                     break;
218                 }
219                 String messageData = data.readLine();
220                 if (messageData == null) {
221                     break;
222                 }
223                 data.readLine();
224                 if (logger.isTraceEnabled()) {
225                     logger.trace("[" + connectionID + "] Server message received: " + event + ";" + messageData.charAt(0));
226                 }
227                 if (event.equals("o")) {
228                     if (messageData.charAt(0) == 'p') {
229                         Serializable message = deserialize(messageData.substring(1));
230                         if (message != null) {
231                             postMessages.add(message);
232                         }
233                     } else if (messageData.charAt(0) == 'b') {
234                         Serializable message = deserialize(messageData.substring(1));
235                         broadcast(message, resource);
236                     }
237 
238                 } else if (event.equals("s")) {
239 
240                     if (messageData.charAt(0) == 'p') {
241                         String message = messageData.substring(1);
242                         postMessages.add(message);
243                     } else if (messageData.charAt(0) == 'b') {
244                         Serializable message = messageData.substring(1);
245                         broadcast(message, resource);
246                     }
247 
248                 } else if (event.equals("c")) {
249 
250                     if (messageData.equals("d")) {
251                         disconnect(resource);
252                     }
253                 }
254             }
255         } catch (IOException ex) {
256             logger.error("[" + connectionID + "] Failed to read", ex);
257         }
258 
259         if (postMessages.size() > 0) {
260             post(postMessages, resource);
261         }
262     }

336     private void doCometImpl(GwtAtmosphereResourceImpl resource) throws IOException {
337 
338         try {
339             // setup the request
340             resource.getWriterImpl().initiate();
341             if (resources == null) {
342                 resources = new ConcurrentHashMap<Integer, GwtAtmosphereResource>(5);
343                 resource.getBroadcaster().getBroadcasterConfig().getScheduledExecutorService().scheduleWithFixedDelay(new Runnable() {
344                     @Override
345                     public void run() {
346                         reapResources();
347                     }
348                 }, 30, 10, TimeUnit.SECONDS);
349             }
350             resources.put(resource.getConnectionID(), resource);
351         } catch (IOException e) {
352             logger.error("Error initiating GwtComet", e);
353             return;
354         }
355 
356         int timeout;
357         try {
358             // call the application code
359             timeout = doComet(resource);
360             if (timeout == -1) {
361                 logger.info("You have set an infinite timeout for your comet connection this is not recommended");
362             }
363         } catch (ServletException e) {
364             logger.error("Error calling doComet()", e);
365 //			resource.getResponseWriter().sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e.getMessage());
366             return;
367         } catch (IOException e) {
368             logger.error("Error calling doComet()", e);
369 //			resource.getResponseWriter().sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e.getMessage());
370             return;
371         }
372 
373         // at this point the application may have spawned threads to process this response
374         // so we have to be careful about concurrency from here on
375         resource.suspend(timeout);
376     }

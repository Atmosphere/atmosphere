60      /* =========================================================== */
61      /*
62        * Input data mangement.
63        *
64        * The window keeps a copy of the previously encountered uncompressed data bytes. In the beginning, the window is
65        * empty, then filled up to a certain limit (a power of 2, at most 32768). Beyond that limit, it is managed as a
66        * circular buffer.
67        *
68        * To each data byte we associate a triplet, consisting of that byte and the next two bytes. We assemble triplets
69        * into 24-bit values, such that the lower 8 bits are the most recently received. Each window slot contains a
70        * triplet; hence, each input data byte appears in three distinct, successive slots in the window.
71        *
72        * Window triplets are linked as hash chains, with the hashLink[] array. The hashTable[] array contains the chain
73        * entry points.
74        *
75        * We also copy input bytes into the ucBuffer[] array (up to some limit). These are used to produce uncompressed
76        * data blocks if it turns out that the compressor cannot find anything better. The size limit is computed so that
77        * when an uncompressed data block must be produced, then the size limit on ucBuffer[] has not been reached. Note:
78        * technically, we could limit ucBuffer[] to the window length and complete the uncompressed block by reconstructing
79        * the input bytes from the symbols in buffer[]. This would be interesting if we used a large buffer[] array. But
80        * such a large symbol buffer is undesirable since it prevents the dynamic Huffman codes from adapting to changes in
81        * the file content type.
82        */
83

105     /**
106      * Hash chains: element "n" in this array contains the backward distance to the previous triplet with the same hash
107      * (or 0, if there is none). Due to the window buffer reuse, the previous triplet may have been overwritten: the
108      * code which walks a chain must make sure that it stays within the window size.
109      */
110     private char[] windowLink;
111 
112     /**
113      * For the hash value "v", <code>hashTable[v]</code> contains the value "j". If <code>j == 0</code>, then there is
114      * no current chain for this hash value; otherwise, the chain head is in the window, at index "j-1". It may happen
115      * that the chain head entry has been overwritten: the insertion code checks the designated entry when linking a new
116      * triplet.
117      */
118     private char[] hashTable;
119 
120     /**
121      * The buffer which receives a copy of the uncompressed data. That buffer is circular.

123     private byte[] ucBuffer;

125     /**
126      * The length of the accumulated data in <code>ucBuffer</code>.
127      */
128     private int ucBufferPtr;

201     /* =========================================================== */
202     /*
203       * Output data management.
204       *
205       * Output symbols are accumulated in a buffer. Each entry is a 32-bit word consisting of: -- literal value in the
206       * literal+length alphabet (9 bits) -- extra length bits (5 bits) -- distance symbol (5 bits) -- extra distance
207       * length (13 bits) (values ordered from least- to most-significant bits)
208       *
209       * If the literal value is not a sequence copy symbol (i.e. it has value 255 or less) then the other fields are
210       * zero. Note that the end-of-block special literal (value 256) is never written in that buffer.
211       *
212       * The lengths of the buffer[] and ucBuffer[] arrays are linked. Basically, if there are bufferLen symbols in
213       * buffer[], then these could be encoded as (at worst) 32bufferLen bits, in a type 1 block (plus the three-bit block
214       * header). Hence, a type 0 block may be used only if the uncompressed data length is no more than 32bufferLen-32
215       * bits, because this is the payload length for an uncompressed block of size 32bufferLen bits (then again excluding
216       * the three-bit header, and the additional byte-alignment padding bits). Therefore, ucBuffer.length needs not be
217       * larger than 4buffer.length (we need 258 more extra bytes to accomodate a currently matched sequence).
218       *
219       * We furthermore limit buffer.length to 16384, thus guaranteeing that when type 0 blocks are selected, no more than
220       * 65532 bytes are to be written out, which fits in a single block. The drawback is that uncompressible data may
221       * fill buffer[] after only 16384 literal bytes, which yields an overhead four times larger than the optimal
222       * overhead. This is not a serious issue: it turns out that zlib has the same overhead and is quite happy with it.
223       *
224       * Actual writes to the output stream use an internal buffer so that unbuffered transport streams can be used.
225       * Compression is inherently a buffered process.
226       */
227

270     /**
271      * Compression level 1: do not apply LZ77; only Huffman codes are computed. This is faster than <code>SPEED</code>
272      * but with a degraded compression ratio.
273      */
274     public static final int HUFF = 1;

276     /**
277      * Compression level 2: optimize for speed.
278      */

310     /**
311      * Build a deflater with the provided compression strategy (either <code>SPEED</code>, <code>MEDIUM</code> or
312      * <code>COMPACT</code>) and the provided window size. The window size is expressed in bits and may range from 9 to
313      * 15 (inclusive). The DEFLATE format uses a 15-bit window; by using a smaller window, the produced stream can be
314      * inflated with less memory (but compression ratio is lowered). If the compression level is 0, then the default
315      * compression level is selected.
316      *
317      * @param level      the compression strategy
318      * @param windowBits the window bit size (9 to 15)
319      */
320     public Deflater(int level, int windowBits) {
321         if (windowBits < 9 || windowBits > 15)
322             throw new IllegalArgumentException("invalid LZ77 window bit length: " + windowBits);
323 
324         /*
325            * The internal buffer should not be too large, because it prevents the Huffman codes from adapting to data
326            * internal changes. When the internal buffer is too small, the block headers and dynamic tree representations
327            * become too expensive. The maximum value is 16384; to use something greater, the type 0 block code in
328            * endBlock() would have to be adapted.
329            */
330         int bufferLen = 16384;
331 
332         int windowLen = 1 << windowBits;
333         window = new int[windowLen];
334         windowLink = new char[windowLen];
335         hashTable = new char[windowLen];
336         maxDistanceTriplet = windowLen - 261;
337         maxDistanceSeq1 = windowLen - 261;
338         maxDistanceSeq2 = windowLen - 261;
339         if (level == 0)
340             level = MEDIUM;
341         switch (level) {
342             case SPEED:
343                 maxChainLengthTriplet = 16;
344                 maxChainLengthSeq1 = 8;
345                 goodSequenceLength = 8;
346                 maxLazyLength = 4;
347                 maxChainLengthSeq2 = 4;
348                 break;
349             case MEDIUM:
350                 maxChainLengthTriplet = 128;
351                 maxChainLengthSeq1 = 128;
352                 goodSequenceLength = 64;
353                 maxLazyLength = 64;
354                 maxChainLengthSeq2 = 32;
355                 break;
356             case COMPACT:
357                 maxChainLengthTriplet = 1024;
358                 maxChainLengthSeq1 = 1024;
359                 goodSequenceLength = 258;
360                 maxLazyLength = 258;
361                 maxChainLengthSeq2 = 1024;
362                 break;
363             case HUFF:
364                 huffOnly = true;
365                 break;
366             default:
367                 throw new IllegalArgumentException("unknown compression level: " + level);
368         }
369         buffer = new int[bufferLen];
370         ucBuffer = new byte[4 * bufferLen + 258];
371         outBuf = new byte[4096];
372     }

397     /**
398      * Compute a 32-bit copy symbol, for the provided sequence length and source distance.
399      *
400      * @param len  the sequence length
401      * @param dist the source sequence distance
402      * @return the copy symbol
403      */
404     private static int makeCopySymbol(int len, int dist) {
405         int symlen, elen;
406         if (len <= 10) {
407             symlen = 254 + len;
408             elen = 0;
409         } else if (len == 258) {
410             symlen = 285;
411             elen = 0;
412         } else {
413             /*
414                 * This may be optimized with a dichotomic search.
415                 */
416             int i;
417             for (i = 9; i < 29; i++)
418                 if (LENGTH[i] > len)
419                     break;
420             symlen = 256 + i;
421             elen = len - LENGTH[i - 1];
422         }
423 
424         int symdist, edist;
425         if (dist <= 4) {
426             symdist = dist - 1;
427             edist = 0;
428         } else {
429             /*
430                 * This may be optimized with a dichotomic search.
431                 */
432             int i;
433             for (i = 5; i < 30; i++)
434                 if (DIST[i] > dist)
435                     break;
436             symdist = i - 1;
437             edist = dist - DIST[i - 1];
438         }
439 
440         return symlen + (elen << 9) + (symdist << 14) + (edist << 19);
441     }

443     /**
444      * Find a previous sequence, identical to the sequence at index <code>orig</code> and length <code>len</code>, such
445      * that this previous sequence is continued by three bytes equal to the provided <code>end</code> value.
446      * <strong>IMPORTANT:</strong> the "original" sequence must actually be longer than <code>len</code> by two bytes,
447      * which match the high two bytes of <code>end</code>.
448      *
449      * @param win     the window buffer
450      * @param winMask the window buffer length, minus one
451      * @param wlink   the window link buffer
452      * @param orig    the original sequence begin index
453      * @param dist    the distance associated with the original sequence
454      * @param len     the original sequence length
455      * @param end     the sequence end triplet
456      * @param maxLen  the maximum explored chain length
457      * @param maxDist the maximum accepted distance
458      * @return the previous longer sequence distance, or 0 if not found
459      */
460     private static int findPreviousSequence(int[] win, int winMask, char[] wlink, int orig, int dist, int len, int end, int maxLen, int maxDist) {
461         int n = orig;
462         int chainLength = maxLen;
463         loop:
464         while (chainLength-- > 0) {
465             int d = wlink[n];
466             if (d == 0)
467                 return 0;
468             dist += d;
469             if (dist > maxDist)
470                 return 0;
471             n = (n - d) & winMask;
472             int tm = len;
473             int j = orig, k = n;
474             if (tm >= 3) {
475                 while (tm >= 3) {
476                     if (win[j] != win[k])
477                         continue loop;
478                     tm -= 3;
479                     j = (j + 3) & winMask;
480                     k = (k + 3) & winMask;
481                 }
482                 switch (tm) {
483                     case 1:
484                         j = (j - 2) & winMask;
485                         k = (k - 2) & winMask;
486                         if (win[j] != win[k])
487                             continue loop;
488                         k = (k + 3) & winMask;
489                         break;
490                     case 2:
491                         j = (j - 1) & winMask;
492                         k = (k - 1) & winMask;
493                         if (win[j] != win[k])
494                             continue loop;
495                         k = (k + 3) & winMask;
496                         break;
497                 }
498             } else {
499                 if (win[j] != win[k])
500                     continue loop;
501                 k = (k + tm) & winMask;
502             }
503             if (win[k] == end)
504                 return dist;
505         }
506         return 0;
507     }

509     /**
510      * Update the <code>ucBuffer</code> array with the provided uncompressed data. This must be done before ending the
511      * block.
512      *
513      * @param buf  the source data buffer
514      * @param off1 the source start offset (inclusive)
515      * @param off2 the source end offset (exclusive)
516      */
517     private void updateUCBuffer(byte[] buf, int off1, int off2) {
518         int uLen = ucBuffer.length;
519         int sInc = (uLen >>> 1);
520         while (off1 < off2) {
521             int free = uLen - ucBufferPtr;
522             int tc = off2 - off1;
523             if (tc > sInc)
524                 tc = sInc;
525             if (tc > free) {
526                 System.arraycopy(buf, off1, ucBuffer, ucBufferPtr, free);
527                 System.arraycopy(buf, off1 + free, ucBuffer, 0, tc - free);
528                 ucBufferPtr = tc - free;
529             } else {
530                 System.arraycopy(buf, off1, ucBuffer, ucBufferPtr, tc);
531                 ucBufferPtr += tc;
532             }
533             off1 += tc;
534         }
535     }

537     /**
538      * Process some more data. When the internal buffer is full, or when the compressor code deems it appropriate, the
539      * data is compressed and sent on the transport stream. This method is most efficient when data is input by big
540      * enough chunks (at least a dozen bytes per call).
541      *
542      * @param buf the data buffer
543      * @param off the data offset
544      * @param len the data length (in bytes)
545      * @throws IOException on I/O error with the transport stream
546      */
547     public void process(byte[] buf, int off, int len) throws IOException {
548         if (len == 0)
549             return;
550         int origOff = off;
551 
552         /*
553            * If in "Huffman only" mode, then we use an alternate loop.
554            */
555         if (huffOnly) {
556             int[] sb = buffer;
557             int sbPtr = bufferPtr;
558             int sbLen = sb.length;
559             while (len-- > 0) {
560                 int bv = buf[off++] & 0xFF;
561                 sb[sbPtr++] = bv;
562                 if (sbPtr == sbLen) {
563                     updateUCBuffer(buf, origOff, off);
564                     origOff = off;
565                     endBlock(false, sbPtr);
566                     sbPtr = 0;
567                 }
568             }
569             bufferPtr = sbPtr;
570             updateUCBuffer(buf, origOff, off);
571             return;
572         }
573 
574         /*
575            * We have some special code for the first two bytes ever.
576            */
577         if (windowState < 2) {
578             int bv = buf[off++] & 0xFF;
579             len--;
580             if (windowState == 0) {
581                 recentBytes = bv;
582                 seqLen = 1;
583                 if (len == 0) {
584                     updateUCBuffer(buf, origOff, off);
585                     return;
586                 }
587                 bv = buf[off++] & 0xFF;
588                 len--;
589             }
590             recentBytes = (recentBytes << 8) | bv;
591             windowState = 2;
592             seqLen = 2;
593             if (len == 0) {
594                 updateUCBuffer(buf, origOff, off);
595                 return;
596             }
597         }
598 
599         /*
600            * We cache most instance fields in local variables. This helps the JIT compiler produce efficient code.
601            */
602         int[] win = window;
603         int winMask = win.length - 1;
604         int winPtr = windowPtr;
605         int recent = recentBytes;
606         char[] wlink = windowLink;
607         char[] ht = hashTable;
608         int htMask = ht.length - 1;
609         int sLen = seqLen;
610         int sPtr = seqPtr;
611         int sDist = seqDist;
612         int[] sb = buffer;
613         int sbPtr = bufferPtr;
614         int sbLen = sb.length;
615         int maxCL0 = maxChainLengthTriplet;
616         int maxDistCL0 = maxDistanceTriplet;
617         int maxCL1 = maxChainLengthSeq1;
618         int maxDistCL1 = maxDistanceSeq1;
619         int goodSLen = goodSequenceLength;
620         int maxLLen = maxLazyLength;
621         int maxCL2 = maxChainLengthSeq2;
622         int maxDistCL2 = maxDistanceSeq2;
623 
624         loop:
625         while (len-- > 0) {
626             int b0 = buf[off++] & 0xFF;
627 
628             /*
629                 * Compute the current triplet.
630                 */
631             int triplet = (recent << 8) | b0;
632             recent = triplet & 0xFFFF;
633 
634             /*
635                 * Update the window and set hash links.
636                 */
637             win[winPtr] = triplet;
638             int h = (triplet + (triplet >>> 4) + (triplet >>> 8) + (triplet >>> 9) - (triplet >>> 16)) & htMask;
639             int link = ht[h] - 1;
640             int dist;
641             if (link < 0) {
642                 dist = 0;
643             } else {
644                 int pv = win[link];
645                 int ph = (pv + (pv >>> 4) + (pv >>> 8) + (pv >>> 9) - (pv >>> 16)) & htMask;
646                 if (ph == h) {
647                     dist = (winPtr - link) & winMask;
648                 } else {
649                     dist = 0;
650                 }
651             }
652             ht[h] = (char) (winPtr + 1);
653             wlink[winPtr] = (char) dist;
654 
655             int thisPtr = winPtr;
656             winPtr = (winPtr + 1) & winMask;
657 
658             /*
659                 * If the current sequence length is 0 or 1, then we do not have a complete triplet yet, and there is
660                 * nothing more to do.
661                 */
662             if (sLen < 2) {
663                 sLen++;
664                 continue loop;
665             }
666 
667             /*
668                 * If we just completed a triplet, then we look for a previous triplet. If we find one, then we begin a
669                 * match sequence; otherwise, we emit a literal for the first byte of our triplet, and we continue. There is
670                 * a corner case when the previous triplet is at the maximum distance: in that situation, we must emit the
671                 * copy symbol immediately.
672                 */
673             if (sLen == 2) {
674                 if (dist == 0) {
675                     sb[sbPtr++] = triplet >>> 16;
676                     if (sbPtr == sbLen) {
677                         updateUCBuffer(buf, origOff, off);
678                         origOff = off;
679                         endBlock(false, sbPtr);
680                         sbPtr = 0;
681                     }
682                     continue loop;
683                 }
684                 sDist = dist;
685                 findTriplet:
686                 do {
687                     int n = link;
688                     int chainLen = maxCL0;
689                     while (chainLen-- > 0) {
690                         if (win[n] == triplet)
691                             break findTriplet;
692                         int d = wlink[n];
693                         if (d == 0)
694                             break;
695                         sDist += d;
696                         if (sDist > maxDistCL0)
697                             break;
698                         n = (n - d) & winMask;
699                     }
700                     sDist = 0;
701                 }
702                 while (false);
703                 if (sDist == 0) {
704                     sb[sbPtr++] = triplet >>> 16;
705                     if (sbPtr == sbLen) {
706                         updateUCBuffer(buf, origOff, off);
707                         origOff = off;
708                         endBlock(false, sbPtr);
709                         sbPtr = 0;
710                     }
711                 } else {
712                     sPtr = (thisPtr - sDist) & winMask;
713                     sLen = 3;
714                     if (sPtr == winPtr) {
715                         sb[sbPtr++] = makeCopySymbol(sLen, sDist);
716                         if (sbPtr == sbLen) {
717                             updateUCBuffer(buf, origOff, off);
718                             origOff = off;
719                             endBlock(false, sbPtr);
720                             sbPtr = 0;
721                         }
722                         sLen = 0;
723                     }
724                 }
725                 continue loop;
726             }
727 
728             /*
729                 * We are currently matching a sequence. We try to augment it. If we can, then we just do that; but we must
730                 * mind the maximum sequence length and also its distance.
731                 */
732             if (win[(thisPtr - sDist) & winMask] == triplet) {
733                 sLen++;
734                 if (sPtr == winPtr || sLen == 258) {
735                     sb[sbPtr++] = makeCopySymbol(sLen, sDist);
736                     if (sbPtr == sbLen) {
737                         updateUCBuffer(buf, origOff, off);
738                         origOff = off;
739                         endBlock(false, sbPtr);
740                         sbPtr = 0;
741                     }
742                     sLen = 0;
743                 }
744                 continue loop;
745             }
746 
747             /*
748                 * The current sequence cannot be agumented. We try to find a longer match in the previous sequences.
749                 */
750             int sDistNew = findPreviousSequence(win, winMask, wlink, sPtr, sDist, sLen - 2, triplet, sLen > goodSLen ? (maxCL1 >>> 2) : maxCL1, maxDistCL1);
751             if (sDistNew > 0) {
752                 sDist = sDistNew;
753                 sPtr = (thisPtr - (sLen - 2) - sDistNew) & winMask;
754                 sLen++;
755                 if (sPtr == winPtr || sLen == 258) {
756                     sb[sbPtr++] = makeCopySymbol(sLen, sDist);
757                     if (sbPtr == sbLen) {
758                         updateUCBuffer(buf, origOff, off);
759                         origOff = off;
760                         endBlock(false, sbPtr);
761                         sbPtr = 0;
762                     }
763                     sLen = 0;
764                 }
765                 continue loop;
766             }
767 
768             /*
769                 * We could not find a longer sequence. We must flush something. We try to find a longer sequence beginning
770                 * at the next byte (that's what RFC 1951 calls lazy matching).
771                 */
772             if (sLen <= maxLLen) {
773                 int refPtr = (thisPtr - (sLen - 2) + 1) & winMask;
774                 int mdc = maxDistCL2;
775                 if (mdc > sDist)
776                     mdc = sDist;
777                 sDistNew = findPreviousSequence(win, winMask, wlink, refPtr, 0, sLen - 3, triplet, maxCL2, mdc);
778                 if (sDistNew > 0) {
779                     sb[sbPtr++] = win[sPtr] >>> 16;
780                     if (sbPtr == sbLen) {
781                         updateUCBuffer(buf, origOff, off);
782                         origOff = off;
783                         endBlock(false, sbPtr);
784                         sbPtr = 0;
785                     }
786                     sDist = sDistNew;
787                     sPtr = (refPtr - sDistNew) & winMask;
788                     if (sPtr == winPtr) {
789                         sb[sbPtr++] = makeCopySymbol(sLen, sDist);
790                         if (sbPtr == sbLen) {
791                             updateUCBuffer(buf, origOff, off);
792                             origOff = off;
793                             endBlock(false, sbPtr);
794                             sbPtr = 0;
795                         }
796                         sLen = 0;
797                     }
798                     continue loop;
799                 }
800             }
801 
802             /*
803                 * The current sequence is finished and we could not find any better. We filter out three-byte sequences
804                 * which are too far: for these sequences, the copy symbol with its distance code and extra bits may be more
805                 * expensive than simply writing out the literal bytes.
806                 *
807                 * The threshold for this test has been determined by compressing many files and seeing how the result
808                 * compares with what gzip produces. It seems that the "right" value is 6144.
809                 */
810             if (sLen == 3 && sDist > 6144) {
811                 int ot = win[sPtr];
812                 for (int k = 16; k >= 0; k -= 8) {
813                     sb[sbPtr++] = (ot >>> k) & 0xFF;
814                     if (sbPtr == sbLen) {
815                         updateUCBuffer(buf, origOff, off);
816                         origOff = off;
817                         endBlock(false, sbPtr);
818                         sbPtr = 0;
819                     }
820                 }
821             } else {
822                 sb[sbPtr++] = makeCopySymbol(sLen, sDist);
823                 if (sbPtr == sbLen) {
824                     updateUCBuffer(buf, origOff, off);
825                     origOff = off;
826                     endBlock(false, sbPtr);
827                     sbPtr = 0;
828                 }
829             }
830             sLen = 1;
831         }
832 
833         /*
834            * Flush out the local copies of the instance fields.
835            */
836         windowPtr = winPtr;
837         recentBytes = recent;
838         seqLen = sLen;
839         seqPtr = sPtr;
840         seqDist = sDist;
841         bufferPtr = sbPtr;
842 
843         /*
844            * Do not forget the original data bytes.
845            */
846         updateUCBuffer(buf, origOff, off);
847     }

882     /*
883       * Huffman trees.
884       *
885       * The optimal Huffman tree cannot always be used, because we have additional constraints on the code lengths. The
886       * generic algorithm is called "package-merge" but it is relatively expensive (O(Ln) where L is the maximum code
887       * length and n is the alphabet size. We rather implement a "tree tweak": we begin with the optimal Huffman tree,
888       * and then we tweak it until it fits in the length constraints (it turns out that zlib uses a very similar trick).
889       *
890       * The tricky point is that once we have sorted the symbol by frequencies, then we just need to compute the number
891       * of codes for each code length. The actual code lengths for all symbols are easy to compute back by giving the
892       * longer codes to the least often encountered symbols (we may somehow obtain a locally optimal yet different tree,
893       * but this is no problem since in fine we will convert the tree to the canonical Huffman codes anyway). That idea
894       * apparently comes from someone called Haruhiko Okumura.
895       *
896       * Tree tweaking is applied when the optimal tree has "overdeep" leaves. Basically, we have a number of non-leaf
897       * nodes at the maximum depth (call "p" that number of nodes). Each is the root for a subtree containing q_i leaves
898       * which are all overdeep (1 <= i <= q). If we have q = \sum_i q_i overdeep leaves, then we must relocate those q
899       * leaves: -- p leaves will use the locations held by the subtree roots at maximum depth; -- q-p leaves will have to
900       * be relocated elsewhere. Hence, a first pass is used to obtain the number of codes for all valid code lengths; in
901       * that pass, the subtree roots at maximum depth are accounted as leaves (this handles the p "natural" relocations),
902       * and the value of q-p is returned. Relocation for those values works thus: to relocate a leaf, find a valid code
903       * of length l (strictly lower than the maximum length) and replace it with two codes of length l+1: the code is
904       * lengthened by one bit, which leaves room for one brother. We iterate over all leaves to relocate, choosing each
905       * time the longest possible code.
906       *
907       * It shall be noted that for the values used in DEFLATE, it is not possible that _all_ optimal codes exceed the
908       * maximum code length. Actually, the tweaking mechanism cannot fail with the parameters which will be used.
909       *
910       * The sorting step uses the Heap Sort, because it is an elegant algorithm which uses O(1) additional space and has
911       * a nice O(n log n) guaranteed worst case. A QuickSort would be faster on average, but I like the Heap Sort better,
912       * and this part is not a CPU bottleneck anyway.
913       */
914 
915     /**
916      * Sort the provided array of integer values (ascending order). Only the values between indexes <code>off</code>
917      * (inclusive) and <code>off + len</code> (exclusive) are touched; other array elements are neither considered nor
918      * modified.
919      *
920      * @param val the array to sort
921      * @param off the array offset
922      * @param len the number of elements to sort
923      */
924     private static void heapSort(int[] val, int off, int len) {
925         if (len <= 1)
926             return;
927 
928         /*
929            * We use virtual indexes, between 1 and len (inclusive). In our internal heap, the two children of node of
930            * virtual index "i" have virtual indexes "2*i" and "2*i+1".
931            */
932         int corr = off - 1;
933 
934         /*
935            * Build the heap by inserting all values at the bottom, and then making them float up as necessary.
936            */
937         for (int i = 2; i <= len; i++) {
938             int j = i;
939             int v = val[corr + j];
940             while (j > 1) {
941                 int k = j >>> 1;
942                 int f = val[corr + k];
943                 if (f > v)
944                     break;
945                 val[corr + j] = f;
946                 val[corr + k] = v;
947                 j = k;
948             }
949         }
950 
951         /*
952            * Repeatedly, extract the heap root (maximum element) and rebuild the heap by promoting the bottom element and
953            * having it sink to its proper place.
954            */
955         for (int i = len; i > 1; i--) {
956             int v = val[corr + i];
957             val[corr + i] = val[corr + 1];
958             val[corr + 1] = v;
959             int j = 1;
960             for (; ; ) {
961                 int kl = j << 1;
962                 int kr = kl + 1;
963                 if (kl >= i)
964                     break;
965                 int si;
966                 int sv;
967                 if (kr >= i) {
968                     si = kl;
969                     sv = val[corr + kl];
970                 } else {
971                     int cl = val[corr + kl];
972                     int cr = val[corr + kr];
973                     if (cl > cr) {
974                         si = kl;
975                         sv = cl;
976                     } else {
977                         si = kr;
978                         sv = cr;
979                     }
980                 }
981                 if (v > sv)
982                     break;
983                 val[corr + j] = sv;
984                 val[corr + si] = v;
985                 j = si;
986             }
987         }
988     }

990     /**
991      * <p>
992      * Compute the optimal Huffman tree, then tweak it so that it fits within the specified maximum code length.
993      * Returned value is the resulting code length for each symbol. From these lengths, the canonical Huffman code can
994      * be rebuilt.
995      * </p>
996      * <p/>
997      * <p>
998      * The alphabet size is assumed to be equal to <code>freq.length</code>.
999      * </p>
1000      * <p/>
1001      * <p>
1002      * The following properties MUST be met:
1003      * <ul>
1004      * <li>frequencies are positive integers (0 is allowed, and qualifies a symbol which does not occur at all);</li>
1005      * <li>the sum of all frequencies must be strictly lower than 2097152 (i.e. that sum must fit in a 21-bit unsigned
1006      * integer field).</li>
1007      * </ul>
1008      * </p>
1009      *
1010      * @param freq       the symbol frequencies
1011      * @param maxCodeLen the maximum code length
1012      * @return the resulting code lengths
1013      */
1014     private static int[] makeHuffmanCodes(int[] freq, int maxCodeLen) {
1015         int alphLen = freq.length;
1016 
1017         /*
1018            * We sort symbols by frequencies. Each value in freqTmp[] consists of: -- symbol value (9 bits) -- flag
1019            * "literal" set (1 bit, equal to 1) -- symbol frequency (21 bits)
1020            *
1021            * Since the frequencies use the upper bits, we can compare those values directly. A side effect is that no two
1022            * values will be considered as equal to each other, even for two symbols which occur with the same frequency;
1023            * this is harmless.
1024            */
1025         int[] freqTmp = new int[alphLen];
1026         for (int i = 0; i < alphLen; i++)
1027             freqTmp[i] = i + (1 << 9) + (freq[i] << 10);
1028         heapSort(freqTmp, 0, alphLen);
1029 
1030         /*
1031            * We skip the values with frequency zero; they will not take part in the tree construction. We handle
1032            * immediately the special case where only one symbol occurs.
1033            */
1034         int freqTmpPtr = 0;
1035         while (freqTmpPtr < alphLen && (freqTmp[freqTmpPtr] >>> 10) == 0)
1036             freqTmpPtr++;
1037         if (freqTmpPtr == alphLen) {
1038             /*
1039                 * No symbol occurs (degenerate case).
1040                 */
1041             return new int[alphLen];
1042         }
1043         if (freqTmpPtr == (alphLen - 1)) {
1044             /*
1045                 * Only onw symbol occurs.
1046                 */
1047             int[] clen = new int[alphLen];
1048             clen[freqTmp[freqTmpPtr] & 0x1FF] = 1;
1049             return clen;
1050         }
1051 
1052         /*
1053            * We use an additional FIFO in which we store the built tree nodes. All nodes make it to that fifo, and we do
1054            * not move elements inside it; rather, we shift the limit indexes. There are at most alphLen-1 nodes in the
1055            * tree (possibly less if some symbols are not used).
1056            */
1057         int[] fifo = new int[alphLen - 1];
1058         int fifoHead = 0, fifoRear = 0;
1059 
1060         /*
1061            * The tree is stored in an array. Each array element specifies a non-leaf node and consists of two 10-bit
1062            * values, for the left and right node children, respectively. Each such 10-bit value specifies either a leaf
1063            * symbol (9-bit value, 10th bit set) or the index for a non-leaf child node (9-bit index, 10th bit cleared).
1064            * The tree root index is stored in rootIndex when the tree is finished.
1065            *
1066            * The tree is complete (no missing child anywhere) and has at most alphLen leaves (since we skip symbols which
1067            * do not occur, the actual tree can be smaller); hence, it may have up to alphLen-1 non-leaf nodes.
1068            */
1069         int[] tree = new int[alphLen - 1];
1070         int treePtr = 0;
1071         int rootIndex;
1072 
1073         /*
1074            * Tree building uses the classical Huffman code construction algorithm. We have two queues, the one in
1075            * freqTmp[] (the yet unprocessed leaves) and the one in fifo[] (the built subtrees). At each step, we take the
1076            * two least frequent elements (not necessarily one from each queue) and assemble them into a new subtree, which
1077            * we insert in the fifo. The algorithm ends when freqTmp[] is empty (all symbols attached) and fifo[] contains
1078            * a single subtree (which is the complete tree).
1079            */
1080         for (; ; ) {
1081             if (freqTmpPtr == alphLen && fifoRear == (fifoHead + 1)) {
1082                 rootIndex = fifo[fifoHead] & 0x1FF;
1083                 break;
1084             }
1085             int n0, n1;
1086             if (fifoRear == fifoHead) {
1087                 n0 = freqTmp[freqTmpPtr++];
1088                 n1 = freqTmp[freqTmpPtr++];
1089             } else if (freqTmpPtr == alphLen) {
1090                 n0 = fifo[fifoHead++];
1091                 n1 = fifo[fifoHead++];
1092             } else {
1093                 int f = fifo[fifoHead];
1094                 int q = freqTmp[freqTmpPtr];
1095                 if (f < q) {
1096                     n0 = f;
1097                     fifoHead++;
1098                 } else {
1099                     n0 = q;
1100                     freqTmpPtr++;
1101                 }
1102                 if (fifoHead == fifoRear) {
1103                     n1 = freqTmp[freqTmpPtr++];
1104                 } else if (freqTmpPtr == alphLen) {
1105                     n1 = fifo[fifoHead++];
1106                 } else {
1107                     f = fifo[fifoHead];
1108                     q = freqTmp[freqTmpPtr];
1109                     if (f < q) {
1110                         n1 = f;
1111                         fifoHead++;
1112                     } else {
1113                         n1 = q;
1114                         freqTmpPtr++;
1115                     }
1116                 }
1117             }
1118             int ni = (n0 & ~0x3FF) + (n1 & ~0x3FF) + treePtr;
1119             fifo[fifoRear++] = ni;
1120             int nv = (n0 & 0x3FF) + ((n1 & 0x3FF) << 10);
1121             tree[treePtr++] = nv;
1122         }
1123 
1124         /*
1125            * Now, we gather the count for each code length, and the number of overdeep leaves which must be relocated.
1126            */
1127         int[] blCount = new int[maxCodeLen + 1];
1128         int overdeep = getCodeLengths(tree, rootIndex, 0, blCount, maxCodeLen);
1129 
1130         /*
1131            * We relocate the overdeep leaves.
1132            *
1133            * "dpi" contains the code length where we find nodes to demote (i.e. for which we lengthen the code) in order
1134            * to find some room for a relocated leaf. That length is kept as long as possible, provided that it is strictly
1135            * lower than maxCodeLen, and that there remain codes with that length.
1136            */
1137         int dpi = maxCodeLen;
1138         while (overdeep-- > 0) {
1139             if (dpi == maxCodeLen) {
1140                 do {
1141                     dpi--;
1142                 }
1143                 while (blCount[dpi] == 0);
1144             }
1145             blCount[dpi]--;
1146             blCount[++dpi] += 2;
1147         }
1148 
1149         /*
1150            * We rebuild the code lengths. We just walk the symbols sorted by frequencies, and give them code lengths,
1151            * according to the counts in blCount[].
1152            */
1153         int[] codeLen = new int[alphLen];
1154         int p = 0;
1155         while ((freqTmp[p] >>> 10) == 0)
1156             p++;
1157         for (int bits = maxCodeLen; bits > 0; bits--) {
1158             for (int k = blCount[bits]; k > 0; k--) {
1159                 int sym = freqTmp[p++] & 0x1FF;
1160                 codeLen[sym] = bits;
1161             }
1162         }
1163 
1164         /*
1165            * We have finished: we computed the code lengths for all symbols. These code lengths are for an optimized (not
1166            * necessarily optimal) Huffman tree which fits in the allowed maximum code length.
1167            */
1168         return codeLen;
1169     }

1171     /**
1172      * Count the code lengths for the provided subtree; the counts are accumulated in the <code>blCount[]</code> array.
1173      * Non-leaf nodes at exactly the maximum depth are accounted as leaves with the maximum code length. The number of
1174      * overdeep leaves which must be relocated is returned.
1175      *
1176      * @param tree       the tree array
1177      * @param idx        the subtree root index (or leaf code)
1178      * @param depth      the subtree root depth
1179      * @param blCount    the code length count accumulator array
1180      * @param maxCodeLen the maximum code length
1181      * @return the number of overdeep leaves to relocate
1182      */
1183     private static int getCodeLengths(int[] tree, int idx, int depth, int[] blCount, int maxCodeLen) {
1184         if ((idx & 0x200) != 0) {
1185             if (depth > maxCodeLen) {
1186                 return 1;
1187             } else {
1188                 blCount[depth]++;
1189                 return 0;
1190             }
1191         }
1192 
1193         int s;
1194         if (depth == maxCodeLen) {
1195             blCount[maxCodeLen]++;
1196             s = -1;
1197         } else {
1198             s = 0;
1199         }
1200         int n = tree[idx];
1201         int l = n & 0x3FF;
1202         int r = (n >>> 10) & 0x3FF;
1203         s += getCodeLengths(tree, l, depth + 1, blCount, maxCodeLen);
1204         s += getCodeLengths(tree, r, depth + 1, blCount, maxCodeLen);
1205         return s;
1206     }

1238     /**
1239      * RLE-compress two Huffman trees (for the literal+length and distance alphabets, represented as arrays of code
1240      * lengths). This results in values from 0 to 18 (5 low bits), where values 16, 17 and 18 have extra bits (bit 5 and
1241      * beyond). The frequencies for the resulting values are also accumulated in the freq[] array.
1242      *
1243      * @param tree1    the first tree
1244      * @param tree1len the first tree actual length
1245      * @param tree2    the second tree
1246      * @param tree2len the second tree actual length
1247      * @param freq     an array which receives frequencies
1248      * @return the compressed trees
1249      */
1250     private static int[] compressTrees(int[] tree1, int tree1len, int[] tree2, int tree2len, int[] freq) {
1251         int inLen = tree1len + tree2len;
1252         int[] in = new int[inLen];
1253         System.arraycopy(tree1, 0, in, 0, tree1len);
1254         System.arraycopy(tree2, 0, in, tree1len, tree2len);
1255 
1256         int ptr = 0;
1257         int[] ct = new int[inLen];
1258         int ctPtr = 0;
1259         while (ptr < inLen) {
1260             int v = in[ptr++];
1261             if (v == 0) {
1262                 int r = 1;
1263                 while (r < 138 && ptr < inLen) {
1264                     if (in[ptr] != 0)
1265                         break;
1266                     r++;
1267                     ptr++;
1268                 }
1269                 switch (r) {
1270                     case 1:
1271                         ct[ctPtr++] = 0;
1272                         freq[0]++;
1273                         break;
1274                     case 2:
1275                         ct[ctPtr++] = 0;
1276                         ct[ctPtr++] = 0;
1277                         freq[0] += 2;
1278                         break;
1279                     default:
1280                         if (r <= 10) {
1281                             ct[ctPtr++] = 17 + ((r - 3) << 5);
1282                             freq[17]++;
1283                         } else {
1284                             ct[ctPtr++] = 18 + ((r - 11) << 5);
1285                             freq[18]++;
1286                         }
1287                         break;
1288                 }
1289             } else {
1290                 int r = 0;
1291                 while (r < 6 && ptr < inLen) {
1292                     if (in[ptr] != v)
1293                         break;
1294                     r++;
1295                     ptr++;
1296                 }
1297                 ct[ctPtr++] = v;
1298                 freq[v]++;
1299                 switch (r) {
1300                     case 0:
1301                         break;
1302                     case 1:
1303                         ct[ctPtr++] = v;
1304                         freq[v]++;
1305                         break;
1306                     case 2:
1307                         ct[ctPtr++] = v;
1308                         ct[ctPtr++] = v;

1313                         freq[16]++;

1317         }

1319         int[] res = new int[ctPtr];
1320         System.arraycopy(ct, 0, res, 0, ctPtr);

1332       * The LZ77 code assembles symbols in the buffer[] array. We compute appropriate Huffman trees, and then produce the
1333       * compressed blocks. We first get all frequencies, compute the dynamic Huffman trees, and deduce the compressed
1334       * block size. We also use the frequencies to know what size we would get with the fixed Huffman trees. We compare
1335       * these two sizes with each other, and with the size for uncompressed blocks; we will use whichever method yields
1336       * the smallest size.
1337       *
1338       * When producing an uncompressed block, we need to rebuild the uncompressed data. The ucBuffer[] array contains the
1339       * first uncompressed bytes for this block; the subsequent bytes can be deduced by using the buffered symbols.
1340       *
1341       * Technically, we could use the same technique for type 2 blocks, in order to know whether a given short sequence
1342       * is worth copying. But this is tricky because if we decide to switch symbols at that time, then the Huffman codes
1343       * themselves have been computed with "wrong" frequencies. Right now, we filter out those sequences heuristically in
1344       * the LZ77 stage.
1345       */
1346 
1347     /**
1348      * Write out some bits (least significant bit exits first).
1349      *
1350      * @param val the bit values
1351      * @param num the number of bits (possibly 0)
1352      * @throws IOException on I/O error with the transport stream
1353      */
1354     private void writeBits(int val, int num) throws IOException {
1355         if (num == 0)
1356             return;
1357         for (; ; ) {
1358             int fs = 8 - outPtr;
1359             int v = outByte | (val << outPtr);
1360             if (fs > num) {
1361                 outByte = v;
1362                 outPtr += num;
1363                 return;
1364             } else if (fs == num) {
1365                 outBuf[outBufPtr++] = (byte) v;
1366                 if (outBufPtr == outBuf.length) {
1367                     out.write(outBuf);
1368                     outBufPtr = 0;
1369                 }
1370                 outByte = 0;
1371                 outPtr = 0;
1372                 return;
1373             }
1374             outBuf[outBufPtr++] = (byte) v;
1375             if (outBufPtr == outBuf.length) {
1376                 out.write(outBuf);
1377                 outBufPtr = 0;
1378             }
1379             val >>>= fs;
1380             num -= fs;
1381             outByte = 0;
1382             outPtr = 0;
1383         }
1384     }

1398     /**
1399      * End the current block and compress it. The <code>bufferPtr</code> field value may be incorrect; the value
1400      * provided in the <code>sbPtr</code> parameter must be used instead. This method resets the <code>bufferPtr</code>
1401      * field to 0.
1402      *
1403      * @param fin   <code>true</code> for the final block
1404      * @param sbPtr the correct value for <code>bufferPtr</code>
1405      * @throws IOException on I/O error with the transport stream
1406      */
1407     private void endBlock(boolean fin, int sbPtr) throws IOException {
1408         if (noOutput) {
1409             bufferPtr = 0;
1410             return;
1411         }
1412 
1413         int[] sb = buffer;
1414 
1415         int[] freqLit = new int[286];
1416         int[] freqDist = new int[30];
1417 
1418         /*
1419            * Do not forget the EOB symbol.
1420            */
1421         freqLit[256] = 1;
1422 
1423         /*
1424            * Get the uncompressed data length; also, gather frequencies.
1425            */
1426         int csU = 0;
1427         for (int i = 0; i < sbPtr; i++) {
1428             int val = sb[i];
1429             int sym = val & 0x1FF;
1430             freqLit[sym]++;
1431             if (sym < 256) {
1432                 csU += 8;
1433                 continue;
1434             }
1435             csU += 8 * (LENGTH[sym - 257] + ((val >>> 9) & 0x1F));
1436             int dist = (val >>> 14) & 0x1F;
1437             freqDist[dist]++;
1438         }
1439 
1440         /*
1441            * Adjust csU to account for the header bytes. Also, save the uncompressed data length (in bytes) in uDataLen.
1442            */
1443         int csUextra = 0;
1444         for (int t = 0; t < csU; t += 65535) {
1445             if (t == 0) {
1446                 if (outPtr > 5) {
1447                     csUextra = 48 - outPtr;
1448                 } else {
1449                     csUextra = 40 - outPtr;
1450                 }
1451             } else {
1452                 csUextra += 40;
1453             }
1454         }
1455         int uDataLen = (csU >>> 3);
1456         csU += csUextra;
1457 
1458         /*
1459            * Compute the dynamic Huffman codes and get the lengths for dynamic and fixed codes.
1460            */
1461         Huff huff = new Huff(freqLit, freqDist);
1462         int csD = huff.getDynamicBitLength();
1463         int csF = huff.getFixedBitLength();
1464 
1465         /*
1466            * We now have the bit lengths for uncompressed blocks (csU), fixed Huffman codes (csF) and dynamic Huffman
1467            * codes (csD). We use the smallest. On equality, we prefer uncompressed blocks over Huffman codes, and fixed
1468            * Huffman codes over dynamic Huffman codes.
1469            */
1470         if (csU <= csF && csU <= csD) {
1471             writeBits(fin ? 1 : 0, 3);
1472             if (outPtr > 0)
1473                 writeBits(0, 8 - outPtr);
1474             writeBits(uDataLen | (~uDataLen << 16), 32);
1475             sendBuffered();
1476             out.write(ucBuffer, 0, uDataLen);
1477         } else if (csF <= csD) {
1478             /*
1479                 * Fixed Huffman codes.
1480                 */
1481 
1482             /*
1483                 * Block header (3 bits).
1484                 */
1485             writeBits(fin ? 3 : 2, 3);
1486 
1487             /*
1488                 * Now, write out the data.
1489                 */
1490             for (int i = 0; i < sbPtr; i++) {
1491                 int val = buffer[i];
1492                 int sym = val & 0x1FF;
1493                 if (sym < 256) {
1494                     writeBits(FIXED_LIT_CODE[sym], sym < 144 ? 8 : 9);
1495                     continue;
1496                 }
1497 
1498                 writeBits(FIXED_LIT_CODE[sym], sym < 280 ? 7 : 8);
1499                 int eLenNum = LENGTH_ENUM[sym - 257];
1500                 if (eLenNum > 0)
1501                     writeBits((val >>> 9) & 0x1F, eLenNum);
1502                 int dist = (val >>> 14) & 0x1F;
1503                 writeBits(FIXED_DIST_CODE[dist], 5);
1504                 int eDistNum = DIST_ENUM[dist];
1505                 if (eDistNum > 0)
1506                     writeBits(val >>> 19, eDistNum);
1507             }
1508 
1509             /*
1510                 * Write out the EOB.
1511                 */
1512             writeBits(FIXED_LIT_CODE[256], 7);
1513         } else {
1514             /*
1515                 * Dynamic Huffman codes.
1516                 */
1517             int[] litCode = huff.getLitCode();
1518             int[] litCodeLen = huff.getLitCodeLen();
1519             int[] distCode = huff.getDistCode();
1520             int[] distCodeLen = huff.getDistCodeLen();
1521             int[] compTrees = huff.getCompTrees();
1522             int compTreesLen = compTrees.length;
1523             int[] ctCode = huff.getCTCode();
1524             int[] ctCodeLen = huff.getCTCodeLen();
1525             int[] permCT = huff.getPermCT();
1526             int permCTLen = permCT.length;
1527 
1528             /*
1529                 * Block header (3 bits).
1530                 */
1531             writeBits(fin ? 5 : 4, 3);
1532 
1533             /*
1534                 * The tree lengths.
1535                 */
1536             writeBits(litCode.length - 257, 5);
1537             writeBits(distCode.length - 1, 5);
1538             writeBits(permCTLen - 4, 4);
1539 
1540             /*
1541                 * The CT tree.
1542                 */
1543             for (int i = 0; i < permCTLen; i++)
1544                 writeBits(permCT[i], 3);
1545 
1546             /*
1547                 * The two compressed trees.
1548                 */
1549             for (int i = 0; i < compTreesLen; i++) {
1550                 int v = compTrees[i];
1551                 int s = v & 0x1F;
1552                 writeBits(ctCode[s], ctCodeLen[s]);
1553                 int ebits;
1554                 switch (s) {
1555                     case 16:
1556                         ebits = 2;
1557                         break;
1558                     case 17:
1559                         ebits = 3;
1560                         break;
1561                     case 18:
1562                         ebits = 7;
1563                         break;
1564                     default:
1565                         continue;
1566                 }
1567                 writeBits((v >>> 5), ebits);
1568             }
1569 
1570             /*
1571                 * Now, write out the data.
1572                 */
1573             for (int i = 0; i < sbPtr; i++) {
1574                 int val = buffer[i];
1575                 int sym = val & 0x1FF;
1576                 writeBits(litCode[sym], litCodeLen[sym]);
1577                 if (sym < 256)
1578                     continue;
1579                 int eLenNum = LENGTH_ENUM[sym - 257];
1580                 if (eLenNum > 0)
1581                     writeBits((val >>> 9) & 0x1F, eLenNum);
1582                 int dist = (val >>> 14) & 0x1F;
1583                 writeBits(distCode[dist], distCodeLen[dist]);
1584                 int eDistNum = DIST_ENUM[dist];
1585                 if (eDistNum > 0)
1586                     writeBits(val >>> 19, eDistNum);
1587             }
1588 
1589             /*
1590                 * Write out the EOB.
1591                 */
1592             writeBits(litCode[256], litCodeLen[256]);
1593         }
1594 
1595         sendBuffered();
1596         bufferPtr = 0;
1597 
1598         /*
1599            * Adjust ucBuffer[]. Some data has been processed, but some may remain (at most 258 bytes, corresponding to the
1600            * currently matched sequence). We must take care: the buffer is circular. We use the fact that the buffer is
1601            * more than twice as large than the maximum amout of data we move around.
1602            */
1603         int uLen = ucBuffer.length;
1604         int uRealLen = ucBufferPtr;
1605         while (uRealLen < uDataLen)
1606             uRealLen += uLen;
1607         if (uDataLen < uRealLen) {
1608             int tm = uRealLen - uDataLen;
1609 
1610             /* DEBUG */
1611             if (tm > 258)
1612                 throw new Error("too much data: " + tm);
1613 
1614             if (tm <= ucBufferPtr) {
1615                 System.arraycopy(ucBuffer, ucBufferPtr - tm, ucBuffer, 0, tm);
1616             } else {
1617                 int fpl = tm - ucBufferPtr;
1618                 System.arraycopy(ucBuffer, 0, ucBuffer, fpl, ucBufferPtr);
1619                 System.arraycopy(ucBuffer, uLen - fpl, ucBuffer, 0, fpl);
1620             }
1621         }
1622         ucBufferPtr = uRealLen - uDataLen;
1623     }

1638         /**
1639          * Build the instance with the provided frequencies for the literal+length and the distance alphabets. The first
1640          * frequency array MUST include the value 1 for the EOB symbol (value 256).
1641          *
1642          * @param freqLit  the literal+length frequencies
1643          * @param freqDist the distance frequencies
1644          */

1646             csD = 17;
1647             csF = 3;
1648

1650             distCodeLen = makeHuffmanCodes(freqDist, 15);

1655                 elen = (i >= 257) ? LENGTH_ENUM[i - 257] : 0;

1683                 * Compute the Huffman tree for the RLE-compressed trees.
1684                 */
1685             ctCodeLen = makeHuffmanCodes(freqCT, 7);
1686             ctCode = makeCanonicalHuff(ctCodeLen, 7);
1687             for (int i = 0; i < 19; i++) {
1688                 int ccl = ctCodeLen[i];
1689                 switch (i) {
1690                     case 16:
1691                         ccl += 2;
1692                         break;
1693                     case 17:
1694                         ccl += 3;
1695                         break;
1696                     case 18:
1697                         ccl += 7;
1698                         break;
1699                 }

1701             }
1702 
1703             /*
1704                 * Compute the permuted tree for the RLE-compressed trees, and its minimal length.
1705                 */

1708             for (int i = 0; i < 19; i++) {

1711                     permCTLen = i + 1;
1712                 permCTtmp[i] = len;
1713             }
1714             permCT = new int[permCTLen];
1715             System.arraycopy(permCTtmp, 0, permCT, 0, permCTLen);
1716             csD += 3 * permCTLen;
1717         }

1811     /*
1812       * Flush handling.
1813       *
1814       * We need some handling for the end of stream. If we have some buffered data, then we may just end the current
1815       * block with the "final" flag set; otherwise, we need a new empty block to set that flag.
1816       *
1817       * We also implement two flush modes. The "partial flush" mimics what zlib does with Z_PARTIAL_FLUSH. Recent
1818       * versions of zlib deprecate that option, and do not document it any more, but it is needed for the implementation
1819       * of some protocols, e.g. OpenSSH. With this flush mode, we add one or two empty blocks (with fixed Huffman trees),
1820       * in order to make sure that the peer has enough compressed data to decompress all meaningful bytes. Whether we
1821       * need to add one or two blocks depends on a computation, which hinges on the idea that zlib uses 9 bits of
1822       * lookahead.
1823       *
1824       * The "sync flush" terminates the current block (if any) and appends an empty "uncompressed data" block. That block
1825       * includes automatic byte alignment, and ends with the four-byte sequence 00 00 FF FF. A common convention is _not
1826       * to include_ that four-byte sequence, in which case the receiver is responsible for adding them. PPP uses this
1827       * convention (see RFC 1979). We implement this mode, using a parameter flag to decide whether the four-byte
1828       * sequence must be included or not.
1829       */

1967     /**
1968      * Build the canonical Huffman codes, given the length of each
1969      * code. <code>null</code> is returned if the code is not
1970      * correct. The returned array is trimmed to its minimal size
1971      * (trailing codes which do not occur are removed). The codes
1972      * are "reversed" (first bit is least significant).
1973      *
1974      * @param codeLen    the code lengths
1975      * @param maxCodeLen the maximum code length
1976      * @return the codes, or <code>null</code>
1977      */
1978     static int[] makeCanonicalHuff(int[] codeLen, int maxCodeLen) {
1979         int alphLen = codeLen.length;
1980         int actualAlphLen = 0;
1981 
1982         /*
1983            * Compute the number of codes for each length
1984            * (by convention, there is no code of length 0).
1985            */
1986         int[] blCount = new int[maxCodeLen + 1];
1987         for (int n = 0; n < alphLen; n++) {
1988             int len = codeLen[n];
1989             if (len < 0 || len > maxCodeLen)
1990                 return null;
1991             if (len > 0) {
1992                 actualAlphLen = n + 1;
1993                 blCount[len]++;
1994             }
1995         }

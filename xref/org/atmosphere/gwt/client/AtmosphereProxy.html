84      protected void initialize() {
85          if (WindowSocket.exists(Window.current(), EVENT_SOCKET)) {
86              throw new IllegalStateException("Only one AtmosphereProxy instance is allowed per window");
87          }
88          
89          eventSocket = new WindowSocket();
90          eventSocket.addHandler(new WindowSocket.MessageHandler() {
91            @Override
92            public void onMessage(Window source, String message) {
93                  AtmosphereProxyEvent event = deserialize(message);
94                  onEvent(source, event, message);
95          }});
96          eventSocket.bind(EVENT_SOCKET);
97          Window current = Window.current();
98          
99          if (current.opener() != null && current != current.opener() 
100                 && WindowSocket.exists(current.opener(), EVENT_SOCKET)) {
101             logger.info("Connecting new client window");
102             parent = current.opener();
103             dispatchEvent(parent, event(EventType.ANNOUNCE_NEW_CHILD));
104         } else {
105             logger.info("Starting master connection");
106             masterConnection = new AtmosphereClient(url, serializer, masterListener);
107             masterConnection.start();
108             parent = null;
109         }
110         
111         com.google.gwt.user.client.Window.addWindowClosingHandler(new ClosingHandler() {
112             @Override
113             public void onWindowClosing(ClosingEvent event) {
114                 AtmosphereProxy.this.onWindowClosing(event);
115             }
116         });
117     }

119     protected void onWindowClosing(ClosingEvent event) {
120         if (parent != null) {
121             dispatchEvent(parent, event(EventType.ANNOUNCE_CHILD_DEATH));
122         }
123         if (windowList.size() > 0) {
124             Window fosterParent = parent != null ? parent : windowList.get(0);
125             JSONArray arr = new JSONArray();
126             List<Window> orphans;
127             if (parent == fosterParent) {
128                 orphans = windowList;
129             } else {
130                 orphans = windowList.subList(1, windowList.size());
131             }
132             int i = 0;
133             for (Window w : orphans) {
134                 arr.set(i++, new JSONObject(w));
135             }
136             if (arr.size() > 0) {
137                 fosterParent.set("orphans", arr.getJavaScriptObject());
138             }
139             if (masterConnection != null) {
140                 dispatchEvent(fosterParent, event(EventType.ELECT_MASTER));
141                 masterConnection.stop();
142                 masterConnection = null;
143             } else if (arr.size() > 0){
144                 dispatchEvent(fosterParent, event(EventType.ADOPT_ORPHANS));
145             }
146         }
147     }

149     protected void onEvent(Window source, AtmosphereProxyEvent event, String rawEvent) {
150         switch (event.getEventType()) {
151             // do not bubble these events
152             // they are specificly meant for us
153             case ANNOUNCE_NEW_CHILD:
154                 logger.fine("Adding new child");
155                 windowList.add(source);
156                 return;
157             case ANNOUNCE_CHILD_DEATH:
158                 logger.fine("Removing child");
159                 windowList.remove(source);
160                 return;
161             case ANNOUNCE_NEW_PARENT:
162                 logger.fine("Set new parent");
163                 parent = source;
164                 return;
165             case ELECT_MASTER:
166                 logger.fine("Become new master");
167                 parent = null;
168                 masterConnection = new AtmosphereClient(url, serializer, masterListener);
169                 // same code for adoption
170             case ADOPT_ORPHANS:
171                 JavaScriptObject orphanArray = Window.current().getObject("orphans");
172                 if (orphanArray != null) {
173                     JSONArray arr = new JSONArray(orphanArray);
174                     // adopt orphans
175                     logger.fine("Adopting " + arr.size() + " orphans");
176                     for (int i=0; i < arr.size(); i++) {
177                         final Window w = (Window)((JSONObject)arr.get(i)).getJavaScriptObject();
178                         windowList.add(w);
179                         Scheduler.get().scheduleDeferred(new Scheduler.ScheduledCommand() {
180                             @Override
181                             public void execute() {
182                                 dispatchEvent(w, event(EventType.ANNOUNCE_NEW_PARENT));
183                             }
184                         });
185                     }
186                     Window.current().remove("orphans");
187                 }
188                 if (masterConnection != null) {
189                     masterConnection.start();
190                 }
191                 return;
192                 
193             // these events need to dispatched to our children as well as to the client listener
194             // they originate from the master connection and need to be propagated top-down
195             case ON_CONNECTED:
196                 clientListener.onConnected(-1, -1);
197                 break;
198             case ON_BEFORE_DISCONNECTED:
199                 clientListener.onBeforeDisconnected();
200                 break;
201             case ON_DISCONNECTED:
202                 clientListener.onDisconnected();
203                 break;
204             case ON_ERROR:
205                 clientListener.onError(new AtmosphereClientException((String)event.getData()), true);
206                 break;
207             case ON_HEARTBEAT:
208                 clientListener.onHeartbeat();
209                 break;
210             case ON_REFRESH:
211                 clientListener.onRefresh();
212                 break;
213             case ON_AFTER_REFRESH:
214                 clientListener.onAfterRefresh();
215                 break;
216             case ON_MESSAGE:
217                 clientListener.onMessage(Collections.singletonList(event.getData()));
218                 break;
219                 
220             // these events originate from the client and are meant for the master connection
221             case POST:
222                 if (masterConnection != null) {
223                     masterConnection.post(event.getData());
224                 } else if (parent != null) {
225                     // we are not the master propagate up
226                     dispatchRawEvent(parent, rawEvent);
227                 } else {
228                     throw new IllegalStateException("Failed to find master connection for post");
229                 }
230                 return;
231             case BROADCAST:
232                 if (masterConnection != null) {
233                     masterConnection.broadcast(event.getData());
234                 } else if (parent != null) {
235                     // we are not the master propagate up
236                     dispatchRawEvent(parent, rawEvent);
237                 } else {
238                     throw new IllegalStateException("Failed to find master connection for broadcast");
239                 }
240                 return;
241         }
242         // propagate message to children
243         dispatchRawEvent(rawEvent);
244     }

268     protected String serialize(Serializable object) {
269         try {
270             SerializationStreamWriter writer = new CommandToStringWriter(null);
271             writer.writeObject(object);
272             return writer.toString();
273         } catch (SerializationException ex) {
274             clientListener.onError(ex, true);
275             throw new RuntimeException("Failed to serialize object", ex);
276         }
277     }
278     
279     protected <T extends Serializable> T deserialize(String data) {
280         try {
281             SerializationStreamReader reader = ClientWriterFactory.createReader(data);
282             return (T) reader.readObject();
283         } catch (SerializationException ex) {
284             clientListener.onError(ex, true);
285             throw new RuntimeException("Failed to deserialize object", ex);
286         }
287     }

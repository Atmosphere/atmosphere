100     private static void maybeInit() {
101         if (!initialized) {
102             if (isFirefox()) {
103                 initWindowHandlers();
104             } else {
105                 initRootHandlers(Document.get().getBody());
106             }
107             initialized = true;
108         }
109     }

111     static String onBeforeUnload() {
112         eventBus.fireEvent(beforeUnloadEvent);
113         return null;

117         eventBus.fireEvent(unloadEvent);

120     private static boolean isFirefox() {
121         String ua = userAgent();
122         return ua.indexOf("gecko") != -1 && ua.indexOf("webkit") == -1;

135     private static void initWindowHandlers() {

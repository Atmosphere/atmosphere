46  /**
47   * This class uses a XmlHttpRequest and onreadystatechange events to process stream events.
48   * <p/>
49   * The main issue with this implementation is that GWT does not generate RECEIVING events from its XMLHTTPRequest. The
50   * implementation of XMLHTTPRequest included in this package overrides that behaviour.
51   * <p/>
52   * Another issues is that the memory required for the XMLHTTPRequest's responseText constantly grows so the server
53   * occasionally disconnects the client who then reestablishes the connection.
54   * <p/>
55   * The protocol for this transport is a '\n' separated transport messages. The different types of transport message are
56   * identified by the first character in the line as follows:
57   * <p/>
58   * ! A connection message followed by the heartbeat duration as an integer
59   * <p/>
60   * ? A clean server disconnection message
61   * <p/>
62   * # A heartbeat message
63   * <p/>
64   * * A padding message to cause the browser to start processing the stream
65   * <p/>
66   * ] A string message that needs unescaping
67   * <p/>
68   * | A string message that does not need unescaping
69   * <p/>
70   * [ A GWT serialized object
71   * <p/>
72   * R, r or f A GWT deRPC object
73   * <p/>
74   * string messages are escaped for '\\' and '\n' characters as '\n' is the message separator.
75   * <p/>
76   * GWT serialized object messages are escaped by GWT so do not need to be escaped by the transport
77   *
78   * @author Richard Zschech
79   */
80  public class HTTPRequestCometTransport extends StreamingProtocolTransport {

83       * Fix dropping RPC request when hitting ESC in firefox
84       * @see http://is.gd/dDeGQ

86      static {
87          Event.addNativePreviewHandler(new NativePreviewHandler() {
88              @Override
89              public void onPreviewNativeEvent(NativePreviewEvent e) {
90                  if (e.getTypeInt() == Event.getTypeInt(KeyDownEvent.getType().getName())) {
91                      NativeEvent nativeEvent = e.getNativeEvent();
92                      if (nativeEvent.getKeyCode() == KeyCodes.KEY_ESCAPE) {
93                          nativeEvent.preventDefault();
94                      }
95                  }
96              }
97          });
98      }

100     private final static int POLLING_INTERVAL = 250;
101 
102     private XMLHttpRequest xmlHttpRequest;
103     private Timer pollingTimer = new Timer() {
104         @Override
105         public void run() {
106             if (xmlHttpRequest != null && read < xmlHttpRequest.getResponseText().length()) {
107                 onReceiving(xmlHttpRequest.getStatus(), xmlHttpRequest.getResponseText());
108             }
109         }
110     };
111 
112     @Override
113     public void connect(int connectionCount) {
114         init();
115         xmlHttpRequest = XMLHttpRequest.create();
116         try {
117             xmlHttpRequest.open("GET", getUrl(connectionCount));
118             xmlHttpRequest.setRequestHeader("Accept", "application/comet");
119             xmlHttpRequest.setOnReadyStateChange(new ReadyStateChangeHandler() {
120                 @Override
121                 public void onReadyStateChange(XMLHttpRequest request) {
122                     if (!aborted) {
123                         switch (request.getReadyState()) {
124                             case XMLHttpRequest.LOADING:
125                                 onReceiving(request.getStatus(), request.getResponseText());
126                                 if (needPolling()) {
127                                     pollingTimer.scheduleRepeating(POLLING_INTERVAL);
128                                 }
129                                 break;
130                             case XMLHttpRequest.DONE:
131                                 onLoaded(request.getStatus(), request.getResponseText());
132                                 pollingTimer.cancel();
133                                 break;
134                         }
135                     } else {
136                         request.clearOnReadyStateChange();
137                         if (request.getReadyState() != XMLHttpRequest.DONE) {
138                             request.abort();
139                         }
140                     }
141                 }
142             });
143             xmlHttpRequest.send();
144         } catch (JavaScriptException e) {
145             if (xmlHttpRequest != null) {
146                 xmlHttpRequest.abort();
147                 xmlHttpRequest = null;
148             }
149             listener.onError(new RequestException(e.getMessage()), false);
150         }
151     }
152 
153     @Override
154     public void disconnect() {
155         aborted = true;
156         expectingDisconnection = true;
157         super.disconnect();
158         if (xmlHttpRequest != null) {
159             if (xmlHttpRequest.getReadyState() >= XMLHttpRequest.HEADERS_RECEIVED) {
160 //                if readystate >= HEADERS_RECEIVED we can abort otherwise wait for this in onReadyStateChange
161                 xmlHttpRequest.clearOnReadyStateChange();
162                 if (xmlHttpRequest.getReadyState() != XMLHttpRequest.DONE) {
163                     listener.onDisconnected();
164                     xmlHttpRequest.abort();
165                 }
166                 xmlHttpRequest = null;
167             } else {
168                 new Timer() {
169                     XMLHttpRequest r = xmlHttpRequest;
170 
171                     @Override
172                     public void run() {
173                         r.clearOnReadyStateChange();
174                         if (r.getReadyState() != XMLHttpRequest.DONE
175                                 && r.getReadyState() != XMLHttpRequest.UNSENT) {
176                             listener.onDisconnected();
177                             r.abort();
178                         }
179                         r = null;
180                     }
181                 }.schedule(5000);
182                 xmlHttpRequest = null;
183             }
184         }
185     }
186 
187     protected boolean needPolling() {
188         String ua = getUserAgent();
189         return ua.contains("opera");
190     }

201     private void onLoaded(int statusCode, String responseText) {
202         xmlHttpRequest.clearOnReadyStateChange();
203         xmlHttpRequest = null;
204         onReceiving(statusCode, responseText, false);
205     }
206 
207     private void onReceiving(int statusCode, String responseText) {
208         onReceiving(statusCode, responseText, true);
209     }

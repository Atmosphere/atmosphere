51      @Override
52      public void connect(int connectionCount) {
53          domain = getDomain(getDocumentDomain(), client.getUrl());
54  
55          StringBuilder html = new StringBuilder("<html>");
56          if (domain != null) {
57              html.append("<script>document.domain='").append(domain).append("'</script>");
58          }
59          html.append("<iframe src=''></iframe></html>");
60  
61          iframe = createIFrame(html.toString());
62  
63          expectingDisconnection = false;
64          String url = getUrl(connectionCount);
65          if (domain != null) {
66              url += "&d=" + domain;
67          }
68          iframe.setSrc(url);
69      }
70  
71      @Override
72      public void disconnect() {
73          // TODO this does not seem to close the connection immediately.
74          expectingDisconnection = true;
75          super.disconnect();
76          iframe.setSrc("");
77          // perhaps this will fix disconnecting properly
78          destroyIFrame();
79          iframe = null;
80          body = null;
81          if (connected) {
82              onDisconnected();
83          }
84          connected = false;
85      }

167     private void collect() {
168         if (body != null) {
169             NodeList<Node> childNodes = body.getChildNodes();
170             if (childNodes.getLength() > 1) {
171                 body.removeChild(childNodes.getItem(0));
172             }
173         }
174     }

176     @SuppressWarnings("unused")
177     private void onMessages(JsArrayString arguments) {
178         collect();
179         int length = arguments.length();
180         List messages = new ArrayList<Serializable>(length);
181         for (int i = 0; i < length; i++) {
182             String message = arguments.get(i);
183             switch (message.charAt(0)) {
184                 case ']':
185                     messages.add(message.substring(1));
186                     break;
187                 case '[':
188                 case 'R':
189                 case 'r':
190                 case 'f':
191                     try {
192                         messages.add(parse(message));
193                     } catch (SerializationException e) {
194                         listener.onError(e, true);
195                     }
196                     break;
197                 default:
198                     listener.onError(new AtmosphereClientException("Invalid message received: " + message), true);
199             }
200         }
201 
202         listener.onMessage(messages);
203     }

205     @SuppressWarnings("unused")
206     private void onConnected(int heartbeat, int connectionID) {
207         connected = true;
208         body = iframe.getContentDocument().getBody();
209         collect();
210         this.connectionId = connectionID;
211         listener.onConnected(heartbeat, connectionID);
212     }
213 
214     private void onDisconnected() {
215         connected = false;
216         body = null;
217         if (expectingDisconnection) {
218             listener.onDisconnected();
219         } else {
220             listener.onError(new AtmosphereClientException("Unexpected disconnection"), false);
221         }
222     }

51  /**
52   * This is the base class for the comet implementations
53   *
54   * @author Richard Zschech

56  public abstract class BaseCometTransport implements CometTransport {
57  
58      protected AtmosphereClient client;
59      protected AtmosphereListener listener;
60      protected ServerTransport serverTransport;
61      protected int connectionId;
62      private static final Logger logger = Logger.getLogger(BaseCometTransport.class.getName());
63  
64      @Override
65      public void disconnect() {
66          listener.onBeforeDisconnected();
67          getServerTransport().disconnect();
68      }
69  
70      @Override
71      public void post(Object message, AsyncCallback<Void> callback) {
72          getServerTransport().post(message, callback);
73      }
74  
75      @Override
76      public void post(List messages, AsyncCallback<Void> callback) {
77          getServerTransport().post(messages, callback);
78      }

80      @Override
81      public void broadcast(Object message) {
82          getServerTransport().broadcast(message);
83      }

87          getServerTransport().broadcast(messages);
88      }

90      @Override
91      public void initiate(AtmosphereClient client, AtmosphereListener listener) {
92          this.client = client;
93          this.listener = listener;
94      }

96      protected ServerTransport getServerTransport() {
97          if (serverTransport == null) {
98              serverTransport = new RPCServerTransport(client.getSerializer());
99          }
100         return serverTransport;
101     }

103     protected class RPCServerTransport extends ServerTransportProtocol {
104 
105         public RPCServerTransport(AtmosphereGWTSerializer serializer) {
106             super(serializer);
107         }
108 
109         @Override
110         void send(String message, final AsyncCallback<Void> callback) {
111             RequestBuilder request = new RequestBuilder(RequestBuilder.POST, serviceUrl());
112             try {
113                 logger.fine("Sending message to server: " + message);
114                 request.sendRequest(message, new RequestCallback() {
115                     @Override
116                     public void onResponseReceived(Request request, Response response) {
117                         // when a connection is abruptly closed (for instance when a user presses F5
118                         // the statuscode seems to be 0, the call could have arrived at the server though
119                         if (response.getStatusCode() != Response.SC_OK
120                                 && response.getStatusCode() != 0) {
121                             logger.log(Level.SEVERE, "Failed to send server message: [" + response.getStatusText() + "," + response.getStatusCode() + "]");
122                             callback.onFailure(new StatusCodeException(response.getStatusCode(), response.getStatusText()));
123                         } else {
124                             callback.onSuccess(null);
125                         }
126                     }
127 
128                     @Override
129                     public void onError(Request request, Throwable exception) {
130                         callback.onFailure(exception);
131                     }
132                 });
133             } catch (RequestException ex) {
134                 callback.onFailure(ex);
135             }
136         }
137 
138         protected String serviceUrl() {
139             int i = client.getUrl().indexOf('?');
140             String serviceUrl = (i > 0 ? client.getUrl().substring(0, i) : client.getUrl())
141                     + "?servertransport=rpcprotocol&connectionID=" + connectionId;
142             
143             serviceUrl = addUrlParameter(serviceUrl, Constants.CLIENT_SERIALZE_MODE_PARAMETER, client.getSerializer().getPushMode().name());
144             return serviceUrl;
145         }
146 
147     }
148 
149     protected Object parse(String message) throws SerializationException {
150         if (message == null || message.isEmpty()) {
151             return null;
152         }
153         AtmosphereGWTSerializer serializer = client.getSerializer();
154         if (serializer == null) {
155             throw new SerializationException("Can not deserialize message without serializer: " + message);
156         } else {
157             return serializer.deserialize(message);
158         }
159     }
160 
161 
162     public String getUrl(int connectionCount) {
163         String url = client.getUrl();
164         if (client.getSerializer() != null) {
165             url = addUrlParameter(url, Constants.CLIENT_DESERIALZE_MODE_PARAMETER, client.getSerializer().getMode().name());
166             url = addUrlParameter(url, Constants.CLIENT_SERIALZE_MODE_PARAMETER, client.getSerializer().getPushMode().name());
167         }
168         String className = getClass().getName();
169         className = className.substring(className.lastIndexOf('.') + 1);
170         String transport = className.substring(0, className.indexOf("CometTransport"));
171         return addUrlParameter(url, "t", Integer.toString((int) (Duration.currentTimeMillis() % Integer.MAX_VALUE), Character.MAX_RADIX).toUpperCase()
172                 + "&c=" + connectionCount
173                 + "&tr=" + transport);
174     }
175     
176     public String addUrlParameter(String url, String name, String value) {
177         return url + (url.contains("?") ? "&" : "?") + name + "=" + value;
178     }

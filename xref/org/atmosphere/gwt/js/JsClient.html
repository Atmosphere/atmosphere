28  public class JsClient implements EntryPoint {
29  
30      @Override
31      public void onModuleLoad() {
32          GWT.create(Client.class);
33          GWT.create(OnConnected.class);
34          GWT.create(OnBeforeDisconnected.class);
35          GWT.create(OnDisconnected.class);
36          GWT.create(OnError.class);
37          GWT.create(OnMessage.class);
38          GWT.create(OnHeartbeat.class);
39  
40          include(GWT.getModuleBaseURL() + "JSON.js");
41  
42          Scheduler.get().scheduleDeferred(new Scheduler.ScheduledCommand() {
43              @Override
44              public void execute() {
45                  onLoadImpl();
46              }
47          });
48      }
49  
50      protected void include(String source) {
51          ScriptElement el = Document.get().createScriptElement();
52          el.setType("text/javascript");
53          el.setSrc(source);
54          Document.get().getBody().appendChild(el);
55      }

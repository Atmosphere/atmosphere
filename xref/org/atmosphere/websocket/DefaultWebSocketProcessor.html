114     public WebSocketProcessor configure(AtmosphereConfig config) {
115         this.framework = config.framework();
116         this.webSocketProtocol = framework.getWebSocketProtocol();
117 
118         String s = framework.getAtmosphereConfig().getInitParameter(RECYCLE_ATMOSPHERE_REQUEST_RESPONSE);
119         if (s != null && Boolean.valueOf(s)) {
120             destroyable = true;
121         } else {
122             destroyable = false;
123         }
124 
125         s = framework.getAtmosphereConfig().getInitParameter(WEBSOCKET_PROTOCOL_EXECUTION);
126         if (s != null && Boolean.valueOf(s)) {
127             executeAsync = true;
128         } else {
129             executeAsync = false;
130         }
131 
132         s = framework.getAtmosphereConfig().getInitParameter(IN_MEMORY_STREAMING_BUFFER_SIZE);
133         if (s != null) {
134             byteBufferMaxSize = Integer.valueOf(s);
135             charBufferMaxSize = byteBufferMaxSize;
136         }
137 
138         if (executeAsync) {
139             asyncExecutor = ExecutorsFactory.getAsyncOperationExecutor(config, "WebSocket");
140         } else {
141             asyncExecutor = VoidExecutorService.VOID;
142         }
143 
144         scheduler = ExecutorsFactory.getScheduler(config);
145         optimizeMapping();
146 
147         closingTime = Long.valueOf(config.getInitParameter(ApplicationConfig.CLOSED_ATMOSPHERE_THINK_TIME, "0"));
148         invokeInterceptors = Boolean.valueOf(config.getInitParameter(INVOKE_ATMOSPHERE_INTERCEPTOR_ON_WEBSOCKET_MESSAGE, "true"));
149         config.startupHook(new AtmosphereConfig.StartupHook() {
150             @Override
151             public void started(final AtmosphereFramework framework) {
152                 if (AsynchronousProcessor.class.isAssignableFrom(framework.getAsyncSupport().getClass())) {
153                     asynchronousProcessor = AsynchronousProcessor.class.cast(framework.getAsyncSupport());
154                 } else {
155                     asynchronousProcessor = new AsynchronousProcessor(framework.getAtmosphereConfig()) {
156                         @Override
157                         public Action service(AtmosphereRequest req, AtmosphereResponse res) throws IOException, ServletException {
158                             return framework.getAsyncSupport().service(req, res);
159                         }
160                     };
161                 }
162             }
163         });
164         return this;
165     }

181     @Override
182     public final void open(final WebSocket webSocket, final AtmosphereRequest request, final AtmosphereResponse response) throws IOException {
183 
184         if (framework.isDestroyed()) return;
185 
186         // TODO: Fix this. Instead add an Interceptor.
187         if (framework.getAtmosphereConfig().handlers().isEmpty()) {
188             synchronized (framework) {
189                 if (handlers.isEmpty()) {
190                     logger.warn("No AtmosphereHandler or WebSocketHandler installed. Adding a default one.");
191                 }
192                 framework.addAtmosphereHandler(ROOT_MASTER, REFLECTOR_ATMOSPHEREHANDLER);
193             }
194         }
195 
196         request.headers(configureHeader(request)).setAttribute(WebSocket.WEBSOCKET_SUSPEND, true);
197 
198         AtmosphereResource r = framework.atmosphereFactory().create(framework.getAtmosphereConfig(),
199                 response,
200                 framework.getAsyncSupport());
201 
202         boolean cleanUpAfterDisconnect = false;
203         try {
204             request.setAttribute(INJECTED_ATMOSPHERE_RESOURCE, r);
205             request.setAttribute(SUSPENDED_ATMOSPHERE_RESOURCE_UUID, r.uuid());
206 
207             if (Utils.firefoxWebSocketEnabled(request)) {
208                 request.setAttribute("firefox", "true");
209             }
210 
211             AtmosphereResourceImpl.class.cast(r).webSocket(webSocket);
212             webSocket.resource(r);
213             webSocketProtocol.onOpen(webSocket);
214             WebSocketHandler proxy = null;
215             if (!handlers.isEmpty()) {
216                 WebSocketHandlerProxy handler = mapper.map(request, handlers);
217                 if (handler == null) {
218                     logger.debug("No WebSocketHandler maps request for {} with mapping {}", request.getRequestURI(), handlers);
219                     throw new AtmosphereMappingException("No AtmosphereHandler maps request for " + request.getRequestURI());
220                 }
221                 proxy = postProcessMapping(webSocket, request, handler);
222             }
223 
224             dispatch(webSocket, request, response);
225 
226             if (proxy != null) {
227                 webSocket.webSocketHandler(proxy).resource().suspend(-1);
228                 proxy.onOpen(webSocket);
229             }
230 
231             request.removeAttribute(INJECTED_ATMOSPHERE_RESOURCE);
232 
233             // Resource can be null if the client disconnect.
234             if (webSocket.resource() != null) {
235                 final Action action = ((AtmosphereResourceImpl) webSocket.resource()).action();
236                 if (action.timeout() != -1 && !framework.getAsyncSupport().getContainerName().contains("Netty")) {
237                     final AtomicReference<Future<?>> f = new AtomicReference();
238                     f.set(scheduler.scheduleAtFixedRate(new Runnable() {
239                         @Override
240                         public void run() {
241                             if (WebSocket.class.isAssignableFrom(webSocket.getClass())
242                                     && System.currentTimeMillis() - WebSocket.class.cast(webSocket).lastWriteTimeStampInMilliseconds() > action.timeout()) {
243                                 asynchronousProcessor.endRequest(((AtmosphereResourceImpl) webSocket.resource()), false);
244                                 f.get().cancel(true);
245                             }
246                         }
247                     }, action.timeout(), action.timeout(), TimeUnit.MILLISECONDS));
248                 }
249             } else {
250                 logger.warn("AtmosphereResource was null");
251                 cleanUpAfterDisconnect = true;
252             }
253             notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent("", CONNECT, webSocket));
254         } catch (AtmosphereMappingException ex) {
255             cleanUpAfterDisconnect = true;
256             throw ex;
257         } catch (IOException ex) {
258             cleanUpAfterDisconnect = true;
259             throw ex;
260         } catch (Exception ex) {
261             logger.trace("onOpen exception", ex);
262             cleanUpAfterDisconnect = true;
263         } finally {
264             if (cleanUpAfterDisconnect) {
265                 logger.warn("Problem opening websocket for {}", r.uuid());
266                 framework.atmosphereFactory().remove(r.uuid());
267 
268                 AtmosphereResourceEventImpl.class.cast(r.getAtmosphereResourceEvent()).setCancelled(true);
269                 AsynchronousProcessor.class.cast(framework.getAsyncSupport()).completeLifecycle(r, true);
270             }
271             webSocket.shiftAttributes();
272         }
273     }

275     protected WebSocketHandler postProcessMapping(WebSocket webSocket, AtmosphereRequest request, WebSocketHandlerProxy w) {
276         WebSocketHandlerProxy p = null;
277         String path = w.path();
278         if (wildcardMapping()) {
279             String pathInfo = null;
280             try {
281                 pathInfo = request.getPathInfo();
282             } catch (IllegalStateException ex) {
283                 // http://java.net/jira/browse/GRIZZLY-1301
284             }
285 
286             if (pathInfo != null) {
287                 path = request.getServletPath() + pathInfo;
288             } else {
289                 path = request.getServletPath();
290             }
291 
292             if (path == null || path.isEmpty()) {
293                 path = "/";
294             }
295 
296             synchronized (handlers) {
297                 p = handlers.get(path);
298                 if (p == null) {
299                     // AtmosphereHandlerService
300                     WebSocketHandlerService a = w.proxied.getClass().getAnnotation(WebSocketHandlerService.class);
301                     if (a != null) {
302                         String targetPath = a.path();
303                         if (targetPath.indexOf("{") != -1 && targetPath.indexOf("}") != -1) {
304                             try {
305                                 boolean singleton = w.proxied.getClass().getAnnotation(Singleton.class) != null;
306 
307                                 if (!singleton) {
308                                     w = new WebSocketHandlerProxy(a.broadcaster(),
309                                             framework.newClassInstance(WebSocketHandler.class, w.proxied.getClass()));
310                                 }
311 
312                                 registerWebSocketHandler(path, new WebSocketHandlerProxy(a.broadcaster(), w));
313                                 request.localAttributes().put(Named.class.getName(), path.substring(targetPath.indexOf("{")));
314 
315                                 p = handlers.get(path);
316                             } catch (Throwable e) {
317                                 logger.warn("Unable to create WebSocketHandler", e);
318                             }
319                         }
320                     }
321                 }
322             }
323         }
324 
325         try {
326             webSocket.resource().setBroadcaster(AnnotationUtil.broadcaster(framework, p != null ? p.broadcasterClazz : w.broadcasterClazz, path));
327         } catch (Exception e) {
328             logger.error("", e);
329         }
330 
331         return p != null ? p : w;
332     }

334     private void dispatch(final WebSocket webSocket, List<AtmosphereRequest> list) {
335         if (list == null) return;
336 
337         for (final AtmosphereRequest r : list) {
338             if (r != null) {
339 
340                 asyncExecutor.execute(new Runnable() {
341                     @Override
342                     public void run() {
343                         AtmosphereResponse w = new AtmosphereResponseImpl(webSocket, r, destroyable);
344                         try {
345                             dispatch(webSocket, r, w);
346                         } finally {
347                             r.destroy();
348                             w.destroy();
349                         }
350                     }
351                 });
352             }
353         }
354     }

356     private WebSocketHandlerProxy webSocketHandlerForMessage(WebSocket webSocket) {
357         AtmosphereResourceImpl impl = AtmosphereResourceImpl.class.cast(webSocket.resource());
358         if (impl != null) {
359             impl.getRequest(false).setAttribute(FrameworkConfig.WEBSOCKET_MESSAGE, "true");
360 
361             try {
362                 Utils.inject(impl);
363             } catch (IllegalAccessException e) {
364                 logger.warn("", e);
365             }
366 
367         }
368         return WebSocketHandlerProxy.class.cast(webSocket.webSocketHandler());
369     }

377     private void invokeInterceptors(WebSocketHandlerProxy webSocketHandler,
378                                     WebSocket webSocket, Object webSocketMessageAsBody, int offset, int length) throws IOException {
379 
380         AtmosphereResourceImpl resource = AtmosphereResourceImpl.class.cast(webSocket.resource());
381         if (resource == null) {
382             return;
383         }
384 
385         AtmosphereRequest request = resource.getRequest(false);
386         if (String.class.isAssignableFrom(webSocketMessageAsBody.getClass())) {
387             request.body(String.class.cast(webSocketMessageAsBody));
388         } else if (Reader.class.isAssignableFrom(webSocketMessageAsBody.getClass())) {
389             request.body(Reader.class.cast(webSocketMessageAsBody));
390         } else if (InputStream.class.isAssignableFrom(webSocketMessageAsBody.getClass())) {
391             request.body(InputStream.class.cast(webSocketMessageAsBody));
392         } else {
393             request.body(new ByteArrayInputStream((byte[]) webSocketMessageAsBody, offset, length));
394         }
395 
396         String path = webSocketHandler.proxied.getClass().isAnnotationPresent(WebSocketHandlerService.class) ?
397                 webSocketHandler.proxied.getClass().getAnnotation(WebSocketHandlerService.class).path() : "/";
398 
399         AtmosphereFramework.AtmosphereHandlerWrapper w = framework.getAtmosphereHandlers().get(framework.normalizePath(path));
400         List<AtmosphereInterceptor> l;
401         if (w == null) {
402             l = framework.interceptors();
403         } else {
404             l = w.interceptors;
405         }
406 
407         try {
408             // Globally defined
409             int tracing = 0;
410             Action a = asynchronousProcessor.invokeInterceptors(l, resource, tracing);
411             if (a.type() != Action.TYPE.CONTINUE && a.type() != Action.TYPE.SKIP_ATMOSPHEREHANDLER) {
412                 return;
413             }
414 
415             //Unit test mock the request and will throw NPE.
416             boolean skipAtmosphereHandler = request.getAttribute(SKIP_ATMOSPHEREHANDLER.name()) != null
417                     ? (Boolean) request.getAttribute(SKIP_ATMOSPHEREHANDLER.name()) : Boolean.FALSE;
418             if (!skipAtmosphereHandler) {
419                 try {
420                     if (String.class.isAssignableFrom(webSocketMessageAsBody.getClass())) {
421                         webSocketHandler.onTextMessage(webSocket, String.class.cast(webSocketMessageAsBody));
422                     } else if (Reader.class.isAssignableFrom(webSocketMessageAsBody.getClass())) {
423                         WebSocketStreamingHandler.class.cast(webSocketHandler.proxied).onTextStream(webSocket, Reader.class.cast(webSocketMessageAsBody));
424                     } else if (InputStream.class.isAssignableFrom(webSocketMessageAsBody.getClass())) {
425                         WebSocketStreamingHandler.class.cast(webSocketHandler.proxied()).onBinaryStream(webSocket, InputStream.class.cast(webSocketMessageAsBody));
426                     } else {
427                         webSocketHandler.onByteMessage(webSocket, (byte[]) webSocketMessageAsBody, offset, length);
428                     }
429                 } catch (IOException t) {
430                     resource.onThrowable(t);
431                     throw t;
432                 }
433             }
434             request.setAttribute(SKIP_ATMOSPHEREHANDLER.name(), Boolean.FALSE);
435         } finally {
436             asynchronousProcessor.postInterceptors(l, resource);
437         }
438     }

440     @Override
441     public void invokeWebSocketProtocol(final WebSocket webSocket, String webSocketMessage) {
442         WebSocketHandlerProxy webSocketHandler = webSocketHandlerForMessage(webSocket);
443 
444         if (webSocketHandler == null) {
445             if (!WebSocketProtocolStream.class.isAssignableFrom(webSocketProtocol.getClass())) {
446                 List<AtmosphereRequest> list = webSocketProtocol.onMessage(webSocket, webSocketMessage);
447                 dispatch(webSocket, list);
448             } else {
449                 logger.debug("The WebServer doesn't support streaming. Wrapping the message as stream.");
450                 invokeWebSocketProtocol(webSocket, new StringReader(webSocketMessage));
451                 return;
452             }
453         } else {
454             if (!WebSocketStreamingHandler.class.isAssignableFrom(webSocketHandler.proxied().getClass())) {
455                 try {
456                     if (invokeInterceptors) {
457                         invokeInterceptors(webSocketHandler, webSocket, webSocketMessage);
458                     } else {
459                         webSocketHandler.onTextMessage(webSocket, webSocketMessage);
460                     }
461                 } catch (Exception ex) {
462                     handleException(ex, webSocket, webSocketHandler);
463                 }
464             } else {
465                 logger.debug("The WebServer doesn't support streaming. Wrapping the message as stream.");
466                 invokeWebSocketProtocol(webSocket, new StringReader(webSocketMessage));
467                 return;
468             }
469         }
470         notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent(webSocketMessage, MESSAGE, webSocket));
471     }

534         }

571     private void handleException(Exception ex, WebSocket webSocket, WebSocketHandler webSocketHandler) {
572         logger.error("", ex);
573         AtmosphereResource r = webSocket.resource();
574         if (r != null) {
575             webSocketHandler.onError(webSocket, new WebSocketException(ex,
576                     new AtmosphereResponseImpl.Builder()
577                             .request(r != null ? AtmosphereResourceImpl.class.cast(r).getRequest(false) : null)
578                             .status(500)
579                             .statusMessage("Server Error").build()));
580         }
581     }

583     /**
584      * Dispatch to request/response to the {@link org.atmosphere.cpr.AsyncSupport} implementation as it was a normal HTTP request.
585      *
586      * @param request a {@link AtmosphereRequest}
587      * @param r       a {@link AtmosphereResponse}
588      */
589     public final void dispatch(WebSocket webSocket, final AtmosphereRequest request, final AtmosphereResponse r) {
590         if (request == null) return;
591 
592         try {
593             framework.doCometSupport(request, r);
594         } catch (Throwable e) {
595             logger.warn("Failed invoking AtmosphereFramework.doCometSupport()", e);
596             webSocketProtocol.onError(webSocket, new WebSocketException(e,
597                     new AtmosphereResponseImpl.Builder()
598                             .request(request)
599                             .status(500)
600                             .statusMessage("Server Error").build()));
601             return;
602         }
603 
604         if (r.getStatus() >= 400) {
605             webSocketProtocol.onError(webSocket, new WebSocketException("Status code higher or equal than 400", r));
606         }
607     }

609     @Override
610     public void close(final WebSocket webSocket, int closeCode) {
611         WebSocketHandler webSocketHandler = webSocket.webSocketHandler();
612         // A message might be in the process of being processed and the websocket gets closed. In that corner
613         // case the webSocket.resource will be set to false and that might cause NPE in some WebSocketProcol implementation
614         // We could potentially synchronize on webSocket but since it is a rare case, it is better to not synchronize.
615         // synchronized (webSocket) {
616 
617         final AtmosphereResourceImpl resource = (AtmosphereResourceImpl) webSocket.resource();
618 
619         if (resource == null) {
620             logger.trace("Already closed {}", webSocket);
621         } else {
622             final boolean allowedToClose = allowedCloseCode(closeCode);
623 
624             final AtmosphereRequest r = resource.getRequest(false);
625             final AtmosphereResponse s = resource.getResponse(false);
626             boolean ff = r.getAttribute("firefox") != null;
627             boolean completeLifecycle = true;
628             try {
629                 webSocketProtocol.onClose(webSocket);
630 
631                 if (webSocketHandler != null) {
632                     webSocketHandler.onClose(webSocket);
633                 }
634 
635                 logger.trace("About to close AtmosphereResource for {} with code {}", resource, closeCode);
636                 if (!resource.getAtmosphereResourceEvent().isClosedByClient() && !resource.getAtmosphereResourceEvent().isClosedByApplication() && !resource.isCancelled()) {
637                     // See https://github.com/Atmosphere/atmosphere/issues/1590
638                     // Better to call onDisconnect that onResume.
639                     if (allowedToClose) {
640                         if (ff || closingTime > 0) {
641                             completeLifecycle = false;
642                             logger.debug("Delaying closing operation for firefox and resource {}", resource.uuid());
643                             ExecutorsFactory.getScheduler(framework.getAtmosphereConfig()).schedule(new Callable<Object>() {
644                                 @Override
645                                 public Object call() throws Exception {
646                                     executeClose(webSocket, 1005);
647                                     finish(webSocket, resource, r, s, !allowedToClose);
648                                     return null;
649                                 }
650                             }, ff ? (closingTime == 0 ? 1000 : closingTime) : closingTime, TimeUnit.MILLISECONDS);
651                             resource.getAndSetPendingClose();
652                         } else {
653                             executeClose(webSocket, closeCode);
654                         }
655                     } else {
656                         logger.debug("Timeout {}", resource.uuid());
657                         asynchronousProcessor.endRequest(AtmosphereResourceImpl.class.cast(webSocket.resource()), false);
658                     }
659                 } else {
660                     logger.trace("Unable to properly complete {}", resource == null ? "null" : resource.uuid());
661                     completeLifecycle = false;
662                 }
663             } finally {
664                 if (completeLifecycle) {
665                     finish(webSocket, resource, r, s, !allowedToClose);
666                 }
667             }
668         }
669     }

671     // Highly bogus based on which I/O layer we are using.
672     private boolean allowedCloseCode(int closeCode) {
673         return closeCode <= 1001 || closeCode > 1004 ? true : false;
674     }

676     private void finish(WebSocket webSocket, AtmosphereResource resource, AtmosphereRequest r, AtmosphereResponse s, boolean closeWebSocket) {
677         // Don't take any risk in case something goes wrong and remove the associated resource.
678         framework.atmosphereFactory().remove(resource.uuid());
679         if (webSocket != null) {
680             try {
681                 r.setAttribute(WebSocket.CLEAN_CLOSE, Boolean.TRUE);
682                 webSocket.resource(null);
683 
684                 if (closeWebSocket) webSocket.close(s);
685             } catch (IOException e) {
686                 logger.trace("", e);
687             }
688         }
689 
690         if (r != null) {
691             r.destroy(true);
692         }
693 
694         if (s != null) {
695             s.destroy(true);
696         }
697     }

699     public void executeClose(WebSocket webSocket, int closeCode) {
700         AtmosphereResource r = webSocket.resource();
701 
702         boolean isClosedByClient = r == null ? true : r.getAtmosphereResourceEvent().isClosedByClient();
703         try {
704             if (r != null) {
705                 asynchronousProcessor.endRequest(AtmosphereResourceImpl.class.cast(r), true);
706             }
707         } finally {
708             if (!isClosedByClient) {
709                 notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent(closeCode, CLOSE, webSocket));
710             }
711         }
712     }

716         boolean shared = framework.isShareExecutorServices();
717         if (asyncExecutor != null && !shared) {
718             asyncExecutor.shutdown();
719         }
720 
721         if (scheduler != null && !shared) {
722             scheduler.shutdown();
723         }

726     @Override
727     public void notifyListener(WebSocket webSocket, WebSocketEventListener.WebSocketEvent event) {
728         AtmosphereResource resource = webSocket.resource();
729         if (resource == null) return;
730 
731         AtmosphereResourceImpl r = AtmosphereResourceImpl.class.cast(resource);
732         for (AtmosphereResourceEventListener l : r.atmosphereResourceEventListener()) {
733             if (WebSocketEventListener.class.isAssignableFrom(l.getClass())) {
734                 try {
735                     switch (event.type()) {
736                         case CONNECT:
737                             WebSocketEventListener.class.cast(l).onConnect(event);
738                             break;
739                         case DISCONNECT:
740                             WebSocketEventListener.class.cast(l).onDisconnect(event);
741                             onDisconnect(event, l);
742                             break;
743                         case CONTROL:
744                             WebSocketEventListener.class.cast(l).onControl(event);
745                             break;
746                         case MESSAGE:
747                             WebSocketEventListener.class.cast(l).onMessage(event);
748                             break;
749                         case HANDSHAKE:
750                             WebSocketEventListener.class.cast(l).onHandshake(event);
751                             break;
752                         case CLOSE:
753                             boolean isClosedByClient = r.getAtmosphereResourceEvent().isClosedByClient();
754                             l.onDisconnect(new AtmosphereResourceEventImpl(r, !isClosedByClient, false, isClosedByClient, null));
755                             onDisconnect(event, l);
756                             WebSocketEventListener.class.cast(l).onClose(event);
757                             break;
758                     }
759                 } catch (Throwable t) {
760                     logger.debug("Listener error {}", t);
761                     try {
762                         WebSocketEventListener.class.cast(l).onThrowable(new AtmosphereResourceEventImpl(r, false, false, t));
763                     } catch (Throwable t2) {
764                         logger.warn("Listener error {}", t2);
765                     }
766                 }
767             } else {
768                 switch (event.type()) {
769                     case CLOSE:
770                         boolean isClosedByClient = r.getAtmosphereResourceEvent().isClosedByClient();
771                         l.onDisconnect(new AtmosphereResourceEventImpl(r, !isClosedByClient, false, isClosedByClient, null));
772                         break;
773                 }
774             }
775         }
776     }

778     private void onDisconnect(WebSocketEvent event, AtmosphereResourceEventListener l) {
779         if (event.webSocket() != null && event.webSocket().resource() != null) {
780             framework.notifyDestroyed(event.webSocket().resource().uuid());
781         }
782         WebSocketEventListener.class.cast(l).onDisconnect(event);
783     }

785     public static final Map<String, String> configureHeader(AtmosphereRequest request) {
786         Map<String, String> headers = new HashMap<String, String>();
787 
788         Enumeration<String> e = request.getParameterNames();
789         String s;
790         while (e.hasMoreElements()) {
791             s = e.nextElement();
792             headers.put(s, request.getParameter(s));
793         }
794 
795         headers.put(HeaderConfig.X_ATMOSPHERE_TRANSPORT, HeaderConfig.WEBSOCKET_TRANSPORT);
796         return headers;
797     }

799     protected void dispatchStream(WebSocket webSocket, InputStream is) throws IOException {
800         int read = 0;
801         ByteBuffer bb = webSocket.bb;
802         try {
803             while (read > -1) {
804                 bb.position(bb.position() + read);
805                 if (bb.remaining() == 0) {
806                     bb = resizeByteBuffer(webSocket);
807                 }
808                 read = is.read(bb.array(), bb.position(), bb.remaining());
809             }
810             bb.flip();
811 
812             invokeWebSocketProtocol(webSocket, bb.array(), 0, bb.limit());
813         } finally {
814             bb.clear();
815         }
816     }

818     protected void dispatchReader(WebSocket webSocket, Reader r) throws IOException {
819         int read = 0;
820         CharBuffer cb = webSocket.cb;
821         try {
822             while (read > -1) {
823                 cb.position(cb.position() + read);
824                 if (cb.remaining() == 0) {
825                     cb = resizeCharBuffer(webSocket);
826                 }
827                 read = r.read(cb.array(), cb.position(), cb.remaining());
828             }
829             cb.flip();
830             invokeWebSocketProtocol(webSocket, cb.toString());
831         } finally {
832             cb.clear();
833         }
834     }

836     private ByteBuffer resizeByteBuffer(WebSocket webSocket) throws IOException {
837         int maxSize = byteBufferMaxSize;
838         ByteBuffer bb = webSocket.bb;
839         if (bb.limit() >= maxSize) {
840             throw new IOException("Message Buffer too small. Use " + StreamingHttpProtocol.class.getName() + " when streaming over websocket.");
841         }
842 
843         long newSize = bb.limit() * 2;
844         if (newSize > maxSize) {
845             newSize = maxSize;
846         }
847 
848         // Cast is safe. newSize < maxSize and maxSize is an int
849         ByteBuffer newBuffer = ByteBuffer.allocate((int) newSize);
850         bb.rewind();
851         newBuffer.put(bb);
852         webSocket.bb = newBuffer;
853         return newBuffer;
854     }

57  /**
58   * Like the {@link org.atmosphere.cpr.AsynchronousProcessor} class, this class is responsible for dispatching WebSocket request to the
59   * proper {@link org.atmosphere.websocket.WebSocket} implementation. This class can be extended in order to support any protocol
60   * running on top  websocket.
61   *
62   * @author Jeanfrancois Arcand
63   */
64  public class DefaultWebSocketProcessor implements WebSocketProcessor, Serializable {
65  
66      private static final Logger logger = LoggerFactory.getLogger(DefaultWebSocketProcessor.class);
67  
68      private final AtmosphereFramework framework;
69      private final WebSocket webSocket;
70      private final WebSocketProtocol webSocketProtocol;
71      private final AtomicBoolean loggedMsg = new AtomicBoolean(false);
72      private final boolean destroyable;
73      private final boolean executeAsync;
74      private final ExecutorService asyncExecutor;
75      private final ExecutorService voidExecutor;
76      private final ScheduledExecutorService scheduler = Executors.newScheduledThreadPool(Runtime.getRuntime().availableProcessors());
77  
78      public DefaultWebSocketProcessor(AtmosphereFramework framework, WebSocket webSocket, WebSocketProtocol webSocketProtocol) {
79          this.webSocket = webSocket;
80          this.framework = framework;
81          this.webSocketProtocol = webSocketProtocol;
82  
83          String s = framework.getAtmosphereConfig().getInitParameter(RECYCLE_ATMOSPHERE_REQUEST_RESPONSE);
84          if (s != null && Boolean.valueOf(s)) {
85              destroyable = true;
86          } else {
87              destroyable = false;
88          }
89  
90          s = framework.getAtmosphereConfig().getInitParameter(WEBSOCKET_PROTOCOL_EXECUTION);
91          if (s != null && Boolean.valueOf(s)) {
92              executeAsync = true;
93          } else {
94              executeAsync = false;
95          }
96          asyncExecutor = Executors.newCachedThreadPool();
97          voidExecutor = VoidExecutorService.VOID;
98      }

100     /**
101      * {@inheritDoc}
102      */
103     @Override
104     public final void open(final AtmosphereRequest request) throws IOException {
105         if (!loggedMsg.getAndSet(true)) {
106             logger.debug("Atmosphere detected WebSocket: {}", webSocket.getClass().getName());
107         }
108 
109         AtmosphereResponse wsr = new AtmosphereResponse(webSocket, request, destroyable);
110         request.headers(configureHeader(request)).setAttribute(WebSocket.WEBSOCKET_SUSPEND, true);
111 
112         AtmosphereResource r = AtmosphereResourceFactory.getDefault().create(framework.getAtmosphereConfig(),
113                 wsr,
114                 framework.getAsyncSupport());
115 
116         request.setAttribute(INJECTED_ATMOSPHERE_RESOURCE, r);
117         request.setAttribute(SUSPENDED_ATMOSPHERE_RESOURCE_UUID, r.uuid());
118 
119         webSocket.resource(r);
120         webSocketProtocol.onOpen(webSocket);
121 
122         dispatch(request, wsr);
123         request.removeAttribute(INJECTED_ATMOSPHERE_RESOURCE);
124 
125         if (webSocket.resource() != null) {
126             final AsynchronousProcessor.AsynchronousProcessorHook hook =
127                     new AsynchronousProcessor.AsynchronousProcessorHook((AtmosphereResourceImpl) webSocket.resource());
128             request.setAttribute(ASYNCHRONOUS_HOOK, hook);
129 
130             final Action action = ((AtmosphereResourceImpl) webSocket.resource()).action();
131             if (action.timeout() != -1 && !framework.getAsyncSupport().getContainerName().contains("Netty")) {
132                 final AtomicReference<Future<?>> f = new AtomicReference();
133                 f.set(scheduler.scheduleAtFixedRate(new Runnable() {
134                     @Override
135                     public void run() {
136                         if (WebSocket.class.isAssignableFrom(webSocket.getClass())
137                                 && System.currentTimeMillis() - WebSocket.class.cast(webSocket).lastWriteTimeStampInMilliseconds() > action.timeout()) {
138                             hook.timedOut();
139                             f.get().cancel(true);
140                         }
141                     }
142                 }, action.timeout(), action.timeout(), TimeUnit.MILLISECONDS));
143             }
144 
145         } else {
146             logger.warn("AtmosphereResource was null");
147         }
148         notifyListener(new WebSocketEventListener.WebSocketEvent("", CONNECT, webSocket));
149     }

153      */
154     @Override
155     public void invokeWebSocketProtocol(String webSocketMessage) {
156         List<AtmosphereRequest> list = webSocketProtocol.onMessage(webSocket, webSocketMessage);
157         dispatch(list);
158         notifyListener(new WebSocketEventListener.WebSocketEvent(webSocketMessage, MESSAGE, webSocket));
159     }
160 
161     private void dispatch(List<AtmosphereRequest> list) {
162         if (list == null) return;
163 
164         for (final AtmosphereRequest r : list) {
165             if (r != null) {
166 
167                 boolean b = r.dispatchRequestAsynchronously();
168                 ExecutorService s = (executeAsync || b) ? asyncExecutor : voidExecutor;
169 
170                 s.execute(new Runnable() {
171                     @Override
172                     public void run() {
173                         AtmosphereResponse w = new AtmosphereResponse(webSocket, r, destroyable);
174                         try {
175                             dispatch(r, w);
176                         } finally {
177                             r.destroy();
178                             w.destroy();
179                         }
180                     }
181                 });
182             }
183         }
184     }

189     @Override
190     public void invokeWebSocketProtocol(byte[] data, int offset, int length) {
191         List<AtmosphereRequest> list = webSocketProtocol.onMessage(webSocket, data, offset, length);
192         dispatch(list);
193         try {
194             notifyListener(new WebSocketEventListener.WebSocketEvent(new String(data, offset, length, "UTF-8"), MESSAGE, webSocket));
195         } catch (UnsupportedEncodingException e) {
196             logger.warn("UnsupportedEncodingException", e);
197 
198         }
199     }

201     /**
202      * Dispatch to request/response to the {@link org.atmosphere.cpr.AsyncSupport} implementation as it was a normal HTTP request.
203      *
204      * @param request a {@link AtmosphereRequest}
205      * @param r       a {@link AtmosphereResponse}
206      */
207     public final void dispatch(final AtmosphereRequest request, final AtmosphereResponse r) {
208         if (request == null) return;
209         try {
210             framework.doCometSupport(request, r);
211         } catch (Throwable e) {
212             logger.warn("Failed invoking AtmosphereFramework.doCometSupport()", e);
213             webSocketProtocol.onError(webSocket, new WebSocketException(e,
214                     new AtmosphereResponse.Builder()
215                             .request(request)
216                             .status(500)
217                             .statusMessage("Server Error").build()));
218             return;
219         }
220 
221         if (r.getStatus() >= 400) {
222             webSocketProtocol.onError(webSocket, new WebSocketException("Status code higher or equal than 400", r));
223         }
224     }

234     /**
235      * {@inheritDoc}
236      */
237     @Override
238     public void close(int closeCode) {
239         logger.trace("WebSocket closed with {}", closeCode);
240         // A message might be in the process of being processed and the websocket gets closed. In that corner
241         // case the webSocket.resource will be set to false and that might cause NPE in some WebSocketProcol implementation
242         // We could potentially synchronize on webSocket but since it is a rare case, it is better to not synchronize.
243         // synchronized (webSocket) {
244 
245         notifyListener(new WebSocketEventListener.WebSocketEvent("", CLOSE, webSocket));
246         AtmosphereResourceImpl resource = (AtmosphereResourceImpl) webSocket.resource();
247 
248         if (resource == null) {
249             logger.warn("Unable to retrieve AtmosphereResource for {}", webSocket);
250         } else {
251             AtmosphereRequest r = resource.getRequest(false);
252             AtmosphereResponse s = resource.getResponse(false);
253             try {
254                 webSocketProtocol.onClose(webSocket);
255 
256                 if (resource != null && resource.isInScope()) {
257                     AsynchronousProcessor.AsynchronousProcessorHook h = (AsynchronousProcessor.AsynchronousProcessorHook)
258                             r.getAttribute(ASYNCHRONOUS_HOOK);
259                     if (h != null) {
260                         if (closeCode == 1000) {
261                             h.timedOut();
262                         } else {
263                             h.closed();
264                         }
265                     } else {
266                         logger.warn("AsynchronousProcessor.AsynchronousProcessorHook was null");
267                     }
268                 }
269             } finally {
270                 if (r != null) {
271                     r.destroy();
272                 }
273 
274                 if (s != null) {
275                     s.destroy();
276                 }
277 
278                 if (webSocket != null) {
279                     webSocket.resource(null);
280                 }
281             }
282         }
283         asyncExecutor.shutdown();
284         voidExecutor.shutdown();
285         scheduler.shutdown();
286     }

296     /**
297      * {@inheritDoc}
298      */
299     @Override
300     public void notifyListener(WebSocketEventListener.WebSocketEvent event) {
301         AtmosphereResource resource = webSocket.resource();
302         if (resource == null) return;
303 
304         AtmosphereResourceImpl r = AtmosphereResourceImpl.class.cast(resource);
305 
306         for (AtmosphereResourceEventListener l : r.atmosphereResourceEventListener()) {
307             if (WebSocketEventListener.class.isAssignableFrom(l.getClass())) {
308                 try {
309                     switch (event.type()) {
310                         case CONNECT:
311                             WebSocketEventListener.class.cast(l).onConnect(event);
312                             break;
313                         case DISCONNECT:
314                             WebSocketEventListener.class.cast(l).onDisconnect(event);
315                             break;
316                         case CONTROL:
317                             WebSocketEventListener.class.cast(l).onControl(event);
318                             break;
319                         case MESSAGE:
320                             WebSocketEventListener.class.cast(l).onMessage(event);
321                             break;
322                         case HANDSHAKE:
323                             WebSocketEventListener.class.cast(l).onHandshake(event);
324                             break;
325                         case CLOSE:
326                             WebSocketEventListener.class.cast(l).onDisconnect(event);
327                             WebSocketEventListener.class.cast(l).onClose(event);
328                             break;
329                     }
330                 } catch (Throwable t) {
331                     logger.debug("Listener error {}", t);
332                     try {
333                         WebSocketEventListener.class.cast(l).onThrowable(new AtmosphereResourceEventImpl(r, false, false, t));
334                     } catch (Throwable t2) {
335                         logger.warn("Listener error {}", t2);
336                     }
337                 }
338             }
339         }
340     }

342     public static final Map<String, String> configureHeader(AtmosphereRequest request) {
343         Map<String, String> headers = new HashMap<String, String>();
344 
345         Enumeration<String> e = request.getParameterNames();
346         String s;
347         while (e.hasMoreElements()) {
348             s = e.nextElement();
349             headers.put(s, request.getParameter(s));
350         }
351 
352         headers.put(HeaderConfig.X_ATMOSPHERE_TRANSPORT, HeaderConfig.WEBSOCKET_TRANSPORT);
353         return headers;
354     }

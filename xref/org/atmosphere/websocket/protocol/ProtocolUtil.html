16  package org.atmosphere.websocket.protocol;
17  
18  import org.atmosphere.cpr.AtmosphereRequest;
19  import org.atmosphere.cpr.AtmosphereRequestImpl;
20  import org.atmosphere.cpr.AtmosphereResource;
21  import org.atmosphere.cpr.AtmosphereResourceImpl;
22  import org.atmosphere.websocket.WebSocket;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import java.util.Map;
27  import java.util.concurrent.ConcurrentHashMap;
28  
29  public class ProtocolUtil {
30      private final static Logger logger = LoggerFactory.getLogger(ProtocolUtil.class);
31  
32      protected static AtmosphereRequestImpl.Builder constructRequest(WebSocket webSocket,
33                                                                  String pathInfo,
34                                                                  String requestURI,
35                                                                  String methodType,
36                                                                  String contentType,
37                                                                  boolean destroyable) {
38  
39          AtmosphereResource resource = webSocket.resource();
40          AtmosphereRequest request = AtmosphereResourceImpl.class.cast(resource).getRequest(false);
41          Map<String, Object> m = attributes(webSocket, request);
42  
43          // We need to create a new AtmosphereRequest as WebSocket message may arrive concurrently on the same connection.
44          AtmosphereRequestImpl.Builder b = (new AtmosphereRequestImpl.Builder()
45                  .request(request)
46                  .method(methodType)
47                  .contentType(contentType == null ? request.getContentType() : contentType)
48                  .attributes(m)
49                  .pathInfo(pathInfo)
50                  .contextPath(request.getContextPath())
51                  .servletPath(request.getServletPath())
52                  .requestURI(requestURI)
53                  .requestURL(request.requestURL())
54                  .destroyable(destroyable)
55                  .headers(request.headersMap())
56                  .session(resource.session()));
57          return b;
58      }
59  
60      private static Map<String, Object> attributes(WebSocket webSocket, AtmosphereRequest request) {
61          Map<String, Object> m = new ConcurrentHashMap<String, Object>();
62          m.putAll(webSocket.attributes());
63          return m;
64      }
65  }

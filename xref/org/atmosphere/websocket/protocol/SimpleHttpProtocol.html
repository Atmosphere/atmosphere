36  /**
37   * Like the {@link org.atmosphere.cpr.AsynchronousProcessor} class, this class is responsible for dispatching WebSocket messages to the
38   * proper {@link org.atmosphere.websocket.WebSocket} implementation by wrapping the Websocket message's bytes within
39   * an {@link javax.servlet.http.HttpServletRequest}.
40   * <p/>
41   * The content-type is defined using {@link org.atmosphere.cpr.ApplicationConfig#WEBSOCKET_CONTENT_TYPE} property
42   * The method is defined using {@link org.atmosphere.cpr.ApplicationConfig#WEBSOCKET_METHOD} property
43   * <p/>
44   *
45   * @author Jeanfrancois Arcand
46   */
47  public class SimpleHttpProtocol implements WebSocketProtocol, Serializable {
48  
49      private static final Logger logger = LoggerFactory.getLogger(SimpleHttpProtocol.class);
50      private String contentType = "text/plain";
51      private String methodType = "POST";
52      private String delimiter = "@@";
53      private boolean destroyable;
54  
55      /**
56       * {@inheritDoc}
57       */
58      @Override
59      public void configure(AtmosphereConfig config) {
60          String contentType = config.getInitParameter(ApplicationConfig.WEBSOCKET_CONTENT_TYPE);
61          if (contentType == null) {
62              contentType = "text/plain";
63          }
64          this.contentType = contentType;
65  
66          String methodType = config.getInitParameter(ApplicationConfig.WEBSOCKET_METHOD);
67          if (methodType == null) {
68              methodType = "POST";
69          }
70          this.methodType = methodType;
71  
72          String delimiter = config.getInitParameter(ApplicationConfig.WEBSOCKET_PATH_DELIMITER);
73          if (delimiter == null) {
74              delimiter = "@@";
75          }
76          this.delimiter = delimiter;
77  
78          String s = config.getInitParameter(ApplicationConfig.RECYCLE_ATMOSPHERE_REQUEST_RESPONSE);
79          if (s != null && Boolean.valueOf(s)) {
80              destroyable = true;
81          } else {
82              destroyable = false;
83          }
84      }
85  
86      /**
87       * {@inheritDoc}
88       */
89      @Override
90      public List<AtmosphereRequest> onMessage(WebSocket webSocket, String d) {
91          AtmosphereResourceImpl resource = (AtmosphereResourceImpl) webSocket.resource();
92          if (resource == null) {
93              logger.error("Invalid state. No AtmosphereResource has been suspended");
94              return null;
95          }
96          String pathInfo = resource.getRequest().getPathInfo();
97          if (d.startsWith(delimiter)) {
98              String[] token = d.split(delimiter);
99              pathInfo = token[1];
100             d = token[2];
101         }
102         Map<String,Object> m = new HashMap<String, Object>();
103         m.put(FrameworkConfig.WEBSOCKET_SUBPROTOCOL, FrameworkConfig.SIMPLE_HTTP_OVER_WEBSOCKET);
104         // Propagate the original attribute to WebSocket message.
105         m.putAll(resource.getRequest().attributes());
106 
107         List<AtmosphereRequest> list = new ArrayList<AtmosphereRequest>();
108 
109         // We need to create a new AtmosphereRequest as WebSocket message may arrive concurrently on the same connection.
110         list.add(new AtmosphereRequest.Builder()
111                 .request(resource.getRequest())
112                 .method(methodType)
113                 .contentType(contentType)
114                 .body(d)
115                 .attributes(m)
116                 .pathInfo(pathInfo)
117                 .destroyable(destroyable)
118                 .headers(resource.getRequest().headersMap())
119                 .build());
120 
121         return list;
122     }
123 
124     /**
125      * {@inheritDoc}
126      */
127     @Override
128     public List<AtmosphereRequest> onMessage(WebSocket webSocket, byte[] d, final int offset, final int length) {
129         return onMessage(webSocket, new String(d, offset, length));
130     }
131 
132     /**
133      * {@inheritDoc}
134      */
135     @Override
136     public void onOpen(WebSocket webSocket) {
137     }
138 
139     /**
140      * {@inheritDoc}
141      */
142     @Override
143     public void onClose(WebSocket webSocket) {
144     }
145 
146     /**
147      * {@inheritDoc}
148      */
149     @Override
150     public void onError(WebSocket webSocket, WebSocketProcessor.WebSocketException t) {
151         logger.warn(t.getMessage() + " Status {} Message {}", t.response().getStatus(), t.response().getStatusMessage());
152     }
153 }

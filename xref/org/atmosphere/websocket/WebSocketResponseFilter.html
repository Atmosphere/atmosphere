16  package org.atmosphere.websocket;

18  /**
19   * Implementation of this interface allow the customization of the WebSocket message before it gets send.
20   */
21  public interface WebSocketResponseFilter {
22  
23      final WebSocketResponseFilter NOOPS_WebSocketResponseFilter = new NoOpsWebSocketResponseFilter();
24  
25      /**
26       * Transform of filter the message, return a new one or the same
27       * @param message the Websocket text message
28       * @return a new message
29       */
30      String filter(String message);
31  
32      /**
33       * Transform of filter the message, return a new one or the same
34       * @param message the Websocket bytes message
35       * @return a new message
36       */
37      byte[] filter(byte[] message);
38  
39      public final static class NoOpsWebSocketResponseFilter implements WebSocketResponseFilter {
40          @Override
41          public String filter(String message) {
42              return message;
43          }
44  
45          @Override
46          public byte[] filter(byte[] message) {
47              return message;
48          }
49      }
50  
51  }

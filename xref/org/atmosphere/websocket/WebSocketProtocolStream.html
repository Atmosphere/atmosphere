16  package org.atmosphere.websocket;
17  
18  import org.atmosphere.cpr.AtmosphereRequest;
19  import org.atmosphere.cpr.AtmosphereRequestImpl;
20  
21  import java.io.InputStream;
22  import java.io.Reader;
23  import java.util.List;
24  
25  /**
26   * A streaming API for WebServer that support WebSocket streaming. When a {@link WebSocketProtocol} implements this interface,
27   * bytes/text will ve streamed instead of read in memory.
28   *
29   * @author Jeanfrancois Arcand
30   */
31  public interface WebSocketProtocolStream extends WebSocketProtocol {
32  
33      /**
34       * Parse the WebSocket stream, and delegate the processing to the {@link org.atmosphere.cpr.AtmosphereFramework#asyncSupport} or
35       * to any existing technology. Invoking  {@link org.atmosphere.cpr.AtmosphereFramework#asyncSupport} will delegate the request processing
36       * to the {@link org.atmosphere.cpr.AtmosphereHandler} implementation. Returning null means this implementation will
37       * handle itself the processing/dispatching of the WebSocket's request;
38       * <br>
39       * As an example, this is how Websocket messages are delegated to the
40       * Jersey runtime.
41       * <br>
42       *
43       * @param webSocket The {@link org.atmosphere.websocket.WebSocket} connection
44       * @param r         a {@link java.io.Reader}
45       * @return a List of {@link AtmosphereRequest}
46       */
47      List<AtmosphereRequest> onTextStream(WebSocket webSocket, Reader r);
48  
49      /**
50       * Parse the WebSocket stream, and delegate the processing to the {@link org.atmosphere.cpr.AtmosphereFramework#asyncSupport} or
51       * to any existing technology. Invoking  {@link org.atmosphere.cpr.AtmosphereFramework#asyncSupport} will delegate the request processing
52       * to the {@link org.atmosphere.cpr.AtmosphereHandler} implementation. Returning null means this implementation will
53       * handle itself the processing/dispatching of the WebSocket's request;
54       * <br>
55       * As an example, this is how Websocket messages are delegated to the
56       * Jersey runtime.
57       * <br>
58       *
59       * @param webSocket The {@link WebSocket} connection
60       * @param stream    a {@link Reader}
61       * @return a List of {@link AtmosphereRequest}
62       */
63      List<AtmosphereRequest> onBinaryStream(WebSocket webSocket, InputStream stream);
64  }

61      public WebSocket(AtmosphereConfig config) {
62          String s = config.getInitParameter(ApplicationConfig.WEBSOCKET_BINARY_WRITE);
63          if (s != null && Boolean.parseBoolean(s)) {
64              binaryWrite = true;
65          } else {
66              binaryWrite = false;
67          }

121     protected byte[] transform(byte[] b, int offset, int length) throws IOException {
122         AtmosphereResponse response = r.getResponse();
123         AsyncIOWriter a = response.getAsyncIOWriter();
124         try {
125             response.asyncIOWriter(buffer);
126             invokeInterceptor(response, b, offset, length);
127             return buffer.stream().toByteArray();
128         } finally {
129             buffer.close(null);
130             response.asyncIOWriter(a);
131         }
132     }

134     @Override
135     public WebSocket write(AtmosphereResponse r, String data) throws IOException {
136         firstWrite.set(true);
137         if (data == null) {
138             logger.error("Cannot write null value for {}", resource());
139             return this;
140         }
141 
142         if (!isOpen()) throw new IOException("Connection remotely closed");
143         logger.trace("WebSocket.write()");
144 
145         boolean transform = filters.size() > 0 && r.getStatus() < 400;
146         if (binaryWrite) {
147             byte[] b = data.getBytes(resource().getResponse().getCharacterEncoding());
148             if (transform) {
149                 b = transform(b, 0, b.length);
150             }
151 
152             if (b != null) {
153                 write(b, 0, b.length);
154             }
155         } else {
156             if (transform) {
157                 byte[] b = data.getBytes(resource().getResponse().getCharacterEncoding());
158                 data = new String(transform(b, 0, b.length), r.getCharacterEncoding());
159             }
160 
161             if (data != null) {
162                 write(data);
163             }
164         }
165         lastWrite = System.currentTimeMillis();
166         return this;
167     }

169     @Override
170     public WebSocket write(AtmosphereResponse r, byte[] data) throws IOException {
171         if (data == null) {
172             logger.error("Cannot write null value for {}", resource());
173             return this;
174         }
175         return write(r, data, 0, data.length);
176     }

178     @Override
179     public WebSocket write(AtmosphereResponse r, byte[] b, int offset, int length) throws IOException {
180         firstWrite.set(true);
181         if (b == null) {
182             logger.error("Cannot write null value for {}", resource());
183             return this;
184         }
185         if (!isOpen()) throw new IOException("Connection remotely closed");
186 
187         logger.trace("WebSocket.write()");
188         boolean transform = filters.size() > 0 && r.getStatus() < 400;
189         if (binaryWrite) {
190             if (transform) {
191                 b = transform(b, offset, length);
192             }
193 
194             if (b != null) {
195                 write(b, 0, length);
196             }
197         } else {
198             String data = null;
199             String charset = r.getCharacterEncoding() == null ? "UTF-8" : r.getCharacterEncoding();
200             if (transform) {
201                 data = new String(transform(b, offset, length), charset);
202             } else {
203                 data = new String(b, offset, length, charset);
204             }
205 
206             if (data != null) {
207                 write(data);
208             }
209         }
210         lastWrite = System.currentTimeMillis();
211         return this;
212     }

230     @Override
231     public WebSocket writeError(AtmosphereResponse r, int errorCode, String message) throws IOException {
232         super.writeError(r, errorCode, message);
233         if (!firstWrite.get()) {
234             logger.debug("The WebSocket handshake succeeded but the dispatched URI failed with status {} : {} " +
235                     "The WebSocket connection is still open and client can continue sending messages.", errorCode + " " + message, retrieveUUID());
236         } else {
237             logger.debug("{} {}", errorCode, message);
238         }
239 
240         return this;
241     }

252         logger.trace("WebSocket.close()");

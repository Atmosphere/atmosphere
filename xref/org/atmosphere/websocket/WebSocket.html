68      public WebSocket(AtmosphereConfig config) {
69          String s = config.getInitParameter(ApplicationConfig.WEBSOCKET_BINARY_WRITE);
70          if (s != null && Boolean.parseBoolean(s)) {
71              binaryWrite = true;
72          } else {
73              binaryWrite = false;
74          }
75          this.config = config;
76      }

108     public WebSocket resource(AtmosphereResource r) {
109 
110         // Make sure we carry what was set at the onOpen stage.
111         if (this.r != null && r != null) {
112             // TODO: This is all over the place and quite ugly (the cast). Need to fix this in 1.1
113             AtmosphereResourceImpl.class.cast(r).cloneState(this.r);
114         }
115         this.r = r;
116         if (r != null) uuid = r.uuid();
117         return this;
118     }

159     protected byte[] transform(byte[] b, int offset, int length) throws IOException {
160         return transform(r.getResponse(), b, offset, length);

163     protected byte[] transform(AtmosphereResponse response, byte[] b, int offset, int length) throws IOException {
164         AsyncIOWriter a = response.getAsyncIOWriter();
165         // NOTE #1961 for now, create a new buffer par transform call and release it after the transform call.
166         //      Alternatively, we may cache the buffer in thread-local and use it while this thread invokes
167         //      multiple writes and release it when this thread invokes the close method.
168         ByteArrayAsyncWriter buffer = new ByteArrayAsyncWriter();
169         try {
170             response.asyncIOWriter(buffer);
171             invokeInterceptor(response, b, offset, length);
172             return buffer.stream().toByteArray();
173         } finally {
174             buffer.close(null);
175             response.asyncIOWriter(a);
176         }
177     }

179     @Override
180     public WebSocket write(AtmosphereResponse r, String data) throws IOException {
181         firstWrite.set(true);
182         if (data == null) {
183             logger.error("Cannot write null value for {}", resource());
184             return this;
185         }
186 
187         if (!isOpen()) throw new IOException("Connection remotely closed for " + uuid);
188         logger.trace("WebSocket.write() {}", data);
189 
190         boolean transform = !filters.isEmpty() && r.getStatus() < 400;
191         if (binaryWrite) {
192             byte[] b = data.getBytes(resource().getResponse().getCharacterEncoding());
193             if (transform) {
194                 b = transform(r, b, 0, b.length);
195             }
196 
197             if (b != null) {
198                 write(b, 0, b.length);
199             }
200         } else {
201             if (transform) {
202                 byte[] b = data.getBytes(resource().getResponse().getCharacterEncoding());
203                 data = new String(transform(r, b, 0, b.length), r.getCharacterEncoding());
204             }
205 
206             if (data != null) {
207                 write(data);
208             }
209         }
210         lastWrite = System.currentTimeMillis();
211         return this;
212     }

214     @Override
215     public WebSocket write(AtmosphereResponse r, byte[] data) throws IOException {
216         if (data == null) {
217             logger.error("Cannot write null value for {}", resource());
218             return this;
219         }
220         return write(r, data, 0, data.length);
221     }

223     @Override
224     public WebSocket write(AtmosphereResponse r, byte[] b, int offset, int length) throws IOException {
225         firstWrite.set(true);
226         if (b == null) {
227             logger.error("Cannot write null value for {}", resource());
228             return this;
229         }
230         if (!isOpen()) throw new IOException("Connection remotely closed for " + uuid);
231 
232         if (logger.isTraceEnabled()) {
233             logger.trace("WebSocket.write() {}", new String(b, offset, length, "UTF-8"));
234         }
235 
236         boolean transform = !filters.isEmpty() && r.getStatus() < 400;
237         if (binaryWrite || resource().forceBinaryWrite()) {
238             if (transform) {
239                 b = transform(r, b, offset, length);
240             }
241 
242             if (b != null) {
243                 write(b, 0, b.length);
244             }
245         } else {
246             String data = null;
247             String charset = r.getCharacterEncoding() == null ? "UTF-8" : r.getCharacterEncoding();
248             if (transform) {
249                 data = new String(transform(r, b, offset, length), charset);
250             } else {
251                 data = new String(b, offset, length, charset);
252             }
253 
254             if (data != null) {
255                 write(data);
256             }
257         }
258         lastWrite = System.currentTimeMillis();
259         return this;
260     }

262     /**
263      * Broadcast, using the {@link org.atmosphere.cpr.AtmosphereResource#getBroadcaster()} the object to all
264      * {@link WebSocket} associated with the {@link org.atmosphere.cpr.Broadcaster}. This method does the same as
265      * websocket.resource().getBroadcaster().broadcast(o).
266      *
267      * @param o An object to broadcast to all WebSockets.
268      */
269     public WebSocket broadcast(Object o) {
270         if (r != null) {
271             r.getBroadcaster().broadcast(o);
272         } else {
273             logger.debug("No AtmosphereResource Associated with this WebSocket.");
274         }
275         return this;

278     @Override
279     public WebSocket writeError(AtmosphereResponse r, int errorCode, String message) throws IOException {
280         super.writeError(r, errorCode, message);
281         if (!firstWrite.get()) {
282             logger.debug("The WebSocket handshake succeeded but the dispatched URI failed with status {} : {} " +
283                     "The WebSocket connection is still open and client can continue sending messages.", errorCode + " " + message, uuid());
284         } else {
285             logger.warn("Unable to write {} {}", errorCode, message);
286         }
287 
288         return this;
289     }

298     @Override
299     public void close(AtmosphereResponse r) throws IOException {
300         logger.trace("WebSocket.close() for {}", uuid);
301 
302         try {
303             // Never trust underlying server.
304             // https://github.com/Atmosphere/atmosphere/issues/1633
305             if (r.request() != null && r.request().getAttribute(CLEAN_CLOSE) == null) {
306                 close();
307             }
308         } catch (Exception ex) {
309             logger.trace("", ex);
310         }
311 
312         try {
313             bb.clear();
314             cb.clear();
315             // NOTE #1961 if the buffer is cached at thread-local, it needs to be released here.
316         } catch (Exception ex) {
317             logger.trace("", ex);
318         }
319     }

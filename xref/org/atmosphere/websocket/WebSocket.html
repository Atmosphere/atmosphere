37  /**
38   * Represent a portable WebSocket implementation which can be used to write message.
39   *
40   * @author Jeanfrancois Arcand
41   */
42  public abstract class WebSocket extends AtmosphereInterceptorWriter {
43  
44      protected static final Logger logger = LoggerFactory.getLogger(WebSocket.class);
45      public final static String WEBSOCKET_INITIATED = WebSocket.class.getName() + ".initiated";
46      public final static String WEBSOCKET_SUSPEND = WebSocket.class.getName() + ".suspend";
47      public final static String WEBSOCKET_RESUME = WebSocket.class.getName() + ".resume";
48      public final static String WEBSOCKET_ACCEPT_DONE = WebSocket.class.getName() + ".acceptDone";
49      public final static String NOT_SUPPORTED = "Websocket protocol not supported";
50      public final static String CLEAN_CLOSE = "Clean_Close";
51  
52      private AtmosphereResource r;
53      protected long lastWrite = 0;
54      protected boolean binaryWrite;
55      private final ByteArrayAsyncWriter buffer = new ByteArrayAsyncWriter();
56      private final AtomicBoolean firstWrite = new AtomicBoolean(false);
57      private final AtmosphereConfig config;
58      private WebSocketHandler webSocketHandler;
59      protected ByteBuffer bb = ByteBuffer.allocate(8192);
60      protected CharBuffer cb = CharBuffer.allocate(8192);
61      protected String uuid = "NUll";
62  
63      public WebSocket(AtmosphereConfig config) {
64          String s = config.getInitParameter(ApplicationConfig.WEBSOCKET_BINARY_WRITE);
65          if (s != null && Boolean.parseBoolean(s)) {
66              binaryWrite = true;
67          } else {
68              binaryWrite = false;
69          }
70          this.config = config;
71      }
72  
73      public AtmosphereConfig config() {
74          return config;
75      }
76  
77      protected WebSocket webSocketHandler(WebSocketHandler webSocketHandler) {
78          this.webSocketHandler = webSocketHandler;
79          return this;
80      }

82      /**
83       * Switch to binary write, or go back to text write. Default is false.
84       * @param binaryWrite true to switch to binary write.
85       * @return
86       */
87      public WebSocket binaryWrite(boolean binaryWrite) {
88          this.binaryWrite = binaryWrite;
89          return this;
90      }
91  
92      protected WebSocketHandler webSocketHandler() {
93          return webSocketHandler;
94      }

96      /**
97       * Associate an {@link AtmosphereResource} to this WebSocket
98       *
99       * @param r an {@link AtmosphereResource} to this WebSocket
100      * @return this
101      */
102     public WebSocket resource(AtmosphereResource r) {
103 
104         // Make sure we carry what was set at the onOpen stage.
105         if (this.r != null && r != null) {
106             // TODO: This is all over the place and quite ugly (the cast). Need to fix this in 1.1
107             AtmosphereResourceImpl.class.cast(r).cloneState(this.r);
108         }
109         this.r = r;
110         if (r != null) uuid = r.uuid();
111         return this;
112     }
113 
114     /**
115      * Return the an {@link AtmosphereResource} used by this WebSocket, or null if the WebSocket has been closed
116      * before the WebSocket message has been processed.
117      *
118      * @return {@link AtmosphereResource}
119      */
120     public AtmosphereResource resource() {
121         return r;
122     }
123 
124     /**
125      * The last time, in milliseconds, a write operation occurred.
126      *
127      * @return this
128      */
129     public long lastWriteTimeStampInMilliseconds() {
130         return lastWrite == -1 ? System.currentTimeMillis() : lastWrite;
131     }
132 
133     protected byte[] transform(byte[] b, int offset, int length) throws IOException {
134         AtmosphereResponse response = r.getResponse();
135         AsyncIOWriter a = response.getAsyncIOWriter();
136         try {
137             response.asyncIOWriter(buffer);
138             invokeInterceptor(response, b, offset, length);
139             return buffer.stream().toByteArray();
140         } finally {
141             buffer.close(null);
142             response.asyncIOWriter(a);
143         }
144     }
145 
146     @Override
147     public WebSocket write(AtmosphereResponse r, String data) throws IOException {
148         firstWrite.set(true);
149         if (data == null) {
150             logger.error("Cannot write null value for {}", resource());
151             return this;
152         }
153 
154         if (!isOpen()) throw new IOException("Connection remotely closed for " + uuid);
155         logger.trace("WebSocket.write() {}", data);
156 
157         boolean transform = filters.size() > 0 && r.getStatus() < 400;
158         if (binaryWrite) {
159             byte[] b = data.getBytes(resource().getResponse().getCharacterEncoding());
160             if (transform) {
161                 b = transform(b, 0, b.length);
162             }
163 
164             if (b != null) {
165                 write(b, 0, b.length);
166             }
167         } else {
168             if (transform) {
169                 byte[] b = data.getBytes(resource().getResponse().getCharacterEncoding());
170                 data = new String(transform(b, 0, b.length), r.getCharacterEncoding());
171             }
172 
173             if (data != null) {
174                 write(data);
175             }
176         }
177         lastWrite = System.currentTimeMillis();
178         return this;
179     }
180 
181     @Override
182     public WebSocket write(AtmosphereResponse r, byte[] data) throws IOException {
183         if (data == null) {
184             logger.error("Cannot write null value for {}", resource());
185             return this;
186         }
187         return write(r, data, 0, data.length);
188     }
189 
190     @Override
191     public WebSocket write(AtmosphereResponse r, byte[] b, int offset, int length) throws IOException {
192         firstWrite.set(true);
193         if (b == null) {
194             logger.error("Cannot write null value for {}", resource());
195             return this;
196         }
197         if (!isOpen()) throw new IOException("Connection remotely closed for " + uuid);
198 
199         if (logger.isTraceEnabled()) {
200             logger.trace("WebSocket.write() {}", new String(b, offset, length, "UTF-8"));
201         }
202 
203         boolean transform = filters.size() > 0 && r.getStatus() < 400;
204         if (binaryWrite || resource().forceBinaryWrite()) {
205             if (transform) {
206                 b = transform(b, offset, length);
207             }
208 
209             if (b != null) {
210                 write(b, 0, b.length);
211             }
212         } else {
213             String data = null;
214             String charset = r.getCharacterEncoding() == null ? "UTF-8" : r.getCharacterEncoding();
215             if (transform) {
216                 data = new String(transform(b, offset, length), charset);
217             } else {
218                 data = new String(b, offset, length, charset);
219             }
220 
221             if (data != null) {
222                 write(data);
223             }
224         }
225         lastWrite = System.currentTimeMillis();
226         return this;
227     }

229     /**
230      * Broadcast, using the {@link org.atmosphere.cpr.AtmosphereResource#getBroadcaster()} the object to all
231      * {@link WebSocket} associated with the {@link org.atmosphere.cpr.Broadcaster}. This method does the same as
232      * websocket.resource().getBroadcaster().broadcast(o).
233      *
234      * @param o An object to broadcast to all WebSockets.
235      */
236     public WebSocket broadcast(Object o) {
237         if (r != null) {
238             r.getBroadcaster().broadcast(o);
239         } else {
240             logger.debug("No AtmosphereResource Associated with this WebSocket.");
241         }
242         return this;
243     }
244 
245     @Override
246     public WebSocket writeError(AtmosphereResponse r, int errorCode, String message) throws IOException {
247         super.writeError(r, errorCode, message);
248         if (!firstWrite.get()) {
249             logger.debug("The WebSocket handshake succeeded but the dispatched URI failed with status {} : {} " +
250                     "The WebSocket connection is still open and client can continue sending messages.", errorCode + " " + message, uuid());
251         } else {
252             logger.warn("Unable to write {} {}", errorCode, message);
253         }
254 
255         return this;
256     }
257 
258     @Override
259     public WebSocket redirect(AtmosphereResponse r, String location) throws IOException {
260         logger.error("WebSocket Redirect not supported");
261         return this;
262     }

264 
265     @Override
266     public void close(AtmosphereResponse r) throws IOException {
267         logger.trace("WebSocket.close() for {}", uuid);
268 
269         try {
270             // Never trust underlying server.
271             // https://github.com/Atmosphere/atmosphere/issues/1633
272             if (r.request() != null && r.request().getAttribute(CLEAN_CLOSE) == null) {
273                 close();
274             }
275         } catch (Exception ex) {
276             logger.trace("", ex);
277         }
278 
279         try {
280             bb.clear();
281             cb.clear();
282             buffer.close(r);
283         } catch (Exception ex) {
284             logger.trace("", ex);
285         }
286     }
287 
288     @Override
289     public WebSocket flush(AtmosphereResponse r) throws IOException {
290         return this;
291     }

300     /**
301      * Use the underlying container's websocket to write the String.
302      *
303      * @param s a websocket String message
304      * @return this
305      * @throws IOException
306      */
307     abstract public WebSocket write(String s) throws IOException;
308 
309     /**
310      * Use the underlying container's websocket to write the byte.
311      *
312      * @param b      a websocket byte message
313      * @param offset start
314      * @param length end
315      * @return this
316      * @throws IOException
317      */
318     abstract public WebSocket write(byte[] b, int offset, int length) throws IOException;
319 
320     /**
321      * Close the underlying WebSocket
322      */
323     abstract public void close();
324 
325     protected String uuid() {
326         return uuid;
327     }
328 
329     public static void notSupported(AtmosphereRequest request, AtmosphereResponse response) throws IOException {
330         response.addHeader(X_ATMOSPHERE_ERROR, WebSocket.NOT_SUPPORTED);
331         response.sendError(501, WebSocket.NOT_SUPPORTED);
332         logger.trace("{} for request {}", WebSocket.NOT_SUPPORTED, request);
333     }

62   * An {@link AnnotationProcessor} that selects between a ServletContextInitializer based scanner, and
63   * a bytecode based scanner based on <a href="https://github.com/rmuller/infomas-asl"></a>.
64   * <p/>
65   *
66   * @author Jeanfrancois Arcand
67   */

105     private AnnotationProcessor delegate;
106     private final AnnotationHandler handler;
107     private final AtomicBoolean coreAnnotationsFound = new AtomicBoolean();
108 
109     private final AnnotationDetector.TypeReporter atmosphereReporter = new AnnotationDetector.TypeReporter() {
110         @SuppressWarnings("unchecked")
111         @Override
112         public Class<? extends Annotation>[] annotations() {
113             return new Class[]{
114                     AtmosphereAnnotation.class
115             };
116         }
117 
118         @Override
119         public void reportTypeAnnotation(Class<? extends Annotation> annotation, String className) {
120             try {
121                 coreAnnotationsFound.set(true);
122                 handler.handleProcessor(loadClass(getClass(), className));
123             } catch (Exception e) {
124                 logger.warn("Error scanning @AtmosphereAnnotation", e);
125             }
126         }
127     };

129     public DefaultAnnotationProcessor() {
130         this.handler = new AnnotationHandler();
131     }
132 
133     @Override
134     public void configure(final AtmosphereConfig config) {
135         ServletContext sc = config.framework().getServletContext();
136 
137         Map<Class<? extends Annotation>, Set<Class<?>>>  annotations= (Map<Class<? extends Annotation>, Set<Class<?>>>) sc.getAttribute(ANNOTATION_ATTRIBUTE);
138         sc.removeAttribute(ANNOTATION_ATTRIBUTE);
139 
140         boolean useByteCodeProcessor = config.getInitParameter(ApplicationConfig.BYTECODE_PROCESSOR, false);
141 
142         boolean scanForAtmosphereAnnotation = false;
143         if (useByteCodeProcessor || annotations == null || annotations.isEmpty()) {
144             delegate = new BytecodeBasedAnnotationProcessor(handler);
145             scanForAtmosphereAnnotation = true;
146         } else {
147             Map<Class<? extends Annotation>, Set<Class<?>>> clone = new HashMap<Class<? extends Annotation>, Set<Class<?>>>();
148             clone.putAll(annotations);
149             delegate = new ServletContainerInitializerAnnotationProcessor(handler, clone, config.framework());
150         }
151         logger.info("AnnotationProcessor {} being used", delegate.getClass());
152 
153         if (scanForAtmosphereAnnotation) {
154             scanForAnnotation(config.framework());
155         }
156 
157         delegate.configure(config.framework().getAtmosphereConfig());
158     }
159 
160     private void scanForAnnotation(AtmosphereFramework f) {
161         List<String> packages = f.customAnnotationPackages();
162         AnnotationDetector detector = new AnnotationDetector(atmosphereReporter);
163         try {
164             if (!packages.isEmpty()) {
165                 for (String p : packages) {
166                     logger.trace("Package {} scanned for @AtmosphereAnnotation", p);
167                     detector.detect(p);
168                 }
169             }
170 
171             // Now look for application defined annotation
172             String path = IOUtils.realPath(f.getServletContext(), f.getHandlersPath());
173             if (path != null) {
174                 detector.detect(new File(path));
175             }
176 
177             String pathLibs =  IOUtils.realPath(f.getServletContext(), f.getLibPath());
178             if (pathLibs != null) {
179                 File libFolder = new File(pathLibs);
180                 File jars[] = libFolder.listFiles(new FilenameFilter() {
181                     @Override
182                     public boolean accept(File arg0, String arg1) {
183                         return arg1.endsWith(".jar");
184                     }
185                 });
186 
187                 if (jars != null) {
188                     for (File file : jars) {
189                         detector.detect(file);
190                     }
191                 }
192             }
193 
194             // JBoss|vfs with APR issue, or any strange containers may fail. This is a hack for them.
195             // https://github.com/Atmosphere/atmosphere/issues/1292
196             if (!coreAnnotationsFound.get()) {
197                 fallbackToManualAnnotatedClasses(getClass(), f, handler);
198             }
199         } catch (IOException e) {
200             logger.warn("Unable to scan annotation", e);
201         } finally {
202             detector.destroy();
203         }
204     }
205 
206     private static void fallbackToManualAnnotatedClasses(Class<?> mainClass, AtmosphereFramework f, AnnotationHandler handler) {
207         logger.warn("Unable to detect annotations. Application may fail to deploy.");
208         f.annotationScanned(true);
209         for (Class a : coreAnnotations) {
210             try {
211                 handler.handleProcessor(loadClass(mainClass, a.getName()));
212             } catch (Exception e) {
213                 logger.trace("", e);
214             }
215         }
216     }

247     private static final class ServletContainerInitializerAnnotationProcessor implements AnnotationProcessor {
248 
249         private final Map<Class<? extends Annotation>, Set<Class<?>>> annotations;
250         private final AtmosphereFramework framework;
251         private final AnnotationHandler handler;
252         /**
253          * This is not great, but we can't differentiate based on the file, so we just do one scan
254          * of everything in the war. It would be nice to change to the API to make this a bit cleaner
255          * but it looks like it is a public API.
256          */
257         private boolean alreadyScanned;
258 
259         private ServletContainerInitializerAnnotationProcessor(AnnotationHandler handler,
260                                                                final Map<Class<? extends Annotation>, Set<Class<?>>> annotations,
261                                                                final AtmosphereFramework framework) {
262             this.annotations = annotations;
263             this.framework = framework;
264             this.handler = handler;
265         }
266 
267         @Override
268         public void configure(final AtmosphereConfig config) {
269         }
270 
271         @Override
272         public AnnotationProcessor scan(final File rootDir) throws IOException {
273             if (alreadyScanned) {
274                 return this;
275             }
276             alreadyScanned = true;
277 
278             Set<Class<?>> atmosphereAnnotatedClasses = annotations.get(AtmosphereAnnotation.class);
279             boolean handleAtmosphereAnnotation = handleAtmosphereAnnotation(atmosphereAnnotatedClasses);
280 
281             for (Map.Entry<Class<? extends Annotation>, Set<Class<?>>> entry : annotations.entrySet()) {
282                 for (Class<?> clazz : entry.getValue()) {
283                     handler.handleAnnotation(framework, entry.getKey(), clazz);
284                 }
285             }
286 
287             if (handleAtmosphereAnnotation) {
288                 scanForCustomAnnotation(atmosphereAnnotatedClasses);
289             }
290             return this;
291         }
292 
293         private boolean handleAtmosphereAnnotation(Set<Class<?>> atmosphereAnnotatedClasses) {
294             boolean scanForCustomizedAnnotation = false;
295             if (atmosphereAnnotatedClasses != null) {
296                 for (Class<?> clazz : atmosphereAnnotatedClasses) {
297                     handler.handleProcessor(clazz);
298                 }
299             } else {
300                 fallbackToManualAnnotatedClasses(getClass(),framework, handler);
301             }
302 
303             // If larger, a custom annotation has been defined.
304             if (atmosphereAnnotatedClasses != null && atmosphereAnnotatedClasses.size() >=
305                     AnnotationScanningServletContainerInitializer.class.getAnnotation(HandlesTypes.class).value().length) {
306                 scanForCustomizedAnnotation = true;
307             }
308             return scanForCustomizedAnnotation;
309         }
310 
311         private void scanForCustomAnnotation(Set<Class<?>> atmosphereAnnotatedClasses) throws IOException {
312 
313             handler.flushCoreAnnotations(atmosphereAnnotatedClasses);
314 
315             BytecodeBasedAnnotationProcessor b = new BytecodeBasedAnnotationProcessor(handler);
316             b.configure(framework.getAtmosphereConfig());
317             String path = framework.getServletContext().getRealPath(framework.getHandlersPath());
318             if (path != null) {
319                 b.scan(new File(path)).destroy();
320             } else {
321                 logger.warn("Unable to scan using File. Scanning classpath");
322                 b.scanAll();
323             }
324         }
325 
326         @Override
327         public AnnotationProcessor scan(final String packageName) throws IOException {
328             Set<Class<?>> atmosphereAnnotatedClasses = annotations.get(AtmosphereAnnotation.class);
329 
330             if (packageName.equals("all") || getClass().getClassLoader().getResource(packageName.replace(".", "/")) != null) {
331                 boolean handleAtmosphereAnnotation = handleAtmosphereAnnotation(atmosphereAnnotatedClasses);
332 
333                 for (Map.Entry<Class<? extends Annotation>, Set<Class<?>>> entry : annotations.entrySet()) {
334                     for (Class<?> clazz : entry.getValue()) {
335                         if (packageName.equals("all") || clazz.getPackage().getName().startsWith(packageName)) {
336                             handler.handleAnnotation(framework, entry.getKey(), clazz);
337                         }
338                     }
339                 }
340 
341                 if (handleAtmosphereAnnotation) {
342                     scanForCustomAnnotation(atmosphereAnnotatedClasses);
343                 }
344             }
345 
346             return this;
347         }
348 
349         @Override
350         public AnnotationProcessor scanAll() throws IOException {
351             return scan("all");
352         }
353 
354         @Override
355         public void destroy() {
356             annotations.clear();
357         }
358     }

360     private static final class BytecodeBasedAnnotationProcessor implements AnnotationProcessor {
361 
362         protected AnnotationDetector detector;
363         protected final AnnotationHandler handler;
364 
365         public BytecodeBasedAnnotationProcessor(AnnotationHandler handler) {
366             this.handler = handler;
367         }
368 
369         @Override
370         public void configure(final AtmosphereConfig config) {
371 
372             final AnnotationDetector.TypeReporter reporter = new AnnotationDetector.TypeReporter() {
373                 @SuppressWarnings("unchecked")
374                 @Override
375                 public Class<? extends Annotation>[] annotations() {
376                     return handler.handledClass();
377                 }
378 
379                 @Override
380                 public void reportTypeAnnotation(Class<? extends Annotation> annotation, String className) {
381                     try {
382                         final Class<?> discoveredClass = loadClass(getClass(), className);
383                         handler.handleAnnotation(config.framework(), annotation, discoveredClass);
384                     } catch (Exception e) {
385                         logger.warn("Could not load discovered class", e);
386                     }
387                 }
388 
389             };
390             detector = new AnnotationDetector(reporter);
391         }
392 
393         @Override
394         public AnnotationProcessor scan(File rootDir) throws IOException {
395             detector.detect(rootDir);
396             return this;
397         }
398 
399         @Override
400         public AnnotationProcessor scan(String packageName) throws IOException {
401             logger.trace("Scanning @Service annotations in {}", packageName);
402             detector.detect(packageName);
403             return this;
404         }
405 
406         @Override
407         public AnnotationProcessor scanAll() throws IOException {
408             detector.detect();
409             return this;
410         }
411 
412         @Override
413         public void destroy() {
414             if (detector != null) detector.destroy();
415         }
416     }

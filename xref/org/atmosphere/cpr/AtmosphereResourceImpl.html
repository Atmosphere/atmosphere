105      *

107      * @param broadcaster       The {@link org.atmosphere.cpr.Broadcaster}.

114     @Override
115     public AtmosphereResource initialize(AtmosphereConfig config, Broadcaster broadcaster,
116                                          AtmosphereRequest req, AtmosphereResponse response,
117                                          AsyncSupport asyncSupport, AtmosphereHandler atmosphereHandler) {
118         this.req = req;
119         this.response = response;
120         this.config = config;
121         this.asyncSupport = asyncSupport;
122         this.atmosphereHandler = atmosphereHandler;
123         this.event = new AtmosphereResourceEventImpl(this);
124 
125         this.broadcaster = broadcaster;
126         uniqueBroadcaster(broadcaster);
127 
128         String s = (String) req.getAttribute(SUSPENDED_ATMOSPHERE_RESOURCE_UUID);
129         if (s == null) {
130             s = response.getHeader(HeaderConfig.X_ATMOSPHERE_TRACKING_ID);
131             if (s == null && req != null) {
132                 String tmp = req.getHeader(HeaderConfig.X_ATMOSPHERE_TRACKING_ID);
133                 s = tmp != null && !tmp.equalsIgnoreCase("0") ? tmp : null;
134             }
135         }
136         uuid = s == null ? config.uuidProvider().generateUuid() : s;
137 
138         if (config.isSupportSession()) {
139             // Keep a reference to an HttpSession in case the associated request get recycled by the underlying container.
140             try {
141                 session = req.getSession(config.getInitParameter(PROPERTY_SESSION_CREATE, true));
142             } catch (NullPointerException ex) {
143                 // http://java.net/jira/browse/GLASSFISH-18856
144                 logger.trace("http://java.net/jira/browse/GLASSFISH-18856", ex);
145             }
146         }
147         transport = configureTransport();
148         closeOnCancel = config.getInitParameter(ApplicationConfig.CLOSE_STREAM_ON_CANCEL, false);
149         return this;
150     }

159     private TRANSPORT configureTransport() {
160         if (req == null) return UNDEFINED;
161 
162         String s = req.getHeader(HeaderConfig.X_ATMOSPHERE_TRANSPORT);
163         if (s == null) return UNDEFINED;
164 
165         if (s.equals(UNDEFINED.name()) && Utils.rawWebSocket(req)) {
166             return TRANSPORT.WEBSOCKET;
167         }
168 
169         s = s.replace("-", "_").toUpperCase();
170         if (TRANSPORT.POLLING.name().equals(s)) {
171             return TRANSPORT.POLLING;
172         } else if (TRANSPORT.LONG_POLLING.name().equals(s)) {
173             return TRANSPORT.LONG_POLLING;
174         } else if (TRANSPORT.STREAMING.name().equals(s)) {
175             return TRANSPORT.STREAMING;
176         } else if (TRANSPORT.JSONP.name().equals(s)) {
177             return TRANSPORT.JSONP;
178         } else if (TRANSPORT.WEBSOCKET.name().equals(s)) {
179             return TRANSPORT.WEBSOCKET;
180         } else if (TRANSPORT.SSE.name().equals(s)) {
181             return TRANSPORT.SSE;
182         } else if (TRANSPORT.AJAX.name().equals(s)) {
183             return TRANSPORT.AJAX;
184         } else if (TRANSPORT.CLOSE.name().equals(s)) {
185             return TRANSPORT.CLOSE;
186         } else {
187             return UNDEFINED;
188         }
189     }

256     @Override
257     public AtmosphereResource resume() {
258 
259         if (!isSuspended()) {
260             logger.trace("AtmosphereResource {} not suspended {}, cannot resume it.", uuid(), action);
261             return this;
262         }
263 
264         try {
265             if (!isResumed.getAndSet(true) && isInScope.get()) {
266                 suspended.set(false);
267                 logger.trace("AtmosphereResource {} is resuming", uuid());
268 
269                 action.type(Action.TYPE.RESUME);
270 
271                 boolean notify = true;
272                 for (Broadcaster b : broadcasters) {
273                     // We need it as Jetty doesn't support timeout
274                     if (!b.isDestroyed() && b instanceof DefaultBroadcaster) {
275                         ((DefaultBroadcaster) b).broadcastOnResume(this);
276                     }
277 
278                     if (notify) {
279                         notify = false;
280                         notifyListeners();
281                     }
282 
283                     if (b.getScope() == Broadcaster.SCOPE.REQUEST) {
284                         logger.debug("Broadcaster's scope is set to request, destroying it {}", b.getID());
285                         b.destroy();
286                     }
287                 }
288                 removeFromAllBroadcasters();
289 
290                 try {
291                     req.setAttribute(ApplicationConfig.RESUMED_ON_TIMEOUT, Boolean.FALSE);
292                 } catch (Exception ex) {
293                     logger.debug("Resume exception: Cannot resume an already resumed/cancelled request", ex);
294                 }
295 
296                 if (req.getAttribute(PRE_SUSPEND) == null) {
297                     asyncSupport.action(this);
298                 }
299             } else {
300                 logger.trace("Already resumed {}", this);
301                 return this;
302             }
303         } catch (Throwable t) {
304             logger.trace("Wasn't able to resume a connection {}", this, t);
305         } finally {
306             unregister();
307             Utils.destroyMeteor(req);
308         }
309         listeners.clear();
310         return this;
311     }

328     @Override
329     public AtmosphereResource suspend(long timeout) {
330 
331         if (event.isSuspended() || disableSuspend) return this;
332 
333         if (config.isSupportSession()
334                 && req.getSession(false) != null
335                 && req.getSession().getMaxInactiveInterval() >= 0
336                 && req.getSession().getMaxInactiveInterval() * 1000 < timeout) {
337             throw new IllegalStateException("Cannot suspend a " +
338                     "response longer than the session timeout. Increase the value of session-timeout in web.xml");
339         }
340 
341         if (Utils.resumableTransport(transport())) {
342             resumeOnBroadcast.set(true);
343         }
344 
345         onPreSuspend(event);
346 
347         // Recheck based on preSuspend
348         if (event.isSuspended() || disableSuspend) return this;
349 
350         if (!event.isResumedOnTimeout()) {
351             suspended.set(true);
352 
353             Enumeration<String> connection = req.getHeaders("Connection");
354             if (connection == null) {
355                 connection = req.getHeaders("connection");
356             }
357 
358             if (connection != null && connection.hasMoreElements()) {
359                 String[] e = connection.nextElement().toString().split(",");
360                 for (String upgrade : e) {
361                     if (upgrade.trim().equalsIgnoreCase(WEBSOCKET_UPGRADE)) {
362                         if (!asyncSupport.supportWebSocket()) {
363                             response.addHeader(X_ATMOSPHERE_ERROR, "Websocket protocol not supported");
364                         } else {
365                             req.setAttribute(FrameworkConfig.TRANSPORT_IN_USE, HeaderConfig.WEBSOCKET_TRANSPORT);
366                         }
367                     }
368                 }
369             }
370 
371             if (req.getHeader(X_ATMOSPHERE_TRANSPORT) == null || transport().equals(UNDEFINED)) {
372                 req.setAttribute(FrameworkConfig.TRANSPORT_IN_USE, LONG_POLLING_TRANSPORT);
373             }
374 
375             req.setAttribute(PRE_SUSPEND, "true");
376             action.type(Action.TYPE.SUSPEND);
377             action.timeout(timeout);
378 
379             // TODO: We can possibly optimize that call by avoiding creating a Broadcaster if we are sure the Broadcaster
380             // is unique.
381             boolean isJersey = req.getAttribute(FrameworkConfig.CONTAINER_RESPONSE) != null;
382 
383             boolean skipCreation = false;
384             if (req.getAttribute(SKIP_BROADCASTER_CREATION) != null) {
385                 skipCreation = true;
386             }
387 
388             Broadcaster broadcaster = getBroadcaster();
389 
390             // Null means SCOPE=REQUEST set by a Meteor
391             if (!skipCreation && (broadcaster == null || broadcaster.getScope() == Broadcaster.SCOPE.REQUEST) && !isJersey) {
392                 String id = broadcaster != null ? broadcaster.getID() : ROOT_MASTER;
393                 Class<? extends Broadcaster> clazz = broadcaster != null ? broadcaster.getClass() : DefaultBroadcaster.class;
394 
395                 broadcaster = config.getBroadcasterFactory().lookup(clazz, id, false);
396                 if (broadcaster == null || !broadcaster.getAtmosphereResources().isEmpty()) {
397                     broadcaster = config.getBroadcasterFactory().lookup(clazz, id + "/" + UUID.randomUUID(), true);
398                 }
399             }
400 
401             broadcaster.addAtmosphereResource(this);
402             if (req.getAttribute(DefaultBroadcaster.CACHED) != null && transport() != null && Utils.resumableTransport(transport())) {
403                 action.type(Action.TYPE.CONTINUE);
404                 // Do nothing because we have found cached message which was written already, and the handler resumed.
405                 logger.debug("Cached message found, not suspending {}", uuid());
406                 return this;
407             }
408             req.removeAttribute(PRE_SUSPEND);
409             register();
410             notifyListeners();
411         }
412         return this;
413     }

449     protected Broadcaster getBroadcaster(boolean autoCreate) {
450         if (broadcaster == null) {
451             throw new IllegalStateException("No Broadcaster associated with this AtmosphereResource.");
452         }
453 
454         String s = config.getInitParameter(ApplicationConfig.RECOVER_DEAD_BROADCASTER);
455         if (s != null) {
456             autoCreate = Boolean.parseBoolean(s);
457         }
458 
459         if (autoCreate && broadcaster.isDestroyed() && config.getBroadcasterFactory() != null) {
460             logger.debug("Broadcaster {} has been destroyed and cannot be re-used. Recreating a new one with the same name. You can turn off this" +
461                     " mechanism by adding, in web.xml, {} set to false", broadcaster.getID(), ApplicationConfig.RECOVER_DEAD_BROADCASTER);
462 
463             Broadcaster.SCOPE scope = broadcaster.getScope();
464             synchronized (this) {
465                 String id = scope != Broadcaster.SCOPE.REQUEST ? broadcaster.getID() : broadcaster.getID() + ".recovered"
466                         + config.uuidProvider().generateUuid();
467 
468                 // Another Thread may have added the Broadcaster.
469                 broadcaster = config.getBroadcasterFactory().lookup(id, true);
470                 broadcaster.setScope(scope);
471                 broadcaster.addAtmosphereResource(this);
472             }
473         }
474         return broadcaster;
475     }

494     protected AtmosphereResource uniqueBroadcaster(Broadcaster newB) {
495         if (newB == null) {
496             return this;
497         }
498         // TODO: performance bottleneck
499         for (Broadcaster b: broadcasters) {
500             if (b.getID() != null && b.getID().equalsIgnoreCase(newB.getID())) {
501                 logger.trace("Duplicate Broadcaster {}", newB);
502                 return this;
503             }
504         }
505         broadcasters.add(newB);
506         return this;
507     }

629     @Override
630     public AtmosphereResource notifyListeners(AtmosphereResourceEvent event) {
631         if (listeners.isEmpty() && config.framework().atmosphereResourceListeners().isEmpty()) {
632             logger.trace("No listener with {}", uuid);
633             return this;
634         }
635         logger.trace("Invoking listener {} for {}", listeners, uuid);
636 
637         Action oldAction = action;
638         try {
639             if (HeartbeatAtmosphereResourceEvent.class.isAssignableFrom(event.getClass())) {
640                 onHeartbeat(event);
641             } else if (event.isClosedByApplication()) {
642                 onClose(event);
643             } else if (event.isCancelled() || event.isClosedByClient()) {
644                 if (!disconnected.getAndSet(true)) {
645                     onDisconnect(event);
646                 } else {
647                     logger.trace("Skipping notification, already disconnected {}", event.getResource().uuid());
648                 }
649             } else if (event.isResuming() || event.isResumedOnTimeout()) {
650                 onResume(event);
651             } else if (!isSuspendEvent.getAndSet(true) && event.isSuspended()) {
652                 onSuspend(event);
653             } else if (event.throwable() != null) {
654                 onThrowable(event);
655             } else {
656                 onBroadcast(event);
657             }
658 
659             if (oldAction.type() != action.type()) {
660                 action().type(Action.TYPE.CREATED);
661             }
662         } catch (Throwable t) {
663             AtmosphereResourceEventImpl.class.cast(event).setThrowable(t);
664             if (event.isSuspended()) {
665                 logger.warn("Exception during suspend() operation {}", t.toString());
666                 logger.debug("", t);
667                 removeFromAllBroadcasters();
668             } else {
669                 logger.debug("Listener error {}", t);
670             }
671 
672             try {
673                 onThrowable(event);
674             } catch (Throwable t2) {
675                 logger.warn("Listener error {}", t2);
676             }
677         }
678         return this;
679     }

721     void onSuspend(AtmosphereResourceEvent e) {
722         for (AtmosphereResourceEventListener r : listeners) {
723             if (disableSuspendEvent) {
724                 if (!AllowInterceptor.class.isAssignableFrom(r.getClass())) {
725                     continue;
726                 }

729         }
730         if (e.getResource() != null) {
731             config.framework().notifySuspended(e.getResource().uuid());
732         }

789     public void cancel() throws IOException {
790         try {
791             if (!isCancelled.getAndSet(true)) {
792                 suspended.set(false);
793                 logger.trace("Cancelling {}", uuid);
794 
795                 if (config.getBroadcasterFactory() != null) {
796                     removeFromAllBroadcasters();
797                     if (transport.equals(TRANSPORT.WEBSOCKET)) {
798                         String parentUUID = (String) req.getAttribute(SUSPENDED_ATMOSPHERE_RESOURCE_UUID);
799                         AtmosphereResource p = config.resourcesFactory().find(parentUUID);
800                         if (p != null) {
801                             p.removeFromAllBroadcasters();
802                         }
803                     }
804                 }
805 
806                 asyncSupport.complete(this);
807 
808                 try {
809                     Object o = req.getAttribute(AtmosphereResourceImpl.METEOR);
810                     if (o != null && Meteor.class.isAssignableFrom(o.getClass())) {
811                         Meteor.class.cast(o).destroy();
812                     }
813                 } catch (Exception ex) {
814                     logger.trace("Meteor exception {}", ex);
815                 }
816 
817                 SessionTimeoutSupport.restoreTimeout(req);
818                 action.type(Action.TYPE.CANCELLED);
819                 if (asyncSupport != null) asyncSupport.action(this);
820                 // We must close the underlying WebSocket as well.
821                 if (AtmosphereResponseImpl.class.isAssignableFrom(response.getClass())) {
822                     if (closeOnCancel) {
823                         AtmosphereResponseImpl.class.cast(response).close();
824                     }
825                     AtmosphereResponseImpl.class.cast(response).destroy();
826                 }
827 
828                 if (AtmosphereRequestImpl.class.isAssignableFrom(req.getClass())) {
829                     if (closeOnCancel) {
830                         AtmosphereRequestImpl.class.cast(req).destroy();
831                     }
832                 }
833                 event.destroy();
834             }
835         } finally {
836             unregister();
837         }
838     }

841         config.resourcesFactory().remove(uuid);
842     }

844     public void _destroy() {
845         try {
846             if (!isCancelled.get()) {
847                 removeFromAllBroadcasters();
848             }
849             broadcasters.clear();
850 
851             unregister();
852             removeEventListeners();
853         } catch (Throwable t) {
854             logger.trace("destroyResource", t);
855         } finally {
856             unregister();
857             webSocket = null;
858         }
859     }

940     public AtmosphereResourceImpl cloneState(AtmosphereResource r) {
941         for (AtmosphereResourceEventListener l : AtmosphereResourceImpl.class.cast(r).atmosphereResourceEventListener()) {
942             addEventListener(l);
943         }
944         AtmosphereResourceImpl.class.cast(r).session(r.session());
945         boolean isFirst = true;
946         for (Broadcaster b : broadcasters) {
947             if (isFirst) {
948                 isFirst = false;
949                 setBroadcaster(b);
950             } else {
951                 addBroadcaster(b);
952             }
953         }
954         atmosphereHandler(r.getAtmosphereHandler());
955         return this;
956     }

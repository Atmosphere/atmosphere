136         beginCompatibleData = createStreamingPadding(padding);

159     public void resume() {
160         // Strangely but possible two thread try to resume at the same time.
161         try {
162         synchronized (event) {
163             if (!event.isResuming() && !event.isResumedOnTimeout() && event.isSuspended() && isInScope) {
164                 action.type = AtmosphereServlet.Action.TYPE.RESUME;
165 
166                 try {
167                     logger.debug("Resuming {}", getRequest());
168                 } catch (Throwable ex) {
169                     // Jetty NPE toString()
170                     // Ignore
171                     // Stop here as the request object as becomes invalid.
172                     return;
173                 }
174 
175                 // We need it as Jetty doesn't support timeout
176                 Broadcaster b = getBroadcaster(false);
177                 if (!b.isDestroyed() && b instanceof DefaultBroadcaster) {
178                     ((DefaultBroadcaster) b).broadcastOnResume(this);
179                 }
180 
181                 notifyListeners();
182                 listeners.clear();
183 
184                 try {
185                     if (!b.isDestroyed()) {
186                         broadcaster.removeAtmosphereResource(this);
187                     }
188                 } catch (IllegalStateException ex) {
189                     logger.warn("Unable to resume", this);
190                     logger.debug(ex.getMessage(), ex);
191                 }
192 
193                 if (b.getScope() == Broadcaster.SCOPE.REQUEST) {
194                     logger.debug("Broadcaster's scope is set to request, destroying it {}", b.getID());
195                     b.destroy();
196                 }
197 
198                 // Resuming here means we need to pull away from all other Broadcaster, if they exists.
199                 if (BroadcasterFactory.getDefault() != null) {
200                     BroadcasterFactory.getDefault().removeAllAtmosphereResource(this);
201                 }
202 
203                 try {
204                     req.setAttribute(ApplicationConfig.RESUMED_ON_TIMEOUT, Boolean.FALSE);
205                 } catch (Exception ex) {
206                     logger.debug("Cannot resume an already resumed/cancelled request");
207                 }
208                 if (req.getAttribute(PRE_SUSPEND) == null) {
209                     cometSupport.action(this);
210                 }
211             } else {
212                 logger.debug("Cannot resume an already resumed/cancelled request {}", getRequest());
213             }
214         }
215         } catch (Throwable t) {
216             logger.trace("Wasn't able to resume a connection {}", this, t);
217         }
218     }

253     public void suspend(long timeout, boolean flushComment) {
254 
255         if (req.getSession(false) != null && req.getSession().getMaxInactiveInterval() != -1 && req.getSession().getMaxInactiveInterval() * 1000 < timeout) {
256             throw new IllegalStateException("Cannot suspend a " +
257                     "response longer than the session timeout. Increase the value of session-timeout in web.xml");
258         }
259 
260         if (req.getAttribute(DefaultBroadcaster.CACHED) != null) {
261             // Do nothing because we have found cached message which was written already, and the handler resumed.
262             req.removeAttribute(DefaultBroadcaster.CACHED);
263             return;
264         }
265 
266         if (!event.isResumedOnTimeout()) {
267 
268             if (req.getHeaders("Connection") != null && req.getHeaders("Connection").hasMoreElements()) {
269                 String[] e = req.getHeaders("Connection").nextElement().split(",");
270                 for (String upgrade : e) {
271                     if (upgrade.trim().equalsIgnoreCase(WEBSOCKET_UPGRADE)) {
272                         if (writeHeaders && !cometSupport.supportWebSocket()) {
273                             response.addHeader(X_ATMOSPHERE_ERROR, "Websocket protocol not supported");
274                         } else {
275                             flushComment = false;
276                         }
277                     }
278                 }
279             }
280 
281             if (writeHeaders && injectCacheHeaders) {
282                 // Set to expire far in the past.
283                 response.setHeader(EXPIRES, "-1");
284                 // Set standard HTTP/1.1 no-cache headers.
285                 response.setHeader(CACHE_CONTROL, "no-store, no-cache, must-revalidate");
286                 // Set standard HTTP/1.0 no-cache header.
287                 response.setHeader(PRAGMA, "no-cache");
288             }
289 
290             if (writeHeaders && enableAccessControl) {
291                 response.setHeader(ACCESS_CONTROL_ALLOW_ORIGIN, "*");
292                 response.setHeader(ACCESS_CONTROL_ALLOW_CREDENTIALS, "true");
293             }
294 
295             if (flushComment) {
296                 write();
297             }
298             req.setAttribute(PRE_SUSPEND, "true");
299             action.type = AtmosphereServlet.Action.TYPE.SUSPEND;
300             action.timeout = timeout;
301 
302             // TODO: We can possibly optimize that call by avoiding creating a Broadcaster if we are sure the Broadcaster
303             // is unique.
304             boolean isJersey = req.getAttribute(FrameworkConfig.CONTAINER_RESPONSE) != null;
305             if (broadcaster.getScope() == Broadcaster.SCOPE.REQUEST && !isJersey) {
306                 String id = broadcaster.getID();
307                 Class<? extends Broadcaster> clazz = broadcaster.getClass();
308                 broadcaster = BroadcasterFactory.getDefault().lookup(clazz, id, false);
309                 if (broadcaster == null || broadcaster.getAtmosphereResources().size() > 0) {
310                     broadcaster = BroadcasterFactory.getDefault().lookup(clazz, id + "/" + UUID.randomUUID(), true);
311                 }
312             }
313 
314             broadcaster.addAtmosphereResource(this);
315             req.removeAttribute(PRE_SUSPEND);
316             notifyListeners();
317         }
318     }

320     void write() {
321         try {
322             if (useWriter && !((Boolean) req.getAttribute(ApplicationConfig.PROPERTY_USE_STREAM))) {
323                 try {
324                     response.getWriter();
325                 } catch (IllegalStateException e) {
326                     return;
327                 }
328 
329                 response.getWriter().write(beginCompatibleData);
330                 response.getWriter().flush();
331             } else {
332                 try {
333                     response.getOutputStream();
334                 } catch (IllegalStateException e) {
335                     return;
336                 }
337 
338                 response.getOutputStream().write(beginCompatibleData.getBytes());
339                 response.getOutputStream().flush();
340             }
341 
342         } catch (Throwable ex) {
343             logger.warn("failed to write to response", ex);
344         }
345     }

375     private Broadcaster getBroadcaster(boolean autoCreate) {
376         if (broadcaster == null) {
377             throw new IllegalStateException("No Broadcaster associated with this AtmosphereResource.");
378         }
379 
380         String s = config.getInitParameter(ApplicationConfig.RECOVER_DEAD_BROADCASTER);
381         if (s != null) {
382             autoCreate = Boolean.parseBoolean(s);
383         }
384 
385         if (autoCreate && broadcaster.isDestroyed() && BroadcasterFactory.getDefault() != null) {
386             logger.warn("Broadcaster {} has been destroyed and cannot be re-used. Recreating a new one with the same name. You can turn off that" +
387                     " mechanism by adding, in web.xml, {} set to false", broadcaster.getID(), ApplicationConfig.RECOVER_DEAD_BROADCASTER);
388 
389             synchronized (this) {
390                 String id = broadcaster.getScope() != Broadcaster.SCOPE.REQUEST ? broadcaster.getID() : broadcaster.getID() + ".recovered" + UUID.randomUUID();
391 
392                 // Another Thread may have added the Broadcaster.
393                 broadcaster = BroadcasterFactory.getDefault().lookup(id, true);
394                 broadcaster.addAtmosphereResource(this);
395             }
396         }
397         return broadcaster;
398     }

483     public static String createStreamingPadding(String padding) {
484         StringBuilder s = new StringBuilder();
485 
486         if (padding == null || padding.equalsIgnoreCase("atmosphere")) {
487             s.append("<!-- ----------------------------------------------------------" +
488                     "------ http://github.com/Atmosphere ----------------------------" +
489                     "-------------------------------------------- -->\n");
490             s.append("<!-- Welcome to the Atmosphere Framework. To work with all the" +
491                     " browsers when suspending connection, Atmosphere must output some" +
492                     " data to makes WebKit based browser working.-->\n");
493             for (int i = 0; i < 10; i++) {
494                 s.append("<!-- ----------------------------------------------------------" +
495                         "---------------------------------------------------------------" +
496                         "-------------------------------------------- -->\n");
497             }
498             s.append("<!-- EOD -->");
499         } else {
500             for (int i = 0; i < 2048; i++) {
501                 s.append(" ");
502             }
503         }
504         return s.toString();
505     }

540     /**
541      * Notify {@link AtmosphereResourceEventListener}.
542      */
543     public void notifyListeners(AtmosphereResourceEvent event) {
544         if (listeners.size() > 0) {
545             logger.trace("Invoking listener with {}", event);
546         } else {
547             return;
548         }
549 
550         if (event.isResuming() || event.isResumedOnTimeout()) {
551             onResume(event);
552         } else if (event.isCancelled()) {
553             onDisconnect(event);
554         } else if (!isSuspendEvent.getAndSet(true) && event.isSuspended()) {
555             onSuspend(event);
556         } else if (event.throwable() != null) {
557             onThrowable(event);
558         } else {
559             onBroadcast(event);
560         }
561     }

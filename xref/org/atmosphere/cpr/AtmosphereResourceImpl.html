146     public void resume() {
147         if (!event.isResuming() && !event.isResumedOnTimeout() && event.isSuspended() && isInScope) {
148             action.type = AtmosphereServlet.Action.TYPE.RESUME;
149 
150             // We need it as Jetty doesn't support timeout
151             Broadcaster b = getBroadcaster();
152             if (b instanceof DefaultBroadcaster) {
153                 ((DefaultBroadcaster) b).broadcastOnResume(this);
154             }
155 
156             notifyListeners();
157             listeners.clear();
158             try {
159                 broadcaster.removeAtmosphereResource(this);
160             } catch (IllegalStateException ex) {
161                 logger.trace(ex.getMessage(), ex);
162             }
163 
164             // Resuming here means we need to pull away from all other Broadcaster, if they exists.
165             if (BroadcasterFactory.getDefault() != null) {
166                 BroadcasterFactory.getDefault().removeAllAtmosphereResource(this);
167             }
168 
169             try {
170                 req.setAttribute(ApplicationConfig.RESUMED_ON_TIMEOUT, Boolean.FALSE);
171             } catch (Exception ex) {
172                 logger.debug("Cannot resume an already resumed/cancelled request");
173             }
174             if (req.getAttribute(PRE_SUSPEND) == null) {
175                 cometSupport.action(this);
176             }
177         } else {
178             logger.debug("Cannot resume an already resumed/cancelled request");
179         }
180     }

215     public void suspend(long timeout, boolean flushComment) {
216 
217         if (req.getSession(false) != null && req.getSession().getMaxInactiveInterval() != -1 && req.getSession().getMaxInactiveInterval() * 1000 < timeout) {
218             throw new IllegalStateException("Cannot suspend a " +
219                     "response longer than the session timeout. Increase the value of session-timeout in web.xml");
220         }
221 
222         if(req.getAttribute(DefaultBroadcaster.CACHED) != null) {
223             // Do nothing because we have found cached message which was written already, and the handler resumed.
224             req.removeAttribute(DefaultBroadcaster.CACHED);
225             return;
226         }
227 
228         if (!event.isResumedOnTimeout()) {
229 
230             if (req.getHeaders("Connection") != null && req.getHeaders("Connection").hasMoreElements()) {
231                 String[] e = req.getHeaders("Connection").nextElement().split(",");
232                 for (String upgrade : e) {
233                     if (upgrade.trim().equalsIgnoreCase("Upgrade")) {
234                         if (writeHeaders && !cometSupport.supportWebSocket()) {
235                             response.addHeader("X-Atmosphere-error", "Websocket protocol not supported");
236                         } else {
237                             flushComment = false;
238                         }
239                     }
240                 }
241             }
242 
243             if (writeHeaders && injectCacheHeaders) {
244                 // Set to expire far in the past.
245                 response.setHeader("Expires", "-1");
246                 // Set standard HTTP/1.1 no-cache headers.
247                 response.setHeader("Cache-Control", "no-store, no-cache, must-revalidate");
248                 // Set standard HTTP/1.0 no-cache header.
249                 response.setHeader("Pragma", "no-cache");
250             }
251 
252             if (writeHeaders && enableAccessControl) {
253                 response.setHeader("Access-Control-Allow-Origin", "*");
254                 response.setHeader("Access-Control-Allow-Credentials", "true");
255             }
256 
257             if (flushComment) {
258                 write();
259             }
260             req.setAttribute(PRE_SUSPEND, "true");
261             action.type = AtmosphereServlet.Action.TYPE.SUSPEND;
262             action.timeout = timeout;
263 
264             // TODO: We can possibly optimize that call by avoiding creating a Broadcaster if we are sure the Broadcaster
265             // is unique.
266             if (broadcaster.getScope() == Broadcaster.SCOPE.REQUEST) {
267                 String id = broadcaster.getID();
268                 Class<? extends Broadcaster> clazz = broadcaster.getClass();
269                 broadcaster = BroadcasterFactory.getDefault().lookup(clazz, id, false);
270                 if (broadcaster == null || broadcaster.getAtmosphereResources().size() > 0) {
271                     broadcaster = BroadcasterFactory.getDefault().lookup(clazz, id + "/" + UUID.randomUUID(), true);
272                 }
273             }
274 
275             broadcaster.addAtmosphereResource(this);
276             req.removeAttribute(PRE_SUSPEND);
277             notifyListeners();
278         }
279     }

281     void write() {
282         try {
283             if (useWriter && !((Boolean) req.getAttribute(ApplicationConfig.PROPERTY_USE_STREAM))) {
284                 try {
285                     response.getWriter();
286                 } catch (IllegalStateException e) {
287                     return;
288                 }
289 
290                 response.getWriter().write(beginCompatibleData);
291                 response.getWriter().flush();
292             } else {
293                 try {
294                     response.getOutputStream();
295                 } catch (IllegalStateException e) {
296                     return;
297                 }
298 
299                 response.getOutputStream().write(beginCompatibleData.getBytes());
300                 response.getOutputStream().flush();
301             }
302 
303         } catch (Throwable ex) {
304             logger.warn("failed to write to response", ex);
305         }
306     }

476     /**
477      * Notify {@link AtmosphereResourceEventListener}.
478      */
479     public void notifyListeners(AtmosphereResourceEvent event) {
480         logger.debug("Invoking listener with {}", event);
481         if (event.isResuming() || event.isResumedOnTimeout()) {
482             onResume(event);
483         } else if (event.isCancelled()) {
484             onDisconnect(event);
485         } else if (!isSuspendEvent.getAndSet(true) && event.isSuspended()) {
486             onSuspend(event);
487         } else if (event.throwable() != null) {
488             onThrowable(event);
489         } else {
490             onBroadcast(event);
491         }
492     }

16  package org.atmosphere.cpr;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  /**
22   * Universe contains static reference to Atmosphere's Factories.
23   * </p>
24   * PLEASE DO NOT USE THIS CLASS IF YOUR APPLICATION CONTAINS WEBFRAGMENTS OR MORE THAN ONCE ATMOSPHERE SERVLET DEFINED
25   * AS THIS CLASS IS USING STATIC INSTANCE.
26   * </p>
27   * <p/>
28   * This is ugly, only here to save your buts, Atmosphere users!
29   *
30   * @author Jeanfrancois Arcand
31   */
32  public class Universe {
33  
34      private static final Logger logger = LoggerFactory.getLogger(Universe.class);
35      private static BroadcasterFactory factory;
36      private static AtmosphereFramework framework;
37      private static AtmosphereResourceFactory resourceFactory;
38      private static AtmosphereResourceSessionFactory sessionFactory;
39      private static DefaultMetaBroadcaster metaBroadcaster;
40  
41      /**
42       * Set the must be unique {@link DefaultMetaBroadcaster}
43       *
44       * @param a {@link DefaultMetaBroadcaster}
45       */
46      public static void metaBroadcaster(DefaultMetaBroadcaster a) {
47          if (metaBroadcaster != null) {
48              logger.warn("More than one Universe configured. Universe class will gives wrong object reference {}", a);
49          }
50          metaBroadcaster = a;
51      }
52  
53      /**
54       * Set the must be unique {@link org.atmosphere.cpr.BroadcasterFactory}
55       *
56       * @param a {@link org.atmosphere.cpr.BroadcasterFactory}
57       */
58      public static void broadcasterFactory(BroadcasterFactory a) {
59          if (factory != null) {
60              logger.warn("More than one Universe configured. Universe class will gives wrong object reference {}", a);
61          }
62          factory = a;
63      }
64  
65      /**
66       * Set the must be unique {@link org.atmosphere.cpr.AtmosphereFramework}
67       *
68       * @param a {@link org.atmosphere.cpr.AtmosphereFramework}
69       */
70      public static void framework(AtmosphereFramework a) {
71          if (framework != null) {
72              logger.warn("More than one Universe configured. Universe class will gives wrong object reference {}", a);
73          }
74          framework = a;
75      }
76  
77      /**
78       * Set the must be unique {@link AtmosphereResourceFactory}
79       *
80       * @param a {@link AtmosphereResourceFactory}
81       */
82      public static void resourceFactory(AtmosphereResourceFactory a) {
83          if (resourceFactory != null) {
84              logger.warn("More than one Universe configured. Universe class will gives wrong object reference {}", a);
85          }
86          resourceFactory = a;
87      }
88  
89      /**
90       * Set the must be unique {@link org.atmosphere.cpr.AtmosphereResourceSessionFactory}
91       *
92       * @param a {@link org.atmosphere.cpr.AtmosphereResourceSessionFactory}
93       */
94      public static void sessionResourceFactory(AtmosphereResourceSessionFactory a) {
95          if (sessionFactory != null) {
96              logger.warn("More than one Universe configured. Universe class will gives wrong object reference {}", a);
97          }
98          sessionFactory = a;
99      }
100 
101     /**
102      * Return the {@link org.atmosphere.cpr.BroadcasterFactory}
103      *
104      * @return the {@link org.atmosphere.cpr.BroadcasterFactory}
105      */
106     public static BroadcasterFactory broadcasterFactory() {
107         return factory;
108     }
109 
110     /**
111      * Return the {@link org.atmosphere.cpr.AtmosphereFramework}
112      *
113      * @return the {@link org.atmosphere.cpr.AtmosphereFramework}
114      */
115     public static AtmosphereFramework framework() {
116         return framework;
117     }
118 
119     /**
120      * Return the {@link AtmosphereResourceFactory}
121      *
122      * @return the {@link AtmosphereResourceFactory}
123      */
124     public static AtmosphereResourceFactory resourceFactory() {
125         return resourceFactory;
126     }
127 
128     /**
129      * Return the {@link org.atmosphere.cpr.AtmosphereResourceSessionFactory}
130      *
131      * @return the {@link org.atmosphere.cpr.AtmosphereResourceSessionFactory}
132      */
133     public static AtmosphereResourceSessionFactory sessionFactory() {
134         return sessionFactory;
135     }
136 
137     /**
138      * Return the {@link DefaultMetaBroadcaster}
139      *
140      * @return the {@link DefaultMetaBroadcaster}
141      */
142     public static DefaultMetaBroadcaster metaBroadcaster() {
143         return metaBroadcaster;
144     }
145 }

39  /**
40   * This class is responsible for creating {@link Broadcaster} instances. You can also add and remove {@link Broadcaster}
41   *
42   * @author Jeanfrancois Arcand
43   * @author Jason Burgess
44   */
45  public class DefaultBroadcasterFactory implements BroadcasterFactory {
46      protected final ConcurrentLinkedQueue<BroadcasterListener> broadcasterListeners = new ConcurrentLinkedQueue<BroadcasterListener>();
47  
48      private static final Logger logger = LoggerFactory.getLogger(DefaultBroadcasterFactory.class);
49  
50      protected final ConcurrentHashMap<Object, Broadcaster> store = new ConcurrentHashMap<Object, Broadcaster>();
51  
52      protected Class<? extends Broadcaster> clazz;
53  
54      protected BroadcasterLifeCyclePolicy policy =
55              new BroadcasterLifeCyclePolicy.Builder().policy(NEVER).build();
56      protected Broadcaster.POLICY defaultPolicy = Broadcaster.POLICY.FIFO;
57      protected int defaultPolicyInteger = -1;
58  
59      protected AtmosphereConfig config;
60      protected final BroadcasterListener lifeCycleListener = new BroadcasterLifecyclePolicyHandler();
61      public static final URI legacyBroadcasterURI = URI.create("http://127.0.0.0");
62  
63      public DefaultBroadcasterFactory() {
64      }

73      public void configure(Class<? extends Broadcaster> clazz, String broadcasterLifeCyclePolicy, AtmosphereConfig c) {
74          this.clazz = clazz;
75          config = c;
76          configure(broadcasterLifeCyclePolicy);
77      }
78  
79      protected void configure(String broadcasterLifeCyclePolicy) {
80  
81          int maxIdleTime = 5 * 60 * 1000;
82          String s = config.getInitParameter(ApplicationConfig.BROADCASTER_LIFECYCLE_POLICY_IDLETIME);
83          if (s != null) {
84              maxIdleTime = Integer.parseInt(s);
85          }
86  
87          s = config.getInitParameter(BROADCASTER_POLICY);
88          if (s != null) {
89              defaultPolicy = s.equalsIgnoreCase(Broadcaster.POLICY.REJECT.name()) ? Broadcaster.POLICY.REJECT : Broadcaster.POLICY.FIFO;
90          }
91  
92          s = config.getInitParameter(BROADCASTER_POLICY_TIMEOUT);
93          if (s != null) {
94              defaultPolicyInteger = Integer.valueOf(s);
95          }
96  
97          if (EMPTY.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
98              policy = new BroadcasterLifeCyclePolicy.Builder().policy(EMPTY).build();
99          } else if (EMPTY_DESTROY.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
100             policy = new BroadcasterLifeCyclePolicy.Builder().policy(EMPTY_DESTROY).build();
101         } else if (IDLE.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
102             policy = new BroadcasterLifeCyclePolicy.Builder().policy(IDLE).idleTimeInMS(maxIdleTime).build();
103         } else if (IDLE_DESTROY.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
104             policy = new BroadcasterLifeCyclePolicy.Builder().policy(IDLE_DESTROY).idleTimeInMS(maxIdleTime).build();
105         } else if (IDLE_RESUME.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
106             policy = new BroadcasterLifeCyclePolicy.Builder().policy(IDLE_RESUME).idleTimeInMS(maxIdleTime).build();
107         } else if (NEVER.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
108             policy = new BroadcasterLifeCyclePolicy.Builder().policy(NEVER).build();
109         } else {
110             logger.warn("Unsupported BroadcasterLifeCyclePolicy policy {}", broadcasterLifeCyclePolicy);
111         }
112 
113         broadcasterListeners.add(lifeCycleListener);
114     }

126     @Override
127     public <T extends Broadcaster> T get(Class<T> c, Object id) {
128 
129         if (id == null) {
130             throw new NullPointerException("id is null");
131         }
132         if (c == null) {
133             throw new NullPointerException("Class is null");
134         }
135 
136         return lookup(c, id, true, true);
137     }
138 
139     protected <T extends Broadcaster> T createBroadcaster(Class<T> c, Object id) throws BroadcasterCreationException {
140         try {
141             T b = config.framework().newClassInstance(c, c);
142             b.initialize(id.toString(), legacyBroadcasterURI, config);
143             b.setSuspendPolicy(defaultPolicyInteger, defaultPolicy);
144 
145             if (b.getBroadcasterConfig() == null) {
146                 b.setBroadcasterConfig(new BroadcasterConfig(config.framework().broadcasterFilters, config, id.toString()).init());
147             }
148 
149             b.setBroadcasterLifeCyclePolicy(policy);
150             if (DefaultBroadcaster.class.isAssignableFrom(clazz)) {
151                 DefaultBroadcaster.class.cast(b).start();
152             }
153 
154             for (BroadcasterListener l : broadcasterListeners) {
155                 b.addBroadcasterListener(l);
156             }
157 
158             logger.trace("Broadcaster {} was created {}", id, b);
159 
160             notifyOnPostCreate(b);
161             return b;
162         } catch (Throwable t) {
163             throw new BroadcasterCreationException(t);
164         }
165     }

201     public <T extends Broadcaster> T lookup(Class<T> c, Object id, boolean createIfNull, boolean unique) {
202         synchronized (c) {
203             logger.trace("About to create {}", id);
204             if (unique && store.get(id) != null) {
205                 throw new IllegalStateException("Broadcaster already exists " + id + ". Use BroadcasterFactory.lookup instead");
206             }
207 
208             T b = (T) store.get(id);
209             logger.trace("Looking in the store using {} returned {}", id, b);
210             if (b != null && !c.isAssignableFrom(b.getClass())) {
211                 String msg = "Invalid lookup class " + c.getName() + ". Cached class is: " + b.getClass().getName();
212                 logger.debug(msg);
213                 throw new IllegalStateException(msg);
214             }
215 
216             if ((b == null && createIfNull) || (b != null && b.isDestroyed())) {
217                 if (b != null) {
218                     logger.trace("Removing destroyed Broadcaster {}", b.getID());
219                     store.remove(b.getID(), b);
220                 }
221 
222                 Broadcaster nb = store.get(id);
223                 if (nb == null) {
224                     nb = createBroadcaster(c, id);
225                     store.put(id, nb);
226                 }
227 
228                 if (nb == null && logger.isTraceEnabled()) {
229                     logger.trace("Added Broadcaster {} . Factory size: {}", id, store.size());
230                 }
231 
232                 b = (T) nb;
233             }
234             return b;
235         }
236     }

238     @Deprecated
239     @Override
240     public void removeAllAtmosphereResource(AtmosphereResource r) {
241         // Remove inside all Broadcaster as well.
242         try {
243             if (!store.isEmpty()) {
244                 for (Broadcaster b : lookupAll()) {
245                     try {
246                         b.removeAtmosphereResource(r);
247                     } catch (IllegalStateException ex) {
248                         logger.trace(ex.getMessage(), ex);
249                     }
250                 }
251             }
252         } catch (Exception ex) {
253             logger.warn(ex.getMessage(), ex);
254         }
255     }

267     @Override
268     public synchronized void destroy() {
269         // Invalid state
270         if (config == null) return;
271 
272         String s = config.getInitParameter(ApplicationConfig.SHARED);
273         if (s != null && s.equalsIgnoreCase("true")) {
274             logger.warn("Factory shared, will not be destroyed. This can possibly cause memory leaks if" +
275                     "Broadcasters were created. Make sure you destroy them manually.");
276             return;
277         }
278 
279         Enumeration<Broadcaster> e = store.elements();
280         Broadcaster b;
281         // We just need one when shared.
282         BroadcasterConfig bc = null;
283         while (e.hasMoreElements()) {
284             try {
285                 b = e.nextElement();
286                 bc = b.getBroadcasterConfig();
287                 bc.forceDestroy();
288                 b.destroy();
289             } catch (Throwable t) {
290                 // Shield us from any bad behaviour
291                 logger.debug("Destroy", t);
292             }
293         }
294         broadcasterListeners.clear();
295         store.clear();
296     }

298     public void notifyOnPostCreate(Broadcaster b) {
299         for (BroadcasterListener l : broadcasterListeners) {
300             try {
301                 l.onPostCreate(b);
302             } catch (Exception ex) {
303                 logger.warn("onPostCreate", ex);
304             }
305         }
306     }
307 
308     @Override
309     public BroadcasterFactory addBroadcasterListener(BroadcasterListener l) {
310         if (!broadcasterListeners.contains(l)) {
311             broadcasterListeners.add(l);
312         }
313 
314         for (Broadcaster b : store.values()) {
315             b.addBroadcasterListener(l);
316         }
317         return this;
318     }
319 
320     @Override
321     public BroadcasterFactory removeBroadcasterListener(BroadcasterListener l) {
322         broadcasterListeners.remove(l);
323         for (Broadcaster b : store.values()) {
324             b.removeBroadcasterListener(l);
325         }
326         return this;
327     }

59  /**
60   * This class is responsible for creating {@link Broadcaster} instance. You can also add and remove {@link Broadcaster}
61   * and lookup using {@link BroadcasterFactory#getDefault()} ()}
62   * from any Classes loaded using the same class loader.
63   *
64   * @author Jeanfrancois Arcand
65   */
66  public class DefaultBroadcasterFactory extends BroadcasterFactory {
67  
68      private static final Logger logger = LoggerFactory.getLogger(DefaultBroadcasterFactory.class);
69  
70      private final ConcurrentHashMap<Object, Broadcaster> store = new ConcurrentHashMap<Object, Broadcaster>();
71      
72      private final Class<? extends Broadcaster> clazz;
73      
74      private BroadcasterLifeCyclePolicy policy =
75              new BroadcasterLifeCyclePolicy.Builder().policy(NEVER).build();
76  
77      protected DefaultBroadcasterFactory(Class<? extends Broadcaster> clazz, String broadcasterLifeCyclePolicy, AtmosphereServlet.AtmosphereConfig c) {
78          this.clazz = clazz;
79          config = c;
80          if (factory == null) {
81              this.factory = this;
82          }
83          configure(broadcasterLifeCyclePolicy);
84      }
85  
86      private void configure(String broadcasterLifeCyclePolicy) {
87  
88          int maxIdleTime = 5 * 60 * 1000;
89          String idleTime = config.getInitParameter(ApplicationConfig.BROADCASTER_LIFECYCLE_POLICY_IDLETIME);
90          if (idleTime != null) {
91              maxIdleTime = Integer.parseInt(idleTime);
92          }
93  
94          if (EMPTY.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
95              policy = new BroadcasterLifeCyclePolicy.Builder().policy(EMPTY).build();
96          } else if (EMPTY_DESTROY.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
97              policy = new BroadcasterLifeCyclePolicy.Builder().policy(EMPTY_DESTROY).build();
98          } else if (IDLE.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
99              policy = new BroadcasterLifeCyclePolicy.Builder().policy(IDLE).idleTimeInMS(maxIdleTime).build();
100         } else if (IDLE_DESTROY.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
101             policy = new BroadcasterLifeCyclePolicy.Builder().policy(IDLE_DESTROY).idleTimeInMS(maxIdleTime).build();
102         } else if (IDLE_RESUME.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
103             policy = new BroadcasterLifeCyclePolicy.Builder().policy(IDLE_RESUME).idleTimeInMS(maxIdleTime).build();
104         } else if (NEVER.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
105             policy = new BroadcasterLifeCyclePolicy.Builder().policy(NEVER).build();
106         } else {
107             logger.warn("Unsupported BroadcasterLifeCyclePolicy policy {}", broadcasterLifeCyclePolicy);
108         }
109     }

125     /**
126      * {@inheritDoc}
127      */
128     public final Broadcaster get(Class<? extends Broadcaster> c, Object id) {
129 
130         if (id == null) {
131             throw new NullPointerException("id is null");
132         }
133         if (c == null) {
134             throw new NullPointerException("Class is null");
135         }
136 
137         if (store.containsKey(id)) {
138             throw new IllegalStateException("Broadcaster already existing " + id + ". Use BroadcasterFactory.lookup instead");
139         }
140 
141         return lookup(c, id, true);
142     }
143 
144     private Broadcaster createBroadcaster(Class<? extends Broadcaster> c, Object id) throws BroadcasterCreationException {
145         try {
146             Broadcaster b = c.getConstructor(String.class, AtmosphereServlet.AtmosphereConfig.class).newInstance(id.toString(), config);
147             InjectorProvider.getInjector().inject(b);
148             b.setBroadcasterConfig(new BroadcasterConfig(AtmosphereServlet.broadcasterFilters, config));
149             b.setBroadcasterLifeCyclePolicy(policy);
150             if (DefaultBroadcaster.class.isAssignableFrom(clazz)) {
151                 DefaultBroadcaster.class.cast(b).start();
152             }
153             return b;
154         } catch (Throwable t) {
155             throw new BroadcasterCreationException(t);
156         }
157     }

198     /**
199      * {@inheritDoc}
200      */
201     @Override
202     public Broadcaster lookup(Class<? extends Broadcaster> c, Object id, boolean createIfNull) {
203         Broadcaster b = store.get(id);
204         if (b != null && !c.isAssignableFrom(b.getClass())) {
205             String msg = "Invalid lookup class " + c.getName() + ". Cached class is: " + b.getClass().getName();
206             logger.debug(msg);
207             throw new IllegalStateException(msg);
208         }
209 
210         if ((b == null && createIfNull) || (b != null && b.isDestroyed())) {
211             if (b != null) {
212                 logger.debug("Removing destroyed Broadcaster {}", b.getID());
213                 store.remove(b.getID(), b);
214             }
215             if (store.putIfAbsent(id, createBroadcaster(c, id)) == null) {
216                 logger.debug("Added Broadcaster {} . Factory size: {}", id, store.size());
217             }
218             b = store.get(id);
219         }
220 
221         return b;
222     }

224     /**
225      * {@inheritDoc}
226      */
227     @Override
228     public void removeAllAtmosphereResource(AtmosphereResource<?, ?> r) {
229         // Remove inside all Broadcaster as well.
230         try {
231             if (store.size() > 0) {
232                 for (Broadcaster b : lookupAll()) {
233                     try {
234                         // Prevent deadlock
235                         if (b.getAtmosphereResources().contains(r)) {
236                             b.removeAtmosphereResource(r);
237                         }
238                     } catch (IllegalStateException ex) {
239                         logger.trace(ex.getMessage(), ex);
240                     }
241                 }
242             }
243         } catch (Exception ex) {
244             logger.warn(ex.getMessage(), ex);
245         }
246     }

256     /**
257      * {@inheritDoc}
258      */
259     public synchronized void destroy() {
260         Enumeration<Broadcaster> e = store.elements();
261         Broadcaster b;
262         // We just need one when shared.
263         BroadcasterConfig bc = null;
264         while (e.hasMoreElements()) {
265             try {
266                 b = e.nextElement();
267                 b.resumeAll();
268                 b.destroy();
269                 bc = b.getBroadcasterConfig();
270             } catch (Throwable t) {
271                 // Shield us from any bad behaviour
272                 logger.trace("Destroy", t);
273             }
274         }
275 
276         try {
277             if (bc != null) bc.forceDestroy();
278         } catch (Throwable t) {
279             logger.trace("Destroy", t);
280 
281         }
282 
283         store.clear();
284         factory = null;
285     }

287     /**
288      * Build a default {@link BroadcasterFactory} returned when invoking {@link #getDefault()} ()}.
289      *
290      * @param clazz A class implementing {@link Broadcaster}
291      * @param c     An instance of {@link AtmosphereServlet.AtmosphereConfig}
292      * @return the default {@link BroadcasterFactory}.
293      * @throws InstantiationException
294      * @throws IllegalAccessException
295      */
296     public static BroadcasterFactory buildAndReplaceDefaultfactory(Class<? extends Broadcaster> clazz, AtmosphereServlet.AtmosphereConfig c)
297             throws InstantiationException, IllegalAccessException {
298 
299         factory = new DefaultBroadcasterFactory(clazz, "NEVER", c);
300         return factory;
301     }

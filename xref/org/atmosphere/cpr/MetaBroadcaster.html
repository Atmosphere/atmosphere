37  /**
38   * Broadcast events to all or a subset of available {@link Broadcaster}s based on their {@link org.atmosphere.cpr.Broadcaster#getID()} value.
39   * This class allows broadcasting events to a set of broadcasters that maps to some String like:
40   * <blockquote><pre>
41   *        // Broadcast the event to all Broadcaster ID starting with /hello
42   *        broadcast("/hello", event)
43   *        // Broadcast the event to all Broadcaster ID
44   *        broaccast("/*", event);
45   * </pre></blockquote>
46   * The rule used is similar to path/URI mapping used by technology like Servlet, Jersey, etc.
47   * <p/>
48   * NOTE: Broadcasters' name must start with / in order to get retrieved by this class.
49   * <p/>
50   * This class is NOT thread safe.
51   * <p/>
52   * If you want to use MetaBroadcaster with Jersey or any framework, make sure all {@link org.atmosphere.cpr.Broadcaster#getID()}
53   * starts with '/'. For example, with Jersey:
54   * <blockquote><pre>
55   *
56   * @author Jeanfrancois Arcand
57   * @Path(RestConstants.STREAMING + "/workspace{wid:/[0-9A-Z]+}")
58   * public class JerseyPubSub {
59   * @PathParam("wid") private Broadcaster topic;
60   * </pre></blockquote>
61   */
62  public class MetaBroadcaster {
63  
64      public static final String MAPPING_REGEX = "[/a-zA-Z0-9-&.*=@_;\\?]+";
65  
66      private final static Logger logger = LoggerFactory.getLogger(MetaBroadcaster.class);
67      private final static MetaBroadcaster metaBroadcaster = new MetaBroadcaster();
68      private final static ConcurrentLinkedQueue<BroadcasterListener> broadcasterListeners = new ConcurrentLinkedQueue<BroadcasterListener>();
69      private final static MetaBroadcasterFuture E = new MetaBroadcasterFuture(Collections.<Broadcaster>emptyList());
70      private MetaBroadcasterCache cache = new NoCache();
71  
72      protected MetaBroadcasterFuture broadcast(final String path, Object message, int time, TimeUnit unit, boolean delay, boolean cacheMessage) {
73          if (BroadcasterFactory.getDefault() != null) {
74              Collection<Broadcaster> c = BroadcasterFactory.getDefault().lookupAll();
75  
76              final Map<String, String> m = new HashMap<String, String>();
77              List<Broadcaster> l = new ArrayList<Broadcaster>();
78              logger.trace("Map {}", path);
79              UriTemplate t = null;
80              try {
81                  t = new UriTemplate(path);
82                  for (Broadcaster b : c) {
83                      logger.trace("Trying to map {} to {}", t, b.getID());
84                      if (t.match(b.getID(), m)) {
85                          l.add(b);
86                      }
87                      m.clear();
88                  }
89              } finally {
90                  if (t != null) t.destroy();
91              }
92  
93              if (l.isEmpty() && cacheMessage) {
94                  if (NoCache.class.isAssignableFrom(cache.getClass())) {
95                      logger.warn("No Broadcaster matches {}. Message {} WILL BE LOST. " +
96                              "Make sure you cache it or make sure the Broadcaster exists before.", path, message);
97                  } else {
98                      cache.cache(path, message);
99                  }
100                 return E;
101             }
102 
103             MetaBroadcasterFuture f = new MetaBroadcasterFuture(l);
104             CompleteListener cl = new CompleteListener(f);
105 
106             for (Broadcaster b : l) {
107                 if (time <= 0) {
108                     f.outerFuture(b.addBroadcasterListener(cl).broadcast(message));
109                 } else if (!delay) {
110                     f.outerFuture(b.scheduleFixedBroadcast(message, time, unit));
111                 } else {
112                     f.outerFuture(b.delayBroadcast(message, time, unit));
113                 }
114             }
115 
116             return f;
117         } else {
118             return E;
119         }
120     }

122     protected MetaBroadcasterFuture map(String path, Object message, int time, TimeUnit unit, boolean delay, boolean cacheMessage) {
123 
124         if (path == null || path.isEmpty()) {
125             throw new NullPointerException();
126         }
127 
128         if (!path.startsWith("/")) {
129             path = "/" + path;
130         }
131 
132         if (path.contains("*")) {
133             path = path.replace("*", MAPPING_REGEX);
134         }
135 
136         if (path.equals("/")) {
137             path += MAPPING_REGEX;
138         }
139 
140         return broadcast(path, message, time, unit, delay, cacheMessage);
141     }

195     private final static class CompleteListener extends BroadcasterListenerAdapter {
196 
197         private final MetaBroadcasterFuture f;
198 
199         private CompleteListener(MetaBroadcasterFuture f) {
200             this.f = f;
201         }
202 
203         @Override
204         public void onPostCreate(Broadcaster b) {
205         }
206 
207         @Override
208         public void onComplete(Broadcaster b) {
209             b.removeBroadcasterListener(this);
210             f.countDown();
211             if (f.isDone()) {
212                 for (BroadcasterListener l : broadcasterListeners) {
213                     try {
214                         l.onComplete(b);
215                     } catch (Exception ex) {
216                         logger.warn("", ex);
217                     }
218                 }
219             }
220         }
221 
222         @Override
223         public void onPreDestroy(Broadcaster b) {
224         }
225     }

227     private final static class MetaBroadcasterFuture implements Future<List<Broadcaster>> {
228 
229         private final CountDownLatch latch;
230         private final List<Broadcaster> l;
231         private boolean isCancelled = false;
232         private final List<Future<?>> outerFuture = new ArrayList<Future<?>>();
233 
234         private MetaBroadcasterFuture(List<Broadcaster> l) {
235             this.latch = new CountDownLatch(l.size());
236             this.l = l;
237         }
238 
239         MetaBroadcasterFuture outerFuture(Future<?> f) {
240             outerFuture.add(f);
241             return this;
242         }
243 
244         @Override
245         public boolean cancel(boolean b) {
246             for (Future<?> f : outerFuture) {
247                 f.cancel(b);
248             }
249 
250             while (latch.getCount() > 0) {
251                 latch.countDown();
252             }
253             isCancelled = true;
254             return isCancelled;
255         }
256 
257         @Override
258         public boolean isCancelled() {
259             return isCancelled;
260         }
261 
262         @Override
263         public boolean isDone() {
264             return latch.getCount() == 0;
265         }
266 
267         @Override
268         public List<Broadcaster> get() throws InterruptedException, ExecutionException {
269             latch.await();
270             return l;
271         }
272 
273         @Override
274         public List<Broadcaster> get(long t, TimeUnit timeUnit) throws InterruptedException, ExecutionException, TimeoutException {
275             latch.await(t, timeUnit);
276             return l;
277         }
278 
279         public void countDown() {
280             latch.countDown();
281         }
282     }

350     /**
351      * Flush the cache every 30 seconds.
352      */
353     public final static class ThirtySecondsCache implements MetaBroadcasterCache, Runnable {
354 
355         private final MetaBroadcaster metaBroadcaster;
356         private final ConcurrentHashMap<String, Object> cache = new ConcurrentHashMap<String, Object>();
357 
358         public ThirtySecondsCache(MetaBroadcaster metaBroadcaster, AtmosphereConfig config) {
359             this.metaBroadcaster = metaBroadcaster;
360             ExecutorsFactory.getScheduler(config).scheduleAtFixedRate(this, 0, 30, TimeUnit.SECONDS);
361         }
362 
363         @Override
364         public MetaBroadcasterCache cache(String path, Object o) {
365             cache.put(path, o);
366             return this;
367         }
368 
369         @Override
370         public MetaBroadcasterCache flushCache() {
371             for (Map.Entry<String, Object> e : cache.entrySet()) {
372                 metaBroadcaster.map(e.getKey(), e.getValue(), -1, null, false, false);
373             }
374             return this;
375         }
376 
377         @Override
378         public void run() {
379             flushCache();
380             cache.clear();
381         }
382     }

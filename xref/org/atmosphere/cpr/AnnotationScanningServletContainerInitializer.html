33  /**
34   * A ServletContainerInitializer that scans for annotations, and places them in a map keyed by annotation type in the
35   * servlet context.
36   *
37   * NOTE: Any new Atmosphere's annotation must be hardcoded here.
38   *
39   * @author Stuart Douglas
40   */
41  @HandlesTypes({
42          AtmosphereHandlerService.class,
43          BroadcasterCacheService.class,
44          BroadcasterFilterService.class,
45          BroadcasterFactoryService.class,
46          BroadcasterService.class,
47          MeteorService.class,
48          WebSocketHandlerService.class,
49          WebSocketProtocolService.class,
50          AtmosphereInterceptorService.class,
51          BroadcasterListenerService.class,
52          AsyncSupportService.class,
53          AsyncSupportListenerService.class,
54          WebSocketProcessorService.class,
55          BroadcasterCacheInspectorService.class,
56          ManagedService.class,
57          AtmosphereService.class,
58          EndpointMapperService.class,
59          BroadcasterCacheListenerService.class,
60          AtmosphereAnnotation.class
61  })
62  public class AnnotationScanningServletContainerInitializer implements ServletContainerInitializer {
63  
64      @Override
65      public void onStartup(final Set<Class<?>> classes, final ServletContext servletContext) throws ServletException {
66          final Map<Class<? extends Annotation>, Set<Class<?>>> classesByAnnotation = new HashMap<Class<? extends Annotation>, Set<Class<?>>>();
67          if (classes != null) {
68              for(final Class<?> clazz : classes) {
69                  for(Annotation annotation : clazz.getAnnotations()) {
70                      Set<Class<?>> classSet = classesByAnnotation.get(annotation.annotationType());
71                      if(classSet == null) {
72                          classesByAnnotation.put(annotation.annotationType(), classSet = new HashSet<Class<?>>());
73                      }
74                      classSet.add(clazz);
75                  }
76              }
77          }
78          servletContext.setAttribute(DefaultAnnotationProcessor.ANNOTATION_ATTRIBUTE, classesByAnnotation);
79      }
80  }

16  package org.atmosphere.cpr;
17  
18  import org.atmosphere.container.BlockingIOCometSupport;
19  
20  import javax.servlet.ServletConfig;
21  import javax.servlet.ServletException;
22  import java.awt.event.ActionEvent;
23  import java.io.IOException;
24  
25  /**
26   * Atmosphere's supported WebServer must implement this interface in order to be auto detected by the
27   * {@link AtmosphereFramework}. If the {@link AtmosphereFramework} fail to detect the {@link AsyncSupport}, it will
28   * use a blocking thread approach to emulate Comet using the {@link BlockingIOCometSupport}.
29   * <p/>
30   * Framework designers or Atmosphere application developers can also add their own implementation of that class by
31   * referencing their class within the atmosphere.xml file:
32   * <p><pre><code>
33   * <&lt;atmosphere-handler ... comet-support="your.class.name"&gt;
34   * </code></pre></p>
35   *
36   * @author Jeanfrancois Arcand
37   */
38  public interface AsyncSupport<E extends AtmosphereResource> {
39  
40      /**
41       * Return the name of the Java Web Server.
42       *
43       * @return the name of the Java Web Server.
44       */
45      public String getContainerName();
46  
47      /**
48       * Initialize the WebServer using the {@link ServletConfig}
49       *
50       * @param sc the {@link ServletConfig}
51       * @throws javax.servlet.ServletException
52       */
53      public void init(ServletConfig sc) throws ServletException;
54  
55      /**
56       * Serve the {@link AtmosphereRequest} and the {@link AtmosphereResponse} and return
57       * the appropriate {@link Action}.
58       *
59       * @param req the {@link AtmosphereRequest}
60       * @param res the {@link AtmosphereResponse}
61       * @return the {@link Action} that was manipulated by the {@link AtmosphereHandler}
62       * @throws java.io.IOException
63       * @throws javax.servlet.ServletException
64       */
65      public Action service(AtmosphereRequest req, AtmosphereResponse res) throws IOException, ServletException;
66  
67      /**
68       * Process an {@link Action} from an {@link ActionEvent} operation like suspend, resume or timed out.
69       *
70       * @param actionEvent An instance of {@link Action}
71       */
72      public void action(E actionEvent);
73  
74      /**
75       * Return true if this implementation supports the websocket protocol.
76       *
77       * @return true if supported
78       */
79      public boolean supportWebSocket();
80  
81      /**
82       * Complete and close the connection associated with an implementation of {@link org.atmosphere.cpr.AtmosphereResource}
83       * @param r {@link org.atmosphere.cpr.AtmosphereResource}
84       * @return this
85       */
86      public AsyncSupport complete(E r);
87  }

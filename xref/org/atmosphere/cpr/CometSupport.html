44  import javax.servlet.ServletConfig;

46  import javax.servlet.http.HttpServletRequest;
47  import javax.servlet.http.HttpServletResponse;
48  import java.awt.event.ActionEvent;

51  /**
52   * Atmosphere's supported WebServer must implement this interface in order
53   * to be auto detected by the {@link AtmosphereServlet}. If the {@link AtmosphereServlet}
54   * fail to detect the {@link CometSupport}, it will use a blocking thread
55   * approach to emulate Comet using the {@link BlockingIOCometSupport}.
56   * <p/>
57   * Framework designer or Atmosphere application developer
58   * can also add their own implementation of that class by referencing their
59   * class within the atmosphere.xml file:
60   * <p><pre><code>
61   * <&lt;atmosphere-handler ... comet-support="your.class.name"&gt;
62   * </code></pre></p>
63   *
64   * @author Jeanfrancois Arcand
65   */
66  public interface CometSupport<E extends AtmosphereResource> {
67  
68      final String MAX_INACTIVE = CometSupport.class.getName() + ".maxInactiveActivity";
69  
70      /**
71       * Return the name of the Java Web Server.
72       *
73       * @return the name of the Java Web Server.
74       */
75      public String getContainerName();
76  
77      /**
78       * Initialize the WebServer using the {@link ServletConfig}
79       *
80       * @param sc the  {@link ServletConfig}
81       * @throws javax.servlet.ServletException
82       */
83      public void init(ServletConfig sc) throws ServletException;
84  
85      /**
86       * Serve the {@link HttpServletRequest} and the {@link HttpServletResponse} and return
87       * the appropriate {@link Action}.
88       *
89       * @param req the {@link HttpServletRequest}
90       * @param res the {@link HttpServletResponse}
91       * @return the {@link Action} that was manipulated by the {@link AtmosphereHandler}
92       * @throws java.io.IOException
93       * @throws javax.servlet.ServletException
94       */
95      public Action service(HttpServletRequest req, HttpServletResponse res)
96              throws IOException, ServletException;
97  
98      /**
99       * Process an {@link Action} from an {@link ActionEvent} operation like suspend, resume or timed out.
100      *
101      * @param actionEvent An instance of {@link AtmosphereServlet.Action}
102      */
103     public void action(E actionEvent);
104 
105     /**
106      * Return true if this implementation supports the websocket protocol.
107      * return true if supported
108      */
109     public boolean supportWebSocket();
110 }

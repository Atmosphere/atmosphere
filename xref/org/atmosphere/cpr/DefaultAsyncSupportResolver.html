90      /**
91       * Convenience method that tests if a class with the given FQN is present on the classpath.
92       *
93       * @param testClass
94       * @return true if the class is present
95       */
96      protected boolean testClassExists(final String testClass) {
97          try {
98              final boolean exists = testClass != null && testClass.length() > 0 && IOUtils.loadClass(null, testClass) != null;
99              logger.debug(exists ? "Found {}" : "Not found {}", testClass);
100             return exists; 
101         } catch (Exception ex) {
102             return false;
103         }
104     }

106     /**
107      * Returns a list of comet support by containers available on the classpath.
108      *
109      * @return
110      */
111     public List<Class<? extends AsyncSupport>> detectContainersPresent() {
112         return new LinkedList<Class<? extends AsyncSupport>>() {
113             {
114                 if (testClassExists(GLASSFISH_V2))
115                     add(GlassFishv2CometSupport.class);
116 
117                 if (testClassExists(JETTY_9))
118                     add(Jetty7CometSupport.class);
119 
120                 if (testClassExists(JETTY_8))
121                     add(Jetty7CometSupport.class);
122 
123                 if (testClassExists(JETTY_7))
124                     add(Jetty7CometSupport.class);
125 
126                 if (testClassExists(JETTY))
127                     add(JettyCometSupport.class);
128 
129                 if (testClassExists(JBOSSWEB))
130                     add(JBossWebCometSupport.class);
131 
132                 if (testClassExists(TOMCAT_7))
133                     add(Tomcat7CometSupport.class);
134 
135                 if (testClassExists(TOMCAT) || testClassExists(JBOSS_5))
136                     add(TomcatCometSupport.class);
137 
138                 if (testClassExists(GRIZZLY))
139                     add(GrizzlyCometSupport.class);
140 
141                 if (testClassExists(GRIZZLY2))
142                     add(Grizzly2CometSupport.class);
143 
144                 if (testClassExists(NETTY))
145                     add(NettyCometSupport.class);
146             }
147         };

150     public List<Class<? extends AsyncSupport>> detectWebSocketPresent(final boolean useNativeIfPossible, final boolean useServlet30Async) {
151 
152         return new LinkedList<Class<? extends AsyncSupport>>() {
153             {
154                 if (useServlet30Async && !useNativeIfPossible) {
155 
156                     if (!suppress356 && testClassExists(JSR356_WEBSOCKET)) {
157                         add(JSR356AsyncSupport.class);
158                     } else {
159 
160                         if (testClassExists(TOMCAT_WEBSOCKET))
161                             add(Tomcat7Servlet30SupportWithWebSocket.class);
162 
163                         if (testClassExists(JETTY_9))
164                             add(Jetty9AsyncSupportWithWebSocket.class);
165 
166                         if (testClassExists(JETTY_8))
167                             add(JettyServlet30AsyncSupportWithWebSocket.class);
168 
169                         if (testClassExists(GRIZZLY2_WEBSOCKET))
170                             add(GlassFishServ30WebSocketSupport.class);
171 
172                         if (testClassExists(GRIZZLY_WEBSOCKET))
173                             add(GrizzlyServlet30WebSocketSupport.class);
174 
175                         if (testClassExists(WEBLOGIC_WEBSOCKET) && !testClassExists(HK2)) {
176                             logger.warn("***************************************************************************************************");
177                             logger.warn("WebLogic WebSocket detected and will be deployed under the hardcoded path <<application-name>>/ws/*");
178                             logger.warn("***************************************************************************************************");
179                             add(WebLogicServlet30WithWebSocket.class);
180                         }
181                     }
182                 } else {
183                     if (!suppress356 && testClassExists(JSR356_WEBSOCKET)) {
184                         add(JSR356AsyncSupport.class);
185                     } else {
186                         if (testClassExists(TOMCAT_WEBSOCKET))
187                             add(Tomcat7AsyncSupportWithWebSocket.class);
188 
189                         if (testClassExists(JETTY_9))
190                             add(Jetty9AsyncSupportWithWebSocket.class);
191 
192                         if (testClassExists(JETTY_8))
193                             add(JettyAsyncSupportWithWebSocket.class);
194 
195                         if (testClassExists(GRIZZLY_WEBSOCKET))
196                             add(GlassFishWebSocketSupport.class);
197 
198                         if (testClassExists(GRIZZLY2_WEBSOCKET))
199                             add(Grizzly2WebSocketSupport.class);
200 
201                         if (testClassExists(JBOSS_AS7_WEBSOCKET))
202                             add(JBossAsyncSupportWithWebSocket.class);
203                     }
204 
205                 }
206             }
207         };
208     }

210     /**
211      * This method is used to determine the default AsyncSupport if all else fails.
212      *
213      * @param preferBlocking
214      * @return
215      */
216     public AsyncSupport defaultCometSupport(final boolean preferBlocking) {
217         if (!preferBlocking && testClassExists(SERVLET_30)) {
218             return new Servlet30CometSupport(config);
219         } else {
220             return new BlockingIOCometSupport(config);
221         }
222     }

224     /**
225      * Given a Class of something that extends AsyncSupport, it tries to return an instance of that class.
226      * <p/>
227      * The class has to have a visible constructor with the signature (@link {AtmosphereConfig}).
228      *
229      * @param targetClass
230      * @return an instance of the specified class or null if the class cannot be instantiated
231      */
232     public AsyncSupport newCometSupport(final Class<? extends AsyncSupport> targetClass) {
233         try {
234             return (AsyncSupport) targetClass.getDeclaredConstructor(new Class[]{AtmosphereConfig.class})
235                     .newInstance(config);
236         } catch (final Exception e) {
237             logger.warn("Failed to create AsyncSupport class: {}, error: {}", targetClass, e);
238 
239             Throwable cause = e.getCause();
240             if (cause != null) {
241                 logger.error("Real error: {}", cause.getMessage(), cause);
242             }
243             return null;
244         }
245     }
246 
247     public AsyncSupport newCometSupport(final String targetClassFQN) {
248         try {
249             ClassLoader cl = Thread.currentThread().getContextClassLoader();
250             return (AsyncSupport) cl.loadClass(targetClassFQN)
251                     .getDeclaredConstructor(new Class[]{AtmosphereConfig.class}).newInstance(config);
252         } catch (final Exception e) {
253             logger.error("Failed to create AsyncSupport class: {}, error: {}", targetClassFQN, e);
254             Throwable cause = e.getCause();
255             if (cause != null) {
256                 logger.error("Real error: {}", cause.getMessage(), cause);
257             }
258             throw new IllegalArgumentException("Unable to create " + targetClassFQN, e);
259         }
260     }

262     /**
263      * This method is the general interface to the outside world.
264      *
265      * @param useNativeIfPossible - should the resolver try to use a native container comet support if present?
266      * @param defaultToBlocking   - should the resolver default to blocking IO comet support?
267      * @return an instance of AsyncSupport
268      */
269     public AsyncSupport resolve(final boolean useNativeIfPossible, final boolean defaultToBlocking) {
270         final AsyncSupport servletAsyncSupport = defaultCometSupport(defaultToBlocking);
271 
272         final AsyncSupport nativeSupport;
273         if (!defaultToBlocking && (useNativeIfPossible ||
274                 servletAsyncSupport.getClass().getName().equals(BlockingIOCometSupport.class.getName()))) {
275             nativeSupport = resolveNativeCometSupport(detectContainersPresent());
276             return nativeSupport == null ? servletAsyncSupport : nativeSupport;
277         }
278         return servletAsyncSupport;
279     }
280 
281     @Override
282     public AsyncSupport resolve(boolean useNativeIfPossible, boolean defaultToBlocking, boolean useServlet30Async) {
283         AsyncSupport cs = null;
284 
285         // Validate the value for old Servlet Container.
286         useServlet30Async = testClassExists(SERVLET_30);
287 
288         if (!defaultToBlocking) {
289             List<Class<? extends AsyncSupport>> l = detectWebSocketPresent(useNativeIfPossible, useServlet30Async);
290 
291             if (!l.isEmpty()) {
292                 cs = resolveWebSocket(l);
293             }
294         }
295 
296         if (cs == null) {
297             AsyncSupport nativeSupport = resolveNativeCometSupport(detectContainersPresent());
298             return nativeSupport == null ? defaultCometSupport(defaultToBlocking) : nativeSupport;
299         } else {
300             return cs;
301         }
302     }
303 
304     public AsyncSupport resolveWebSocket(final java.util.List<Class<? extends AsyncSupport>> available) {
305         if (available == null || available.isEmpty()) return null;
306         else return newCometSupport(available.get(0));
307     }

309     /**
310      * This method is called to determine which native comet support to the used.
311      *
312      * @param available
313      * @return the result of @link {resolveMultipleNativeSupportConflict} if there are more than 1 item in the list of available ontainers
314      */
315     protected AsyncSupport resolveNativeCometSupport(final java.util.List<Class<? extends AsyncSupport>> available) {
316         if (available == null || available.isEmpty()) return null;
317         else if (available.size() == 1) return newCometSupport(available.get(0));
318         else return resolveMultipleNativeSupportConflict(available);
319     }
320 
321     /**
322      * This method is called if there are more than one potential native container in scope.
323      *
324      * @return a AsyncSupport instance
325      */
326     protected AsyncSupport resolveMultipleNativeSupportConflict(final List<Class<? extends AsyncSupport>> available) {
327         final StringBuilder b = new StringBuilder("Found multiple containers, please specify which one to use: ");
328         for (Class<? extends AsyncSupport> cs : available) {
329             b.append((cs != null) ? cs.getCanonicalName() : "null").append(", ");
330         }
331 
332         b.append(" until you do, Atmosphere will use:" + available.get(0));
333         logger.warn("{}", b.toString());
334         return newCometSupport(available.get(0));
335     }

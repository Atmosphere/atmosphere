40  /**
41   * An Atmosphere's response representation. An AtmosphereResponse can be used to construct bi-directional asynchronous
42   * application. If the underlying transport is a WebSocket or if its associated {@link AtmosphereResource} has been

44   * <br/>
45   * This object can delegates the write operation to {@link AsyncIOWriter}.
46   */

236         if (!delegateToNativeResponse || forceAsyncIOWriter) {
237             setStatus(sc, msg);
238 
239             // Prevent StackOverflow
240             boolean b = forceAsyncIOWriter;
241             forceAsyncIOWriter = false;
242             asyncIOWriter.writeError(this, sc, msg);
243             forceAsyncIOWriter = b;
244         } else {
245             if (!_r().isCommitted()) {
246                 _r().sendError(sc, msg);
247             } else {
248                 logger.warn("Committed error code {} {}", sc, msg);
249             }
250         }

554     private void writeStatusAndHeaders() throws java.io.IOException {
555         if (writeStatusAndHeader.getAndSet(false) && !forceAsyncIOWriter) {
556             asyncIOWriter.write(this, constructStatusAndHeaders());
557         }
558     }

560     private String constructStatusAndHeaders() {
561         StringBuffer b = new StringBuffer("HTTP/1.1")
562                 .append(" ")
563                 .append(status)
564                 .append(" ")
565                 .append(statusMessage)
566                 .append("\n");
567 
568         b.append("Content-Type").append(":").append(headers.get("Content-Type") == null ? contentType : headers.get("Content-Type")).append("\n");
569         if (contentLength != -1) {
570             b.append("Content-Length").append(":").append(contentLength).append("\n");
571         }
572 
573         for (String s : headers().keySet()) {
574             if (!s.equalsIgnoreCase("Content-Type")) {
575                 b.append(s).append(":").append(headers.get(s)).append("\n");
576             }
577         }
578         b.deleteCharAt(b.length() - 1);
579         b.append("\r\n\r\n");
580         return b.toString();
581     }

814     public AtmosphereResponse write(String data) {

864         boolean isUsingStream = (Boolean) request().getAttribute(PROPERTY_USE_STREAM);
865         try {
866             if (isUsingStream) {

868                     getOutputStream().write(data, offset, length);

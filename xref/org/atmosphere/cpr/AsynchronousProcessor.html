78  /**
79   * Base class which implement the semantics of suspending and resuming of a
80   * Comet/WebSocket Request.

82   * @author Jeanfrancois Arcand

84  public abstract class AsynchronousProcessor implements AsyncSupport<AtmosphereResourceImpl> {

86      private static final Logger logger = LoggerFactory.getLogger(AsynchronousProcessor.class);
87      protected static final Action timedoutAction = new Action(Action.TYPE.TIMEOUT);
88      protected static final Action cancelledAction = new Action(Action.TYPE.CANCELLED);
89      protected final AtmosphereConfig config;
90      protected final ConcurrentHashMap<AtmosphereRequest, AtmosphereResource>
91              aliveRequests = new ConcurrentHashMap<AtmosphereRequest, AtmosphereResource>();
92      private boolean trackActiveRequest = false;
93      private final ScheduledExecutorService closedDetector = Executors.newScheduledThreadPool(1);

99      @Override
100     public void init(ServletConfig sc) throws ServletException {
101 
102         String maxInactive = sc.getInitParameter(MAX_INACTIVE) != null ? sc.getInitParameter(MAX_INACTIVE) :
103                 config.getInitParameter(MAX_INACTIVE);
104 
105         if (maxInactive != null) {
106             trackActiveRequest = true;
107             final long maxInactiveTime = Long.parseLong(maxInactive);
108             if (maxInactiveTime <= 0) return;
109 
110             closedDetector.scheduleAtFixedRate(new Runnable() {
111                 public void run() {
112                     for (AtmosphereRequest req : aliveRequests.keySet()) {
113                         long l = (Long) req.getAttribute(MAX_INACTIVE);
114                         if (l > 0 && System.currentTimeMillis() - l > maxInactiveTime) {
115                             try {
116                                 logger.debug("Close detector disconnecting {}. Current size {}", req, aliveRequests.size());
117                                 AtmosphereResourceImpl r = (AtmosphereResourceImpl) aliveRequests.remove(req);
118                                 cancelled(req, r.getResponse(false));
119                             } catch (Throwable e) {
120                                 logger.warn("closedDetector", e);
121                             } finally {
122                                 try {
123                                     req.setAttribute(MAX_INACTIVE, (long) -1);
124                                 } catch (Throwable t) {
125                                     logger.trace("closedDetector", t);
126                                 }
127                             }
128                         }
129                     }
130                 }
131             }, 0, 1, TimeUnit.SECONDS);
132         }
133     }

168     /**
169      * Invoke the {@link AtmosphereHandler#onRequest} method.
170      *
171      * @param req the {@link AtmosphereRequest}
172      * @param res the {@link AtmosphereResponse}
173      * @return action the Action operation.
174      * @throws java.io.IOException
175      * @throws javax.servlet.ServletException
176      */
177     Action action(AtmosphereRequest req, AtmosphereResponse res) throws IOException, ServletException {
178 
179         if (Utils.webSocketEnabled(req) && !supportWebSocket()) {
180             res.setStatus(501);
181             res.addHeader(X_ATMOSPHERE_ERROR, "Websocket protocol not supported");
182             res.flushBuffer();
183             return new Action();
184         }
185 
186         if (config.handlers().isEmpty()) {
187             logger.error("No AtmosphereHandler found. Make sure you define it inside web/atmosphere.xml or annotate using @AtmosphereHandlerService");
188             throw new AtmosphereMappingException("No AtmosphereHandler found. Make sure you define it inside web/atmosphere.xml or annotate using @AtmosphereHandlerService");
189         }
190 
191         if (res.request() == null) {
192             res.request(req);
193         }
194 
195         if (supportSession()) {
196             // Create the session needed to support the Resume
197             // operation from disparate requests.
198             HttpSession session = req.getSession(true);
199             // Do not allow times out.
200             SessionTimeoutSupport.setupTimeout(session);
201         }
202 
203         req.setAttribute(FrameworkConfig.SUPPORT_SESSION, supportSession());
204 
205         AtmosphereHandlerWrapper handlerWrapper = map(req);
206         // Check Broadcaster state. If destroyed, replace it.
207         Broadcaster b = handlerWrapper.broadcaster;
208         if (b.isDestroyed()) {
209             synchronized (handlerWrapper) {
210                 config.getBroadcasterFactory().remove(b, b.getID());
211                 handlerWrapper.broadcaster = config.getBroadcasterFactory().get(b.getID());
212             }
213         }
214 
215         AtmosphereResourceImpl resource = (AtmosphereResourceImpl) req.getAttribute(FrameworkConfig.INJECTED_ATMOSPHERE_RESOURCE);
216         if (resource == null) {
217             // TODO: cast is dangerous
218             resource = (AtmosphereResourceImpl)
219                     AtmosphereResourceFactory.create(config, handlerWrapper.broadcaster, res, this, handlerWrapper.atmosphereHandler);
220         } else {
221             resource.setBroadcaster(handlerWrapper.broadcaster).atmosphereHandler(handlerWrapper.atmosphereHandler);
222         }
223 
224         req.setAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE, resource);
225         req.setAttribute(FrameworkConfig.ATMOSPHERE_HANDLER, handlerWrapper.atmosphereHandler);
226 
227         LinkedList<AtmosphereInterceptor> c = config.framework().interceptors();
228         Action a;
229         for (AtmosphereInterceptor arc : c) {
230             a = arc.inspect(resource);
231             if (a == null || a.type() != Action.TYPE.CONTINUE) {
232                 logger.trace("Interceptor {} interrupted the dispatch with {}", arc, a);
233                 return a;
234             }
235         }
236 
237         try {
238             handlerWrapper.atmosphereHandler.onRequest(resource);
239         } catch (IOException t) {
240             resource.onThrowable(t);
241             throw t;
242         }
243 
244         if (trackActiveRequest && resource.getAtmosphereResourceEvent().isSuspended() && req.getAttribute(FrameworkConfig.CANCEL_SUSPEND_OPERATION) == null) {
245             req.setAttribute(MAX_INACTIVE, System.currentTimeMillis());
246             aliveRequests.put(req, resource);
247         }
248         return resource.action();
249     }

260     protected AtmosphereHandlerWrapper map(String path) {
261         AtmosphereHandlerWrapper atmosphereHandlerWrapper = config.handlers().get(path);
262 
263         if (atmosphereHandlerWrapper == null) {
264             final Map<String, String> m = new HashMap<String, String>();
265             for (Map.Entry<String, AtmosphereHandlerWrapper> e : config.handlers().entrySet()) {
266                 UriTemplate t = new UriTemplate(e.getKey());
267                 logger.trace("Trying to map {} to {}", t, path);
268                 if (t.match(path, m)) {
269                     atmosphereHandlerWrapper = e.getValue();
270                     logger.trace("Mapped {} to {}", t, e.getValue().atmosphereHandler);
271                     break;
272                 }
273             }
274         }
275         return atmosphereHandlerWrapper;
276     }

278     /**
279      * Return the {@link AtmosphereHandler} mapped to the passed servlet-path.
280      *
281      * @param req the {@link AtmosphereResponse}
282      * @return the {@link AtmosphereHandler} mapped to the passed servlet-path.
283      * @throws javax.servlet.ServletException
284      */
285     protected AtmosphereHandlerWrapper map(AtmosphereRequest req) throws ServletException {
286         String path;
287         if (req.getPathInfo() != null) {
288             path = req.getServletPath() + req.getPathInfo();
289         } else {
290             path = req.getServletPath();
291         }
292 
293         if (path == null || path.isEmpty()) {
294             path = "/";
295         }
296 
297         AtmosphereHandlerWrapper atmosphereHandlerWrapper = map(path + (path.endsWith("/") ? "all" : "/all"));
298         if (atmosphereHandlerWrapper == null) {
299             // (2) First, try exact match
300             atmosphereHandlerWrapper = map(path);
301 
302             if (atmosphereHandlerWrapper == null) {
303                 // (3) Wildcard
304                 atmosphereHandlerWrapper = map(path + "*");
305 
306                 // (4) try without a path
307                 if (atmosphereHandlerWrapper == null) {
308                     String p = path.lastIndexOf("/") == 0 ? "/" : path.substring(0, path.lastIndexOf("/"));
309                     while (p.length() > 0) {
310                         atmosphereHandlerWrapper = map(p);
311 
312                         // (3.1) Try path wildcard
313                         if (atmosphereHandlerWrapper != null) {
314                             break;
315                         }
316                         p = p.substring(0, p.lastIndexOf("/"));
317                     }
318                 }
319             }
320         }
321 
322         if (atmosphereHandlerWrapper == null) {
323             throw new AtmosphereMappingException("No AtmosphereHandler maps request for " + path);
324         }
325         config.getBroadcasterFactory().add(atmosphereHandlerWrapper.broadcaster,
326                 atmosphereHandlerWrapper.broadcaster.getID());
327         return atmosphereHandlerWrapper;
328     }

349     /**
350      * All proprietary Comet based {@link Servlet} must invoke the timedout
351      * method when the underlying WebServer time out the {@link AtmosphereResponse}.
352      * The returned value, of type
353      * {@link Action}, tells the proprietary Comet {@link Servlet}
354      * to resume (again), suspended or do nothing with the current {@link AtmosphereResponse}.
355      *
356      * @param request  the {@link AtmosphereRequest}
357      * @param response the {@link AtmosphereResponse}
358      * @return action the Action operation.
359      * @throws java.io.IOException
360      * @throws javax.servlet.ServletException
361      */
362     public Action timedout(AtmosphereRequest request, AtmosphereResponse response)
363             throws IOException, ServletException {
364 
365         AtmosphereResourceImpl r = null;
366 
367         try {
368             SessionTimeoutSupport.restoreTimeout(request);
369 
370             if (trackActiveRequest) {
371                 long l = (Long) request.getAttribute(MAX_INACTIVE);
372                 if (l == -1) {
373                     // The closedDetector closed the connection.
374                     return timedoutAction;
375                 }
376                 request.setAttribute(MAX_INACTIVE, (long) -1);
377             }
378 
379             logger.debug("Timing out the connection for request {}", request);
380 
381             // Something went wrong.
382             if (request == null || response == null) {
383                 logger.warn("Invalid Request/Response: {}/{}", request, response);
384                 return timedoutAction;
385             }
386 
387             r = (AtmosphereResourceImpl) request.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
388 
389             if (r != null && r.getAtmosphereResourceEvent().isSuspended()) {
390                 r.getAtmosphereResourceEvent().setIsResumedOnTimeout(true);
391 
392                 Broadcaster b = r.getBroadcaster();
393                 if (b instanceof DefaultBroadcaster) {
394                     ((DefaultBroadcaster) b).broadcastOnResume(r);
395                 }
396 
397                 if (request.getAttribute(ApplicationConfig.RESUMED_ON_TIMEOUT) != null) {
398                     r.getAtmosphereResourceEvent().setIsResumedOnTimeout(
399                             (Boolean) request.getAttribute(ApplicationConfig.RESUMED_ON_TIMEOUT));
400                 }
401 
402                 invokeAtmosphereHandler(r);
403             }
404         } catch (Throwable t) {
405             logger.error("failed to timeout resource {}", r, t);
406         } finally {
407             config.framework().notify(Action.TYPE.TIMEOUT, request, response);
408             try {
409                 if (r != null) {
410                     r.notifyListeners();
411                     r.setIsInScope(false);
412                     r.cancel();
413                 }
414             } catch (Throwable t) {
415                 logger.trace("timedout", t);
416             } finally {
417 
418                 try {
419                     response.getOutputStream().close();
420                 } catch (Throwable t) {
421                     try {
422                         response.getWriter().close();
423                     } catch (Throwable t2) {
424                     }
425                 }
426 
427                 if (r != null) {
428                     destroyResource(r);
429                 }
430             }
431         }
432 
433         return timedoutAction;
434     }

436     void invokeAtmosphereHandler(AtmosphereResourceImpl r) throws IOException {
437         if (!r.isInScope()) return;
438 
439         AtmosphereRequest req = r.getRequest(false);
440         String disableOnEvent = r.getAtmosphereConfig().getInitParameter(ApplicationConfig.DISABLE_ONSTATE_EVENT);
441         r.getAtmosphereResourceEvent().setMessage(r.writeOnTimeout());
442         try {
443             if (disableOnEvent == null || !disableOnEvent.equals(String.valueOf(true))) {
444                 AtmosphereHandler atmosphereHandler =
445                         (AtmosphereHandler)
446                                 req.getAttribute(FrameworkConfig.ATMOSPHERE_HANDLER);
447 
448                 synchronized (r) {
449                     atmosphereHandler.onStateChange(r.getAtmosphereResourceEvent());
450 
451                     Meteor m = (Meteor) req.getAttribute(AtmosphereResourceImpl.METEOR);
452                     if (m != null) {
453                         m.destroy();
454                     }
455                 }
456                 req.removeAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
457             }
458         } catch (IOException ex) {
459             try {
460                 r.onThrowable(ex);
461             } catch (Throwable t) {
462                 logger.warn("failed calling onThrowable()", ex);
463             }
464         }
465     }

467     public static void destroyResource(AtmosphereResource r) {
468         if (r == null) return;
469 
470         try {
471             r.removeEventListeners();
472             try {
473                 AtmosphereResourceImpl.class.cast(r).getBroadcaster(false).removeAtmosphereResource(r);
474             } catch (IllegalStateException ex) {
475                 logger.trace(ex.getMessage(), ex);
476             }
477             if (BroadcasterFactory.getDefault() != null) {
478                 BroadcasterFactory.getDefault().removeAllAtmosphereResource(r);
479             }
480         } catch (Throwable t) {
481             logger.trace("destroyResource", t);
482         }
483     }

485     /**
486      * All proprietary Comet based {@link Servlet} must invoke the cancelled
487      * method when the underlying WebServer detect that the client closed
488      * the connection.
489      *
490      * @param req the {@link AtmosphereRequest}
491      * @param res the {@link AtmosphereResponse}
492      * @return action the Action operation.
493      * @throws java.io.IOException
494      * @throws javax.servlet.ServletException
495      */
496     public Action cancelled(AtmosphereRequest req, AtmosphereResponse res)
497             throws IOException, ServletException {
498 
499         synchronized (req) {
500             SessionTimeoutSupport.restoreTimeout(req);
501 
502             AtmosphereResourceImpl r = null;
503             try {
504                 if (trackActiveRequest) {
505                     long l = (Long) req.getAttribute(MAX_INACTIVE);
506                     if (l == -1) {
507                         // The closedDetector closed the connection.
508                         return timedoutAction;
509                     }
510                     req.setAttribute(MAX_INACTIVE, (long) -1);
511                 }
512 
513                 logger.debug("Cancelling the connection for request {}", req);
514 
515                 r = (AtmosphereResourceImpl) req.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
516                 if (r != null) {
517                     r.getAtmosphereResourceEvent().setCancelled(true);
518                     invokeAtmosphereHandler(r);
519 
520                     try {
521                         r.getResponse().getOutputStream().close();
522                     } catch (Throwable t) {
523                         try {
524                             r.getResponse().getWriter().close();
525                         } catch (Throwable t2) {
526                         }
527                     }
528                 }
529             } catch (Throwable ex) {
530                 // Something wrong happenned, ignore the exception
531                 logger.debug("failed to cancel resource: " + r, ex);
532             } finally {
533                 config.framework().notify(Action.TYPE.CANCELLED, req, res);
534                 try {
535                     if (r != null) {
536                         r.notifyListeners();
537                         r.setIsInScope(false);
538                         r.cancel();
539                     }
540                 } catch (Throwable t) {
541                     logger.trace("cancel", t);
542                 } finally {
543                     if (r != null) {
544                         destroyResource(r);
545                     }
546                 }
547             }
548         }
549 
550         return cancelledAction;
551     }

553     protected void shutdown() {
554         closedDetector.shutdownNow();
555         for (AtmosphereResource resource : aliveRequests.values()) {
556             try {
557                 resource.resume();
558             } catch (Throwable t) {
559                 // Something wrong happenned, ignore the exception
560                 logger.debug("failed on resume: " + resource, t);
561             }
562         }
563     }

569     /**
570      * An Callback class that can be used by Framework integrator to handle the close/timedout/resume life cycle
571      * of an {@link AtmosphereResource}. This class support only support {@link AsyncSupport} implementation that
572      * extends {@link AsynchronousProcessor}
573      */
574     public final static class AsynchronousProcessorHook {
575 
576         private final AtmosphereResourceImpl r;
577 
578         public AsynchronousProcessorHook(AtmosphereResourceImpl r) {
579             this.r = r;
580             if (!AsynchronousProcessor.class.isAssignableFrom(r.asyncSupport.getClass())) {
581                 throw new IllegalStateException("AsyncSupport must extends AsynchronousProcessor");
582             }
583         }
584 
585         public void closed() {
586             try {
587                 ((AsynchronousProcessor) r.asyncSupport).cancelled(r.getRequest(false), r.getResponse(false));
588             } catch (IOException e) {
589                 logger.debug("", e);
590             } catch (ServletException e) {
591                 logger.debug("", e);
592             }
593         }
594 
595         public void timedOut() {
596             try {
597                 ((AsynchronousProcessor) r.asyncSupport).timedout(r.getRequest(false), r.getResponse(false));
598             } catch (IOException e) {
599                 logger.debug("", e);
600             } catch (ServletException e) {
601                 logger.debug("", e);
602             }
603         }
604 
605         public void resume() {
606             ((AsynchronousProcessor) r.asyncSupport).action(r);
607         }
608     }

118     /**
119      * Invoke the {@link AtmosphereHandler#onRequest} method.
120      *
121      * @param req the {@link AtmosphereRequest}
122      * @param res the {@link AtmosphereResponse}
123      * @return action the Action operation.
124      * @throws java.io.IOException
125      * @throws javax.servlet.ServletException
126      */
127     Action action(AtmosphereRequest req, AtmosphereResponse res) throws IOException, ServletException {
128 
129         if (!Utils.properProtocol(req)) {
130             logger.debug("Invalid request state.");
131             res.setStatus(501);
132             res.addHeader(X_ATMOSPHERE_ERROR, "Websocket protocol not supported");
133             res.flushBuffer();
134             return new Action();
135         }
136 
137         if (Utils.webSocketEnabled(req) && !supportWebSocket()) {
138             logger.warn("Websocket protocol not supported");
139             res.setStatus(501);
140             res.addHeader(X_ATMOSPHERE_ERROR, "Websocket protocol not supported");
141             res.flushBuffer();
142             return new Action();
143         }
144 
145         if (config.handlers().isEmpty()) {
146             logger.error("No AtmosphereHandler found. Make sure you define it inside WEB-INF/atmosphere.xml or annotate using @___Service");
147             throw new AtmosphereMappingException("No AtmosphereHandler found. Make sure you define it inside WEB-INF/atmosphere.xml or annotate using @___Service");
148         }
149 
150         if (res.request() == null) {
151             res.request(req);
152         }
153 
154         if (supportSession()) {
155             // Create the session needed to support the Resume
156             // operation from disparate requests.
157             HttpSession s = req.getSession(config.getInitParameter(PROPERTY_SESSION_CREATE, true));
158 
159             // https://github.com/Atmosphere/atmosphere/issues/2034
160             try {
161                 if (s != null && s.isNew()) {
162                     s.setAttribute(FrameworkConfig.BROADCASTER_FACTORY, config.getBroadcasterFactory());
163                 }
164             } catch(IllegalStateException ex) {
165                 AtmosphereResourceImpl r = AtmosphereResourceImpl.class.cast(req.resource());
166                 logger.warn("Session Expired for {}. Closing the connection", req.uuid(), ex);
167                 if (r != null) {
168                     logger.trace("Ending request for {}", r.uuid());
169                     endRequest(r, true);
170                     return Action.CANCELLED;
171                 } else {
172                     logger.trace("Sending error for {}", req.uuid());
173                     res.setStatus(500);
174                     res.addHeader(X_ATMOSPHERE_ERROR, "Session expired");
175                     res.flushBuffer();
176                     return new Action();
177                 }
178             }
179         }
180 
181         req.setAttribute(FrameworkConfig.SUPPORT_SESSION, supportSession());
182 
183         int tracing = 0;
184 
185         AtmosphereHandlerWrapper handlerWrapper = map(req);
186         if (config.getBroadcasterFactory() == null) {
187             logger.error("Atmosphere is misconfigured and will not work. BroadcasterFactory is null");
188             return Action.CANCELLED;
189         }
190         AtmosphereResourceImpl resource = configureWorkflow(null, handlerWrapper, req, res);
191 
192         String v = req.getHeader(HeaderConfig.X_ATMO_BINARY);
193         if (v != null) {
194             resource.forceBinaryWrite(Boolean.valueOf(v));
195         }
196 
197         // handler interceptor lists
198         LinkedList<AtmosphereInterceptor> invokedInterceptors = handlerWrapper.interceptors;
199 
200         try {
201             Action a = invokeInterceptors(invokedInterceptors, resource, tracing);
202             if (a.type() != Action.TYPE.CONTINUE && a.type() != Action.TYPE.SKIP_ATMOSPHEREHANDLER) {
203                 return a;
204             }
205             // Remap occured.
206             if (req.getAttribute(FrameworkConfig.NEW_MAPPING) != null) {
207                 req.removeAttribute(FrameworkConfig.NEW_MAPPING);
208                 handlerWrapper = map(req);
209                 if (handlerWrapper == null) {
210                     logger.debug("Remap {}", resource.uuid());
211                     throw new AtmosphereMappingException("Invalid state. No AtmosphereHandler maps request for " + req.getRequestURI());
212                 }
213                 resource = configureWorkflow(resource, handlerWrapper, req, res);
214                 resource.setBroadcaster(handlerWrapper.broadcaster);
215             }
216 
217             //Unit test mock the request and will throw NPE.
218             boolean skipAtmosphereHandler = req.getAttribute(SKIP_ATMOSPHEREHANDLER.name()) != null
219                     ? (Boolean) req.getAttribute(SKIP_ATMOSPHEREHANDLER.name()) : Boolean.FALSE;
220             if (!skipAtmosphereHandler) {
221                 try {
222                     logger.trace("\t Last: {}", handlerWrapper.atmosphereHandler.getClass().getName());
223                     handlerWrapper.atmosphereHandler.onRequest(resource);
224                 } catch (IOException t) {
225                     resource.onThrowable(t);
226                     throw t;
227                 }
228             }
229         } finally{
230             postInterceptors(handlerWrapper != null? handlerWrapper.interceptors: invokedInterceptors, resource);
231         }
232 
233         Action action = resource.action();
234         if (supportSession() && allowSessionTimeoutRemoval() && action.type().equals(Action.TYPE.SUSPEND)) {
235             // Do not allow times out.
236             SessionTimeoutSupport.setupTimeout(config, req.getSession(config.getInitParameter(ApplicationConfig.PROPERTY_SESSION_CREATE, true)));
237         }
238         logger.trace("Action for {} was {} with transport " + req.getHeader(X_ATMOSPHERE_TRANSPORT), req.resource() != null ? req.resource().uuid() : "null", action);
239         return action;
240     }

242     private AtmosphereResourceImpl configureWorkflow(AtmosphereResourceImpl resource,
243                                                      AtmosphereHandlerWrapper handlerWrapper,
244                                                      AtmosphereRequest req, AtmosphereResponse res) {
245 
246         // Check Broadcaster state. If destroyed, replace it.
247         Broadcaster b = handlerWrapper.broadcaster;
248         if (b.isDestroyed()) {
249             BroadcasterFactory f = config.getBroadcasterFactory();
250             synchronized (f) {
251                 f.remove(b, b.getID());
252                 try {
253                     handlerWrapper.broadcaster = f.get(b.getID());
254                 } catch (IllegalStateException ex) {
255                     // Something wrong occurred, let's not fail and loookup the value
256                     logger.trace("", ex);
257                     // fallback to lookup
258                     handlerWrapper.broadcaster = f.lookup(b.getID(), true);
259                 }
260             }
261         }
262 
263         if (resource == null) {
264             resource = (AtmosphereResourceImpl) req.getAttribute(FrameworkConfig.INJECTED_ATMOSPHERE_RESOURCE);
265         }
266 
267         if (resource == null) {
268             // TODO: cast is dangerous
269             resource = (AtmosphereResourceImpl)
270                     config.resourcesFactory().create(config, handlerWrapper.broadcaster, res, this, handlerWrapper.atmosphereHandler);
271         } else {
272             // TODO: REDESIGN, UGLY.
273             try {
274                 // Make sure it wasn't set before
275                 resource.getBroadcaster();
276             } catch (IllegalStateException ex) {
277                 resource.setBroadcaster(handlerWrapper.broadcaster);
278             }
279             resource.atmosphereHandler(handlerWrapper.atmosphereHandler);
280         }
281 
282         req.setAttribute(ATMOSPHERE_RESOURCE, resource);
283         req.setAttribute(ATMOSPHERE_HANDLER_WRAPPER, handlerWrapper);
284         req.setAttribute(SKIP_ATMOSPHEREHANDLER.name(), Boolean.FALSE);
285 
286         return resource;
287     }

289     protected void shutdown() {
290         Collection<AtmosphereResource> c = config.resourcesFactory().findAll();
291         for (AtmosphereResource r : c) {
292             try {
293                 r.close();
294             } catch (IOException e) {
295                 logger.trace("", e);
296             }
297         }
298     }

309     private String path(AtmosphereRequest request) {
310         String path;
311         String pathInfo = null;
312         try {
313             pathInfo = request.getPathInfo();
314         } catch (IllegalStateException ex) {
315             // http://java.net/jira/browse/GRIZZLY-1301
316         }
317 
318         if (pathInfo != null) {
319             path = request.getServletPath() + pathInfo;
320         } else {
321             path = request.getServletPath();
322         }
323 
324         if (path == null || path.isEmpty()) {
325             path = "/";
326         }
327         return path;
328     }

330     public Action invokeInterceptors(List<AtmosphereInterceptor> c, AtmosphereResource r, int tracing) {
331         Action a = Action.CONTINUE;
332         for (AtmosphereInterceptor arc : c) {
333 
334             if (!AtmosphereResourceImpl.class.cast(r).isInScope()) {
335                 logger.warn("Request closed during processing {} and transport {}", r.uuid(), r.transport());
336                 return Action.CANCELLED;
337             }
338 
339             try {
340                 a = arc.inspect(r);
341             } catch (Exception ex) {
342                 logger.error("Interceptor {} crashed. Processing will continue with other interceptor.", arc, ex);
343                 continue;
344             }
345 
346             if (a == null) {
347                 logger.trace("Action was null for {}", arc);
348                 a = Action.CANCELLED;
349             }
350 
351             boolean skip = a.type() == SKIP_ATMOSPHEREHANDLER;
352             if (skip) {
353                 logger.trace("AtmosphereInterceptor {} asked to skip the AtmosphereHandler for {}", arc, r.uuid());
354                 r.getRequest().setAttribute(SKIP_ATMOSPHEREHANDLER.name(), Boolean.TRUE);
355             }
356 
357             if (a.type() != Action.TYPE.CONTINUE) {
358                 logger.trace("Interceptor {} interrupted the dispatch for {} with " + a, arc, r.uuid());
359                 return a;
360             }
361 
362             if (logger.isTraceEnabled()) {
363                 logger.trace("\t {}: {} for {}", new String[]{String.valueOf(tracing++), arc.getClass().getName(), r.uuid()});
364             }
365         }
366         return a;
367     }

369     public void postInterceptors(List<AtmosphereInterceptor> c, AtmosphereResource r) {
370         AtmosphereInterceptor arc = null;
371         for (int i = c.size() - 1; i > -1; i--) {
372             try {
373                 arc = c.get(i);
374                 arc.postInspect(r);
375             } catch (Exception ex) {
376                 logger.error("Interceptor {} crashed. Processing will continue with other interceptor.", arc, ex);
377                 continue;
378             }
379         }
380     }

383      * Return the {@link AtmosphereHandler} mapped to the passed servlet-path.
384      *
385      * @param req the {@link AtmosphereResponse}
386      * @return the {@link AtmosphereHandler} mapped to the passed servlet-path.
387      * @throws javax.servlet.ServletException
388      */
389     protected AtmosphereHandlerWrapper map(AtmosphereRequest req) throws ServletException {
390         AtmosphereHandlerWrapper atmosphereHandlerWrapper = mapper.map(req, config.handlers());
391         if (atmosphereHandlerWrapper == null) {
392             logger.debug("No AtmosphereHandler maps request for {} with mapping {}", req.getRequestURI(), config.handlers());
393             throw new AtmosphereMappingException("No AtmosphereHandler maps request for " + req.getRequestURI());
394         }
395 
396         return atmosphereHandlerWrapper;
397     }

399     /**
400      * All proprietary Comet based {@link Servlet} must invoke the resume method when the Atmosphere's application
401      * decide to resume the {@link AtmosphereResponse}. The returned value, of type {@link Action}, tells the
402      * proprietary Comet {@link Servlet} to resume (again), suspended or do nothing with the current {@link AtmosphereResponse}.
403      *
404      * @param request  the {@link AtmosphereRequest}
405      * @param response the {@link AtmosphereResponse}
406      * @return action the Action operation.
407      * @throws java.io.IOException
408      * @throws javax.servlet.ServletException
409      */
410     public Action resumed(AtmosphereRequest request, AtmosphereResponse response)
411             throws IOException, ServletException {
412 
413         AtmosphereResourceImpl r =
414                 (AtmosphereResourceImpl) request.getAttribute(ATMOSPHERE_RESOURCE);
415 
416         if (r == null) return Action.CANCELLED; // We are cancelled already
417 
418         AtmosphereHandler atmosphereHandler = r.getAtmosphereHandler();
419 
420         AtmosphereResourceEvent event = r.getAtmosphereResourceEvent();
421         if (event != null && event.isResuming() && !event.isCancelled()) {
422             synchronized (r) {
423                 atmosphereHandler.onStateChange(event);
424             }
425         }
426         return Action.RESUME;
427     }

448     /**
449      * Cancel or times out an {@link AtmosphereResource} by invoking it's associated {@link AtmosphereHandler#onStateChange(AtmosphereResourceEvent)}
450      *
451      * @param r         an {@link AtmosphereResource}
452      * @param cancelled true if cancelled, false if timedout
453      * @return true if the operation was executed.
454      */
455     public boolean completeLifecycle(final AtmosphereResource r, boolean cancelled) {
456         if (r != null && !r.isCancelled() && !AtmosphereResourceImpl.class.cast(r).getAndSetInClosingPhase()) {
457             logger.trace("Finishing lifecycle for AtmosphereResource {}", r.uuid());
458             final AtmosphereResourceImpl impl = AtmosphereResourceImpl.class.cast(r);
459             try {
460                 if (impl.isCancelled()) {
461                     logger.debug("{} is already cancelled", impl.uuid());
462                     return false;
463                 }
464 
465                 AtmosphereResourceEventImpl e = impl.getAtmosphereResourceEvent();
466                 // https://github.com/Atmosphere/atmosphere/issues/1756
467                 // Do not go inside Atmosphere once undeployed to avoid all kind of issues.
468                 if (config.framework().isDestroyed()) {
469                     cancelled = true;
470                 }
471 
472                 if (!e.isClosedByClient()) {
473                     if (cancelled) {
474                         e.setCancelled(cancelled);
475                     } else {
476                         e.setIsResumedOnTimeout(true);
477                         Broadcaster b = r.getBroadcaster();
478                         if (b instanceof DefaultBroadcaster) {
479                             ((DefaultBroadcaster) b).broadcastOnResume(r);
480                         }
481 
482                         // TODO: Was it there for legacy reason?
483                         // impl.getAtmosphereResourceEvent().setIsResumedOnTimeout(impl.resumeOnBroadcast());
484                     }
485                 }
486                 invokeAtmosphereHandler(impl);
487             } catch (Throwable ex) {
488                 // Something wrong happened, ignore the exception
489                 logger.error("Failed to cancel resource: {}", impl.uuid(), ex);
490             } finally {
491                 try {
492                     impl.notifyListeners();
493                     // Legacy code #1894 that may breaks Jersey when set to false on some old server.
494                     if (closeOnCancel) {
495                         try {
496                             impl.getResponse(false).getOutputStream().close();
497                         } catch (Throwable t) {
498                             try {
499                                 impl.getResponse(false).getWriter().close();
500                             } catch (Throwable t2) {
501                             }
502                         }
503                     }
504                     impl.setIsInScope(false);
505                     impl.cancel();
506                 } catch (Throwable t) {
507                     logger.debug("completeLifecycle", t);
508                 } finally {
509                     impl._destroy();
510                 }
511             }
512             return true;
513         } else {
514             logger.trace("AtmosphereResource {} was already cancelled or gc", r != null ? r.uuid() : "null");
515             return false;
516         }
517     }

519     /**
520      * Invoke the associated {@link AtmosphereHandler}. This method must be synchronized on an AtmosphereResource.
521      *
522      * @param r a {@link AtmosphereResourceImpl}
523      * @throws IOException
524      */
525     protected void invokeAtmosphereHandler(AtmosphereResourceImpl r) throws IOException {
526         AtmosphereRequest req = r.getRequest(false);
527         try {
528             // Rely on isInScope instead of synchronization https://github.com/Atmosphere/atmosphere/issues/1865
529             if (r.isInScope()) {
530                 String disableOnEvent = r.getAtmosphereConfig().getInitParameter(ApplicationConfig.DISABLE_ONSTATE_EVENT);
531                 r.getAtmosphereResourceEvent().setMessage(r.writeOnTimeout());
532                 try {
533                     if (disableOnEvent == null || !disableOnEvent.equals(String.valueOf(true))) {
534                         AtmosphereHandler atmosphereHandler = r.getAtmosphereHandler();
535 
536                         if (atmosphereHandler != null && r.isInScope()) {
537 
538                             try {
539                                 Utils.inject(r);
540                             } catch (IllegalAccessException e) {
541                                 logger.warn("",e);
542                             }
543 
544                             atmosphereHandler.onStateChange(r.getAtmosphereResourceEvent());
545                         }
546                     }
547                 } catch (IOException ex) {
548                     try {
549                         r.onThrowable(ex);
550                     } catch (Throwable t) {
551                         logger.warn("failed calling onThrowable()", ex);
552                     }
553                 }
554             } else {
555                 logger.trace("AtmosphereResource out of scope {}", r.uuid());
556                 return;
557             }
558         } finally {
559             Utils.destroyMeteor(req);
560         }
561     }

563     /**
564      * All proprietary Comet based {@link Servlet} must invoke the cancelled method when the underlying WebServer
565      * detect that the client closed the connection.
566      *
567      * @param req the {@link AtmosphereRequest}
568      * @param res the {@link AtmosphereResponse}
569      * @return action the Action operation.
570      * @throws java.io.IOException
571      * @throws javax.servlet.ServletException
572      */
573     public Action cancelled(final AtmosphereRequest req, final AtmosphereResponse res)
574             throws IOException, ServletException {
575 
576         logger.trace("Cancelling {}", req);
577         final AtmosphereResourceImpl r = AtmosphereResourceImpl.class.cast(req.resource());
578         // Leave a chance to the client to send the disconnect message before processing the connection
579         if (closingTime > 0) {
580             ExecutorsFactory.getScheduler(config).schedule(new Callable<Object>() {
581 
582                 @Override
583                 public Object call() throws Exception {
584                     endRequest(r, true);
585                     return null;
586                 }
587             }, closingTime, TimeUnit.MILLISECONDS);
588         } else {
589             if (completeLifecycle(req.resource(), true)) {
590                 config.framework().notify(Action.TYPE.CANCELLED, req, res);
591             }
592         }
593         return cancelledAction;
594     }
595 
596     public void endRequest(AtmosphereResourceImpl r, boolean cancel) {
597         if (completeLifecycle(r, cancel)) {
598             config.framework().notify(Action.TYPE.CANCELLED, r.getRequest(false), r.getResponse(false));
599         }
600     }

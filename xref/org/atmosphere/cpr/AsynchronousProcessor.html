59  /**
60   * Base class which implement the semantics of suspending and resuming of a
61   * Comet Request.
62   *
63   * @author Jeanfrancois Arcand
64   */
65  public abstract class AsynchronousProcessor implements CometSupport<AtmosphereResourceImpl> {
66  
67      private static final Logger logger = LoggerFactory.getLogger(AsynchronousProcessor.class);
68  
69      protected static final Action timedoutAction = new Action(Action.TYPE.TIMEOUT);
70      protected static final Action cancelledAction = new Action(Action.TYPE.CANCELLED);
71      private static final int DEFAULT_SESSION_TIMEOUT = 1800;
72  
73      protected final AtmosphereConfig config;
74  
75      protected final ConcurrentHashMap<HttpServletRequest, AtmosphereResource<HttpServletRequest, HttpServletResponse>>
76              aliveRequests = new ConcurrentHashMap<HttpServletRequest, AtmosphereResource<HttpServletRequest, HttpServletResponse>>();
77  
78      private final ScheduledExecutorService closedDetector = Executors.newScheduledThreadPool(1);
79  
80      public AsynchronousProcessor(AtmosphereConfig config) {
81          this.config = config;
82      }

84      @Override
85      public void init(ServletConfig sc) throws ServletException {
86  
87          String maxInactive = sc.getInitParameter(MAX_INACTIVE) != null ? sc.getInitParameter(MAX_INACTIVE) :
88                  config.getInitParameter(MAX_INACTIVE);
89          if (maxInactive != null) {
90              final long maxInactiveTime = Long.parseLong(maxInactive);
91              if (maxInactiveTime <= 0) return;
92  
93              closedDetector.scheduleAtFixedRate(new Runnable() {
94                  public void run() {
95                      for (HttpServletRequest req : aliveRequests.keySet()) {
96                          long l = (Long) req.getAttribute(MAX_INACTIVE);
97                          if (l > 0 && System.currentTimeMillis() - l > maxInactiveTime) {
98                              try {
99                                  cancelled(req, aliveRequests.get(req).getResponse());
100                                 req.setAttribute(MAX_INACTIVE, (long) -1);
101                             } catch (IOException e) {
102                             } catch (ServletException e) {
103                             }
104                         }
105                     }
106                 }
107             }, 0, 1, TimeUnit.SECONDS);
108         }
109     }

145     /**
146      * Invoke the {@link AtmosphereHandler#onRequest} method.
147      *
148      * @param req the {@link HttpServletRequest}
149      * @param res the {@link HttpServletResponse}
150      * @return action the Action operation.
151      * @throws java.io.IOException
152      * @throws javax.servlet.ServletException
153      */
154     Action action(HttpServletRequest req, HttpServletResponse res)
155             throws IOException, ServletException {
156 
157         boolean webSocketEnabled = false;
158         if (req.getHeaders("Connection") != null && req.getHeaders("Connection").hasMoreElements()) {
159             String[] e = req.getHeaders("Connection").nextElement().split(",");
160             for (String upgrade : e) {
161                 if (upgrade.equalsIgnoreCase("Upgrade")) {
162                     webSocketEnabled = true;
163                     break;
164                 }
165             }
166         }
167 
168         if (webSocketEnabled && !supportWebSocket()) {
169             res.setStatus(501);
170             res.addHeader("X-Atmosphere-error", "Websocket protocol not supported");
171             res.flushBuffer();
172             return new Action();
173         }
174 
175         if (config.handlers().isEmpty()) {
176             logger.error("No AtmosphereHandler found. Make sure you define it inside META-INF/atmosphere.xml");
177             throw new ServletException("No AtmosphereHandler found. Make sure you define it inside META-INF/atmosphere.xml");
178         }
179 
180         if (supportSession()) {
181             // Create the session needed to support the Resume
182             // operation from disparate requests.
183             HttpSession session = req.getSession(true);
184             // Do not allow times out.
185             if (session.getMaxInactiveInterval() == DEFAULT_SESSION_TIMEOUT) {
186                 session.setMaxInactiveInterval(-1);
187             }
188         }
189 
190         req.setAttribute(FrameworkConfig.SUPPORT_SESSION, supportSession());
191 
192         AtmosphereHandlerWrapper handlerWrapper = map(req);
193         AtmosphereResourceImpl resource = new AtmosphereResourceImpl(config, handlerWrapper.broadcaster, req, res, this, handlerWrapper.atmosphereHandler);
194 
195         handlerWrapper.broadcaster.getBroadcasterConfig().setAtmosphereConfig(config);
196 
197         req.setAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE, resource);
198         req.setAttribute(FrameworkConfig.ATMOSPHERE_HANDLER, handlerWrapper.atmosphereHandler);
199 
200         try {
201             handlerWrapper.atmosphereHandler.onRequest(resource);
202         } catch (IOException t) {
203             resource.onThrowable(t);
204             throw t;
205         }
206 
207         if (resource.getAtmosphereResourceEvent().isSuspended()) {
208             req.setAttribute(MAX_INACTIVE, System.currentTimeMillis());
209             aliveRequests.put(req, resource);
210         }
211         return resource.action();
212     }

221     /**
222      * Return the {@link AtmosphereHandler} mapped to the passed servlet-path.
223      *
224      * @param req the {@link HttpServletResponse}
225      * @return the {@link AtmosphereHandler} mapped to the passed servlet-path.
226      * @throws javax.servlet.ServletException
227      */
228     protected AtmosphereHandlerWrapper map(HttpServletRequest req) throws ServletException {
229         String path = req.getServletPath();
230         if (path == null || path.length() == 0) {
231             path = "/";
232         }
233 
234         AtmosphereHandlerWrapper atmosphereHandlerWrapper = config.handlers().get(path);
235         if (atmosphereHandlerWrapper == null) {
236             // Try the /*
237             if (!path.endsWith("/")) {
238                 path += "/*";
239             } else {
240                 path += "*";
241             }
242             atmosphereHandlerWrapper = config.handlers().get(path);
243             if (atmosphereHandlerWrapper == null) {
244                 atmosphereHandlerWrapper = config.handlers().get("/*");
245                 if (atmosphereHandlerWrapper == null) {
246 
247                     if (req.getPathInfo() != null) {
248                         // Try appending the pathInfo
249                         path = req.getServletPath() + req.getPathInfo();
250                     }
251 
252                     atmosphereHandlerWrapper = config.handlers().get(path);
253                     if (atmosphereHandlerWrapper == null) {
254                         // Last chance
255                         if (!path.endsWith("/")) {
256                             path += "/*";
257                         } else {
258                             path += "*";
259                         }
260                         // Try appending the pathInfo
261                         atmosphereHandlerWrapper = config.handlers().get(path);
262                         if (atmosphereHandlerWrapper == null) {
263                             logger.warn("No AtmosphereHandler maps request for {}", path);
264                             for (String m : config.handlers().keySet()) {
265                                 logger.warn("\tAtmosphereHandler registered: {}", m);
266                             }
267                             throw new ServletException("No AtmosphereHandler maps request for " + path);
268                         }
269                     }
270                 }
271             }
272         }
273         config.getBroadcasterFactory().add(atmosphereHandlerWrapper.broadcaster,
274                 atmosphereHandlerWrapper.broadcaster.getID());
275         return atmosphereHandlerWrapper;
276     }

297     /**
298      * All proprietary Comet based {@link Servlet} must invoke the timedout
299      * method when the underlying WebServer time out the {@link HttpServletResponse}.
300      * The returned value, of type
301      * {@link AtmosphereServlet.Action}, tells the proprietary Comet {@link Servlet}
302      * to resume (again), suspended or do nothing with the current {@link HttpServletResponse}.
303      *
304      * @param request  the {@link HttpServletRequest}
305      * @param response the {@link HttpServletResponse}
306      * @return action the Action operation.
307      * @throws java.io.IOException
308      * @throws javax.servlet.ServletException
309      */
310     public Action timedout(HttpServletRequest request, HttpServletResponse response)
311             throws IOException, ServletException {
312 
313         AtmosphereResourceImpl re;
314         long l = (Long) request.getAttribute(MAX_INACTIVE);
315         if (l == -1) {
316             // The closedDetector closed the connection.
317             return timedoutAction;
318         }
319         request.setAttribute(MAX_INACTIVE, (long) -1);
320 
321         // Something went wrong.
322         if (request == null || response == null) {
323             logger.warn("Invalid Request/Response: {}/{}", request, response);
324             return timedoutAction;
325         }
326 
327         re = (AtmosphereResourceImpl) request.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
328 
329         if (re != null) {
330             re.getAtmosphereResourceEvent().setIsResumedOnTimeout(true);
331 
332             Broadcaster b = re.getBroadcaster();
333             if (b instanceof DefaultBroadcaster) {
334                 ((DefaultBroadcaster) b).broadcastOnResume(re);
335             }
336 
337             if (re.getRequest().getAttribute(ApplicationConfig.RESUMED_ON_TIMEOUT) != null) {
338                 re.getAtmosphereResourceEvent().setIsResumedOnTimeout(
339                         (Boolean) re.getRequest().getAttribute(ApplicationConfig.RESUMED_ON_TIMEOUT));
340             }
341             invokeAtmosphereHandler(re);
342         }
343 
344         return timedoutAction;
345     }

347     void invokeAtmosphereHandler(AtmosphereResourceImpl r) throws IOException {
348         HttpServletRequest req = r.getRequest();
349         HttpServletResponse response = r.getResponse();
350         String disableOnEvent = r.getAtmosphereConfig().getInitParameter(ApplicationConfig.DISABLE_ONSTATE_EVENT);
351 
352         try {
353             if (!r.getResponse().equals(response)) {
354                 logger.warn("Invalid response: {}", response);
355             } else if (disableOnEvent == null || !disableOnEvent.equals(String.valueOf(true))) {
356                 AtmosphereHandler<HttpServletRequest, HttpServletResponse> atmosphereHandler =
357                         (AtmosphereHandler<HttpServletRequest, HttpServletResponse>)
358                                 req.getAttribute(FrameworkConfig.ATMOSPHERE_HANDLER);
359                 atmosphereHandler.onStateChange(r.getAtmosphereResourceEvent());
360             } else {
361                 r.getResponse().flushBuffer();
362             }
363         } catch (IOException ex) {
364             try {
365                 r.onThrowable(ex);
366             } catch (Throwable t) {
367                 logger.warn("failed calling onThrowable()", ex);
368             }
369         } finally {
370             try {
371                 aliveRequests.remove(req);
372                 r.notifyListeners();
373             } finally {
374                 destroyResource(r);
375             }
376         }
377     }

379     private void destroyResource(AtmosphereResourceImpl r) {
380         r.removeEventListeners();
381         try {
382             r.getBroadcaster().removeAtmosphereResource(r);
383         } catch (IllegalStateException ex) {
384             logger.trace(ex.getMessage(), ex);
385         }
386         if (BroadcasterFactory.getDefault() != null) {
387             BroadcasterFactory.getDefault().removeAllAtmosphereResource(r);
388         }
389     }

391     /**
392      * All proprietary Comet based {@link Servlet} must invoke the cancelled
393      * method when the underlying WebServer detect that the client closed
394      * the connection.
395      *
396      * @param req the {@link HttpServletRequest}
397      * @param res the {@link HttpServletResponse}
398      * @return action the Action operation.
399      * @throws java.io.IOException
400      * @throws javax.servlet.ServletException
401      */
402     public Action cancelled(HttpServletRequest req, HttpServletResponse res)
403             throws IOException, ServletException {
404 
405         AtmosphereResourceImpl re = null;
406         long l = (Long) req.getAttribute(MAX_INACTIVE);
407         if (l == -1) {
408             // The closedDetector closed the connection.
409             return timedoutAction;
410         }
411         req.setAttribute(MAX_INACTIVE, (long) -1);
412 
413         try {
414             re = (AtmosphereResourceImpl) req.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
415             if (re != null) {
416                 re.getAtmosphereResourceEvent().setCancelled(true);
417                 invokeAtmosphereHandler(re);
418                 re.setIsInScope(false);
419             }
420         } catch (Throwable ex) {
421             // Something wrong happenned, ignore the exception
422             logger.debug("failed to cancel resource: " + re, ex);
423         } finally {
424             try {
425                 aliveRequests.remove(req);
426                 if (re != null) {
427                     re.notifyListeners();
428                 }
429             } finally {
430                 if (re != null) {
431                     destroyResource(re);
432                 }
433             }
434         }
435 
436         return cancelledAction;
437     }

439     void shutdown() {
440         closedDetector.shutdownNow();
441         for (AtmosphereResource<HttpServletRequest, HttpServletResponse> resource : aliveRequests.values()) {
442             try {
443                 resource.resume();
444             } catch (Throwable t) {
445                 // Something wrong happenned, ignore the exception
446                 logger.debug("failed on resume: " + resource, t);
447             }
448         }
449     }

228     /**
229      * Add a {@link BroadcastFilter}
230      *
231      * @param e {@link BroadcastFilter}
232      * @return true if added.
233      */
234     public boolean addFilter(BroadcastFilter e) {
235         if (filters.contains(e) || checkDuplicateFilter(e)) return false;
236 
237         if (e instanceof BroadcastFilterLifecycle) {
238             ((BroadcastFilterLifecycle) e).init();
239         }
240 
241         if (e instanceof PerRequestBroadcastFilter) {
242             perRequestFilters.add((PerRequestBroadcastFilter) e);
243         }
244 
245         return filters.offer(e);
246     }
247 
248     private boolean checkDuplicateFilter(BroadcastFilter e) {
249         for (BroadcastFilter f : filters) {
250             if (f.getClass().isAssignableFrom(e.getClass())) {
251                 return true;
252             }
253         }
254         return false;
255     }

257     public void destroy() {
258         if (broadcasterCache != null) {
259             broadcasterCache.stop();
260         }
261 
262         if (!isExecutorShared && executorService != null) {
263             executorService.shutdownNow();
264         }
265         if (!isAsyncExecutorShared && asyncWriteService != null) {
266             asyncWriteService.shutdownNow();
267         }
268         if (!isExecutorShared && defaultExecutorService != null) {
269             defaultExecutorService.shutdownNow();
270         }
271         if (!isAsyncExecutorShared && defaultAsyncWriteService != null) {
272             defaultAsyncWriteService.shutdownNow();
273         }
274 
275         if (scheduler != null) {
276             scheduler.shutdownNow();
277         }
278 
279         for (BroadcastFilter f : filters) {
280             if (f instanceof BroadcastFilterLifecycle) {
281                 ((BroadcastFilterLifecycle) f).destroy();
282             }
283         }
284         removeAllFilters();
285     }

287     /**
288      * Remove a {@link BroadcastFilter}
289      *
290      * @param filter {@link BroadcastFilter}
291      * @return true if removed
292      */
293     public boolean removeFilter(BroadcastFilter filter) {
294 
295         if (filter instanceof BroadcastFilterLifecycle) {
296             ((BroadcastFilterLifecycle) filter).destroy();
297         }
298 
299         if (filter instanceof PerRequestBroadcastFilter) {
300             perRequestFilters.remove(filter);
301         }
302 
303         return filters.remove(filter);
304     }

329     public boolean hasPerRequestFilters() {
330         if (filters.isEmpty()) {
331             return false;
332         } else {
333             for (BroadcastFilter b : filters) {
334                 if (PerRequestBroadcastFilter.class.isAssignableFrom(b.getClass())) {
335                     return true;
336                 }
337             }
338         }
339         return false;

342     /**
343      * Invoke {@link BroadcastFilter} in the other they were added.
344      *
345      * @param object the broadcasted object.
346      * @return BroadcastAction that tell Atmosphere to invoke the next filter or not.
347      */
348     protected BroadcastAction filter(Object object) {
349         BroadcastAction transformed = new BroadcastAction(object);
350         for (BroadcastFilter mf : filters) {
351             synchronized (mf) {
352                 transformed = mf.filter(object, transformed.message());
353                 if (transformed == null || transformed.action() == BroadcastAction.ACTION.ABORT) {
354                     return transformed;
355                 }
356             }
357         }
358         return transformed;
359     }

361     /**
362      * Invoke {@link BroadcastFilter} in the other they were added, with a unique {@link javax.servlet.http.HttpServletRequest}
363      *
364      * @param request {@link javax.servlet.http.HttpServletRequest}
365      * @param object  the broadcasted object.
366      * @return BroadcastAction that tell Atmosphere to invoke the next filter or not.
367      */
368     protected BroadcastAction filter(HttpServletRequest request, HttpServletResponse response, Object object) {
369         BroadcastAction transformed = new BroadcastAction(object);
370         for (PerRequestBroadcastFilter mf : perRequestFilters) {
371             synchronized (mf) {
372                 transformed = mf.filter(request, response, transformed.message());
373                 if (transformed == null || transformed.action() == BroadcastAction.ACTION.ABORT) {
374                     return transformed;
375                 }
376             }
377         }
378         return transformed;
379     }

462     void configureBroadcasterFilter(String[] list) {
463         for (String broadcastFilter : list) {
464             BroadcastFilter bf = null;
465             try {
466                 bf = BroadcastFilter.class
467                         .cast(Thread.currentThread().getContextClassLoader().loadClass(broadcastFilter).newInstance());
468             } catch (InstantiationException e) {
469                 logger.warn("Error trying to instantiate BroadcastFilter: " + broadcastFilter, e);
470             } catch (IllegalAccessException e) {
471                 logger.warn("Error trying to instantiate BroadcastFilter: " + broadcastFilter, e);
472             } catch (ClassNotFoundException e) {
473                 try {
474                     bf = BroadcastFilter.class
475                             .cast(BroadcastFilter.class.getClassLoader().loadClass(broadcastFilter).newInstance());
476                 } catch (InstantiationException e1) {
477                 } catch (IllegalAccessException e1) {
478                 } catch (ClassNotFoundException e1) {
479                     logger.warn("Error trying to instantiate BroadcastFilter: " + broadcastFilter, e);
480                 }
481             }
482             if (bf != null) {
483                 InjectorProvider.getInjector().inject(bf);
484                 addFilter(bf);
485             }
486 
487         }
488     }

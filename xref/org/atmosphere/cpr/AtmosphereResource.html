16  package org.atmosphere.cpr;
17  
18  import javax.servlet.http.HttpSession;
19  import java.io.IOException;
20  import java.util.List;
21  import java.util.concurrent.TimeUnit;
22  
23  /**
24   * An {@link AtmosphereResource} encapsulates the mechanism to {@link #suspend()}, {@link #resume()} and
25   * broadcast ({@link #getBroadcaster()}) messages among suspended resources. {@link AtmosphereResource}s are passed to
26   * an instance of {@link AtmosphereHandler} at runtime.
27   *
28   * @author Jeanfrancois Arcand
29   */
30  public interface AtmosphereResource {
31  
32      enum TRANSPORT {POLLING, LONG_POLLING, STREAMING, WEBSOCKET, JSONP, UNDEFINED, SSE, AJAX, HTMLFILE, CLOSE}
33  
34      /**
35       * Return the current {@link TRANSPORT}. The transport value is retrieved using the {@link HeaderConfig#X_ATMOSPHERE_TRANSPORT}
36       * header value.
37       */
38      TRANSPORT transport();
39  
40      /**
41       * Set to true to resume the response after the first broadcast. False by default.
42       *
43       * @param resumeOnBroadcast
44       */
45      AtmosphereResource resumeOnBroadcast(boolean resumeOnBroadcast);
46  
47      /**
48       * Return true if the {@link org.atmosphere.cpr.AtmosphereResource#suspend()} has been invoked.
49       *
50       * @return true if the {@link org.atmosphere.cpr.AtmosphereResource#suspend()} has been invoked
51       */
52      boolean isSuspended();
53  
54      /**
55       * Return true if this AtmosphereResource is resumed after the first broadcast.
56       *
57       * @see org.atmosphere.cpr.AtmosphereResource#resumeOnBroadcast()
58       */
59      boolean resumeOnBroadcast();
60  
61      /**
62       * Return true if this object has been resumed.
63       *
64       * @return true if this object has been resumed
65       */
66      boolean isResumed();
67  
68      /**
69       * Return true if this object has been cancelled.
70       *
71       * @return true if this object has been cancelled
72       */
73      boolean isCancelled();
74  
75      /**
76       * Complete the {@link AtmosphereResponse} and finish/commit it. If the {@link AtmosphereResponse} is in the
77       * process of being resumed, invoking this method has no effect.
78       */
79      AtmosphereResource resume();
80  
81      /**
82       * Suspend the {@link AtmosphereResponse} indefinitely.
83       * Suspending a {@link AtmosphereResponse} will tell the underlying container to avoid recycling objects associated
84       * with the current instance, and also to avoid committing a response.
85       * <p/>
86       * The Framework will output some HTML comments when suspending the response in order to make sure all browsers
87       * work well with suspended responses.
88       */
89      AtmosphereResource suspend();
90  
91      /**
92       * Suspend the {@link AtmosphereResponse}. Suspending a {@link AtmosphereResponse} will tell the underlying
93       * container to avoid recycling objects associated with the current instance, and also to avoid committing response.
94       * Invoking this method when a request is being timed out (e.g. {@link AtmosphereResourceEvent#isResumedOnTimeout}
95       * returns true) has no effect.
96       * <p/>
97       *
98       * @param timeout The maximum amount of time, in milliseconds, a {@link AtmosphereResponse} can be suspended. When
99       *                the timeout expires, the {@link AtmosphereResponse} will be automatically resumed and committed.
100      *                Usage of any methods of a {@link AtmosphereResponse} that times out will throw an {@link IllegalStateException}.
101      */
102     AtmosphereResource suspend(long timeout);
103 
104     /**
105      * Suspend the {@link AtmosphereResponse}. Suspending a {@link AtmosphereResponse} will tell the underlying
106      * container to avoid recycling objects associated with the current instance, and also to avoid committing response.
107      * Invoking this method when a request is being timed out (e.g. {@link AtmosphereResourceEvent#isResumedOnTimeout}
108      * returns true) has no effect.
109      * <p/>
110      *
111      * @param timeout  The maximum amount of time a {@link AtmosphereResponse} can be suspended. When the timeout
112      *                 expires, the {@link AtmosphereResponse} will be automatically resumed and committed. Usage of any
113      *                 methods of a {@link AtmosphereResponse} that times out will throw an {@link IllegalStateException}.
114      * @param timeunit The time unit of the timeout value
115      */
116 
117     AtmosphereResource suspend(long timeout, TimeUnit timeunit);
118 
119     /**
120      * Return the underlying {@link AtmosphereRequest} request.
121      *
122      * @return {@link AtmosphereRequest} the underlying request.
123      */
124     AtmosphereRequest getRequest();
125 
126     /**
127      * Return the {@link AtmosphereResponse}.
128      *
129      * @return {@link AtmosphereResponse} the underlying response.
130      */
131     AtmosphereResponse getResponse();
132 
133     /**
134      * Return the {@link AtmosphereConfig}.
135      *
136      * @return the {@link AtmosphereConfig}
137      */
138     AtmosphereConfig getAtmosphereConfig();
139 
140     /**
141      * Return the first added {@link Broadcaster}.
142      *
143      * @return the current {@link Broadcaster}
144      */
145     Broadcaster getBroadcaster();
146 
147     /**
148      * Return an unmodifiable list of {@link Broadcaster}s associated with this resource
149      *
150      * @return an unmodifiable list of {@link Broadcaster}
151      */
152     List<Broadcaster> broadcasters();
153 
154     /**
155      * Remove this {@link org.atmosphere.cpr.AtmosphereResource} from all {@link org.atmosphere.cpr.Broadcaster}
156      *
157      * @return this
158      */
159     public AtmosphereResource removeFromAllBroadcasters();
160 
161     /**
162      * Set the first {@link Broadcaster} associated with this resource. This {@link org.atmosphere.cpr.Broadcaster}
163      * will be returned when {@link #getBroadcaster()} is invoked.
164      *
165      * @param broadcaster
166      * @return this
167      */
168     AtmosphereResource setBroadcaster(Broadcaster broadcaster);
169 
170     /**
171      * Add/Associate a {@link org.atmosphere.cpr.Broadcaster} with this resource.
172      *
173      * @param broadcaster
174      * @return this
175      */
176     AtmosphereResource addBroadcaster(Broadcaster broadcaster);
177 
178     /**
179      * Remove a {@link org.atmosphere.cpr.Broadcaster} with this resource.
180      *
181      * @param broadcaster
182      * @return this
183      */
184     AtmosphereResource removeBroadcaster(Broadcaster broadcaster);
185 
186     /**
187      * Set the {@link Serializer} to use when {@link AtmosphereResource#write} execute the operation.
188      * By default, the {@link Serializer} is null.
189      *
190      * @param s the {@link Serializer}
191      * @return this
192      */
193     AtmosphereResource setSerializer(Serializer s);
194 
195     /**
196      * Write the String. If {@link #resumeOnBroadcast()} is true, the underlying connection will be resumed (@link #resume());
197      *
198      * @param s
199      * @return this
200      */
201     AtmosphereResource write(String s);
202 
203     /**
204      * Write the bytes If {@link #resumeOnBroadcast()} is true, the underlying connection will be resumed (@link #resume());
205      *
206      * @param s
207      * @return this
208      */
209     AtmosphereResource write(byte[] s);
210 
211     /**
212      * Get the {@link Serializer} or null if not defined.
213      *
214      * @return the {@link Serializer} or null if not defined
215      */
216     Serializer getSerializer();
217 
218     /**
219      * Return the current {@link AtmosphereResourceEvent}.
220      */
221     AtmosphereResourceEvent getAtmosphereResourceEvent();
222 
223     /**
224      * Return the associated {@link AtmosphereHandler} associated with this resource.
225      *
226      * @return the associated {@link AtmosphereHandler} associated with this resource
227      */
228     AtmosphereHandler getAtmosphereHandler();
229 
230     /**
231      * Set a message that will be written when the resource times out. Can be an {@link Object} or {@link java.util.concurrent.Callable}.
232      *
233      * @return this
234      */
235     AtmosphereResource writeOnTimeout(Object o);
236 
237     /**
238      * Return the object that will be written when the resource times out.
239      */
240     Object writeOnTimeout();
241 
242     /**
243      * Return the unique ID associated with this AtmosphereResource.
244      *
245      * @return the unique ID associated with this AtmosphereResource
246      */
247     String uuid();
248 
249     /**
250      * Add a {@link AtmosphereResourceEventListener}.
251      *
252      * @param e an instance of AtmosphereResourceEventListener
253      * @return this
254      */
255     AtmosphereResource addEventListener(AtmosphereResourceEventListener e);
256 
257     /**
258      * Remove a {@link AtmosphereResourceEventListener}.
259      *
260      * @param e
261      * @return this
262      */
263     AtmosphereResource removeEventListener(AtmosphereResourceEventListener e);
264 
265     /**
266      * Remove all {@link AtmosphereResourceEventListener}s.
267      *
268      * @return this
269      */
270     AtmosphereResource removeEventListeners();
271 
272     /**
273      * Notify all {@link AtmosphereResourceEventListener}.
274      *
275      * @param e an instance of AtmosphereResourceEvent
276      * @return this
277      */
278     AtmosphereResource notifyListeners(AtmosphereResourceEvent e);
279 
280     /**
281      * Notify all {@link AtmosphereResourceEventListener}s.
282      *
283      * @return this
284      */
285     AtmosphereResource notifyListeners();
286 
287     /**
288      * Return the {@link HttpSession} if supported, null if not
289      *
290      * @return the {@link HttpSession} if supported, null if not
291      */
292     HttpSession session();
293 
294     /**
295      * Return the {@link HttpSession} if supported, and creates it if not already created.
296      *
297      * @return the {@link HttpSession} if supported, and creates it if not already created
298      */
299     HttpSession session(boolean create);
300 
301     /**
302      * Close the underlying connection. Invoking this method will close the underlying connection and resume the
303      * {@link AtmosphereResource}}.
304      */
305     void close() throws IOException;
306 
307     /**
308      * Force binary write and never write String value.
309      * return this
310      */
311     AtmosphereResource forceBinaryWrite(boolean force);
312 
313     /**
314      * Return true when binary write is forced.
315      *
316      * @return true when binary write is forced.
317      */
318     boolean forceBinaryWrite();
319 
320     /**
321      * Initialize an {@link AtmosphereResource}.
322      *
323      * @param config            The {@link org.atmosphere.cpr.AtmosphereConfig}
324      * @param broadcaster       The {@link org.atmosphere.cpr.Broadcaster}.
325      * @param req               The {@link AtmosphereRequest}
326      * @param response          The {@link AtmosphereResource}
327      * @param asyncSupport      The {@link AsyncSupport}
328      * @param atmosphereHandler The {@link AtmosphereHandler}
329      * @return this
330      */
331     public AtmosphereResource initialize(AtmosphereConfig config, Broadcaster broadcaster,
332                                          AtmosphereRequest req, AtmosphereResponse response,
333                                          AsyncSupport asyncSupport, AtmosphereHandler atmosphereHandler);
334 }

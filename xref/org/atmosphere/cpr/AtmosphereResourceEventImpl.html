64  public class AtmosphereResourceEventImpl implements AtmosphereResourceEvent {

66      // Was the remote connection closed.
67      private final AtomicBoolean isCancelled = new AtomicBoolean(false);
68      // Is Resumed on Timeout?
69      private final AtomicBoolean isResumedOnTimeout = new AtomicBoolean(false);
70      private Throwable throwable;
71      // The current message
72      protected Object message;

75      public AtmosphereResourceEventImpl(AtmosphereResourceImpl resource) {
76          this.resource = resource;

80      public AtmosphereResourceEventImpl(AtmosphereResourceImpl resource, boolean isCancelled,
81                                         boolean isResumedOnTimeout) {
82          this.isCancelled.set(isCancelled);
83          this.isResumedOnTimeout.set(isResumedOnTimeout);
84          this.resource = resource;
85          this.throwable = null;
86      }
87  
88      public AtmosphereResourceEventImpl(AtmosphereResourceImpl resource, boolean isCancelled,
89                                         boolean isResumedOnTimeout,
90                                         Throwable throwable) {
91          this.isCancelled.set(isCancelled);
92          this.isResumedOnTimeout.set(isResumedOnTimeout);
93          this.resource = resource;
94          this.throwable = throwable;
95      }

97      /**
98       * {@inheritDoc}

100     public boolean isResuming() {

126         this.message = message;
127         return this;
128     }

133     public boolean isResumedOnTimeout() {
134         return isResumedOnTimeout.get();

138      * {@inheritDoc}

140     public boolean isCancelled() {
141         return isCancelled.get();
142     }

144     protected AtmosphereResourceEvent setCancelled(boolean isCancelled) {
145         check();
146         resource.action().type(Action.TYPE.CANCELLED);
147         this.isCancelled.set(isCancelled);
148         return this;
149     }
150 
151     protected AtmosphereResourceEvent setIsResumedOnTimeout(boolean isResumedOnTimeout) {
152         check();
153         resource.action().type(Action.TYPE.TIMEOUT);
154         this.isResumedOnTimeout.set(isResumedOnTimeout);
155         return this;
156     }

158     @Override
159     public boolean equals(Object o) {
160         if (this == o) return true;
161         if (o == null || getClass() != o.getClass()) return false;
162 
163         AtmosphereResourceEventImpl that = (AtmosphereResourceEventImpl) o;
164 
165         if (isCancelled != null ? !isCancelled.equals(that.isCancelled) : that.isCancelled != null) return false;
166         if (isResumedOnTimeout != null ? !isResumedOnTimeout.equals(that.isResumedOnTimeout) : that.isResumedOnTimeout != null)
167             return false;
168         if (message != null ? !message.equals(that.message) : that.message != null) return false;
169         if (resource != null ? !resource.equals(that.resource) : that.resource != null) return false;
170         if (throwable != null ? !throwable.equals(that.throwable) : that.throwable != null) return false;
171 
172         return true;
173     }

178         result = 31 * result + (isResumedOnTimeout != null ? isResumedOnTimeout.hashCode() : 0);
179         result = 31 * result + (throwable != null ? throwable.hashCode() : 0);
180         result = 31 * result + (message != null ? message.hashCode() : 0);
181         result = 31 * result + (resource != null ? resource.hashCode() : 0);
182         return result;
183     }

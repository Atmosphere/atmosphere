16  package org.atmosphere.cpr;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  import java.io.IOException;
22  import java.util.ArrayList;
23  
24  /**
25   * An {@link AsyncIOWriter} that delegates the write operation to it's {@link AsyncIOInterceptor}. If no
26   * AsyncIOInterceptor are specified, this class does nothing and the response's will never get written.
27   *
28   * @author Jeanfrancois Arcand
29   */
30  public class AtmosphereInterceptorWriter extends AsyncIOWriterAdapter {
31  
32      private static final Logger logger = LoggerFactory.getLogger(AsyncIOWriterAdapter.class);
33      private final AtmosphereResponse response;
34  
35      private final ArrayList<AsyncIOInterceptor> filters = new ArrayList<AsyncIOInterceptor>();
36  
37      public AtmosphereInterceptorWriter(AtmosphereResponse response) {
38          this.response = response;
39      }
40  
41      @Override
42      public AsyncIOWriter redirect(String location) throws IOException {
43          response.sendRedirect(location);
44          return this;
45      }
46  
47      @Override
48      public AsyncIOWriter writeError(int errorCode, String message) throws IOException {
49          response.sendError(errorCode);
50          return this;
51      }
52  
53      @Override
54      public AsyncIOWriter write(String data) throws IOException {
55          for (AsyncIOInterceptor i : filters) {
56              i.intercept(response, data);
57          }
58          return this;
59      }
60  
61      @Override
62      public AsyncIOWriter write(byte[] data) throws IOException {
63          for (AsyncIOInterceptor i : filters) {
64              i.intercept(response, data);
65          }
66          return this;
67      }
68  
69      @Override
70      public AsyncIOWriter write(byte[] data, int offset, int length) throws IOException {
71          for (AsyncIOInterceptor i : filters) {
72              i.intercept(response, data, offset, length);
73          }
74          return this;
75      }
76  
77      @Override
78      public void close() throws IOException {
79          response.closeStreamOrWriter();
80      }
81  
82      @Override
83      public AsyncIOWriter flush() throws IOException {
84          response.flushBuffer();
85          return this;
86      }
87  
88      public AtmosphereInterceptorWriter interceptor(AsyncIOInterceptor filter) {
89          filters.add(filter);
90          return this;
91      }
92  }

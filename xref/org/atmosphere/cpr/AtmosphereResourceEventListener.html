53  package org.atmosphere.cpr;
54  
55  /**
56   * Receive notification when a resume, client disconnect or broadcast events
57   * occurs.
58   *
59   * @author Jeanfrancois Arcand
60   */
61  public interface AtmosphereResourceEventListener {
62  
63      /**
64       * Invoked when the {@link AtmosphereResource#suspend} has been completed and the response
65       * considered as suspended.
66       *
67       * @param event a {@link org.atmosphere.cpr.AtmosphereResourceEvent}
68       */
69      void onSuspend(AtmosphereResourceEvent event);
70  
71      /**
72       * Invoked when the {@link AtmosphereResource#resume} is invoked or when the
73       * suspend's time out expires.
74       *
75       * @param event a {@link org.atmosphere.cpr.AtmosphereResourceEvent}
76       */
77      void onResume(AtmosphereResourceEvent event);
78  
79      /**
80       * Invoked when the remote connection gets closed.
81       *
82       * @param event a {@link org.atmosphere.cpr.AtmosphereResourceEvent}
83       */
84      void onDisconnect(AtmosphereResourceEvent event);
85  
86      /**
87       * Invoked when a {@link Broadcaster#broadcast} occurs.
88       *
89       * @param event a {@link org.atmosphere.cpr.AtmosphereResourceEvent}
90       */
91      void onBroadcast(AtmosphereResourceEvent event);
92  
93  
94      /**
95       * Invoked when an operations failed to execute for an unknown reason like : IOException because the client
96       * remotly closed the connection, a broken connection, etc.
97       *
98       * @param event a {@link AtmosphereResourceEvent}
99       */
100     void onThrowable(AtmosphereResourceEvent event);
101 
102 }

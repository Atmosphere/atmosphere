66  /**
67   * An Atmosphere request representation. An {@link AtmosphereRequest} is a two-way communication channel between the
68   * client and the server. If the {@link AtmosphereRequestImpl#isDestroyable()} is set to false, or if its
69   * associated {@link AtmosphereResource} has been suspended, this object can be re-used at any moment between requests.
70   * You can use its associated {@link AtmosphereResponse} to write bytes at any moment, making this object bi-directional.
71   * <br/>
72   *
73   * @author Jeanfrancois Arcand
74   */

106     private ServletInputStream configureStream() {
107         if (bis == null && !streamSet.getAndSet(true)) {
108             if (b.inputStream != null) {
109                 bis = new IS(b.inputStream);
110             } else if (b.reader == null) {
111                 if (b.body.dataBytes != null) {
112                     bis = new ByteInputStream(b.body.dataBytes, b.body.offset, b.body.length);
113                 } else if (b.body.data != null) {
114                     try {
115                         byte[] bytes = b.body.data.getBytes("UTF-8");
116                         bis = new ByteInputStream(bytes, 0, bytes.length);
117                     } catch (UnsupportedEncodingException e) {
118                         logger.trace("", e);
119                     }
120                 }
121             } else {
122                 bis = new IS(new ReaderInputStream(b.reader));
123             }
124         }
125         return bis;
126     }
127 
128     private BufferedReader configureReader() {
129         if (br == null && !readerSet.getAndSet(false)) {
130             if (b.reader != null) {
131                 br = new BufferedReader(b.reader);
132             } else if (b.inputStream == null) {
133                 try {
134                     if (b.body.dataBytes != null) {
135                         br = new BufferedReader(new StringReader(new String(b.body.dataBytes, b.body.offset, b.body.length, b.encoding)));
136                     } else if (b.body.data != null) {
137                         br = new BufferedReader(new StringReader(b.body.data));
138                     }
139                 } catch (UnsupportedEncodingException e) {
140                     throw new RuntimeException(e);
141                 }
142             } else {
143                 br = new BufferedReader(new InputStreamReader(b.inputStream));
144             }
145         }
146         return br;
147     }

175     private String toQs() {
176         StringBuilder q = new StringBuilder();
177         for (Map.Entry<String, String[]> e : b.queryStrings.entrySet()) {
178             for (String k : e.getValue()) {
179                 q.append(e.getKey()).append("=").append(k).append("&");
180             }
181         }
182         if (q.length() > 0) q.deleteCharAt(q.length() - 1);
183         return q.toString();
184     }

236     @Override
237     public Enumeration getHeaders(String name) {
238 
239         ArrayList list = new ArrayList<String>();
240         // Never override the parent Request
241         if (!name.equalsIgnoreCase("content-type")) {
242             list = Collections.list(b.request.getHeaders(name));
243         }
244 
245         if (name.equalsIgnoreCase("content-type")) {
246             String s = getContentType();
247             if (s != null) {
248                 list.add(s);
249             }
250         } else {
251             if (b.headers.get(name) != null) {
252                 list.add(b.headers.get(name));
253             }
254 
255             if (isNotNoOps()) {
256                 if (list.isEmpty() && name.startsWith(X_ATMOSPHERE)) {
257                     if (attributeWithoutException(b.request, name) != null) {
258                         list.add(attributeWithoutException(b.request, name));
259                     }
260                 }
261             }
262         }
263         return Collections.enumeration(list);
264     }

271     @Override
272     public Enumeration<String> getHeaderNames() {
273         Set list = new HashSet();
274         list.addAll(b.headers.keySet());
275 
276         list.addAll(Collections.list(b.request.getHeaderNames()));
277         if (b.request != null) {
278             Enumeration e = b.request.getAttributeNames();
279             while (e.hasMoreElements()) {
280                 String name = e.nextElement().toString();
281                 if (name.startsWith(X_ATMOSPHERE)) {
282                     list.add(name);
283                 }
284             }
285         }
286 
287         if (b.contentType != null) {
288             list.add("Content-Type");
289         }
290         return Collections.enumeration(list);
291     }

342     @Override
343     public String getHeader(String s, boolean checkCase) {
344 
345         if ("content-type".equalsIgnoreCase(s)) {
346             return getContentType();
347         }
348 
349         String name = b.request.getHeader(s);
350         if (name == null) {
351             if (b.headers.get(s) != null) {
352                 return b.headers.get(s);
353             }
354 
355             if (s.startsWith(X_ATMOSPHERE) && isNotNoOps()) {
356                 // Craziness with Struts 2 who wraps String attribute as BigDecimal
357                 // https://github.com/Atmosphere/atmosphere/issues/1367
358                 Object o = attributeWithoutException(b.request, s);
359                 if (o == null || String.class.isAssignableFrom(o.getClass())) {
360                     name = String.class.cast(o);
361                 } else {
362                     try {
363                         if (HttpServletRequestWrapper.class.isAssignableFrom(b.request.getClass())) {
364                             HttpServletRequest hsr = HttpServletRequestWrapper.class.cast(b.request);
365                             while (hsr instanceof HttpServletRequestWrapper) {
366                                 hsr = (HttpServletRequest) ((HttpServletRequestWrapper) hsr).getRequest();
367                                 o = attributeWithoutException(hsr, s);
368                                 if (o == null || String.class.isAssignableFrom(o.getClass())) {
369                                     name = String.class.cast(o);
370                                     break;
371                                 }
372                             }
373                         }
374                     } catch (Exception ex) {
375                         logger.warn("", ex);
376                     }
377                 }
378             }
379         }
380 
381         if (name == null && checkCase) {
382             return getHeader(s.toLowerCase(), false);
383         }
384 
385         if (name == null && "connection".equalsIgnoreCase(s)) {
386             return "keep-alive";
387         }
388 
389         return name;
390     }

420     @Override
421     public String[] getParameterValues(String s) {
422         String[] list = b.request.getParameterValues(s);
423         if (list != null && b.queryStrings.get(s) != null) {
424             String[] newList = b.queryStrings.get(s);
425             if (!Arrays.deepEquals(list, newList)) {
426                 String[] s1 = new String[list.length + newList.length];
427                 System.arraycopy(list, 0, s1, 0, list.length);
428                 System.arraycopy(newList, 0, s1, list.length, newList.length);
429                 return s1;
430             } else {
431                 return list;
432             }
433         } else {
434             return list == null ? b.queryStrings.get(s) : list;
435         }
436     }

486     @Override
487     public AtmosphereRequest queryString(String qs) {
488 
489         if (qs == null) return this;
490 
491         if (!qs.isEmpty()) {
492             QueryStringDecoder decoder = new QueryStringDecoder(getRequestURI() + "?" + qs);
493             Map<String, List<String>> m = decoder.getParameters();
494             Map<String, String[]> newM = new HashMap<String, String[]>();
495             for (Map.Entry<String, List<String>> q : m.entrySet()) {
496                 newM.put(q.getKey(), q.getValue().toArray(new String[q.getValue().size()]));
497             }
498             b.queryStrings(newM);
499         }
500         b.queryString = qs;
501         return this;
502     }

674     @Override
675     public HttpSession getSession(boolean create) {
676         if (b.webSocketFakeSession != null) {
677             return b.webSocketFakeSession;
678         }
679 
680         if (resource() != null) {
681             // UGLY, but we need to prevent looping here.
682             HttpSession session = AtmosphereResourceImpl.class.cast(resource()).session;
683             try {
684                 if (session != null) {
685                     // check if session is valid (isNew() will throw if not)
686                     session.isNew();
687                     return session;
688                 }
689             } catch (IllegalStateException e) {
690                 // session has been invalidated
691             }
692         }
693 
694         try {
695             return b.request.getSession(create);
696         } catch (java.lang.IllegalStateException ex) {
697             // Jetty
698             if (ex.getMessage() != null && ex.getMessage().equals("No SessionManager")) {
699                 return null;
700             }
701             throw ex;
702         } catch (NullPointerException ex) {
703             // GLASSFISH http://java.net/jira/browse/GLASSFISH-18856
704             try {
705                 return b.request.getSession(create);
706             } catch (Exception e) {
707                 logger.trace("", ex);
708                 return null;
709             }
710         } catch (RuntimeException ex) {
711             // https://github.com/Atmosphere/atmosphere/issues/1974
712             logger.trace("", ex);
713             if (ex.getMessage() != null && ex.getMessage().contains("SESN0007E")) {
714                 return null;
715             }
716             throw ex;
717         }
718     }

1422      * Copy the HttpServletRequest content inside an AtmosphereRequest. By default the returned AtmosphereRequest
1423      * is not destroyable.
1424      *
1425      * @param request {@link HttpServletRequest}
1426      * @return an {@link AtmosphereRequest}
1427      */
1428     public final static AtmosphereRequest cloneRequest(HttpServletRequest request, boolean loadInMemory, boolean copySession, boolean isDestroyable, boolean createSession) {
1429         Builder b;
1430         HttpServletRequest r;
1431 
1432         Cookie[] cs = request.getCookies();
1433         Set<Cookie> hs = Collections.synchronizedSet(new HashSet());
1434         if (cs != null) {
1435             for (Cookie c : cs) {
1436                 hs.add(c);
1437             }
1438         }
1439 
1440         boolean isWrapped = false;
1441         if (AtmosphereRequestImpl.class.isAssignableFrom(request.getClass())) {
1442             b = AtmosphereRequestImpl.class.cast(request).b;
1443             isWrapped = true;
1444         } else {
1445             b = new Builder();
1446             b.request(request);
1447         }
1448 
1449         HttpSession session = request.getSession(false);
1450         if (copySession) {
1451             session = request.getSession(createSession);
1452             if (session != null) {
1453                 session = new FakeHttpSession(session);
1454             } else {
1455                 session = new FakeHttpSession("", null, System.currentTimeMillis(), -1);
1456             }
1457         }
1458 
1459         b.servletPath(request.getServletPath())
1460                 .pathInfo(request.getPathInfo())
1461                 .contextPath(request.getContextPath())
1462                 .requestURI(request.getRequestURI())
1463                 .requestURL(request.getRequestURL().toString())
1464                 .method(request.getMethod())
1465                 .serverName(request.getServerName())
1466                 .serverPort(request.getServerPort())
1467                 .remoteAddr(request.getRemoteAddr())
1468                 .remoteHost(request.getRemoteHost())
1469                 .remotePort(request.getRemotePort())
1470                 .destroyable(isDestroyable)
1471                 .cookies(hs)
1472                 .session(session)
1473                 .principal(request.getUserPrincipal())
1474                 .authType(request.getAuthType())
1475                 .isSSecure(request.isSecure());
1476 
1477         if (loadInMemory) {
1478             String s = (String) attributeWithoutException(request, FrameworkConfig.THROW_EXCEPTION_ON_CLONED_REQUEST);
1479             boolean throwException = s != null && Boolean.parseBoolean(s);
1480             r = new NoOpsRequest(throwException);
1481             if (isWrapped) {
1482                 load(b.request, b);
1483             } else {
1484                 load(request, b);
1485             }
1486             b.request(r);
1487         }
1488 
1489         return isWrapped ? AtmosphereRequestImpl.class.cast(request) : b.build();

2045     private static void load(HttpServletRequest request, Builder b) {
2046         Enumeration<String> e = request.getHeaderNames();
2047         String s;
2048         while (e.hasMoreElements()) {
2049             s = e.nextElement();
2050             b.headers.put(s, request.getHeader(s));
2051         }
2052 
2053         e = request.getAttributeNames();
2054         while (e.hasMoreElements()) {
2055             s = e.nextElement();
2056             b.localAttributes.put(s, attributeWithoutException(request, s));
2057         }
2058 
2059         e = request.getParameterNames();
2060         while (e.hasMoreElements()) {
2061             s = e.nextElement();
2062             b.queryStrings.put(s, request.getParameterValues(s));
2063         }
2064         b.queryString = request.getQueryString();
2065 
2066         Enumeration<Locale> l = request.getLocales();
2067         while (l.hasMoreElements()) {
2068             b.locale(l.nextElement());
2069         }

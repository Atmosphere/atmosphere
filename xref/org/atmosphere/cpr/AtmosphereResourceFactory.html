30  /**
31   * A Factory used to manage {@link AtmosphereResource} instances. You can use this factory to create, remove and find
32   * {@link AtmosphereResource} instances that are associated with one or several {@link Broadcaster}s.
33   *
34   * @author Jeanfrancois Arcand
35   */
36  public final class AtmosphereResourceFactory {

38      private final static Logger logger = LoggerFactory.getLogger(AtmosphereResourceFactory.class);
39      private final static AtmosphereResourceFactory factory = new AtmosphereResourceFactory();
40      private final static Broadcaster noOps = (Broadcaster)
41              Proxy.newProxyInstance(Broadcaster.class.getClassLoader(), new Class[]{Broadcaster.class},
42                      new InvocationHandler() {
43                          @Override
44                          public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
45                              return null;
46                          }
47                      });
48      private final static AtmosphereHandler noOpsHandler = (AtmosphereHandler)
49              Proxy.newProxyInstance(AtmosphereHandler.class.getClassLoader(), new Class[]{AtmosphereHandler.class},
50                      new InvocationHandler() {
51                          @Override
52                          public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
53                              return null;
54                          }
55                      });
56      private final static AtmosphereHandler voidAtmosphereHandler = new AbstractReflectorAtmosphereHandler() {
57          @Override
58          public void onRequest(AtmosphereResource resource) throws IOException {
59          }

61          @Override

63          }

153     /**
154      * Create an {@link AtmosphereResource} associated with the uuid.
155      *
156      * @param config an {@link AtmosphereConfig}
157      * @param uuid   a String representing a UUID
158      * @return
159      */
160     public final AtmosphereResource create(AtmosphereConfig config, String uuid) {
161         AtmosphereResponse response = AtmosphereResponse.newInstance();
162         response.setHeader(HeaderConfig.X_ATMOSPHERE_TRACKING_ID, uuid);
163         return create(config,
164                 noOps,
165                 AtmosphereRequest.newInstance(),
166                 response,
167                 config.framework().getAsyncSupport(),
168                 noOpsHandler);
169     }

171     /**
172      * Remove the {@link AtmosphereResource} from all instances of {@link Broadcaster}.
173      *
174      * @param uuid the {@link org.atmosphere.cpr.AtmosphereResource#uuid()}
175      * @return the {@link AtmosphereResource}, or null if not found.
176      */
177     public final AtmosphereResource remove(String uuid) {
178         AtmosphereResource r = find(uuid);
179         if (r != null) {
180             r.getAtmosphereConfig().getBroadcasterFactory().removeAllAtmosphereResource(r);
181         }
182         return r;
183     }

185     /**
186      * Find an {@link AtmosphereResource} based on its {@link org.atmosphere.cpr.AtmosphereResource#uuid()}.
187      *
188      * @param uuid the {@link org.atmosphere.cpr.AtmosphereResource#uuid()}
189      * @return the {@link AtmosphereResource}, or null if not found.
190      */
191     public final AtmosphereResource find(String uuid) {
192         Collection<Broadcaster> l = BroadcasterFactory.getDefault().lookupAll();
193         for (Broadcaster b : l) {
194             for (AtmosphereResource r : b.getAtmosphereResources()) {
195                 if (r.uuid().equalsIgnoreCase(uuid)) {
196                     return r;
197                 }
198             }
199         }
200         return null;
201     }

203     /**
204      * Return all {@link Broadcaster} associated with a {@link AtmosphereResource#uuid}, e.g for which
205      * {@link Broadcaster#addAtmosphereResource(AtmosphereResource)} has been called. Note that this
206      * method is not synchronized and may not return all the {@link Broadcaster} in case
207      * {@link Broadcaster#addAtmosphereResource(AtmosphereResource)} is being called concurrently.
208      *
209      * @param uuid the {@link org.atmosphere.cpr.AtmosphereResource#uuid()}
210      * @return all {@link Broadcaster} associated with a {@link AtmosphereResource#uuid}
211      */
212     public final Set<Broadcaster> broadcasters(String uuid) {
213         Collection<Broadcaster> l = BroadcasterFactory.getDefault().lookupAll();
214         Set<Broadcaster> h = new HashSet<Broadcaster>();        for (Broadcaster b : l) {
215             for (AtmosphereResource r : b.getAtmosphereResources()) {
216                 if (r.uuid().equalsIgnoreCase(uuid)) {
217                     h.add(b);
218                 }
219             }
220         }
221         return h;
222     }

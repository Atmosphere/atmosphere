38  package org.atmosphere.cpr;
39  
40  import java.util.concurrent.CountDownLatch;
41  import java.util.concurrent.ExecutionException;
42  import java.util.concurrent.Future;
43  import java.util.concurrent.TimeUnit;
44  import java.util.concurrent.TimeoutException;
45  
46  /**
47   * Simple {@link Future} that can be used when awiting for a {@link Broadcaster} to finish
48   * it's broadcast operations to {@link AtmosphereHandler}
49   *
50   * @author Jeanfrancois Arcand
51   */
52  public class BroadcasterFuture<E> implements Future {
53  
54      private final CountDownLatch latch;
55  
56      private boolean isCancelled = false;
57  
58      private boolean isDone = false;
59  
60      private final E msg;
61  
62      private final Future<?> innerFuture;
63  
64      public BroadcasterFuture(E msg) {
65          this(null, msg);
66      }
67  
68      public BroadcasterFuture(Future<?> innerFuture, E msg) {
69          this.msg = msg;
70          this.innerFuture = innerFuture;
71          if (innerFuture == null) {
72              latch = new CountDownLatch(1);
73          } else {
74              latch = null;
75          }
76      }
77  
78      /**
79       * {@inheritDoc}
80       */
81      @Override
82      public boolean cancel(boolean b) {
83  
84          if (innerFuture != null) {
85              return innerFuture.cancel(b);
86          }
87  
88          if (latch.getCount() == 1) {
89              latch.countDown();
90              isCancelled = true;
91          }
92          return isCancelled;
93      }
94  
95      /**
96       * {@inheritDoc}
97       */
98      @Override
99      public boolean isCancelled() {
100 
101         if (innerFuture != null) {
102             return innerFuture.isCancelled();
103         }
104 
105         return isCancelled;
106     }
107 
108     /**
109      * {@inheritDoc}
110      */
111     @Override
112     public boolean isDone() {
113 
114         if (innerFuture != null) {
115             return innerFuture.isDone();
116         }
117 
118         isDone = true;
119         return isDone;
120     }
121 
122     /**
123      * Invoked when a {@link Broadcaster} completed it broadcast operation.
124      */
125     public void done() {
126         isDone = true;
127         if (latch != null) {
128             latch.countDown();
129         }
130     }
131 
132     /**
133      * {@inheritDoc}
134      */
135     @Override
136     public E get() throws InterruptedException, ExecutionException {
137         if (innerFuture != null) {
138             return (E) innerFuture.get();
139         }
140 
141         latch.await();
142         return msg;
143     }
144 
145     /**
146      * {@inheritDoc}
147      */
148     @Override
149     public E get(long l, TimeUnit tu) throws InterruptedException, ExecutionException, TimeoutException {
150 
151         if (innerFuture != null) {
152             return (E) innerFuture.get();
153         }
154 
155         latch.await(l, tu);
156         return msg;
157     }
158 }

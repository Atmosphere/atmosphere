53  package org.atmosphere.cpr;
54  
55  import java.util.concurrent.CopyOnWriteArrayList;
56  import java.util.concurrent.CountDownLatch;
57  import java.util.concurrent.ExecutionException;
58  import java.util.concurrent.Future;
59  import java.util.concurrent.TimeUnit;
60  import java.util.concurrent.TimeoutException;
61  
62  /**
63   * Simple {@link Future} that can be used when awaiting for a {@link Broadcaster} to finish
64   * it's broadcast operations to {@link AtmosphereHandler}
65   *
66   * @author Jeanfrancois Arcand
67   */
68  public class BroadcasterFuture<E> implements Future {
69  
70      private final CountDownLatch latch;
71      private boolean isCancelled = false;
72      private boolean isDone = false;
73      private final E msg;
74      private final Future<?> innerFuture;
75      private final CopyOnWriteArrayList<BroadcasterListener> listeners;
76      private final Broadcaster broadcaster;
77  
78      public BroadcasterFuture(E msg, CopyOnWriteArrayList<BroadcasterListener> listeners, Broadcaster b) {
79          this(null, msg, listeners, b);
80      }
81  
82      public BroadcasterFuture(Future<?> innerFuture, E msg,
83                               CopyOnWriteArrayList<BroadcasterListener> listeners, Broadcaster b) {
84          this(innerFuture, msg, 1, listeners, b);
85      }
86  
87      public BroadcasterFuture(E msg, int latchCount,
88                               CopyOnWriteArrayList<BroadcasterListener> listeners, Broadcaster b) {
89          this(null, msg, latchCount, listeners, b);
90      }
91  
92      public BroadcasterFuture(Future<?> innerFuture, E msg, int latchCount,
93                               CopyOnWriteArrayList<BroadcasterListener> listeners, Broadcaster b) {
94          this.msg = msg;
95          this.innerFuture = innerFuture;
96          this.broadcaster = b;
97          if (innerFuture == null) {
98              latch = new CountDownLatch(latchCount);
99          } else {
100             latch = null;
101         }
102         this.listeners = listeners;
103     }
104 
105     /**
106      * {@inheritDoc}
107      */
108     @Override
109     public boolean cancel(boolean b) {
110 
111         if (innerFuture != null) {
112             return innerFuture.cancel(b);
113         }
114         isCancelled = true;
115         while (latch.getCount() > 0) {
116             latch.countDown();
117         }
118         notifyListener();
119         return isCancelled;
120     }
121 
122     /**
123      * {@inheritDoc}
124      */
125     @Override
126     public boolean isCancelled() {
127 
128         if (innerFuture != null) {
129             return innerFuture.isCancelled();
130         }
131 
132         return isCancelled;
133     }
134 
135     /**
136      * {@inheritDoc}
137      */
138     @Override
139     public boolean isDone() {
140 
141         if (innerFuture != null) {
142             return innerFuture.isDone();
143         }
144 
145         isDone = true;
146         return isDone;
147     }
148 
149     /**
150      * Invoked when a {@link Broadcaster} completed it broadcast operation.
151      */
152     public BroadcasterFuture<E> done() {
153         isDone = true;
154         if (latch != null) {
155             latch.countDown();
156         }
157 
158         if (latch.getCount() == 0) {
159             notifyListener();
160         }
161         return this;
162     }
163 
164     /**
165      * {@inheritDoc}
166      */
167     @Override
168     public E get() throws InterruptedException, ExecutionException {
169         if (innerFuture != null) {
170             return (E) innerFuture.get();
171         }
172 
173         latch.await();
174         return msg;
175 
176     }
177 
178     void notifyListener() {
179         for (BroadcasterListener b : listeners) {
180             b.onComplete(broadcaster);
181         }
182     }
183 
184     /**
185      * {@inheritDoc}
186      */
187     @Override
188     public E get(long l, TimeUnit tu) throws InterruptedException, ExecutionException, TimeoutException {
189 
190         if (innerFuture != null) {
191             return (E) innerFuture.get();
192         }
193 
194         latch.await(l, tu);
195         return msg;
196     }
197 }

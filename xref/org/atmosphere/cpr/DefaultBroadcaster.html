87  /**
88   * {@link Broadcaster} implementation.
89   * <p/>
90   * Broadcast messages to suspended response using the caller's Thread.
91   * This basic {@link Broadcaster} use an {@link java.util.concurrent.ExecutorService}
92   * to broadcast message, hence the broadcast operation is asynchronous. Make sure
93   * you block on {@link #broadcast(Object)}.get()} if you need synchronous operations.
94   *
95   * @author Jeanfrancois Arcand
96   */

171     public synchronized void destroy() {
172         if (destroyed.getAndSet(true)) return;
173 
174         notifyDestroyListener();
175 
176         try {
177             logger.trace("Broadcaster {} is being destroyed and cannot be re-used", getID());
178 
179             if (BroadcasterFactory.getDefault() != null) {
180                 BroadcasterFactory.getDefault().remove(this, this.getID());
181             }
182 
183             if (currentLifecycleTask != null) {
184                 currentLifecycleTask.cancel(true);
185             }
186             started.set(false);
187 
188             releaseExternalResources();
189             if (notifierFuture != null) {
190                 notifierFuture.cancel(true);
191             }
192 
193             if (asyncWriteFuture != null) {
194                 asyncWriteFuture.cancel(true);
195             }
196 
197             if (bc != null) {
198                 bc.destroy();
199             }
200 
201             if (broadcasterCache != null) {
202                 broadcasterCache.stop();
203             }
204             resources.clear();
205             broadcastOnResume.clear();
206             messages.clear();
207             asyncWriteQueue.clear();
208             delayedBroadcast.clear();
209         } catch (Throwable t) {
210             logger.error("Unexpected exception during Broadcaster destroy {}", getID(), t);
211         }
212     }

224     public void setScope(SCOPE scope) {
225         if (destroyed.get()) {
226             logger.debug(DESTROYED, getID(), "setScope");
227             return;
228         }
229 
230         this.scope = scope;
231         if (scope != SCOPE.REQUEST) {
232             return;
233         }
234 
235         logger.debug("Changing broadcaster scope for {}. This broadcaster will be destroyed.", getID());
236         synchronized (resources) {
237             try {
238                 // Next, we need to create a new broadcaster per resource.
239                 for (AtmosphereResource resource : resources) {
240                     Broadcaster b = BroadcasterFactory.getDefault()
241                             .get(getClass(), getClass().getSimpleName() + "/" + UUID.randomUUID());
242 
243                     if (DefaultBroadcaster.class.isAssignableFrom(this.getClass())) {
244                         BroadcasterCache cache = bc.getBroadcasterCache().getClass().newInstance();
245                         InjectorProvider.getInjector().inject(cache);
246                         DefaultBroadcaster.class.cast(b).broadcasterCache = cache;
247                         DefaultBroadcaster.class.cast(b).getBroadcasterConfig().setBroadcasterCache(cache);
248                     }
249                     resource.setBroadcaster(b);
250                     b.setScope(SCOPE.REQUEST);
251                     if (resource.getAtmosphereResourceEvent().isSuspended()) {
252                         b.addAtmosphereResource(resource);
253                     }
254                     logger.debug("Resource {} not using broadcaster {}", resource, b.getID());
255                 }
256 
257                 // Do not destroy because this is a new Broadcaster
258                 if (resources.isEmpty()) {
259                     return;
260                 }
261 
262                 destroy();
263             } catch (Exception e) {
264                 logger.error("Failed to set request scope for current resources", e);
265             }
266         }
267     }

279     public synchronized void setID(String id) {
280         if (id == null) {
281             id = getClass().getSimpleName() + "/" + UUID.randomUUID();
282         }
283 
284         if (BroadcasterFactory.getDefault() == null)
285             return; // we are shutdown or destroyed, but someone still reference
286 
287         Broadcaster b = BroadcasterFactory.getDefault().lookup(this.getClass(), id);
288         if (b != null && b.getScope() == SCOPE.REQUEST) {
289             throw new IllegalStateException("Broadcaster ID already assigned to SCOPE.REQUEST. Cannot change the id");
290         } else if (b != null) {
291             return;
292         }
293 
294         BroadcasterFactory.getDefault().remove(this, name);
295         this.name = id;
296         BroadcasterFactory.getDefault().add(this, name);
297     }

331     @Override
332     public void setBroadcasterLifeCyclePolicy(final BroadcasterLifeCyclePolicy lifeCyclePolicy) {
333         this.lifeCyclePolicy = lifeCyclePolicy;
334         if (currentLifecycleTask != null) {
335             currentLifecycleTask.cancel(false);
336         }
337 
338         if (bc != null && bc.getScheduledExecutorService() == null) {
339             logger.error("No Broadcaster's SchedulerExecutorService has been configured on {}. BroadcasterLifeCyclePolicy won't work.", getID());
340             return;
341         }
342 
343         if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE
344                 || lifeCyclePolicy.getLifeCyclePolicy() == IDLE_RESUME
345                 || lifeCyclePolicy.getLifeCyclePolicy() == IDLE_DESTROY) {
346 
347             recentActivity.set(false);
348 
349             int time = lifeCyclePolicy.getTimeout();
350             if (time == -1) {
351                 throw new IllegalStateException("BroadcasterLifeCyclePolicy time is not set");
352             }
353 
354             final AtomicReference<Future<?>> ref = new AtomicReference<Future<?>>();
355             currentLifecycleTask = bc.getScheduledExecutorService().scheduleAtFixedRate(new Runnable() {
356 
357                 @Override
358                 public void run() {
359                     try {
360 
361                         // Check for activity since the last execution.
362                         if (recentActivity.getAndSet(false)) {
363                             return;
364                         } else if (resources.isEmpty()) {
365                             if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE) {
366                                 notifyEmptyListener();
367                                 notifyIdleListener();
368 
369                                 releaseExternalResources();
370                                 logger.debug("Applying BroadcasterLifeCyclePolicy IDLE policy to Broadcaster {}", getID());
371                             } else if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE_DESTROY) {
372                                 notifyEmptyListener();
373                                 notifyIdleListener();
374 
375                                 destroy(false);
376                                 logger.debug("Applying BroadcasterLifeCyclePolicy IDLE_DESTROY policy to Broadcaster {}", getID());
377                             }
378                         } else if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE_RESUME) {
379                             notifyIdleListener();
380 
381                             destroy(true);
382                             logger.debug("Applying BroadcasterLifeCyclePolicy IDLE_RESUME policy to Broadcaster {}", getID());
383                         }
384                     } catch (Throwable t) {
385                         if (destroyed.get()) {
386                             logger.trace("Scheduled BroadcasterLifeCyclePolicy exception", t);
387                         } else {
388                             logger.warn("Scheduled BroadcasterLifeCyclePolicy exception", t);
389                         }
390                     }
391                 }
392 
393                 void destroy(boolean resume) {
394 
395                     if (resume) {
396                         logger.info("All AtmosphereResource will now be resumed from Broadcaster {}", getID());
397                         resumeAll();
398                     }
399 
400                     DefaultBroadcaster.this.destroy();
401                     /**
402                      * The value may be null if the timeout is too low. Hopefully next execution will
403                      * cancel the task properly.
404                      */
405                     if (ref.get() != null) {
406                         currentLifecycleTask.cancel(true);
407                     }
408                 }
409 
410             }, time, time, lifeCyclePolicy.getTimeUnit());
411             ref.set(currentLifecycleTask);
412         }
413     }

442     @Override
443     public <T> Future<T> awaitAndBroadcast(T t, long time, TimeUnit timeUnit) {
444         if (resources.isEmpty()) {
445             synchronized (awaitBarrier) {
446                 try {
447                     logger.trace("Awaiting for AtmosphereResource for {} {}", time, timeUnit);
448                     awaitBarrier.wait(translateTimeUnit(time, timeUnit));
449                 } catch (Throwable e) {
450                     logger.warn("awaitAndBroadcast", e);
451                     return null;
452                 }
453             }
454         }
455         return broadcast(t);
456     }

492     protected Runnable getBroadcastHandler() {
493         return new Runnable() {
494             public void run() {
495                 Entry msg = null;
496                 while (started.get()) {
497                     try {
498                         msg = messages.poll(10, TimeUnit.SECONDS);
499                         if (msg == null) {
500                             if (destroyed.get()) {
501                                 return;
502                             } else {
503                                 continue;
504                             }
505                         }
506                         push(msg);
507                     } catch (InterruptedException ex) {
508                         return;
509                     } catch (Throwable ex) {
510                         if (!started.get() || destroyed.get()) {
511                             logger.trace("Failed to submit broadcast handler runnable on shutdown for Broadcaster {}", getID(), ex);
512                             return;
513                         } else {
514                             logger.warn("This message {} will be lost", msg);
515                             logger.debug("Failed to submit broadcast handler runnable to for Broadcaster {}", getID(), ex);
516                         }
517                     }
518                 }
519             }
520         };
521     }
522 
523     protected void start() {
524         if (!started.getAndSet(true)) {
525             broadcasterCache = bc.getBroadcasterCache();
526             broadcasterCache.start();
527 
528             setID(name);
529             notifierFuture = bc.getExecutorService().submit(getBroadcastHandler());
530             asyncWriteFuture = bc.getAsyncWriteService().submit(getAsyncWriteHandler());
531         }
532     }

534     protected void push(Entry entry) {
535         push(entry, true);
536     }
537 
538     protected void push(Entry entry, boolean rec) {
539 
540         if (destroyed.get()) {
541             return;
542         }
543         // We need to synchronize t make sure there is no suspend operation retrieving cached messages concurrently.
544         // https://github.com/Atmosphere/atmosphere/issues/170
545         synchronized (concurrentSuspendBroadcast) {
546             recentActivity.set(true);
547 
548             String prevMessage = entry.message.toString();
549             if (rec && !delayedBroadcast.isEmpty()) {
550                 Iterator<Entry> i = delayedBroadcast.iterator();
551                 StringBuilder b = new StringBuilder();
552                 while (i.hasNext()) {
553                     Entry e = i.next();
554                     e.future.cancel(true);
555                     try {
556                         // Append so we do a single flush
557                         if (e.message instanceof String
558                                 && entry.message instanceof String) {
559                             b.append(e.message);
560                         } else {
561                             push(e, false);
562                         }
563                     } finally {
564                         i.remove();
565                     }
566                 }
567 
568                 if (b.length() > 0) {
569                     entry.message = b.append(entry.message).toString();
570                 }
571             }
572 
573             Object finalMsg = translate(entry.message);
574 
575             if (finalMsg == null) {
576                 logger.trace("Broascast message was null {}", finalMsg);
577                 return;
578             }
579 
580             Object prevM = entry.originalMessage;
581             entry.originalMessage = (entry.originalMessage != entry.message ? translate(entry.originalMessage) : finalMsg);
582 
583             if (entry.originalMessage == null) {
584                 logger.trace("Broascast message was null {}", prevM);
585                 return;
586             }
587 
588             entry.message = finalMsg;
589 
590             if (resources.isEmpty()) {
591                 logger.debug("Broadcaster {} doesn't have any associated resource. Message will be cached in the configured BroadcasterCache", getID());
592 
593                 AtmosphereResource r = null;
594                 if (entry.multipleAtmoResources != null && AtmosphereResource.class.isAssignableFrom(entry.multipleAtmoResources.getClass())) {
595                     r = AtmosphereResource.class.cast(entry.multipleAtmoResources);
596                 }
597                 trackBroadcastMessage(r, cacheStrategy == BroadcasterCache.STRATEGY.AFTER_FILTER ? entry.message : entry.originalMessage);
598 
599                 if (entry.future != null) {
600                     entry.future.done();
601                 }
602                 return;
603             }
604 
605             try {
606                 if (entry.multipleAtmoResources == null) {
607                     for (AtmosphereResource r : resources) {
608                         finalMsg = perRequestFilter(r, entry);
609 
610                         if (finalMsg == null) {
611                             logger.debug("Skipping broadcast delivery resource {} ", r);
612                             continue;
613                         }
614 
615                         if (entry.writeLocally) {
616                             queueWriteIO(r, finalMsg, entry);
617                         }
618                     }
619                 } else if (entry.multipleAtmoResources instanceof AtmosphereResource) {
620                     finalMsg = perRequestFilter((AtmosphereResource) entry.multipleAtmoResources, entry);
621 
622                     if (finalMsg == null) {
623                         logger.debug("Skipping broadcast delivery resource {} ", entry.multipleAtmoResources);
624                         return;
625                     }
626 
627                     if (entry.writeLocally) {
628                         queueWriteIO((AtmosphereResource) entry.multipleAtmoResources, finalMsg, entry);
629                     }
630                 } else if (entry.multipleAtmoResources instanceof Set) {
631                     Set<AtmosphereResource> sub = (Set<AtmosphereResource>) entry.multipleAtmoResources;
632                     for (AtmosphereResource r : sub) {
633                         finalMsg = perRequestFilter(r, entry);
634 
635                         if (finalMsg == null) {
636                             logger.debug("Skipping broadcast delivery resource {} ", r);
637                             continue;
638                         }
639 
640                         if (entry.writeLocally) {
641                             queueWriteIO(r, finalMsg, entry);
642                         }
643                     }
644                 }
645                 entry.message = prevMessage;
646             } catch (InterruptedException ex) {
647                 logger.debug(ex.getMessage(), ex);
648             }
649         }
650     }
651 
652     protected void queueWriteIO(AtmosphereResource r, Object finalMsg, Entry entry) throws InterruptedException {
653         asyncWriteQueue.put(new AsyncWriteToken(r, finalMsg, entry.future, entry.originalMessage));
654     }
655 
656     protected Object perRequestFilter(AtmosphereResource r, Entry msg) {
657         Object finalMsg = msg.message;
658 
659         if (AtmosphereResourceImpl.class.isAssignableFrom(r.getClass())) {
660             synchronized (r) {
661                 if (isAtmosphereResourceValid(r)) {
662                     if (bc.hasPerRequestFilters()) {
663                         BroadcastAction a = bc.filter(r, msg.message, msg.originalMessage);
664                         if (a.action() == BroadcastAction.ACTION.ABORT) {
665                             return null;
666                         }
667                         if (a.message() != msg.originalMessage) {
668                             finalMsg = a.message();
669                         }
670                     }
671                 } else {
672                     // The resource is no longer valid.
673                     removeAtmosphereResource(r);
674                     BroadcasterFactory.getDefault().removeAllAtmosphereResource(r);
675                 }
676 
677                 if (cacheStrategy == BroadcasterCache.STRATEGY.AFTER_FILTER) {
678                     trackBroadcastMessage(r, finalMsg);
679                 }
680             }
681         }
682         return finalMsg;
683     }

685     private Object translate(Object msg) {
686         if (Callable.class.isAssignableFrom(msg.getClass())) {
687             try {
688                 return Callable.class.cast(msg).call();
689             } catch (Exception e) {
690                 logger.warn("Callable exception", e);
691                 return null;
692             }
693         }
694         return msg;
695     }

697     protected void executeAsyncWrite(final AsyncWriteToken token) {
698         boolean notifyListeners = true;
699         boolean lostCandidate = false;
700 
701         final AtmosphereResourceEventImpl event = (AtmosphereResourceEventImpl) token.resource.getAtmosphereResourceEvent();
702         final AtmosphereResourceImpl r = AtmosphereResourceImpl.class.cast(token.resource);
703         try {
704             event.setMessage(token.msg);
705 
706             // Make sure we cache the message in case the AtmosphereResource has been cancelled, resumed or the client disconnected.
707             if (!isAtmosphereResourceValid(r)) {
708                 removeAtmosphereResource(r);

715             } catch (Throwable t) {

717                         " closed and will be added to the configured BroadcasterCache for later retrieval", event);

721                 removeAtmosphereResource(r);

723                 event.setCancelled(true);
724                 event.setThrowable(t);
725                 r.setIsInScope(false);
726                 lostCandidate = true;
727                 return;

731             broadcast(r, event);

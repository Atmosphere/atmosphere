89  /**
90   * {@link Broadcaster} implementation.
91   * <p/>
92   * Broadcast messages to suspended response using the caller's Thread.
93   * This basic {@link Broadcaster} use an {@link java.util.concurrent.ExecutorService}
94   * to broadcast message, hence the broadcast operation is asynchronous. Make sure
95   * you block on {@link #broadcast(Object)}.get()} if you need synchronous operations.
96   *
97   * @author Jeanfrancois Arcand
98   */

171     /**
172      * {@inheritDoc}
173      */
174     public synchronized void destroy() {
175         if (destroyed.getAndSet(true)) return;
176 
177         notifyOnPreDestroy();
178         notifyDestroyListener();
179 
180         try {
181             logger.trace("Broadcaster {} is being destroyed and cannot be re-used", getID());
182 
183             if (BroadcasterFactory.getDefault() != null) {
184                 BroadcasterFactory.getDefault().remove(this, this.getID());
185             }
186 
187             if (currentLifecycleTask != null) {
188                 currentLifecycleTask.cancel(true);
189             }
190             started.set(false);
191 
192             releaseExternalResources();
193             if (notifierFuture != null) {
194                 notifierFuture.cancel(true);
195             }
196 
197             if (asyncWriteFuture != null) {
198                 asyncWriteFuture.cancel(true);
199             }
200 
201             if (bc != null) {
202                 bc.destroy();
203             }
204 
205             if (broadcasterCache != null) {
206                 broadcasterCache.stop();
207             }
208             resources.clear();
209             broadcastOnResume.clear();
210             messages.clear();
211             asyncWriteQueue.clear();
212             delayedBroadcast.clear();
213         } catch (Throwable t) {
214             logger.error("Unexpected exception during Broadcaster destroy {}", getID(), t);
215         }
216     }

225     /**
226      * {@inheritDoc}
227      */
228     public void setScope(SCOPE scope) {
229         if (destroyed.get()) {
230             logger.debug(DESTROYED, getID(), "setScope");
231             return;
232         }
233 
234         this.scope = scope;
235         if (scope != SCOPE.REQUEST) {
236             return;
237         }
238 
239         logger.debug("Changing broadcaster scope for {}. This broadcaster will be destroyed.", getID());
240         synchronized (resources) {
241             try {
242                 // Next, we need to create a new broadcaster per resource.
243                 for (AtmosphereResource resource : resources) {
244                     Broadcaster b = BroadcasterFactory.getDefault()
245                             .get(getClass(), getClass().getSimpleName() + "/" + UUID.randomUUID());
246 
247                     if (DefaultBroadcaster.class.isAssignableFrom(this.getClass())) {
248                         BroadcasterCache cache = bc.getBroadcasterCache().getClass().newInstance();
249                         InjectorProvider.getInjector().inject(cache);
250                         DefaultBroadcaster.class.cast(b).broadcasterCache = cache;
251                         DefaultBroadcaster.class.cast(b).getBroadcasterConfig().setBroadcasterCache(cache);
252                     }
253                     resource.setBroadcaster(b);
254                     b.setScope(SCOPE.REQUEST);
255                     if (resource.getAtmosphereResourceEvent().isSuspended()) {
256                         b.addAtmosphereResource(resource);
257                     }
258                     logger.debug("Resource {} not using broadcaster {}", resource, b.getID());
259                 }
260 
261                 // Do not destroy because this is a new Broadcaster
262                 if (resources.isEmpty()) {
263                     return;
264                 }
265 
266                 destroy();
267             } catch (Exception e) {
268                 logger.error("Failed to set request scope for current resources", e);
269             }
270         }
271     }

283     public synchronized void setID(String id) {
284         if (id == null) {
285             id = getClass().getSimpleName() + "/" + UUID.randomUUID();
286         }
287 
288         if (BroadcasterFactory.getDefault() == null)
289             return; // we are shutdown or destroyed, but someone still reference
290 
291         Broadcaster b = BroadcasterFactory.getDefault().lookup(this.getClass(), id);
292         if (b != null && b.getScope() == SCOPE.REQUEST) {
293             throw new IllegalStateException("Broadcaster ID already assigned to SCOPE.REQUEST. Cannot change the id");
294         } else if (b != null) {
295             return;
296         }
297 
298         BroadcasterFactory.getDefault().remove(this, name);
299         this.name = id;
300         BroadcasterFactory.getDefault().add(this, name);
301     }

315             for (AtmosphereResource r : resources) {

335     @Override
336     public void setBroadcasterLifeCyclePolicy(final BroadcasterLifeCyclePolicy lifeCyclePolicy) {
337         this.lifeCyclePolicy = lifeCyclePolicy;
338         if (currentLifecycleTask != null) {
339             currentLifecycleTask.cancel(false);
340         }
341 
342         if (bc != null && bc.getScheduledExecutorService() == null) {
343             logger.error("No Broadcaster's SchedulerExecutorService has been configured on {}. BroadcasterLifeCyclePolicy won't work.", getID());
344             return;
345         }
346 
347         if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE
348                 || lifeCyclePolicy.getLifeCyclePolicy() == IDLE_RESUME
349                 || lifeCyclePolicy.getLifeCyclePolicy() == IDLE_DESTROY) {
350 
351             recentActivity.set(false);
352 
353             int time = lifeCyclePolicy.getTimeout();
354             if (time == -1) {
355                 throw new IllegalStateException("BroadcasterLifeCyclePolicy time is not set");
356             }
357 
358             final AtomicReference<Future<?>> ref = new AtomicReference<Future<?>>();
359             currentLifecycleTask = bc.getScheduledExecutorService().scheduleAtFixedRate(new Runnable() {
360 
361                 @Override
362                 public void run() {
363                     try {
364 
365                         // Check for activity since the last execution.
366                         if (recentActivity.getAndSet(false)) {
367                             return;
368                         } else if (resources.isEmpty()) {
369                             if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE) {
370                                 notifyEmptyListener();
371                                 notifyIdleListener();
372 
373                                 releaseExternalResources();
374                                 logger.debug("Applying BroadcasterLifeCyclePolicy IDLE policy to Broadcaster {}", getID());
375                             } else if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE_DESTROY) {
376                                 notifyEmptyListener();
377                                 notifyIdleListener();
378 
379                                 destroy(false);
380                                 logger.debug("Applying BroadcasterLifeCyclePolicy IDLE_DESTROY policy to Broadcaster {}", getID());
381                             }
382                         } else if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE_RESUME) {
383                             notifyIdleListener();
384 
385                             destroy(true);
386                             logger.debug("Applying BroadcasterLifeCyclePolicy IDLE_RESUME policy to Broadcaster {}", getID());
387                         }
388                     } catch (Throwable t) {
389                         if (destroyed.get()) {
390                             logger.trace("Scheduled BroadcasterLifeCyclePolicy exception", t);
391                         } else {
392                             logger.warn("Scheduled BroadcasterLifeCyclePolicy exception", t);
393                         }
394                     }
395                 }
396 
397                 void destroy(boolean resume) {
398 
399                     if (resume) {
400                         logger.info("All AtmosphereResource will now be resumed from Broadcaster {}", getID());
401                         resumeAll();
402                     }
403 
404                     DefaultBroadcaster.this.destroy();
405                     /**
406                      * The value may be null if the timeout is too low. Hopefully next execution will
407                      * cancel the task properly.
408                      */
409                     if (ref.get() != null) {
410                         currentLifecycleTask.cancel(true);
411                     }
412                 }
413 
414             }, time, time, lifeCyclePolicy.getTimeUnit());
415             ref.set(currentLifecycleTask);
416         }
417     }

446     @Override
447     public <T> Future<T> awaitAndBroadcast(T t, long time, TimeUnit timeUnit) {
448         if (resources.isEmpty()) {
449             synchronized (awaitBarrier) {
450                 try {
451                     logger.trace("Awaiting for AtmosphereResource for {} {}", time, timeUnit);
452                     awaitBarrier.wait(translateTimeUnit(time, timeUnit));
453                 } catch (Throwable e) {
454                     logger.warn("awaitAndBroadcast", e);
455                     return null;
456                 }
457             }
458         }
459         return broadcast(t);
460     }

514     protected Runnable getBroadcastHandler() {
515         return new Runnable() {
516             public void run() {
517                 Entry msg = null;
518                 while (started.get()) {
519                     try {
520                         msg = messages.poll(10, TimeUnit.SECONDS);
521                         if (msg == null) {
522                             if (!destroyed.get()) {
523                                 bc.getAsyncWriteService().submit(this);
524                                 return;
525                             }
526                         }
527                         push(msg);
528                     } catch (InterruptedException ex) {
529                         return;
530                     } catch (Throwable ex) {
531                         if (!started.get() || destroyed.get()) {
532                             logger.trace("Failed to submit broadcast handler runnable on shutdown for Broadcaster {}", getID(), ex);
533                             return;
534                         } else {
535                             logger.warn("This message {} will be lost", msg);
536                             logger.debug("Failed to submit broadcast handler runnable to for Broadcaster {}", getID(), ex);
537                         }
538                     }
539                 }
540             }
541         };
542     }
543 
544     protected void start() {
545         if (!started.getAndSet(true)) {
546             broadcasterCache = bc.getBroadcasterCache();
547             broadcasterCache.start();
548 
549             setID(name);
550             notifierFuture = bc.getExecutorService().submit(getBroadcastHandler());
551             asyncWriteFuture = bc.getAsyncWriteService().submit(getAsyncWriteHandler());
552         }
553     }

555     protected void push(Entry entry) {
556         push(entry, true);
557     }
558 
559     protected void push(Entry entry, boolean rec) {
560 
561         if (destroyed.get()) {
562             return;
563         }
564         // We need to synchronize t make sure there is no suspend operation retrieving cached messages concurrently.
565         // https://github.com/Atmosphere/atmosphere/issues/170
566         synchronized (concurrentSuspendBroadcast) {
567             recentActivity.set(true);
568 
569             String prevMessage = entry.message.toString();
570             if (rec && !delayedBroadcast.isEmpty()) {
571                 Iterator<Entry> i = delayedBroadcast.iterator();
572                 StringBuilder b = new StringBuilder();
573                 while (i.hasNext()) {
574                     Entry e = i.next();
575                     e.future.cancel(true);
576                     try {
577                         // Append so we do a single flush
578                         if (e.message instanceof String
579                                 && entry.message instanceof String) {
580                             b.append(e.message);
581                         } else {
582                             push(e, false);
583                         }
584                     } finally {
585                         i.remove();
586                     }
587                 }
588 
589                 if (b.length() > 0) {
590                     entry.message = b.append(entry.message).toString();
591                 }
592             }
593 
594             Object finalMsg = translate(entry.message);
595 
596             if (finalMsg == null) {
597                 logger.trace("Broascast message was null {}", finalMsg);
598                 return;
599             }
600 
601             Object prevM = entry.originalMessage;
602             entry.originalMessage = (entry.originalMessage != entry.message ? translate(entry.originalMessage) : finalMsg);
603 
604             if (entry.originalMessage == null) {
605                 logger.trace("Broascast message was null {}", prevM);
606                 return;
607             }
608 
609             entry.message = finalMsg;
610 
611             if (resources.isEmpty()) {
612                 logger.debug("Broadcaster {} doesn't have any associated resource. Message will be cached in the configured BroadcasterCache", getID());
613 
614                 AtmosphereResource r = null;
615                 if (entry.multipleAtmoResources != null && AtmosphereResource.class.isAssignableFrom(entry.multipleAtmoResources.getClass())) {
616                     r = AtmosphereResource.class.cast(entry.multipleAtmoResources);
617                 }
618                 trackBroadcastMessage(r, cacheStrategy == BroadcasterCache.STRATEGY.AFTER_FILTER ? entry.message : entry.originalMessage);
619 
620                 if (entry.future != null) {
621                     entry.future.done();
622                 }
623                 return;
624             }
625 
626             try {
627                 if (entry.multipleAtmoResources == null) {
628                     for (AtmosphereResource r : resources) {
629                         finalMsg = perRequestFilter(r, entry);
630 
631                         if (finalMsg == null) {
632                             logger.debug("Skipping broadcast delivery resource {} ", r);
633                             continue;
634                         }
635 
636                         if (entry.writeLocally) {
637                             queueWriteIO(r, finalMsg, entry);
638                         }
639                     }
640                 } else if (entry.multipleAtmoResources instanceof AtmosphereResource) {
641                     finalMsg = perRequestFilter((AtmosphereResource) entry.multipleAtmoResources, entry);
642 
643                     if (finalMsg == null) {
644                         logger.debug("Skipping broadcast delivery resource {} ", entry.multipleAtmoResources);
645                         return;
646                     }
647 
648                     if (entry.writeLocally) {
649                         queueWriteIO((AtmosphereResource) entry.multipleAtmoResources, finalMsg, entry);
650                     }
651                 } else if (entry.multipleAtmoResources instanceof Set) {
652                     Set<AtmosphereResource> sub = (Set<AtmosphereResource>) entry.multipleAtmoResources;
653 
654                     if (sub.size() != 0) {
655                         for (AtmosphereResource r : sub) {
656                             finalMsg = perRequestFilter(r, entry);
657 
658                             if (finalMsg == null) {
659                                 logger.debug("Skipping broadcast delivery resource {} ", r);
660                                 continue;
661                             }
662 
663                             if (entry.writeLocally) {
664                                 queueWriteIO(r, finalMsg, entry);
665                             }
666                         }
667                     } else {
668                         // See https://github.com/Atmosphere/atmosphere/issues/572
669                         if (cacheStrategy == BroadcasterCache.STRATEGY.AFTER_FILTER) {
670                             trackBroadcastMessage(null, finalMsg);
671                         }
672                     }
673                 }
674                 entry.message = prevMessage;
675             } catch (InterruptedException ex) {
676                 logger.debug(ex.getMessage(), ex);
677             }
678         }
679     }

681     protected void queueWriteIO(AtmosphereResource r, Object finalMsg, Entry entry) throws InterruptedException {
682         asyncWriteQueue.put(new AsyncWriteToken(r, finalMsg, entry.future, entry.originalMessage));
683     }
684 
685     protected Object perRequestFilter(AtmosphereResource r, Entry msg) {
686 
687         // A broadcaster#broadcast(msg,Set) may contains null value.
688         if (r == null) {
689             logger.trace("Null AtmosphereResource passed inside a Set");
690             return msg.message;
691         }
692 
693         Object finalMsg = msg.message;
694 
695         if (AtmosphereResourceImpl.class.isAssignableFrom(r.getClass())) {
696             synchronized (r) {
697                 if (isAtmosphereResourceValid(r)) {
698                     if (bc.hasPerRequestFilters()) {
699                         BroadcastAction a = bc.filter(r, msg.message, msg.originalMessage);
700                         if (a.action() == BroadcastAction.ACTION.ABORT) {
701                             return null;
702                         }
703                         if (a.message() != msg.originalMessage) {
704                             finalMsg = a.message();
705                         }
706                     }
707                 } else {
708                     // The resource is no longer valid.
709                     removeAtmosphereResource(r);

733     protected void executeAsyncWrite(final AsyncWriteToken token) {
734         boolean notifyListeners = true;
735         boolean lostCandidate = false;
736 
737         final AtmosphereResourceEventImpl event = (AtmosphereResourceEventImpl) token.resource.getAtmosphereResourceEvent();
738         final AtmosphereResourceImpl r = AtmosphereResourceImpl.class.cast(token.resource);
739         r.getRequest().setAttribute(getID(), token.future);
740         try {
741             event.setMessage(token.msg);
742 
743             // Make sure we cache the message in case the AtmosphereResource has been cancelled, resumed or the client disconnected.
744             if (!isAtmosphereResourceValid(r)) {
745                 removeAtmosphereResource(r);
746                 lostCandidate = true;
747                 return;
748             }
749 
750             try {
751                 r.getRequest().setAttribute(MAX_INACTIVE, System.currentTimeMillis());
752             } catch (Throwable t) {
753                 logger.warn("Invalid AtmosphereResource state {}. The connection has been remotely" +
754                         " closed and will be added to the configured BroadcasterCache for later retrieval", event);
755                 logger.trace("If you are using Tomcat 7.0.22 and lower, your most probably hitting http://is.gd/NqicFT");
756                 logger.trace("", t);
757                 // The Request/Response associated with the AtmosphereResource has already been written and commited
758                 removeAtmosphereResource(r, false);
759                 BroadcasterFactory.getDefault().removeAllAtmosphereResource(r);
760                 event.setCancelled(true);
761                 event.setThrowable(t);
762                 r.setIsInScope(false);
763                 lostCandidate = true;
764                 return;
765             }
766 
767             r.getRequest().setAttribute(ASYNC_TOKEN, token);
768             broadcast(r, event);
769         } finally {
770             if (notifyListeners) {
771                 r.notifyListeners();
772             }
773 
774             if (token.future != null) {
775                 token.future.done();
776             }
777 
778             if (lostCandidate) {
779                 cacheLostMessage(r, token);
780             }
781             token.destroy();
782         }

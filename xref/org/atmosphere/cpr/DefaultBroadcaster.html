66  /**
67   * {@link Broadcaster} implementation.
68   * <p/>
69   * Broadcast messages to suspended response using the caller's Thread.
70   * This basic {@link Broadcaster} use an {@link java.util.concurrent.ExecutorService}
71   * to broadcast message, hence the broadcast operation is asynchronous. Make sure
72   * you block on {@link #broadcast(Object)}.get()} if you need synchronous operations.
73   *
74   * @author Jeanfrancois Arcand
75   */

118     /**
119      * {@inheritDoc}
120      */
121     public synchronized void destroy() {
122 
123         if (destroyed.get()) return;
124 
125         started.set(false);
126         destroyed.set(true);
127         releaseExternalResources();
128         if (notifierFuture != null) {
129             notifierFuture.cancel(true);
130         }
131 
132         if (asyncWriteFuture != null) {
133             asyncWriteFuture.cancel(true);
134         }
135 
136         if (bc != null) {
137             bc.destroy();
138         }
139 
140         if (broadcasterCache != null) {
141             broadcasterCache.stop();
142         }
143         resources.clear();
144         broadcastOnResume.clear();
145         messages.clear();
146         asyncWriteQueue.clear();
147         delayedBroadcast.clear();
148         broadcasterCache = null;
149 
150         if (BroadcasterFactory.getDefault() != null) {
151             BroadcasterFactory.getDefault().remove(this, name);
152         }
153 
154         if (currentLifecycleTask != null) {
155             currentLifecycleTask.cancel(true);
156         }
157     }

169     public void setScope(SCOPE scope) {
170         this.scope = scope;
171         if (scope != SCOPE.REQUEST) {
172             return;
173         }
174 
175         try {
176             for (AtmosphereResource<?, ?> resource : resources) {
177                 Broadcaster b = BroadcasterFactory.getDefault()
178                         .get(getClass(), getClass().getSimpleName() + "/" + UUID.randomUUID());
179 
180                 if (DefaultBroadcaster.class.isAssignableFrom(this.getClass())) {
181                     BroadcasterCache cache = bc.getBroadcasterCache().getClass().newInstance();
182                     InjectorProvider.getInjector().inject(cache);
183                     DefaultBroadcaster.class.cast(b).broadcasterCache = cache;
184                 }
185                 resource.setBroadcaster(b);
186                 if (resource.getAtmosphereResourceEvent().isSuspended()) {
187                     b.addAtmosphereResource(resource);
188                 }
189             }
190 
191             if (!resources.isEmpty()) {
192                 destroy();
193             }
194         } catch (Exception e) {
195             logger.error("failed to set request scope for current resources", e);
196         }
197     }

210         if (id == null) {
211             id = getClass().getSimpleName() + "/" + UUID.randomUUID();
212         }
213 
214         Broadcaster b = BroadcasterFactory.getDefault().lookup(this.getClass(), id);
215         if (b != null && b.getScope() == SCOPE.REQUEST) {
216             throw new IllegalStateException("Broadcaster ID already assigned to SCOPE.REQUEST. Cannot change the id");
217         }
218 
219         BroadcasterFactory.getDefault().remove(this, name);
220         this.name = id;
221         BroadcasterFactory.getDefault().add(this, name);
222     }

250     @Override
251     public void setBroadcasterLifeCyclePolicy(final BroadcasterLifeCyclePolicy lifeCyclePolicy) {
252         this.lifeCyclePolicy = lifeCyclePolicy;
253         if (currentLifecycleTask != null) {
254             currentLifecycleTask.cancel(false);
255         }
256 
257         if (lifeCyclePolicy.getLifeCyclePolicy() == BroadcasterLifeCyclePolicy.ATMOSPHERE_RESOURCE_POLICY.IDLE
258                 || lifeCyclePolicy.getLifeCyclePolicy() == BroadcasterLifeCyclePolicy.ATMOSPHERE_RESOURCE_POLICY.IDLE_DESTROY) {
259 
260             int time = lifeCyclePolicy.getTimeout();
261             if (time == -1) {
262                 throw new IllegalStateException("BroadcasterLifeCyclePolicy time is not set");
263             }
264 
265             final AtomicReference<Future<?>> ref = new AtomicReference<Future<?>>();
266             currentLifecycleTask = bc.getScheduledExecutorService().scheduleAtFixedRate(new Runnable() {
267 
268                 @Override
269                 public void run() {
270                     try {
271                         if (resources.isEmpty()) {
272                             notifyEmptyListener();
273                             notifyIdleListener();
274 
275                             if (lifeCyclePolicy.getLifeCyclePolicy() == BroadcasterLifeCyclePolicy.ATMOSPHERE_RESOURCE_POLICY.IDLE) {
276                                 releaseExternalResources();
277                                 logger.debug("Applying BroadcasterLifeCyclePolicy IDLE policy");
278                             } else {
279                                 notifyDestroyListener();
280 
281                                 destroy();
282                                 /**
283                                  * The value may be null if the timeout is too low. Hopefully next execution will
284                                  * cancel the task properly.
285                                  */
286                                 if (ref.get() != null) {
287                                     currentLifecycleTask.cancel(true);
288                                 }
289 
290                                 logger.debug("Applying BroadcasterLifeCyclePolicy IDLE_DESTROY policy");
291                             }
292                         }
293                     } catch (Throwable t) {
294                         logger.warn("Scheduled BroadcasterLifeCyclePolicy exception", t);
295                     }
296                 }
297 
298             }, time, time, lifeCyclePolicy.getTimeUnit());
299             ref.set(currentLifecycleTask);
300         }
301     }

313     public class Entry {
314 
315         public Object message;
316         public Object multipleAtmoResources;
317         public BroadcasterFuture<?> future;
318         public boolean writeLocally;
319         public Object originalMessage;
320 
321         public Entry(Object message, Object multipleAtmoResources, BroadcasterFuture<?> future, Object originalMessage) {
322             this.message = message;
323             this.multipleAtmoResources = multipleAtmoResources;
324             this.future = future;
325             this.writeLocally = true;
326             this.originalMessage = originalMessage;
327         }
328 
329         public Entry(Object message, Object multipleAtmoResources, BroadcasterFuture<?> future, boolean writeLocally) {
330             this.message = message;
331             this.multipleAtmoResources = multipleAtmoResources;
332             this.future = future;
333             this.writeLocally = writeLocally;
334         }
335 
336         @Override
337         public String toString() {
338             return "Entry{" +
339                     "message=" + message +
340                     ", multipleAtmoResources=" + multipleAtmoResources +
341                     ", future=" + future +
342                     '}';
343         }

346     protected Runnable getBroadcastHandler() {
347         return new Runnable() {
348             public void run() {
349                 Entry msg;
350                 while (started.get()) {
351                     try {
352                         msg = messages.take();
353                         push(msg);
354                     } catch (Throwable ex) {
355                         if (!started.get()) {
356                             logger.trace("failed to submit broadcast handler runnable to broadcast executor service on shutdown", ex);
357                         } else {
358                             logger.debug("failed to submit broadcast handler runnable to broadcast executor service", ex);
359                         }
360                     }
361                 }
362             }
363         };
364     }
365 
366     protected void start() {
367         if (!started.getAndSet(true)) {
368 
369             broadcasterCache = bc.getBroadcasterCache();
370             broadcasterCache.start();
371 
372             notifierFuture = bc.getExecutorService().submit(getBroadcastHandler());
373             asyncWriteFuture = bc.getAsyncWriteService().submit(getAsyncWriteHandler());
374         }
375     }
376 
377     protected void push(Entry entry) {
378 
379         if (destroyed.get()) {
380             return;
381         }
382 
383         String prevMessage = entry.message.toString();
384         if (!delayedBroadcast.isEmpty()) {
385             Iterator<Entry> i = delayedBroadcast.iterator();
386             StringBuilder b = new StringBuilder();
387             while (i.hasNext()) {
388                 Entry e = i.next();
389                 e.future.cancel(true);
390                 try {
391                     // Append so we do a single flush
392                     if (e.message instanceof String
393                             && entry.message instanceof String) {
394                         b.append(e.message);
395                     } else {
396                         push(e);
397                     }
398                 } finally {
399                     i.remove();
400                 }
401             }
402 
403             if (b.length() > 0) {
404                 entry.message = b.append(entry.message).toString();
405             }
406         }
407 
408         if (resources.isEmpty()) {
409             trackBroadcastMessage(null, entry.message);
410             if (entry.future != null) {
411                 entry.future.done();
412             }
413             return;
414         }
415 
416         Object finalMsg = translate(entry.message);
417         entry.message = finalMsg;
418         try {
419             if (entry.multipleAtmoResources == null) {
420                 for (AtmosphereResource<?, ?> r : resources) {
421                     finalMsg = perRequestFilter(r, entry);
422                     if (entry.writeLocally) {
423                         queueWriteIO(r, finalMsg, entry);
424                     }
425                 }
426             } else if (entry.multipleAtmoResources instanceof AtmosphereResource<?, ?>) {
427                 finalMsg = perRequestFilter((AtmosphereResource<?, ?>) entry.multipleAtmoResources, entry);
428 
429                 if (entry.writeLocally) {
430                     queueWriteIO((AtmosphereResource<?, ?>) entry.multipleAtmoResources, finalMsg, entry);
431                 }
432             } else if (entry.multipleAtmoResources instanceof Set) {
433                 Set<AtmosphereResource<?, ?>> sub = (Set<AtmosphereResource<?, ?>>) entry.multipleAtmoResources;
434                 for (AtmosphereResource<?, ?> r : sub) {
435                     finalMsg = perRequestFilter(r, entry);
436                     if (entry.writeLocally) {
437                         queueWriteIO(r, finalMsg, entry);
438                     }
439                 }
440             }
441             entry.message = prevMessage;
442         } catch (InterruptedException ex) {
443             logger.debug(ex.getMessage(), ex);
444         }
445     }
446 
447     protected void queueWriteIO(AtmosphereResource<?, ?> r, Object finalMsg, Entry entry) throws InterruptedException {
448         asyncWriteQueue.put(new AsyncWriteToken(r, finalMsg, entry.future));
449     }
450 
451     protected Object perRequestFilter(AtmosphereResource<?, ?> r, Entry msg) {
452         Object finalMsg = msg.message;
453 
454         if (AtmosphereResourceImpl.class.isAssignableFrom(r.getClass())) {
455             if (AtmosphereResourceImpl.class.cast(r).isInScope()) {
456                 if (r.getRequest() instanceof HttpServletRequest && bc.hasPerRequestFilters()) {
457                     Object message = msg.originalMessage;
458                     BroadcastAction a = bc.filter((HttpServletRequest) r.getRequest(), (HttpServletResponse) r.getResponse(), message);
459                     if (a.action() == BroadcastAction.ACTION.ABORT
460                             || a.message() != msg.originalMessage) {
461                         finalMsg = a.message();
462                     }
463                 }
464                 trackBroadcastMessage(r, finalMsg);
465             } else {
466                 // The resource is no longer valid.
467                 removeAtmosphereResource(r);
468                 BroadcasterFactory.getDefault().removeAllAtmosphereResource(r);
469             }
470 
471         }
472         return finalMsg;
473     }
474 
475     private Object translate(Object msg) {
476         if (Callable.class.isAssignableFrom(msg.getClass())) {
477             try {
478                 return Callable.class.cast(msg).call();
479             } catch (Exception e) {
480                 logger.error("failed to cast message: " + msg, e);
481             }
482         }
483         return msg;
484     }
485 
486     protected void executeAsyncWrite(final AtmosphereResource<?, ?> resource, final Object msg, final BroadcasterFuture future) {
487         if (resource.getAtmosphereResourceEvent().isCancelled()) {
488             return;
489         }
490 
491         final AtmosphereResourceEvent event = resource.getAtmosphereResourceEvent();
492         event.setMessage(msg);
493 
494         try {
495             if (resource.getAtmosphereResourceEvent() != null && !resource.getAtmosphereResourceEvent().isCancelled()
496                     && HttpServletRequest.class.isAssignableFrom(resource.getRequest().getClass())) {
497                 HttpServletRequest.class.cast(resource.getRequest())
498                         .setAttribute(CometSupport.MAX_INACTIVE, System.currentTimeMillis());
499             }
500         } catch (Exception t) {
501             // Shield us from any corrupted Request
502             logger.debug("Preventing corruption of a recycled request: resource" + resource, event);
503             removeAtmosphereResource(resource);
504             BroadcasterFactory.getDefault().removeAllAtmosphereResource(resource);
505             return;
506         }
507 
508         broadcast(resource, event);
509         if (resource instanceof AtmosphereEventLifecycle) {
510             ((AtmosphereEventLifecycle) resource).notifyListeners();
511         }
512 
513         if (future != null) {
514             future.done();
515         }
516     }
517 
518     protected Runnable getAsyncWriteHandler() {
519         return new Runnable() {
520             public void run() {
521                 AsyncWriteToken token;
522                 try {
523                     token = asyncWriteQueue.take();
524                     // Leader/follower
525                     synchronized (token.resource) {
526                         // If the resource is no longer in scope, skip the processing.
527                         if (AtmosphereResourceImpl.class.cast(token.resource).isInScope()) {
528                             bc.getAsyncWriteService().submit(this);
529                             executeAsyncWrite(token.resource, token.msg, token.future);
530                         }
531                     }
532                 } catch (Throwable ex) {
533                     if (!started.get()) {
534                         logger.trace("failed to submit async write task on shutdown", ex);
535                     } else {
536                         logger.debug("failed to submit async write task", ex);
537                     }
538                 }
539             }
540         };
541     }

543     protected void checkCachedAndPush(final AtmosphereResource<?, ?> r, final AtmosphereResourceEvent e) {
544         retrieveTrackedBroadcast(r, e);
545         if (e.getMessage() instanceof List && !((List) e.getMessage()).isEmpty()) {
546             HttpServletRequest.class.cast(r.getRequest()).setAttribute(CACHED, "true");
547             broadcast(r, e);
548         }
549     }
550 
551     protected boolean retrieveTrackedBroadcast(final AtmosphereResource<?, ?> r, final AtmosphereResourceEvent e) {
552         List<?> missedMsg = broadcasterCache.retrieveFromCache(r);
553         if (!missedMsg.isEmpty()) {
554             e.setMessage(missedMsg);
555             return true;
556         }
557         return false;
558     }
559 
560     protected void trackBroadcastMessage(final AtmosphereResource<?, ?> r, Object msg) {
561         broadcasterCache.addToCache(r, msg);
562     }
563 
564     protected void broadcast(final AtmosphereResource<?, ?> r, final AtmosphereResourceEvent e) {
565         try {
566             r.getAtmosphereHandler().onStateChange(e);
567         } catch (IOException ex) {
568             if (AtmosphereResourceImpl.class.isAssignableFrom(r.getClass())) {
569                 AtmosphereResourceImpl.class.cast(r).notifyListeners(e);
570             }
571             onException(ex, r);
572         } catch (RuntimeException ex) {
573             if (AtmosphereResourceImpl.class.isAssignableFrom(r.getClass())) {
574                 AtmosphereResourceImpl.class.cast(r).notifyListeners(e);
575             }
576             onException(ex, r);
577         }
578     }
579 
580     protected void onException(Throwable t, final AtmosphereResource<?, ?> r) {
581         logger.debug("onException()", t);
582 
583         if (r instanceof AtmosphereEventLifecycle) {
584             ((AtmosphereEventLifecycle) r)
585                     .notifyListeners(new AtmosphereResourceEventImpl((AtmosphereResourceImpl) r, true, false, t));
586             ((AtmosphereEventLifecycle) r).removeEventListeners();
587         }
588 
589         /**
590          * Make sure we resume the connection on every IOException.
591          */
592         bc.getAsyncWriteService().execute(new Runnable() {
593             @Override
594             public void run() {
595                 r.resume();
596             }
597         });
598 
599     }

610     @Override
611     public <T> Future<T> broadcast(T msg) {
612 
613         if (destroyed.get()) {
614             logger.error("This Broadcaster has been destroyed and cannot be used");
615             return null;
616         }
617 
618         start();
619         Object newMsg = filter(msg);
620         if (newMsg == null) return null;
621 
622         BroadcasterFuture<Object> f = new BroadcasterFuture<Object>(newMsg);
623         messages.offer(new Entry(newMsg, null, f, msg));
624         return f;
625     }

629      *
630      * @param msg
631      * @return
632      */
633     protected Object filter(Object msg) {
634         BroadcastAction a = bc.filter(msg);
635         if (a.action() == BroadcastAction.ACTION.ABORT || msg == null)
636             return null;
637         else
638             return a.message();
639     }

681     protected void broadcastOnResume(AtmosphereResource<?, ?> r) {
682         Iterator<Entry> i = broadcastOnResume.iterator();
683         while (i.hasNext()) {
684             Entry e = i.next();
685             e.multipleAtmoResources = r;
686             push(e);
687         }
688

690             broadcastOnResume.clear();
691         }

714     /**
715      * {@inheritDoc}
716      */
717     @Override
718     public AtmosphereResource<?, ?> addAtmosphereResource(AtmosphereResource<?, ?> r) {
719 
720         if (destroyed.get()) {
721             logger.error("This Broadcaster has been destroyed and cannot be used");
722             return r;
723         }
724 
725         if (scope == SCOPE.REQUEST && requestScoped.getAndSet(true)) {
726             throw new IllegalStateException("Broadcaster " + this
727                     + " cannot be used as its scope is set to REQUEST");
728         }
729 
730         // To avoid excessive synchronization, we allow resources.size() to get larger that maxSuspendResource
731         if (maxSuspendResource.get() > 0 && resources.size() >= maxSuspendResource.get()) {
732             // Resume the first in.
733             if (policy == POLICY.FIFO) {
734                 // TODO handle null return from poll()
735                 AtmosphereResource<?, ?> resource = resources.poll();
736                 try {
737                     logger.warn("Too many resource. Forcing resume of {} ", resource);
738                     resource.resume();
739                 } catch (Throwable t) {
740                     logger.warn("failed to resume resource {} ", resource, t);
741                 }
742             } else if (policy == POLICY.REJECT) {
743                 throw new RejectedExecutionException(String.format("Maximum suspended AtmosphereResources %s", maxSuspendResource));
744             }
745         }
746 
747         if (resources.contains(r)) {
748             return r;
749         }
750 
751         // Re-add yourself
752         if (resources.isEmpty()) {
753             BroadcasterFactory.getDefault().add(this, name);
754         }
755 
756         resources.add(r);
757         checkCachedAndPush(r, r.getAtmosphereResourceEvent());
758         return r;
759     }

766

779             notifyEmptyListener();
780             if (lifeCyclePolicy.getLifeCyclePolicy() == BroadcasterLifeCyclePolicy.ATMOSPHERE_RESOURCE_POLICY.EMPTY) {

16  package org.atmosphere.cpr;
17  
18  /**
19   * A listener that can be used to track {@link AsyncSupport} events like SUSPEND, RESUME, TIMEOUT, DESTROYED and CLOSED
20   *
21   * @author Jeanfrancois Arcand
22   */
23  public interface AsyncSupportListener {
24      /**
25       * Invoked when an {@link AtmosphereResource} gets suspended.
26       * @param request an {@link AtmosphereRequest}
27       * @param response an {@link AtmosphereResponse}
28       */
29      void onSuspend(AtmosphereRequest request, AtmosphereResponse response);
30      /**
31       * Invoked when an {@link AtmosphereResource} gets resumed
32       * @param request an {@link AtmosphereRequest}
33       * @param response an {@link AtmosphereResponse}
34       */
35      void onResume(AtmosphereRequest request, AtmosphereResponse response);
36      /**
37       * Invoked when an {@link AtmosphereResource} gets timed out
38       * @param request an {@link AtmosphereRequest}
39       * @param response an {@link AtmosphereResponse}
40       */
41      void onTimeout(AtmosphereRequest request, AtmosphereResponse response);
42      /**
43       * Invoked when an {@link AtmosphereResource} gets closed
44       * @param request an {@link AtmosphereRequest}
45       * @param response an {@link AtmosphereResponse}
46       */
47      void onClose(AtmosphereRequest request, AtmosphereResponse response);
48      /**
49       * Invoked when an {@link AtmosphereResource} gets destroyed
50       * @param request an {@link AtmosphereRequest}
51       * @param response an {@link AtmosphereResponse}
52       */
53      void onDestroyed(AtmosphereRequest request, AtmosphereResponse response);
54  }

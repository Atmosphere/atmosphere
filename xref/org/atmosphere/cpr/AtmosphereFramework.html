314      * This API is exposed to allow embedding an Atmosphere application.
315      *

317      * @param h       implementation of an {@link AtmosphereHandler}
318      * @param l       An array of {@link AtmosphereInterceptor}.

321         if (!mapping.startsWith("/")) {
322             mapping = "/" + mapping;
323         }
324         AtmosphereHandlerWrapper w = new AtmosphereHandlerWrapper(broadcasterFactory, h, mapping);
325         w.interceptors = l;
326         addMapping(mapping, w);
327 
328         if (isInit) {
329             initHandlerInterceptors(w);
330         } else {
331             logger.info("Installed AtmosphereHandler {} mapped to context-path: {}", h.getClass().getName(), mapping);
332             logger.info("Installed the following AtmosphereInterceptor mapped to AtmosphereHandler {}", h.getClass().getName());
333             if (l.size() > 0) {
334                 for (AtmosphereInterceptor s : l) {
335                     logger.info("\t{} : {}", s.getClass().getName(), s);
336                 }
337             }
338         }
339         return this;
340     }

354     private AtmosphereFramework addMapping(String path, AtmosphereHandlerWrapper w) {
355         // We are using JAXRS mapping algorithm.
356         if (path.contains("*")) {
357             path = path.replace("*", mappingRegex);
358         }
359 
360         if (path.endsWith("/")) {
361             path = path + mappingRegex;
362         }
363 
364         InjectorProvider.getInjector().inject(w.atmosphereHandler);
365         atmosphereHandlers.put(path, w);
366         return this;
367     }

1613         if (s != null) {

1634

1839     protected Map<String, String> configureQueryStringAsRequest(AtmosphereRequest request) {
1840         Map<String, String> headers = new HashMap<String, String>();
1841 
1842         StringBuilder q = new StringBuilder();
1843         try {
1844             String qs = request.getQueryString();
1845             if (qs != null && !qs.isEmpty()) {
1846                 String[] params = qs.split("&");
1847                 String[] s;
1848                 for (String p : params) {
1849                     s = p.split("=");
1850                     if (s[0].equalsIgnoreCase("Content-Type")) {
1851                         // Use the one set by the user first.
1852                         if (request.getContentType() == null ||
1853                                 !request.getContentType().equalsIgnoreCase(s.length > 1 ? s[1] : "")) {
1854                             request.contentType(s.length > 1 ? s[1] : "");
1855                         }
1856                     }
1857                     if (!s[0].isEmpty()
1858                             && !s[0].toLowerCase().startsWith("x-atmo")
1859                             && !s[0].equalsIgnoreCase("x-cache-date")
1860                             && !s[0].equalsIgnoreCase("Content-Type")
1861                             && !s[0].equalsIgnoreCase("_")) {
1862                         q.append(s[0]).append("=").append(s.length > 1 ? s[1] : "").append("&");
1863                     }
1864                     headers.put(s[0], s.length > 1 ? s[1] : "");
1865                 }
1866             }
1867         } catch (Exception ex) {
1868             logger.error("Unable to parse query string", ex);
1869         }
1870         if (q.length() > 0) q.deleteCharAt(q.length() - 1);
1871         request.queryString(q.toString());
1872 
1873         logger.trace("Query String translated to headers {}", headers);
1874         return headers;
1875     }

293     public static final class AtmosphereHandlerWrapper {
294 
295         public final AtmosphereHandler atmosphereHandler;
296         public Broadcaster broadcaster;
297         public String mapping;
298         public final LinkedList<AtmosphereInterceptor> interceptors = new LinkedList<AtmosphereInterceptor>();
299         public boolean create;

301         private final boolean wilcardMapping;
302 
303         public AtmosphereHandlerWrapper(BroadcasterFactory broadcasterFactory, final AtmosphereHandler atmosphereHandler, String mapping,
304                                         final AtmosphereConfig config) {
305             this.atmosphereHandler = atmosphereHandler;
306 
307             try {
308                 if (broadcasterFactory != null) {
309                     this.broadcaster = broadcasterFactory.lookup(mapping, true);
310                 } else {
311                     this.mapping = mapping;
312                 }
313             } catch (Exception t) {
314                 throw new RuntimeException(t);
315             }
316             wilcardMapping = mapping.contains("{") && mapping.contains("}");
317             hookInjection(config);
318         }
319 
320         void hookInjection(final AtmosphereConfig config) {
321             config.startupHook(new AtmosphereConfig.StartupHook() {
322                 @Override
323                 public void started(AtmosphereFramework framework) {
324                     needRequestScopedInjection = Utils.requestScopedInjection(config, atmosphereHandler);
325                 }
326             });
327         }
328 
329         public AtmosphereHandlerWrapper(final AtmosphereHandler atmosphereHandler, Broadcaster broadcaster,
330                                         final AtmosphereConfig config) {
331             this.atmosphereHandler = atmosphereHandler;
332             this.broadcaster = broadcaster;
333             hookInjection(config);
334             wilcardMapping = false;
335         }
336 
337         @Override
338         public String toString() {
339 
340             StringBuilder b = new StringBuilder();
341             for (int i = 0; i < interceptors.size(); i++) {
342                 b.append("\n\t").append(i).append(": ").append(interceptors.get(i).getClass().getName());
343             }
344 
345             return "\n atmosphereHandler"
346                     + "\n\t" + atmosphereHandler
347                     + "\n interceptors" +
348                     b.toString()
349                     + "\n broadcaster"
350                     + "\t" + broadcaster;
351         }
352 
353         public boolean needRequestScopedInjection() {
354             return needRequestScopedInjection;
355         }
356 
357         public boolean wildcardMapping() {
358             return wilcardMapping;
359         }

436         /**
437          * <p>
438          * Install the classes.
439          * </p>
440          *
441          * @author Guillaume DROUET
442          * @version 1.0
443          * @since 2.2.0
444          */
445         private static class InstallMetaServiceProcedure implements MetaServiceProcedure {
446 
447             /**
448              * {@inheritDoc}
449              */
450             @Override
451             public void apply(final AtmosphereFramework fwk, final Class c) throws Exception {
452                 if (AtmosphereInterceptor.class.isAssignableFrom(c)) {
453                     fwk.interceptor(fwk.newClassInstance(AtmosphereInterceptor.class, c));
454                 } else if (Broadcaster.class.isAssignableFrom(c)) {
455                     fwk.setDefaultBroadcasterClassName(c.getName());
456                 } else if (BroadcasterListener.class.isAssignableFrom(c)) {
457                     fwk.addBroadcasterListener(fwk.newClassInstance(BroadcasterListener.class, c));
458                 } else if (BroadcasterCache.class.isAssignableFrom(c)) {
459                     fwk.setBroadcasterCacheClassName(c.getName());
460                 } else if (BroadcastFilter.class.isAssignableFrom(c)) {
461                     fwk.broadcasterFilters.add(c.getName());
462                 } else if (BroadcasterCacheInspector.class.isAssignableFrom(c)) {
463                     fwk.inspectors.add(fwk.newClassInstance(BroadcasterCacheInspector.class, c));
464                 } else if (AsyncSupportListener.class.isAssignableFrom(c)) {
465                     fwk.asyncSupportListeners.add(fwk.newClassInstance(AsyncSupportListener.class, c));
466                 } else if (AsyncSupport.class.isAssignableFrom(c)) {
467                     fwk.setAsyncSupport(fwk.newClassInstance(AsyncSupport.class, c));
468                 } else if (BroadcasterCacheListener.class.isAssignableFrom(c)) {
469                     fwk.broadcasterCacheListeners.add(fwk.newClassInstance(BroadcasterCacheListener.class, c));
470                 } else if (BroadcasterConfig.FilterManipulator.class.isAssignableFrom(c)) {
471                     fwk.filterManipulators.add(fwk.newClassInstance(BroadcasterConfig.FilterManipulator.class, c));
472                 } else if (WebSocketProtocol.class.isAssignableFrom(c)) {
473                     fwk.webSocketProtocolClassName = c.getName();
474                 } else if (WebSocketProcessor.class.isAssignableFrom(c)) {
475                     fwk.webSocketProcessorClassName = c.getName();
476                 } else if (AtmosphereResourceFactory.class.isAssignableFrom(c)) {
477                     fwk.setAndConfigureAtmosphereResourceFactory(fwk.newClassInstance(AtmosphereResourceFactory.class, c));
478                 } else if (AtmosphereFrameworkListener.class.isAssignableFrom(c)) {
479                     fwk.frameworkListener(fwk.newClassInstance(AtmosphereFrameworkListener.class, c));
480                 } else if (WebSocketFactory.class.isAssignableFrom(c)) {
481                     fwk.webSocketFactory(fwk.newClassInstance(WebSocketFactory.class, c));
482                 } else if (AtmosphereFramework.class.isAssignableFrom(c)) {
483                     // No OPS
484                 } else if (EndpointMapper.class.isAssignableFrom(c)) {
485                    fwk.endPointMapper(fwk.newClassInstance(EndpointMapper.class, c));
486                 } else {
487                     logger.warn("{} is not a framework service that could be installed", c.getName());
488                 }
489             }
490         }

2463     protected Map<String, String> configureQueryStringAsRequest(AtmosphereRequest request) {
2464         Map<String, String> headers = new HashMap<String, String>();
2465 
2466         StringBuilder q = new StringBuilder();
2467         try {
2468             String qs = request.getQueryString();
2469             if (qs != null && !qs.isEmpty()) {
2470                 String[] params = qs.split("&");
2471                 String[] s;
2472                 for (String p : params) {
2473                     s = p.split("=", 2);
2474                     final String header = s[0];
2475                     final String value = s.length > 1 ? s[1] : "";
2476 
2477                     if (header.equalsIgnoreCase("Content-Type")) {
2478                         // Use the one set by the user first.
2479                         if (request.getContentType() == null ||
2480                                 !request.getContentType().equalsIgnoreCase(s.length > 1 ? value : "")) {
2481                             request.contentType(s.length > 1 ? URLDecoder.decode(value, "UTF-8") : "");
2482                         }
2483                     }
2484                     if (!header.isEmpty()
2485                             && !header.toLowerCase().startsWith("x-atmo")
2486                             && !header.equalsIgnoreCase(HeaderConfig.X_HEARTBEAT_SERVER)
2487                             && !header.equalsIgnoreCase("Content-Type")
2488                             && !header.equalsIgnoreCase("_")) {
2489                         q.append(header).append("=").append(s.length > 1 ? value : "").append("&");
2490                     }
2491                     headers.put(header, s.length > 1 ? value : "");
2492                 }
2493             }
2494         } catch (Exception ex) {
2495             logger.error("Unable to parse query string", ex);
2496         }
2497         String disallowModifyQueryString = config.getInitParameter(ApplicationConfig.DISALLOW_MODIFY_QUERYSTRING);

2826         String path = handlersPath != DEFAULT_HANDLER_PATH ? handlersPath : realPath(sc, handlersPath);
2827         try {
2828             annotationProcessor = newClassInstance(AnnotationProcessor.class,
2829                     (Class<AnnotationProcessor>) IOUtils.loadClass(getClass(), annotationProcessorClassName));
2830             logger.info("Atmosphere is using {} for processing annotation", annotationProcessorClassName);
2831 
2832             annotationProcessor.configure(config);
2833 
2834             if (!packages.isEmpty()) {
2835                 for (String s : packages) {
2836                     annotationProcessor.scan(s);
2837                 }
2838             }
2839 
2840             // Second try.
2841             if (!annotationFound) {
2842                 if (path != null) {
2843                     annotationProcessor.scan(new File(path));

2848                 if (pathLibs != null) {

2854                             return arg1.endsWith(".jar");
2855                         }
2856                     });
2857

75  /**
76   * The {@link AtmosphereServlet} acts as a dispatcher for {@link AtmosphereHandler}
77   * defined in META-INF/atmosphere.xml, or if atmosphere.xml is missing, all classes
78   * that implements {@link AtmosphereHandler} will be discovered and mapped using
79   * the class's name.
80   * <p/>
81   * This {@link Servlet} can be defined inside an application's web.xml using the following:
82   * <blockquote><pre>
83   *  &lt;servlet&gt;
84   *      &lt;description&gt;AtmosphereServlet&lt;/description&gt;
85   *      &lt;servlet-name&gt;AtmosphereServlet&lt;/servlet-name&gt;
86   *      &lt;servlet-class&gt;org.atmosphere.cpr.AtmosphereServlet&lt;/servlet-class&gt;
87   *      &lt;load-on-startup&gt;0 &lt;/load-on-startup&gt;
88   *  &lt;/servlet&gt;
89   *  &lt;servlet-mapping&gt;
90   *      &lt;servlet-name&gt;AtmosphereServlet&lt;/servlet-name&gt;
91   *      &lt;url-pattern&gt;/Atmosphere &lt;/url-pattern&gt;
92   *  &lt;/servlet-mapping&gt;
93   * </pre></blockquote>
94   * You can force this Servlet to use native API of the Web Server instead of
95   * the Servlet 3.0 Async API you are deploying on by adding
96   * <blockquote><pre>
97   *  &lt;init-param&gt;
98   *      &lt;param-name&gt;org.atmosphere.useNative&lt;/param-name&gt;
99   *      &lt;param-value&gt;true&lt;/param-value&gt;
100  *  &lt;/init-param&gt;
101  * </pre></blockquote>
102  * You can force this Servlet to use one Thread per connection instead of
103  * native API of the Web Server you are deploying on by adding
104  * <blockquote><pre>
105  *  &lt;init-param&gt;
106  *      &lt;param-name&gt;org.atmosphere.useBlocking&lt;/param-name&gt;
107  *      &lt;param-value&gt;true&lt;/param-value&gt;
108  *  &lt;/init-param&gt;
109  * </pre></blockquote>
110  * You can also define {@link Broadcaster}by adding:
111  * <blockquote><pre>
112  *  &lt;init-param&gt;
113  *      &lt;param-name&gt;org.atmosphere.cpr.broadcasterClass&lt;/param-name&gt;
114  *      &lt;param-value&gt;class-name&lt;/param-value&gt;
115  *  &lt;/init-param&gt;
116  * </pre></blockquote>
117  * You can also for Atmosphere to use {@link java.io.OutputStream} for all write operations.
118  * <blockquote><pre>
119  *  &lt;init-param&gt;
120  *      &lt;param-name&gt;org.atmosphere.useStream&lt;/param-name&gt;
121  *      &lt;param-value&gt;true&lt;/param-value&gt;
122  *  &lt;/init-param&gt;
123  * </pre></blockquote>
124  * You can also configure {@link org.atmosphere.cpr.BroadcasterCache} that persist message when Browser is disconnected.
125  * <blockquote><pre>
126  *  &lt;init-param&gt;
127  *      &lt;param-name&gt;org.atmosphere.cpr.broadcasterCacheClass&lt;/param-name&gt;
128  *      &lt;param-value&gt;class-name&lt;/param-value&gt;
129  *  &lt;/init-param&gt;
130  * </pre></blockquote>
131  * You can also configure Atmosphere to use http session or not
132  * <blockquote><pre>
133  *  &lt;init-param&gt;
134  *      &lt;param-name&gt;org.atmosphere.cpr.sessionSupport&lt;/param-name&gt;
135  *      &lt;param-value&gt;false&lt;/param-value&gt;
136  *  &lt;/init-param&gt;
137  * </pre></blockquote>
138  * You can also configure {@link BroadcastFilter} that will be applied at all newly created {@link Broadcaster}
139  * <blockquote><pre>
140  *  &lt;init-param&gt;
141  *      &lt;param-name&gt;org.atmosphere.cpr.broadcastFilterClasses&lt;/param-name&gt;
142  *      &lt;param-value&gt;BroadcastFilter class name separated by coma&lt;/param-value&gt;
143  *  &lt;/init-param&gt;
144  * </pre></blockquote>
145  * All the property available are defined in {@link ApplicationConfig}
146  * The Atmosphere Framework can also be used as a Servlet Filter ({@link AtmosphereFilter}).
147  * <p/>
148  * If you are planning to use JSP, Servlet or JSF, you can instead use the
149  * {@link MeteorServlet}, which allow the use of {@link Meteor} inside those
150  * components.
151  *
152  * @author Jeanfrancois Arcand
153  */

296     /**
297      * Hack to support Tomcat AIO like other WebServer. This method is invoked
298      * by Tomcat when it detect a {@link Servlet} implements the interface
299      * {@link CometProcessor} without invoking {@link Servlet#service}
300      *
301      * @param cometEvent the {@link CometEvent}
302      * @throws java.io.IOException
303      * @throws javax.servlet.ServletException
304      */
305     public void event(CometEvent cometEvent) throws IOException, ServletException {
306         HttpServletRequest req = cometEvent.getHttpServletRequest();
307         HttpServletResponse res = cometEvent.getHttpServletResponse();
308         req.setAttribute(TomcatCometSupport.COMET_EVENT, cometEvent);
309 
310         if (!framework.getAsyncSupport().supportWebSocket()) {
311             if (!framework.isCometSupportSpecified && !framework.isCometSupportConfigured.getAndSet(true)) {
312                 synchronized (framework.asyncSupport) {
313                     if (!framework.asyncSupport.getClass().equals(TomcatCometSupport.class)) {
314                         logger.warn("TomcatCometSupport is enabled, switching to it");
315                         framework.asyncSupport = new TomcatCometSupport(framework.config);
316                     }
317                 }
318             }
319         }
320 
321         framework.doCometSupport(AtmosphereRequest.wrap(req), AtmosphereResponse.wrap(res));
322 
323         String transport = cometEvent.getHttpServletRequest().getParameter(HeaderConfig.X_ATMOSPHERE_TRANSPORT);
324         if (transport != null && transport.equalsIgnoreCase(HeaderConfig.WEBSOCKET_TRANSPORT)) {
325             cometEvent.close();
326         }
327     }

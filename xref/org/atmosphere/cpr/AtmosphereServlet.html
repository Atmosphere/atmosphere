95  /**
96   * The {@link AtmosphereServlet} acts as a dispatcher for {@link AtmosphereHandler}
97   * defined in META-INF/atmosphere.xml, or if atmosphere.xml is missing, all classes
98   * that implements {@link AtmosphereHandler} will be discovered and mapped using
99   * the class's name.
100  * <p/>
101  * This {@link Servlet} can be defined inside an application's web.xml using the following:
102  * <blockquote><pre>
103  *  &lt;servlet&gt;
104  *      &lt;description&gt;AtmosphereServlet&lt;/description&gt;
105  *      &lt;servlet-name&gt;AtmosphereServlet&lt;/servlet-name&gt;
106  *      &lt;servlet-class&gt;org.atmosphere.cpr.AtmosphereServlet&lt;/servlet-class&gt;
107  *      &lt;load-on-startup&gt;0 &lt;/load-on-startup&gt;
108  *  &lt;/servlet&gt;
109  *  &lt;servlet-mapping&gt;
110  *      &lt;servlet-name&gt;AtmosphereServlet&lt;/servlet-name&gt;
111  *      &lt;url-pattern&gt;/Atmosphere &lt;/url-pattern&gt;
112  *  &lt;/servlet-mapping&gt;
113  * </pre></blockquote>
114  * You can force this Servlet to use native API of the Web Server instead of
115  * the Servlet 3.0 Async API you are deploying on by adding
116  * <blockquote><pre>
117  *  &lt;init-param&gt;
118  *      &lt;param-name&gt;org.atmosphere.useNative&lt;/param-name&gt;
119  *      &lt;param-value&gt;true&lt;/param-value&gt;
120  *  &lt;/init-param&gt;
121  * </pre></blockquote>
122  * You can force this Servlet to use one Thread per connection instead of
123  * native API of the Web Server you are deploying on by adding
124  * <blockquote><pre>
125  *  &lt;init-param&gt;
126  *      &lt;param-name&gt;org.atmosphere.useBlocking&lt;/param-name&gt;
127  *      &lt;param-value&gt;true&lt;/param-value&gt;
128  *  &lt;/init-param&gt;
129  * </pre></blockquote>
130  * You can also define {@link Broadcaster}by adding:
131  * <blockquote><pre>
132  *  &lt;init-param&gt;
133  *      &lt;param-name&gt;org.atmosphere.cpr.broadcasterClass&lt;/param-name&gt;
134  *      &lt;param-value&gt;class-name&lt;/param-value&gt;
135  *  &lt;/init-param&gt;
136  * </pre></blockquote>
137  * You can also for Atmosphere to use {@link java.io.OutputStream} for all write operations.
138  * <blockquote><pre>
139  *  &lt;init-param&gt;
140  *      &lt;param-name&gt;org.atmosphere.useStream&lt;/param-name&gt;
141  *      &lt;param-value&gt;true&lt;/param-value&gt;
142  *  &lt;/init-param&gt;
143  * </pre></blockquote>
144  * You can also configure {@link org.atmosphere.cpr.BroadcasterCache} that persist message when Browser is disconnected.
145  * <blockquote><pre>
146  *  &lt;init-param&gt;
147  *      &lt;param-name&gt;org.atmosphere.cpr.broadcasterCacheClass&lt;/param-name&gt;
148  *      &lt;param-value&gt;class-name&lt;/param-value&gt;
149  *  &lt;/init-param&gt;
150  * </pre></blockquote>
151  * You can also configure Atmosphere to use http session or not
152  * <blockquote><pre>
153  *  &lt;init-param&gt;
154  *      &lt;param-name&gt;org.atmosphere.cpr.sessionSupport&lt;/param-name&gt;
155  *      &lt;param-value&gt;false&lt;/param-value&gt;
156  *  &lt;/init-param&gt;
157  * </pre></blockquote>
158  * You can also configure {@link BroadcastFilter} that will be applied at all newly created {@link Broadcaster}
159  * <blockquote><pre>
160  *  &lt;init-param&gt;
161  *      &lt;param-name&gt;org.atmosphere.cpr.broadcastFilterClasses&lt;/param-name&gt;
162  *      &lt;param-value&gt;BroadcastFilter class name separated by coma&lt;/param-value&gt;
163  *  &lt;/init-param&gt;
164  * </pre></blockquote>
165  * All the property available are defined in {@link ApplicationConfig}
166  * The Atmosphere Framework can also be used as a Servlet Filter ({@link AtmosphereFilter}).
167  * <p/>
168  * If you are planning to use JSP, Servlet or JSF, you can instead use the
169  * {@link MeteorServlet}, which allow the use of {@link Meteor} inside those
170  * components.
171  *
172  * @author Jeanfrancois Arcand
173  */

492     @Override
493     public void init(final ServletConfig sc) throws ServletException {
494         try {
495             super.init(sc);
496 
497             ServletContextHolder.register(this);
498 
499             ServletConfig scFacade = new ServletConfig() {
500 
501                 public String getServletName() {
502                     return sc.getServletName();
503                 }
504 
505                 public ServletContext getServletContext() {
506                     return sc.getServletContext();
507                 }
508 
509                 public String getInitParameter(String name) {
510                     String param = sc.getInitParameter(name);
511                     if (param == null) {
512                         return initParams.get(name);
513                     }
514                     return param;
515                 }
516 
517                 public Enumeration<String> getInitParameterNames() {
518                     return sc.getInitParameterNames();
519                 }
520             };
521             doInitParams(scFacade);
522             configureDefaultBroadcasterFactory();
523             doInitParamsForWebSocket(scFacade);
524             loadConfiguration(scFacade);
525 
526             autoDetectContainer();
527             configureBroadcaster(sc.getServletContext());
528             configureWebDotXmlAtmosphereHandler(sc);
529             cometSupport.init(scFacade);
530             initAtmosphereHandler(scFacade);
531 
532             logger.info("Using broadcaster class: {}", broadcasterClassName);
533             logger.info("Atmosphere Framework {} started.", Version.getRawVersion());
534         } catch (Throwable t) {
535             logger.error("failed to initialize atmosphere framework", t);
536 
537             if (t instanceof ServletException) {
538                 throw (ServletException) t;
539             }
540 
541             throw new ServletException(t.getCause());
542         }
543     }

545     protected void configureWebDotXmlAtmosphereHandler(ServletConfig sc) {
546         String s = sc.getInitParameter(ATMOSPHERE_HANDLER);
547         if (s != null) {
548             ClassLoader cl = Thread.currentThread().getContextClassLoader();
549             try {
550                 addAtmosphereHandler("/*", (AtmosphereHandler<?, ?>) cl.loadClass(s).newInstance());
551             } catch (Exception ex) {
552                 logger.warn("Unable to load WebSocketHandle instance", ex);
553             }
554         }
555         logger.info("Installed AtmosphereHandler {} mapped to context-path: /*", s);
556     }

558     protected void configureBroadcaster(ServletContext sc) throws ClassNotFoundException, InstantiationException, IllegalAccessException {
559 
560         if (broadcasterFactoryClassName != null) {
561             logger.info("Using BroadcasterFactory class: {}", broadcasterFactoryClassName);
562 
563             broadcasterFactory = (BroadcasterFactory) Thread.currentThread().getContextClassLoader()
564                     .loadClass(broadcasterFactoryClassName).newInstance();
565         }
566 
567         if (broadcasterFactory == null) {
568             Class<? extends Broadcaster> bc =
569                     (Class<? extends Broadcaster>) Thread.currentThread().getContextClassLoader()
570                             .loadClass(broadcasterClassName);
571 
572             broadcasterFactory = new DefaultBroadcasterFactory(bc, broadcasterLifeCyclePolicy);
573         }
574 
575         // http://java.net/jira/browse/ATMOSPHERE-157
576         sc.setAttribute(BroadcasterFactory.class.getName(), broadcasterFactory);
577 
578         config.broadcasterFactory = broadcasterFactory;
579         BroadcasterFactory.setBroadcasterFactory(broadcasterFactory, config);
580         InjectorProvider.getInjector().inject(broadcasterFactory);
581 
582         Iterator<Entry<String, AtmosphereHandlerWrapper>> i = atmosphereHandlers.entrySet().iterator();
583         AtmosphereHandlerWrapper w;
584         Entry<String, AtmosphereHandlerWrapper> e;
585         while (i.hasNext()) {
586             e = i.next();
587             w = e.getValue();
588             BroadcasterConfig broadcasterConfig = new BroadcasterConfig(broadcasterFilters, config);
589 
590             if (w.broadcaster == null) {
591                 w.broadcaster = broadcasterFactory.get();
592             } else {
593                 w.broadcaster.setBroadcasterConfig(broadcasterConfig);
594                 if (broadcasterCacheClassName != null) {
595                     BroadcasterCache cache = (BroadcasterCache) Thread.currentThread().getContextClassLoader()
596                             .loadClass(broadcasterCacheClassName).newInstance();
597                     InjectorProvider.getInjector().inject(cache);
598                     broadcasterConfig.setBroadcasterCache(cache);
599                 }
600             }
601             w.broadcaster.setID(e.getKey());
602         }
603     }

617     /**
618      * Read init param from web.xml and apply them.
619      *
620      * @param sc {@link ServletConfig}
621      */
622     protected void doInitParams(ServletConfig sc) {
623         String s = sc.getInitParameter(PROPERTY_NATIVE_COMETSUPPORT);
624         if (s != null) {
625             useNativeImplementation = Boolean.parseBoolean(s);
626             if (useNativeImplementation) isCometSupportSpecified = true;
627         }
628         s = sc.getInitParameter(PROPERTY_BLOCKING_COMETSUPPORT);
629         if (s != null) {
630             useBlockingImplementation = Boolean.parseBoolean(s);
631             if (useBlockingImplementation) isCometSupportSpecified = true;
632         }
633         s = sc.getInitParameter(PROPERTY_USE_STREAM);
634         if (s != null) {
635             useStreamForFlushingComments = Boolean.parseBoolean(s);
636         }
637         s = sc.getInitParameter(PROPERTY_COMET_SUPPORT);
638         if (s != null) {
639             cometSupport = new DefaultCometSupportResolver(config).newCometSupport(s);
640             isCometSupportSpecified = true;
641         }
642         s = sc.getInitParameter(BROADCASTER_CLASS);
643         if (s != null) {
644             broadcasterClassName = s;
645             isBroadcasterSpecified = true;
646         }
647         s = sc.getInitParameter(BROADCASTER_CACHE);
648         if (s != null) {
649             broadcasterCacheClassName = s;
650         }

659         } else {

665         }

667         if (s != null) {

671         if (s != null) {

681         try {
682             URL url = sc.getServletContext().getResource("/WEB-INF/classes/");
683             URLClassLoader urlC = new URLClassLoader(new URL[]{url},
684                     Thread.currentThread().getContextClassLoader());
685             loadAtmosphereDotXml(sc.getServletContext().

693                 }
694             }
695         } catch (Throwable t) {
696             throw new ServletException(t);
697         }

701      * Auto-detect Jersey when no atmosphere.xml file are specified.
702      *
703      * @param sc {@link ServletConfig}
704      * @return true if Jersey classes are detected
705      * @throws ClassNotFoundException
706      */
707     protected boolean detectSupportedFramework(ServletConfig sc) throws ClassNotFoundException, IllegalAccessException,
708             InstantiationException, NoSuchMethodException, InvocationTargetException {
709 
710         ClassLoader cl = Thread.currentThread().getContextClassLoader();
711         String broadcasterClassNameTmp = null;
712         
713         try {
714             cl.loadClass(JERSEY_CONTAINER);
715             
716             if (!isBroadcasterSpecified){
717                 broadcasterClassNameTmp = lookupDefaultBroadcasterType();
718 
719                 cl.loadClass(broadcasterClassNameTmp);
720             }
721             useStreamForFlushingComments = true;
722         } catch (Throwable t) {
723             return false;
724         }
725 
726         logger.warn("Missing META-INF/atmosphere.xml but found the Jersey runtime. Starting Jersey");
727 
728         // Jersey will handle itself the headers.
729         initParams.put(WRITE_HEADERS, "false");
730 
731         ReflectorServletProcessor rsp = new ReflectorServletProcessor();
732         if (broadcasterClassNameTmp!=null) broadcasterClassName = broadcasterClassNameTmp;
733         rsp.setServletClassName(JERSEY_CONTAINER);
734         sessionSupport(false);
735         initParams.put(DISABLE_ONSTATE_EVENT, "true");
736 
737         String mapping = sc.getInitParameter(PROPERTY_SERVLET_MAPPING);
738         if (mapping == null) {
739             mapping = "/*";
740         }
741         Class<? extends Broadcaster> bc = (Class<? extends Broadcaster>) cl.loadClass(broadcasterClassName);
742 
743         Broadcaster b = BroadcasterFactory.getDefault().get(bc, mapping);
744 
745         addAtmosphereHandler(mapping, rsp, b);
746         return true;
747     }

784         if (cometSupport != null && AsynchronousProcessor.class.isAssignableFrom(cometSupport.getClass())) {
785             ((AsynchronousProcessor) cometSupport).shutdown();
786         }

805     /**
806      * Load AtmosphereHandler defined under META-INF/atmosphere.xml
807      *
808      * @param stream The input stream we read from.
809      * @param c      The classloader
810      */
811     protected void loadAtmosphereDotXml(InputStream stream, URLClassLoader c)
812             throws IOException, ServletException {
813 
814         if (stream == null) {
815             return;
816         }
817 
818         AtmosphereConfigReader reader = new AtmosphereConfigReader(stream);
819 
820         Map<String, String> atmosphereHandlerNames = reader.getAtmosphereHandlers();
821         Set<Entry<String, String>> entries = atmosphereHandlerNames.entrySet();
822         for (Entry<String, String> entry : entries) {
823             AtmosphereHandler handler;
824             String handlerClassName = entry.getValue();
825             String handlerPath = entry.getKey();
826 
827             try {
828                 if (!handlerClassName.equals(ReflectorServletProcessor.class.getName())) {
829                     handler = (AtmosphereHandler) c.loadClass(handlerClassName).newInstance();
830                     InjectorProvider.getInjector().inject(handler);
831                 } else {
832                     handler = new ReflectorServletProcessor();
833                 }
834 
835                 logger.info("Installed AtmosphereHandler {} mapped to context-path: {}", handler, handlerPath);
836 
837                 AtmosphereHandlerWrapper wrapper = new AtmosphereHandlerWrapper(handler);
838                 atmosphereHandlers.put(handlerPath, wrapper);
839                 boolean isJersey = false;
840                 for (Property p : reader.getProperty(handlerPath)) {
841                     if (p.value != null && p.value.indexOf("jersey") != -1) {
842                         isJersey = true;
843                         initParams.put(DISABLE_ONSTATE_EVENT, "true");
844                         useStreamForFlushingComments = true;
845                         broadcasterClassName = lookupDefaultBroadcasterType();
846                     }
847                     IntrospectionUtils.setProperty(handler, p.name, p.value);
848                 }
849 
850                 config.supportSession = !isJersey;
851 
852                 if (!reader.supportSession().equals("")) {
853                     sessionSupport(Boolean.valueOf(reader.supportSession()));
854                 }
855 
856                 for (Property p : reader.getProperty(handlerPath)) {
857                     IntrospectionUtils.addProperty(handler, p.name, p.value);
858                 }
859 
860                 String broadcasterClass = reader.getBroadcasterClass(handlerPath);
861                 /**
862                  * If there is more than one AtmosphereHandler defined, their Broadcaster
863                  * may clash each other with the BroadcasterFactory. In that case we will use the
864                  * last one defined.
865                  */
866                 if (broadcasterClass != null) {
867                     broadcasterClassName = broadcasterClass;
868                     ClassLoader cl = Thread.currentThread().getContextClassLoader();
869                     Class<? extends Broadcaster> bc = (Class<? extends Broadcaster>) cl.loadClass(broadcasterClassName);
870                     wrapper.broadcaster = BroadcasterFactory.getDefault().get(bc, handlerPath);
871                 }
872 
873                 String bc = reader.getBroadcasterCache(handlerPath);
874                 if (bc != null) {
875                     broadcasterCacheClassName = bc;
876                 }
877 
878                 if (reader.getCometSupportClass() != null) {
879                     cometSupport = (CometSupport) c.loadClass(reader.getCometSupportClass())
880                             .getDeclaredConstructor(new Class[]{AtmosphereConfig.class})
881                             .newInstance(new Object[]{config});
882                 }
883 
884                 if (reader.getBroadcastFilterClasses() != null) {
885                     broadcasterFilters = reader.getBroadcastFilterClasses();
886                 }
887 
888             } catch (Throwable t) {
889                 logger.warn("unable to load AtmosphereHandler class: " + handlerClassName, t);
890                 throw new ServletException(t);
891             }
892         }
893     }

940      * Auto detect instance of {@link AtmosphereHandler} in case META-INF/atmosphere.xml
941      * is missing.
942      *
943      * @param servletContext {@link ServletContext}
944      * @param classloader    {@link URLClassLoader} to load the class.
945      * @throws java.net.MalformedURLException
946      * @throws java.net.URISyntaxException
947      */
948     protected void autoDetectAtmosphereHandlers(ServletContext servletContext, URLClassLoader classloader)
949             throws MalformedURLException, URISyntaxException {
950         logger.info("Auto detecting atmosphere handlers in WEB-INF/classes");
951 
952         String realPath = servletContext.getRealPath(WEB_INF_CLASSES);
953 
954         // Weblogic bug
955         if (realPath == null) {
956             URL u = servletContext.getResource(WEB_INF_CLASSES);
957             if (u == null) return;
958             realPath = u.getPath();
959         }
960 
961         loadAtmosphereHandlersFromPath(classloader, realPath);
962     }

964     protected void loadAtmosphereHandlersFromPath(URLClassLoader classloader, String realPath) {
965         File file = new File(realPath);
966 
967         if (file.isDirectory()) {
968             getFiles(file);
969 
970             for (String className : possibleAtmosphereHandlersCandidate) {
971                 try {
972                     className = className.replace('\\', '/');
973                     className = className.replaceFirst("^.*/(WEB-INF|target)/(test-)?classes/(.*)\\.class", "$3").replace("/", ".");
974                     Class<?> clazz = classloader.loadClass(className);
975 
976                     if (AtmosphereHandler.class.isAssignableFrom(clazz)) {
977                         AtmosphereHandler handler = (AtmosphereHandler) clazz.newInstance();
978                         InjectorProvider.getInjector().inject(handler);
979                         atmosphereHandlers.put("/" + handler.getClass().getSimpleName(),
980                                 new AtmosphereHandlerWrapper(handler, null));
981                         logger.info("Installed AtmosphereHandler {} mapped to context-path: {}", handler, handler.getClass().getName());
982                     }
983                 } catch (Throwable t) {
984                     logger.trace("failed to load class as an AtmosphereHandler: " + className, t);
985                 }
986             }
987         }
988     }

1116     public Action doCometSupport(HttpServletRequest req, HttpServletResponse res)

1118         req.setAttribute(BROADCASTER_FACTORY, broadcasterFactory);

1120         req.setAttribute(BROADCASTER_CLASS, broadcasterClassName);
1121

1126             if (ex.getMessage() != null && ex.getMessage().startsWith("Tomcat failed")) {
1127                 if (!isFilter) {

1129                             ex.getMessage());
1130                     logger.warn("Using BlockingIOCometSupport.");
1131                 }
1132 
1133                 cometSupport = new BlockingIOCometSupport(config);
1134                 service(req, res);
1135             } else {
1136                 logger.error("AtmosphereServlet exception", ex);
1137                 throw ex;
1138             }
1139         }
1140         return null;
1141     }

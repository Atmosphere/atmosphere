118 /**
119  * The {@link AtmosphereServlet} acts as a dispatcher for {@link AtmosphereHandler}
120  * defined in META-INF/atmosphere.xml, or if atmosphere.xml is missing, all classes
121  * that implements {@link AtmosphereHandler} will be discovered and mapped using
122  * the class's name.
123  * <p/>
124  * This {@link Servlet} can be defined inside an application's web.xml using the following:
125  * <blockquote><pre>
126  *  &lt;servlet&gt;
127  *      &lt;description&gt;AtmosphereServlet&lt;/description&gt;
128  *      &lt;servlet-name&gt;AtmosphereServlet&lt;/servlet-name&gt;
129  *      &lt;servlet-class&gt;org.atmosphere.cpr.AtmosphereServlet&lt;/servlet-class&gt;
130  *      &lt;load-on-startup&gt;0 &lt;/load-on-startup&gt;
131  *  &lt;/servlet&gt;
132  *  &lt;servlet-mapping&gt;
133  *      &lt;servlet-name&gt;AtmosphereServlet&lt;/servlet-name&gt;
134  *      &lt;url-pattern&gt;/Atmosphere &lt;/url-pattern&gt;
135  *  &lt;/servlet-mapping&gt;
136  * </pre></blockquote>
137  * You can force this Servlet to use native API of the Web Server instead of
138  * the Servlet 3.0 Async API you are deploying on by adding
139  * <blockquote><pre>
140  *  &lt;init-param&gt;
141  *      &lt;param-name&gt;org.atmosphere.useNative&lt;/param-name&gt;
142  *      &lt;param-value&gt;true&lt;/param-value&gt;
143  *  &lt;/init-param&gt;
144  * </pre></blockquote>
145  * You can force this Servlet to use one Thread per connection instead of
146  * native API of the Web Server you are deploying on by adding
147  * <blockquote><pre>
148  *  &lt;init-param&gt;
149  *      &lt;param-name&gt;org.atmosphere.useBlocking&lt;/param-name&gt;
150  *      &lt;param-value&gt;true&lt;/param-value&gt;
151  *  &lt;/init-param&gt;
152  * </pre></blockquote>
153  * You can also define {@link Broadcaster}by adding:
154  * <blockquote><pre>
155  *  &lt;init-param&gt;
156  *      &lt;param-name&gt;org.atmosphere.cpr.broadcasterClass&lt;/param-name&gt;
157  *      &lt;param-value&gt;class-name&lt;/param-value&gt;
158  *  &lt;/init-param&gt;
159  * </pre></blockquote>
160  * You can also for Atmosphere to use {@link java.io.OutputStream} for all write operations.
161  * <blockquote><pre>
162  *  &lt;init-param&gt;
163  *      &lt;param-name&gt;org.atmosphere.useStream&lt;/param-name&gt;
164  *      &lt;param-value&gt;true&lt;/param-value&gt;
165  *  &lt;/init-param&gt;
166  * </pre></blockquote>
167  * You can also configure {@link org.atmosphere.cpr.BroadcasterCache} that persist message when Browser is disconnected.
168  * <blockquote><pre>
169  *  &lt;init-param&gt;
170  *      &lt;param-name&gt;org.atmosphere.cpr.broadcasterCacheClass&lt;/param-name&gt;
171  *      &lt;param-value&gt;class-name&lt;/param-value&gt;
172  *  &lt;/init-param&gt;
173  * </pre></blockquote>
174  * You can also configure Atmosphere to use http session or not
175  * <blockquote><pre>
176  *  &lt;init-param&gt;
177  *      &lt;param-name&gt;org.atmosphere.cpr.sessionSupport&lt;/param-name&gt;
178  *      &lt;param-value&gt;false&lt;/param-value&gt;
179  *  &lt;/init-param&gt;
180  * </pre></blockquote>
181  * You can also configure {@link BroadcastFilter} that will be applied at all newly created {@link Broadcaster}
182  * <blockquote><pre>
183  *  &lt;init-param&gt;
184  *      &lt;param-name&gt;org.atmosphere.cpr.broadcastFilterClasses&lt;/param-name&gt;
185  *      &lt;param-value&gt;BroadcastFilter class name separated by coma&lt;/param-value&gt;
186  *  &lt;/init-param&gt;
187  * </pre></blockquote>
188  * All the property available are defined in {@link ApplicationConfig}
189  * The Atmosphere Framework can also be used as a Servlet Filter ({@link AtmosphereFilter}).
190  * <p/>
191  * If you are planning to use JSP, Servlet or JSF, you can instead use the
192  * {@link MeteorServlet}, which allow the use of {@link Meteor} inside those
193  * components.
194  *
195  * @author Jeanfrancois Arcand
196  */

393         BroadcasterFactory.setBroadcasterFactory(new DefaultBroadcasterFactory(bc, broadcasterLifeCyclePolicy), config);

515     @Override
516     public void init(final ServletConfig sc) throws ServletException {
517         try {
518             super.init(sc);
519 
520             ServletContextHolder.register(this);
521 
522             ServletConfig scFacade = new ServletConfig() {
523 
524                 public String getServletName() {
525                     return sc.getServletName();
526                 }
527 
528                 public ServletContext getServletContext() {
529                     return sc.getServletContext();
530                 }
531 
532                 public String getInitParameter(String name) {
533                     String param = sc.getInitParameter(name);
534                     if (param == null) {
535                         return initParams.get(name);
536                     }
537                     return param;
538                 }
539 
540                 public Enumeration<String> getInitParameterNames() {
541                     return sc.getInitParameterNames();
542                 }
543             };
544             doInitParams(scFacade);
545             configureDefaultBroadcasterFactory();
546             doInitParamsForWebSocket(scFacade);
547             loadConfiguration(scFacade);
548 
549             autoDetectContainer();
550             configureBroadcaster(sc.getServletContext());
551             configureWebDotXmlAtmosphereHandler(sc);
552             cometSupport.init(scFacade);
553             initAtmosphereHandler(scFacade);
554 
555             logger.info("Using broadcaster class: {}", broadcasterClassName);
556             logger.info("Atmosphere Framework {} started.", Version.getRawVersion());
557         } catch (Throwable t) {
558             logger.error("failed to initialize atmosphere framework", t);
559 
560             if (t instanceof ServletException) {
561                 throw (ServletException) t;
562             }
563 
564             throw new ServletException(t.getCause());
565         }
566     }

568     protected void configureWebDotXmlAtmosphereHandler(ServletConfig sc) {
569         String s = sc.getInitParameter(ATMOSPHERE_HANDLER);
570         if (s != null) {
571             ClassLoader cl = Thread.currentThread().getContextClassLoader();
572             try {
573                 addAtmosphereHandler("/*", (AtmosphereHandler<?, ?>) cl.loadClass(s).newInstance());
574             } catch (Exception ex) {
575                 logger.warn("Unable to load WebSocketHandle instance", ex);
576             }
577         }
578         logger.info("Installed AtmosphereHandler {} mapped to context-path: /*", s);
579     }

581     protected void configureBroadcaster(ServletContext sc) throws ClassNotFoundException, InstantiationException, IllegalAccessException {
582 
583         if (broadcasterFactoryClassName != null) {
584             logger.info("Using BroadcasterFactory class: {}", broadcasterFactoryClassName);
585 
586             broadcasterFactory = (BroadcasterFactory) Thread.currentThread().getContextClassLoader()
587                     .loadClass(broadcasterFactoryClassName).newInstance();
588         }
589 
590         if (broadcasterFactory == null) {
591             Class<? extends Broadcaster> bc =
592                     (Class<? extends Broadcaster>) Thread.currentThread().getContextClassLoader()
593                             .loadClass(broadcasterClassName);
594 
595             broadcasterFactory = new DefaultBroadcasterFactory(bc, broadcasterLifeCyclePolicy);
596         }
597 
598         // http://java.net/jira/browse/ATMOSPHERE-157
599         sc.setAttribute(BroadcasterFactory.class.getName(), broadcasterFactory);
600 
601         config.broadcasterFactory = broadcasterFactory;
602         BroadcasterFactory.setBroadcasterFactory(broadcasterFactory, config);
603         InjectorProvider.getInjector().inject(broadcasterFactory);
604 
605         Iterator<Entry<String, AtmosphereHandlerWrapper>> i = atmosphereHandlers.entrySet().iterator();
606         AtmosphereHandlerWrapper w;
607         Entry<String, AtmosphereHandlerWrapper> e;
608         while (i.hasNext()) {
609             e = i.next();
610             w = e.getValue();
611             BroadcasterConfig broadcasterConfig = new BroadcasterConfig(broadcasterFilters, config);
612 
613             if (w.broadcaster == null) {
614                 w.broadcaster = broadcasterFactory.get();
615             } else {
616                 w.broadcaster.setBroadcasterConfig(broadcasterConfig);
617                 if (broadcasterCacheClassName != null) {
618                     BroadcasterCache cache = (BroadcasterCache) Thread.currentThread().getContextClassLoader()
619                             .loadClass(broadcasterCacheClassName).newInstance();
620                     InjectorProvider.getInjector().inject(cache);

704         try {
705             URL url = sc.getServletContext().getResource("/WEB-INF/classes/");

716                 }
717             }
718         } catch (Throwable t) {
719             throw new ServletException(t);

724      * Auto-detect Jersey when no atmosphere.xml file are specified.
725      *
726      * @param sc {@link ServletConfig}
727      * @return true if Jersey classes are detected
728      * @throws ClassNotFoundException
729      */
730     protected boolean detectSupportedFramework(ServletConfig sc) throws ClassNotFoundException, IllegalAccessException,
731             InstantiationException, NoSuchMethodException, InvocationTargetException {
732 
733         ClassLoader cl = Thread.currentThread().getContextClassLoader();
734         String broadcasterClassNameTmp = null;
735         
736         try {
737             cl.loadClass(JERSEY_CONTAINER);
738             
739             if (!isBroadcasterSpecified){
740                 broadcasterClassNameTmp = lookupDefaultBroadcasterType();
741 
742                 cl.loadClass(broadcasterClassNameTmp);
743             }
744             useStreamForFlushingComments = true;
745         } catch (Throwable t) {
746             return false;
747         }
748 
749         logger.warn("Missing META-INF/atmosphere.xml but found the Jersey runtime. Starting Jersey");
750 
751         // Jersey will handle itself the headers.
752         initParams.put(WRITE_HEADERS, "false");
753 
754         ReflectorServletProcessor rsp = new ReflectorServletProcessor();
755         if (broadcasterClassNameTmp!=null) broadcasterClassName = broadcasterClassNameTmp;
756         rsp.setServletClassName(JERSEY_CONTAINER);
757         sessionSupport(false);
758         initParams.put(DISABLE_ONSTATE_EVENT, "true");
759 
760         String mapping = sc.getInitParameter(PROPERTY_SERVLET_MAPPING);
761         if (mapping == null) {
762             mapping = "/*";
763         }
764         Class<? extends Broadcaster> bc = (Class<? extends Broadcaster>) cl.loadClass(broadcasterClassName);
765 
766         Broadcaster b = BroadcasterFactory.getDefault().get(bc, mapping);
767 
768         addAtmosphereHandler(mapping, rsp, b);
769         return true;
770     }

828     /**
829      * Load AtmosphereHandler defined under META-INF/atmosphere.xml
830      *
831      * @param stream The input stream we read from.
832      * @param c      The classloader
833      */
834     protected void loadAtmosphereDotXml(InputStream stream, URLClassLoader c)
835             throws IOException, ServletException {
836 
837         if (stream == null) {
838             return;
839         }
840 
841         AtmosphereConfigReader reader = new AtmosphereConfigReader(stream);
842 
843         Map<String, String> atmosphereHandlerNames = reader.getAtmosphereHandlers();
844         Set<Entry<String, String>> entries = atmosphereHandlerNames.entrySet();
845         for (Entry<String, String> entry : entries) {
846             AtmosphereHandler handler;
847             String handlerClassName = entry.getValue();
848             String handlerPath = entry.getKey();
849 
850             try {
851                 if (!handlerClassName.equals(ReflectorServletProcessor.class.getName())) {
852                     handler = (AtmosphereHandler) c.loadClass(handlerClassName).newInstance();
853                     InjectorProvider.getInjector().inject(handler);
854                 } else {
855                     handler = new ReflectorServletProcessor();
856                 }
857 
858                 logger.info("Installed AtmosphereHandler {} mapped to context-path: {}", handler, handlerPath);
859 
860                 AtmosphereHandlerWrapper wrapper = new AtmosphereHandlerWrapper(handler);
861                 atmosphereHandlers.put(handlerPath, wrapper);
862                 boolean isJersey = false;
863                 for (Property p : reader.getProperty(handlerPath)) {
864                     if (p.value != null && p.value.indexOf("jersey") != -1) {
865                         isJersey = true;
866                         initParams.put(DISABLE_ONSTATE_EVENT, "true");
867                         useStreamForFlushingComments = true;
868                         broadcasterClassName = lookupDefaultBroadcasterType();
869                     }
870                     IntrospectionUtils.setProperty(handler, p.name, p.value);
871                 }
872 
873                 config.supportSession = !isJersey;
874 
875                 if (!reader.supportSession().equals("")) {
876                     sessionSupport(Boolean.valueOf(reader.supportSession()));
877                 }
878 
879                 for (Property p : reader.getProperty(handlerPath)) {
880                     IntrospectionUtils.addProperty(handler, p.name, p.value);
881                 }
882 
883                 String broadcasterClass = reader.getBroadcasterClass(handlerPath);
884                 /**
885                  * If there is more than one AtmosphereHandler defined, their Broadcaster
886                  * may clash each other with the BroadcasterFactory. In that case we will use the
887                  * last one defined.
888                  */
889                 if (broadcasterClass != null) {
890                     broadcasterClassName = broadcasterClass;
891                     ClassLoader cl = Thread.currentThread().getContextClassLoader();
892                     Class<? extends Broadcaster> bc = (Class<? extends Broadcaster>) cl.loadClass(broadcasterClassName);
893                     wrapper.broadcaster = BroadcasterFactory.getDefault().get(bc, handlerPath);
894                 }
895 
896                 String bc = reader.getBroadcasterCache(handlerPath);
897                 if (bc != null) {
898                     broadcasterCacheClassName = bc;
899                 }
900 
901                 if (reader.getCometSupportClass() != null) {
902                     cometSupport = (CometSupport) c.loadClass(reader.getCometSupportClass())
903                             .getDeclaredConstructor(new Class[]{AtmosphereConfig.class})
904                             .newInstance(new Object[]{config});
905                 }
906 
907                 if (reader.getBroadcastFilterClasses() != null) {
908                     broadcasterFilters = reader.getBroadcastFilterClasses();
909                 }
910 
911             } catch (Throwable t) {
912                 logger.warn("unable to load AtmosphereHandler class: " + handlerClassName, t);
913                 throw new ServletException(t);
914             }
915         }
916     }

987     protected void loadAtmosphereHandlersFromPath(URLClassLoader classloader, String realPath) {
988         File file = new File(realPath);
989 
990         if (file.isDirectory()) {
991             getFiles(file);
992 
993             for (String className : possibleAtmosphereHandlersCandidate) {
994                 try {
995                     className = className.replace('\\', '/');
996                     className = className.replaceFirst("^.*/(WEB-INF|target)/(test-)?classes/(.*)\\.class", "$3").replace("/", ".");
997                     Class<?> clazz = classloader.loadClass(className);
998 
999                     if (AtmosphereHandler.class.isAssignableFrom(clazz)) {
1000                         AtmosphereHandler handler = (AtmosphereHandler) clazz.newInstance();
1001                         InjectorProvider.getInjector().inject(handler);
1002                         atmosphereHandlers.put("/" + handler.getClass().getSimpleName(),
1003                                 new AtmosphereHandlerWrapper(handler, null));
1004                         logger.info("Installed AtmosphereHandler {} mapped to context-path: {}", handler, handler.getClass().getName());
1005                     }
1006                 } catch (Throwable t) {
1007                     logger.trace("failed to load class as an AtmosphereHandler: " + className, t);
1008                 }
1009             }
1010         }
1011     }

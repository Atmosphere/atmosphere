47  import org.slf4j.Logger;
48  import org.slf4j.LoggerFactory;

50  import javax.servlet.ServletException;
51  import javax.servlet.http.HttpServletRequest;
52  import javax.servlet.http.HttpServletResponse;
53  import java.io.IOException;
54  import java.util.concurrent.ConcurrentLinkedQueue;

56  /**
57   * Comet Portable Runtime implementation on top of Tomcat AIO.
58   *
59   * @author Jeanfrancois Arcand
60   */
61  public class TomcatCometSupport extends AsynchronousProcessor {
62  
63      private static final Logger logger = LoggerFactory.getLogger(TomcatCometSupport.class);
64  
65      public static final String COMET_EVENT = "CometEvent";
66  
67      private static final IllegalStateException unableToDetectComet
68              = new IllegalStateException(unableToDetectComet());
69  
70      // Client disconnection is broken on Tomcat.
71      private final ConcurrentLinkedQueue<CometEvent> resumed
72              = new ConcurrentLinkedQueue<CometEvent>();
73  
74      public TomcatCometSupport(AtmosphereConfig config) {
75          super(config);
76      }
77  
78      /**
79       * Invoked by the Tomcat AIO when a Comet request gets detected.
80       *
81       * @param req the {@link javax.servlet.http.HttpServletRequest}
82       * @param res the {@link HttpServletResponse}
83       * @throws java.io.IOException
84       * @throws javax.servlet.ServletException
85       */
86      public Action service(HttpServletRequest req, HttpServletResponse res)
87              throws IOException, ServletException {
88  
89          CometEvent event = (CometEvent) req.getAttribute(COMET_EVENT);
90  
91          // Comet is not enabled.
92          if (event == null) {
93              throw unableToDetectComet;
94          }
95  
96          Action action = null;
97          // For now, we are just interested in CometEvent.READ
98          if (event.getEventType() == EventType.BEGIN) {
99              action = suspended(req, res);
100             if (action.type == Action.TYPE.SUSPEND) {
101                 logger.debug("Suspending response: {}", res);
102 
103                 // Do nothing except setting the times out
104                 try {
105                     if (action.timeout != -1) {
106                         event.setTimeout((int) action.timeout);
107                     } else {
108                         event.setTimeout(Integer.MAX_VALUE);
109                     }
110                 } catch (UnsupportedOperationException ex) {
111                     // Swallow s Tomcat APR isn't supporting time out
112                     // TODO: Must implement the same functionality using a Scheduler
113                 }
114             } else if (action.type == Action.TYPE.RESUME) {
115                 logger.debug("Resuming response: {}", res);
116                 event.close();
117             } else {
118                 event.close();
119             }
120         } else if (event.getEventType() == EventType.READ) {
121             // Not implemented
122         } else if (event.getEventSubType() == CometEvent.EventSubType.CLIENT_DISCONNECT) {
123             logger.debug("Client closed connection: response: {}", res);
124 
125             if (!resumed.remove(event)) {
126                 logger.debug("Client closed connection: response: {}", res);
127                 action = cancelled(req, res);
128             } else {
129                 logger.debug("Cancelling response: {}", res);
130             }
131 
132             event.close();
133         } else if (event.getEventSubType() == CometEvent.EventSubType.TIMEOUT) {
134             logger.debug("Timing out response: {}", res);
135 
136             action = timedout(req, res);
137             event.close();
138         } else if (event.getEventType() == EventType.ERROR) {
139             event.close();
140         } else if (event.getEventType() == EventType.END) {
141             if (!resumed.remove(event)) {
142                 logger.debug("Client closed connection: response: {}", res);
143                 action = cancelled(req, res);
144             } else {
145                 logger.debug("Cancelling response: {}", res);
146             }
147 
148             event.close();
149         }
150         return action;
151     }
152 
153     /**
154      * {@inheritDoc}
155      */
156     @Override
157     public void action(AtmosphereResourceImpl resource) {
158         super.action(resource);
159         if (resource.action().type == Action.TYPE.RESUME && resource.isInScope()) {
160             try {
161                 CometEvent event = (CometEvent) resource.getRequest().getAttribute(COMET_EVENT);
162                 if (event == null) return;
163                 resumed.offer(event);
164 
165                 // Resume without closing the underlying suspended connection.
166                 if (config.getInitParameter(ApplicationConfig.RESUME_AND_KEEPALIVE) == null
167                         || config.getInitParameter(ApplicationConfig.RESUME_AND_KEEPALIVE).equalsIgnoreCase("false")) {
168                     event.close();
169                 }
170             } catch (IOException ex) {
171                 logger.debug("action failed", ex);
172             }
173         }
174     }
175 
176     @Override
177     public Action cancelled(HttpServletRequest req, HttpServletResponse res)
178             throws IOException, ServletException {
179 
180         Action action = super.cancelled(req, res);
181         if (req.getAttribute(MAX_INACTIVE) != null && Long.class.cast(req.getAttribute(MAX_INACTIVE)) == -1) {
182             CometEvent event = (CometEvent) req.getAttribute(COMET_EVENT);
183             if (event == null) return action;
184             resumed.offer(event);
185             event.close();
186         }
187         return action;
188     }
189 
190     /**
191      * Tomcat was unable to detect Atmosphere's CometProcessor implementation.
192      *
193      * @return an error message describing how to fix the issue.
194      */
195     private static String unableToDetectComet() {
196         StringBuilder sb = new StringBuilder();
197         sb.append("Tomcat failed to detect this is a Comet application because context.xml ");
198         sb.append("is missing or the Http11NioProtocol Connector is not enabled.");
199         sb.append("\nIf that's not the case, you can also remove META-INF/context.xml and WEB-INF/lib/atmosphere-compat-tomcat.jar");
200         return sb.toString();
201     }

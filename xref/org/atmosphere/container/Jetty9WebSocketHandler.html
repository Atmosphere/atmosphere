26      public Jetty9WebSocketHandler(HttpServletRequest request, AtmosphereFramework framework, WebSocketProcessor webSocketProcessor) {
27          this.framework = framework;
28          this.request = cloneRequest(request);
29          this.webSocketProcessor = webSocketProcessor;
30      }

32      private AtmosphereRequest cloneRequest(final HttpServletRequest request) {
33          try {
34              AtmosphereRequest r = AtmosphereRequest.wrap(request);
35              return AtmosphereRequest.cloneRequest(r, false, framework.getAtmosphereConfig().isSupportSession(), false);
36          } catch (Exception ex) {
37              logger.error("", ex);
38              throw new RuntimeException("Invalid WebSocket Request");
39          }
40      }

42      @Override
43      public void onWebSocketBinary(byte[] data, int offset, int length) {
44          logger.trace("WebSocket.onMessage (bytes)");
45          webSocketProcessor.invokeWebSocketProtocol(webSocket, data, offset, length);
46      }

48      @Override
49      public void onWebSocketClose(int closeCode, String s) {
50          logger.trace("onClose {}:{}", closeCode, s);
51          try {
52              webSocketProcessor.close(webSocket, closeCode);
53          } finally {
54              request.destroy();
55          }
56      }
57  
58      @Override
59      public void onWebSocketConnect(Session session) {
60          logger.trace("WebSocket.onOpen.");
61          webSocket = new Jetty9WebSocket(session, framework.getAtmosphereConfig());
62          try {
63              webSocketProcessor.open(webSocket, request, AtmosphereResponse.newInstance(framework.getAtmosphereConfig(), request, webSocket));
64          } catch (Exception e) {
65              logger.warn("Failed to connect to WebSocket", e);
66          }
67      }
68  
69      @Override
70      public void onWebSocketError(Throwable e) {
71          logger.error("", e);
72          webSocketProcessor.notifyListener(webSocket,
73                  new WebSocketEventListener.WebSocketEvent(e, WebSocketEventListener.WebSocketEvent.TYPE.EXCEPTION, webSocket));
74      }
75  
76      @Override
77      public void onWebSocketText(String s) {

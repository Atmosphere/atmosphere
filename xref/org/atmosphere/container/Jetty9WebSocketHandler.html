16  package org.atmosphere.container;
17  
18  import org.atmosphere.container.version.Jetty9WebSocket;
19  import org.atmosphere.cpr.AtmosphereFramework;
20  import org.atmosphere.cpr.AtmosphereRequest;
21  import org.atmosphere.cpr.AtmosphereRequestImpl;
22  import org.atmosphere.cpr.AtmosphereResponseImpl;
23  import org.atmosphere.websocket.WebSocket;
24  import org.atmosphere.websocket.WebSocketProcessor;
25  import org.eclipse.jetty.websocket.api.Session;
26  import org.eclipse.jetty.websocket.api.WebSocketListener;
27  import org.eclipse.jetty.websocket.servlet.ServletUpgradeRequest;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import javax.servlet.http.HttpServletRequest;
32  import java.lang.reflect.Field;
33  
34  import static org.atmosphere.cpr.ApplicationConfig.PROPERTY_SESSION_CREATE;
35  
36  public class Jetty9WebSocketHandler implements WebSocketListener {
37  
38      private static final Logger logger = LoggerFactory.getLogger(Jetty9WebSocketHandler.class);
39  
40      private AtmosphereRequest request;
41      private final AtmosphereFramework framework;
42      private final WebSocketProcessor webSocketProcessor;
43      private WebSocket webSocket;
44      private static final boolean jetty93Up;
45  
46      static {
47          Exception ex = null;
48          try {
49              Class.forName("org.eclipse.jetty.websocket.api.WebSocketFrameListener");
50          } catch (ClassNotFoundException e) {
51              ex = e;
52          } finally {
53              jetty93Up = ex == null ? true : false;
54          }
55      }
56  
57      public Jetty9WebSocketHandler(HttpServletRequest request, AtmosphereFramework framework, WebSocketProcessor webSocketProcessor) {
58          this.framework = framework;
59          this.request = cloneRequest(request);
60          this.webSocketProcessor = webSocketProcessor;
61      }
62  
63      private AtmosphereRequest cloneRequest(final HttpServletRequest request) {
64          try {
65              AtmosphereRequest r = AtmosphereRequestImpl.wrap(request);
66              return AtmosphereRequestImpl.cloneRequest(r, false, false, false, framework.getAtmosphereConfig().getInitParameter(PROPERTY_SESSION_CREATE, true));
67          } catch (Exception ex) {
68              logger.error("", ex);
69              throw new RuntimeException("Invalid WebSocket Request");
70          }
71      }
72  
73      @Override
74      public void onWebSocketBinary(byte[] data, int offset, int length) {
75          logger.trace("WebSocket.onMessage (bytes)");
76          webSocketProcessor.invokeWebSocketProtocol(webSocket, data, offset, length);
77      }
78  
79      @Override
80      public void onWebSocketClose(int closeCode, String s) {
81          logger.trace("onClose {}:{}", closeCode, s);
82          try {
83              webSocketProcessor.close(webSocket, closeCode);
84          } finally {
85              request.destroy();
86          }
87      }
88  
89      @Override
90      public void onWebSocketConnect(Session session) {
91          logger.trace("WebSocket.onOpen.");
92          webSocket = new Jetty9WebSocket(session, framework.getAtmosphereConfig());
93  
94          /**
95           * https://github.com/Atmosphere/atmosphere/issues/1998
96           * The Original Jetty Request will be recycled, hence we must loads its content in memory. We can't do that before
97           * as it break Jetty 9.3.0 upgrade process.
98           *
99           * This is a performance regression from 9.2 as we need to clone again the request. 9.3.0+ should use jsr 356!
100          */
101         if (jetty93Up) {
102             HttpServletRequest r = originalRequest(session);
103             if (r != null) {
104                 // We close except the session which we can still reach.
105                 request = AtmosphereRequestImpl.cloneRequest(r, true, false, false, framework.getAtmosphereConfig().getInitParameter(PROPERTY_SESSION_CREATE, true));
106             } else {
107                 // Bad Bad Bad
108                 request = AtmosphereRequestImpl.cloneRequest(r, true, true, false, framework.getAtmosphereConfig().getInitParameter(PROPERTY_SESSION_CREATE, true));
109             }
110         }
111 
112         try {
113             webSocketProcessor.open(webSocket, request, AtmosphereResponseImpl.newInstance(framework.getAtmosphereConfig(), request, webSocket));
114         } catch (Exception e) {
115             logger.warn("Failed to connect to WebSocket", e);
116         }
117     }
118 
119     @Override
120     public void onWebSocketError(Throwable e) {
121         logger.error("{}", e);
122         onWebSocketClose(1006, "Unexpected error");
123     }
124 
125     @Override
126     public void onWebSocketText(String s) {
127         logger.trace("WebSocket.onMessage (bytes)");
128         webSocketProcessor.invokeWebSocketProtocol(webSocket, s);
129     }
130 
131     private HttpServletRequest originalRequest(Session session) {
132         try {
133             // Oh boy...
134             ServletUpgradeRequest request = (ServletUpgradeRequest) session.getUpgradeRequest();
135             Field[] fields = ServletUpgradeRequest.class.getDeclaredFields();
136             for (Field f : fields) {
137                 f.setAccessible(true);
138                 Object o = f.get(request);
139                 if (o instanceof HttpServletRequest) {
140                     return HttpServletRequest.class.cast(o);
141                 }
142             }
143         } catch (Exception ex) {
144             logger.error("", ex);
145         }
146         return null;
147     }

41  /**
42   * Comet Portable Runtime implementation on top of Grizzly 1.5 and up.
43   *
44   * @author Jeanfrancois Arcand
45   */
46  public class Grizzly2CometSupport extends AsynchronousProcessor {
47  
48      private static final Logger logger = LoggerFactory.getLogger(Grizzly2CometSupport.class);
49  
50      private static final String ATMOSPHERE = "/atmosphere";
51  
52      private String atmosphereCtx = "";
53  
54      public Grizzly2CometSupport(AtmosphereConfig config) {
55          super(config);
56      }
57  
58      /**
59       * Init Grizzly's {@link org.glassfish.grizzly.comet.CometContext} that will be used to suspend and
60       * resume the response.
61       *
62       * @param sc the {@link javax.servlet.ServletContext}
63       * @throws javax.servlet.ServletException
64       */
65      @Override
66      public void init(ServletConfig sc) throws ServletException {
67          super.init(sc);
68  
69          atmosphereCtx = sc.getServletContext().getContextPath() + ATMOSPHERE;
70  
71          CometEngine cometEngine = CometEngine.getEngine();
72          CometContext context = cometEngine.register(atmosphereCtx);
73          context.setExpirationDelay(-1);
74          logger.debug("Created CometContext for atmosphere context: {}", atmosphereCtx);
75      }
76  
77      /**
78       * {@inheritDoc}
79       */
80      public Action service(AtmosphereRequest req, AtmosphereResponse res)
81              throws IOException, ServletException {
82  
83          CometContext ctx = CometEngine.getEngine().getCometContext(atmosphereCtx);
84          Action action = suspended(req, res);
85          if (action.type() == Action.TYPE.SUSPEND) {
86              suspend(ctx, action, req, res);
87          } else if (action.type() == Action.TYPE.RESUME) {
88              resume(req, ctx);
89          }
90          return action;
91      }

93      /**
94       * Suspend the response
95       *
96       * @param ctx
97       * @param action
98       * @param req
99       * @param res
100      */
101     private void suspend(CometContext ctx, Action action, AtmosphereRequest req, AtmosphereResponse res) {
102         VoidCometHandler c = new VoidCometHandler(req, res);
103         ctx.setExpirationDelay(action.timeout());
104         ctx.addCometHandler(c);
105         req.setAttribute(ATMOSPHERE, c.hashCode());
106         ctx.addAttribute("Time", System.currentTimeMillis());
107 
108         if (supportSession()) {
109             // Store as well in the session in case the resume operation
110             // happens outside the AtmosphereHandler.onStateChange scope.
111             req.getSession().setAttribute(ATMOSPHERE, c.hashCode());
112         }
113     }
114 
115     /**
116      * Resume the underlying response,
117      *
118      * @param req an {@link org.atmosphere.cpr.AtmosphereRequest}
119      * @param ctx a {@link org.glassfish.grizzly.comet.CometContext}
120      */
121     private void resume(AtmosphereRequest req, CometContext ctx) {
122 
123         if (req.getAttribute(ATMOSPHERE) == null) {
124             return;
125         }
126 
127         CometHandler handler = getCometHandler(ctx, (Integer) req.getAttribute(ATMOSPHERE));
128         req.removeAttribute(ATMOSPHERE);
129 
130         if (handler == null && supportSession() && req.getSession(false) != null) {
131             handler = getCometHandler(ctx, (Integer) req.getSession(false).getAttribute(ATMOSPHERE));
132             req.getSession().removeAttribute(ATMOSPHERE);
133         }
134 
135         if (handler != null && (config.getInitParameter(ApplicationConfig.RESUME_AND_KEEPALIVE) == null
136                 || config.getInitParameter(ApplicationConfig.RESUME_AND_KEEPALIVE).equalsIgnoreCase("false"))) {
137             try {
138                 ctx.resumeCometHandler(handler);
139             } catch (IOException e) {
140                logger.error("Error resuming CometHandler", e);
141             }
142         }
143     }
144 
145     /**
146      * {@inheritDoc}
147      */
148     @Override
149     public void action(AtmosphereResourceImpl r) {
150         super.action(r);
151         if (r.action().type() == Action.TYPE.RESUME && r.isInScope()) {
152             CometContext ctx = CometEngine.getEngine().getCometContext(atmosphereCtx);
153             resume(r.getRequest(), ctx);
154         }
155     }
156 
157     @Override
158     public Action cancelled(AtmosphereRequest req, AtmosphereResponse res)
159             throws IOException, ServletException {
160 
161         Action action = super.cancelled(req, res);
162         if (req.getAttribute(MAX_INACTIVE) != null && Long.class.cast(req.getAttribute(MAX_INACTIVE)) == -1) {
163             resume(req, CometEngine.getEngine().getCometContext(atmosphereCtx));
164         }
165         return action;
166     }
167 
168     private static CometHandler getCometHandler(CometContext context,
169                                                 int handlerId) {
170         List<CometHandler> handlers = context.getCometHandlers();
171         for (CometHandler handler : handlers) {
172             if (handler.hashCode() == handlerId) {
173                 return handler;
174             }
175         }
176         return null;
177     }

179     /**
180      * Void {@link org.glassfish.grizzly.comet.CometHandler}, which delegate the processing of the
181      * {@link org.atmosphere.cpr.AtmosphereRequest} to an {@link org.atmosphere.cpr.AtmosphereResourceImpl}.
182      */
183     private class VoidCometHandler implements CometHandler {
184 
185         AtmosphereRequest req;
186         AtmosphereResponse res;
187         private Response grizzlyResponse;
188         CometContext cometContext;
189 
190         public VoidCometHandler(AtmosphereRequest req, AtmosphereResponse res) {
191             this.req = req;
192             this.res = res;
193         }

198         public void attach(Object o) {
199         }

204         public void onEvent(CometEvent ce) throws IOException {
205         }

210         public void onInitialize(CometEvent ce) throws IOException {
211         }

214          * {@inheritDoc}

217         }
218 
219         /**
220          * {@inheritDoc}
221          */
222         public synchronized void onInterrupt(CometEvent ce) throws IOException {
223             long timeStamp = (Long) ce.getCometContext().getAttribute("Time");
224             try {
225                 if (ce.getCometContext().getExpirationDelay() > 0
226                         && (System.currentTimeMillis() - timeStamp) >= ce.getCometContext().getExpirationDelay()) {
227                     timedout(req, res);
228                 } else {
229                     cancelled(req, res);
230                 }
231             } catch (ServletException ex) {
232                 logger.warn("onInterrupt() encountered exception", ex);
233             }
234         }
235 
236         @Override
237         public Response getResponse() {
238             return grizzlyResponse;
239         }
240 
241         @Override
242         public void setResponse(Response response) {
243             grizzlyResponse = response;
244         }

246         @Override
247         public CometContext getCometContext() {
248             return cometContext;
249         }
250 
251         @Override
252         public void setCometContext(CometContext cometContext) {
253             this.cometContext = cometContext;
254         }
255     }

16  package org.atmosphere.container;
17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.ApplicationConfig;
20  import org.atmosphere.cpr.AsyncSupport;
21  import org.atmosphere.cpr.AsynchronousProcessor;
22  import org.atmosphere.cpr.AtmosphereConfig;
23  import org.atmosphere.cpr.AtmosphereRequest;
24  import org.atmosphere.cpr.AtmosphereResourceImpl;
25  import org.atmosphere.cpr.AtmosphereResponse;
26  import org.atmosphere.cpr.FrameworkConfig;
27  import org.eclipse.jetty.continuation.Continuation;
28  import org.eclipse.jetty.continuation.ContinuationListener;
29  import org.eclipse.jetty.continuation.ContinuationSupport;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import javax.servlet.ServletException;
34  import javax.servlet.ServletRequest;
35  import java.io.IOException;
36  
37  /**
38   * Comet Portable Runtime implementation on top of Jetty's Continuation.
39   *
40   * @author Jeanfrancois Arcand
41   */
42  public class Jetty7CometSupport extends AsynchronousProcessor {
43  
44      private static final Logger logger = LoggerFactory.getLogger(Jetty7CometSupport.class);
45  
46      public Jetty7CometSupport(AtmosphereConfig config) {
47          super(config);
48      }
49  
50      @Override
51      public Action service(final AtmosphereRequest req, final AtmosphereResponse res) throws IOException, ServletException {
52          Action action = null;
53  
54          Continuation c = (Continuation) req.getAttribute(Continuation.class.getName());
55  
56          if (c == null || c.isInitial()) {
57              action = suspended(req, res);
58              if (action.type() == Action.TYPE.SUSPEND && req.getAttribute(FrameworkConfig.CANCEL_SUSPEND_OPERATION) == null) {
59                  c = getContinuation(req);
60                  req.setAttribute(Continuation.class.getName(), c);
61  
62                  if (action.timeout() != -1) {
63                      c.setTimeout(action.timeout());
64                  } else {
65                      // Jetty 7 does something really weird if you set it to
66                      // Long.MAX_VALUE, which is to resume automatically.
67                      c.setTimeout(Integer.MAX_VALUE);
68                  }
69  
70                  c.setAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE, req.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE));
71                  c.addContinuationListener(new ContinuationListener() {
72  
73                      @Override
74                      public void onComplete(Continuation continuation) {
75                          AtmosphereResourceImpl r = (AtmosphereResourceImpl) req.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
76                          if (r != null) {
77                              try {
78                                  r.cancel();
79                              } catch (IOException e) {
80                              }
81                          }
82                      }
83  
84                      @Override
85                      public void onTimeout(Continuation continuation) {
86                          AtmosphereResourceImpl r = (AtmosphereResourceImpl) req.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
87                          if (r != null) {
88                              try {
89                                  timedout(r.getRequest(), r.getResponse());
90                              } catch (Throwable t) {
91                                  logger.error("", t);
92                              }
93                          } else {
94                              logger.trace("AtmosphereResource was null");
95                          }
96                          try {
97                              continuation.complete();
98                          } catch (Throwable t) {
99                          }
100                     }
101                 });
102 
103                 if (req.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE) != null) {
104                     c.suspend(res);
105                 }
106             } else if (action.type() == Action.TYPE.RESUME) {
107                 // If resume occurs during a suspend operation, stop processing.
108                 Boolean resumeOnBroadcast = (Boolean) req.getAttribute(ApplicationConfig.RESUME_ON_BROADCAST);
109                 if (resumeOnBroadcast != null && resumeOnBroadcast) {
110                     return action;
111                 }
112 
113                 c = (Continuation) req.getAttribute(Continuation.class.getName());
114                 if (c != null) {
115                     if (c.isSuspended()) {
116                         try {
117                             c.complete();
118                         } catch (IllegalStateException ex) {
119                             logger.trace("Continuation.complete()", ex);
120                         } finally {
121                             resumed(req, res);
122                         }
123                     }
124                 }
125             }
126         }
127         return action;
128     }
129 
130     protected Continuation getContinuation(AtmosphereRequest req) {
131         return ContinuationSupport.getContinuation(req);
132     }
133 
134     @Override
135     public void action(AtmosphereResourceImpl r) {
136         super.action(r);
137         complete(r);
138     }
139 
140     @Override
141     public AsyncSupport complete(AtmosphereResourceImpl r) {
142         ServletRequest request = r.getRequest(false);
143         while (request != null) {
144             Continuation c = (Continuation) request.getAttribute(Continuation.class.getName());
145             if (c != null) {
146                 try {
147                     if (c.isSuspended()) {
148                         c.complete();
149                     }
150                 } catch (IllegalStateException ex) {
151                     logger.trace("c.complete()", ex);
152                 } finally {
153                     r.getRequest(false).setAttribute(FrameworkConfig.CANCEL_SUSPEND_OPERATION, true);
154                 }
155                 request.removeAttribute(Continuation.class.getName());
156                 return this;
157             } else {
158                 return this;
159             }
160         }
161         return this;
162     }
163 }

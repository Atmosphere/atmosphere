16  package org.atmosphere.container;
17  
18  import org.atmosphere.container.version.WebLogicWebSocket;
19  import org.atmosphere.cpr.ApplicationConfig;
20  import org.atmosphere.cpr.AtmosphereConfig;
21  import org.atmosphere.cpr.AtmosphereRequest;
22  import org.atmosphere.cpr.AtmosphereRequestImpl;
23  import org.atmosphere.cpr.AtmosphereResponseImpl;
24  import org.atmosphere.cpr.WebSocketProcessorFactory;
25  import org.atmosphere.websocket.WebSocket;
26  import org.atmosphere.websocket.WebSocketEventListener;
27  import org.atmosphere.websocket.WebSocketProcessor;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  import weblogic.websocket.ClosingMessage;
31  import weblogic.websocket.WSHandshakeRequest;
32  import weblogic.websocket.WSHandshakeResponse;
33  import weblogic.websocket.WebSocketConnection;
34  import weblogic.websocket.WebSocketContext;
35  import weblogic.websocket.WebSocketListener;
36  
37  import javax.servlet.ServletContext;
38  import java.io.IOException;
39  
40  import static org.atmosphere.cpr.ApplicationConfig.PROPERTY_SESSION_CREATE;
41  
42  @weblogic.websocket.annotation.WebSocket(pathPatterns = "/ws/*", timeout = -1, maxMessageSize = 8192)
43  public class WeblogicWebSocketHandler implements WebSocketListener {
44  
45      private final Logger logger = LoggerFactory.getLogger(WeblogicWebSocketHandler.class);
46      private WebSocketProcessor webSocketProcessor;
47      private AtmosphereConfig config;
48      private Integer maxTextBufferSize;
49      private int webSocketWriteTimeout;
50      private final ThreadLocal<WSHandshakeRequest> request = new ThreadLocal<WSHandshakeRequest>();
51  
52      @Override
53      public void init(WebSocketContext webSocketContext) {
54          logger.info("WebSocketContext initialized {}", webSocketContext);
55          configure(webSocketContext.getServletContext());
56      }
57  
58      @Override
59      public void destroy() {
60  
61      }
62  
63      @Override
64      public boolean accept(WSHandshakeRequest wsHandshakeRequest, WSHandshakeResponse wsHandshakeResponse) {
65          request.set(wsHandshakeRequest);
66          return true;
67      }
68  
69      @Override
70      public void onOpen(WebSocketConnection webSocketConnection) {
71          if (webSocketWriteTimeout != -1)
72              webSocketConnection.getWebSocketContext().setTimeoutSecs(webSocketWriteTimeout);
73          if (maxTextBufferSize != -1) webSocketConnection.getWebSocketContext().setMaxMessageSize(maxTextBufferSize);
74  
75          WebSocket webSocket = new WebLogicWebSocket(webSocketConnection, config);
76          // TODO: Dangerous
77          webSocketConnection.getWebSocketContext().getServletContext().setAttribute(webSocketConnection.toString(), webSocket);
78  
79          AtmosphereRequest ar = AtmosphereRequestImpl.cloneRequest(request.get(), true, true, true, config.getInitParameter(PROPERTY_SESSION_CREATE, true));
80          // https://github.com/Atmosphere/atmosphere/issues/1854
81          // We need to force processing of the query string.
82          ar.queryString(ar.getQueryString());
83          request.set(null);
84          try {
85              webSocketProcessor.open(webSocket, ar, AtmosphereResponseImpl.newInstance(config, ar, webSocket));
86          } catch (IOException e) {
87              logger.error("{}", e);
88          }
89  
90      }
91  
92      @Override
93      public void onMessage(WebSocketConnection webSocketConnection, String s) {
94          WebSocket webSocket = (WebSocket) webSocketConnection.getWebSocketContext().getServletContext().getAttribute(webSocketConnection.toString());
95  
96          webSocketProcessor.invokeWebSocketProtocol(webSocket, s);
97      }
98  
99      @Override
100     public void onMessage(WebSocketConnection webSocketConnection, byte[] bytes) {
101         WebSocket webSocket = (WebSocket) webSocketConnection.getWebSocketContext().getServletContext().getAttribute(webSocketConnection.toString());
102 
103         webSocketProcessor.invokeWebSocketProtocol(webSocket, bytes, 0, bytes.length);
104     }
105 
106     @Override
107     public void onFragment(WebSocketConnection webSocketConnection, boolean b, String s) {
108         logger.trace("Warning, Fragment not supported");
109         onMessage(webSocketConnection, s);
110     }
111 
112     @Override
113     public void onFragment(WebSocketConnection webSocketConnection, boolean b, byte[] bytes) {
114         logger.trace("Warning, Fragment not supported");
115         onMessage(webSocketConnection, bytes);
116     }
117 
118     @Override
119     public void onPing(WebSocketConnection webSocketConnection, byte[] bytes) {
120         logger.trace("Warning, Fragment not supported");
121     }
122 
123     @Override
124     public void onPong(WebSocketConnection webSocketConnection, byte[] bytes) {
125         logger.trace("Warning, Fragment not supported");
126     }
127 
128     @Override
129     public void onTimeout(WebSocketConnection webSocketConnection) {
130         WebSocket webSocket = (WebSocket) webSocketConnection.getWebSocketContext().getServletContext().getAttribute(webSocketConnection.toString());
131 
132         webSocketProcessor.close(webSocket, 1000);
133     }
134 
135     @Override
136     public void onError(WebSocketConnection webSocketConnection, Throwable throwable) {
137         WebSocket webSocket = (WebSocket) webSocketConnection.getWebSocketContext().getServletContext().getAttribute(webSocketConnection.toString());
138         webSocketConnection.getWebSocketContext().getServletContext().removeAttribute(webSocketConnection.toString());
139 
140         webSocketProcessor.notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent<Throwable>(throwable, WebSocketEventListener.WebSocketEvent.TYPE.EXCEPTION, webSocket));
141     }
142 
143     @Override
144     public void onClose(WebSocketConnection webSocketConnection, ClosingMessage closingMessage) {
145         WebSocket webSocket = (WebSocket) webSocketConnection.getWebSocketContext().getServletContext().getAttribute(webSocketConnection.toString());
146         webSocketConnection.getWebSocketContext().getServletContext().removeAttribute(webSocketConnection.toString());
147 
148         webSocketProcessor.close(webSocket, closingMessage.getStatusCode());
149     }
150 
151     private void configure(ServletContext servletContext) {
152         config = (AtmosphereConfig) servletContext.getAttribute(AtmosphereConfig.class.getName());
153         webSocketProcessor = WebSocketProcessorFactory.getDefault().getWebSocketProcessor(config.framework());
154 
155         String s = config.getInitParameter(ApplicationConfig.WEBSOCKET_IDLETIME);
156         if (s != null) {
157             webSocketWriteTimeout = Integer.valueOf(s);
158         } else {
159             webSocketWriteTimeout = -1;
160         }
161 
162         s = config.getInitParameter(ApplicationConfig.WEBSOCKET_MAXTEXTSIZE);
163         if (s != null) {
164             maxTextBufferSize = Integer.valueOf(s);
165         } else {
166             maxTextBufferSize = -1;
167         }
168     }
169 }

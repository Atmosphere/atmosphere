37  package org.atmosphere.container.version;
38  
39  import org.atmosphere.websocket.WebSocketAdapter;
40  import org.atmosphere.websocket.WebSocket;
41  import org.eclipse.jetty.websocket.WebSocket.Outbound;
42  import org.slf4j.Logger;
43  import org.slf4j.LoggerFactory;
44  
45  import java.io.IOException;
46  
47  /**
48   * Jetty 7.1/2 & 8 < M3 WebSocket support.
49   *
50   * @author Jeanfrancois Arcand
51   */
52  public class JettyWebSocket extends WebSocketAdapter {
53  
54      private static final Logger logger = LoggerFactory.getLogger(JettyWebSocket.class);
55      private final Outbound outbound;
56      private final byte frame = 0x00;
57  
58      public JettyWebSocket(Outbound outbound) {
59          this.outbound = outbound;
60      }
61  
62      /**
63       * {@inheritDoc}
64       */
65      @Override
66      public void writeError(int errorCode, String message) throws IOException {
67          logger.debug("{} {}", errorCode, message);
68      }
69  
70      /**
71       * {@inheritDoc}
72       */
73      @Override
74      public void redirect(String location) throws IOException {
75          logger.error("redirect not supported");
76      }
77  
78      /**
79       * {@inheritDoc}
80       */
81      @Override
82      public void write(String data) throws IOException {
83          if (!outbound.isOpen()) throw new IOException("Connection remotely closed");
84          logger.trace("WebSocket.write()");
85          outbound.sendMessage(frame, data);
86      }
87  
88      /**
89       * {@inheritDoc}
90       */
91      @Override
92      public void write(byte[] data) throws IOException {
93          if (!outbound.isOpen()) throw new IOException("Connection remotely closed");
94          logger.trace("WebSocket.write()");
95          outbound.sendMessage(frame, data, 0, data.length);
96      }
97  
98      /**
99       * {@inheritDoc}
100      */
101     @Override
102     public void write(byte[] data, int offset, int length) throws IOException {
103         if (!outbound.isOpen()) throw new IOException("Connection remotely closed");
104         logger.trace("WebSocket.write()");
105         outbound.sendMessage(frame, data, offset, length);
106     }
107 
108     /**
109      * {@inheritDoc}
110      */
111     @Override
112     public void close() throws IOException {
113         outbound.disconnect();
114     }
115 
116 }

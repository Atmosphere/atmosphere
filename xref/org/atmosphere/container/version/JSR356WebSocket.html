21  import org.atmosphere.cpr.AtmosphereResource;

24  import org.atmosphere.websocket.WebSocket;

26  import org.slf4j.LoggerFactory;

28  import javax.websocket.SendHandler;
29  import javax.websocket.SendResult;
30  import javax.websocket.Session;
31  import java.io.IOException;
32  import java.nio.ByteBuffer;
33  import java.util.concurrent.Semaphore;
34  
35  /**
36   * Asynchronous based {@link Session} websocket
37   *
38   * @author Jeanfrancois Arcand
39   */
40  public class JSR356WebSocket extends WebSocket {
41  
42      private final Logger logger = LoggerFactory.getLogger(JSR356WebSocket.class);
43      private final Session session;
44      private final Semaphore semaphore = new Semaphore(1, true);
45  
46      public JSR356WebSocket(Session session, AtmosphereConfig config) {
47          super(config);
48          this.session = session;
49          // https://issues.apache.org/bugzilla/show_bug.cgi?id=56026
50          String s = config.getInitParameter(ApplicationConfig.WEBSOCKET_IDLETIME);
51          if (s != null) {
52              session.getAsyncRemote().setSendTimeout(Integer.valueOf(s));
53          }
54      }
55  
56      @Override
57      public boolean isOpen() {
58          return session.isOpen();
59      }
60  
61      @Override
62      public WebSocket write(String s) throws IOException {
63          try {
64              try {
65                  semaphore.acquireUninterruptibly();
66                  session.getAsyncRemote().sendText(s, new WriteResult(resource(), s, semaphore));
67              } catch (IllegalStateException e) {
68                  semaphore.release();
69                  throw e;
70              }
71          } catch (NullPointerException e) {
72              patchGlassFish(e);
73          }
74          return this;
75      }
76  
77      @Override
78      public WebSocket write(byte[] data, int offset, int length) throws IOException {
79          try {
80              semaphore.acquireUninterruptibly();
81              ByteBuffer b = ByteBuffer.wrap(data, offset, length);
82              session.getAsyncRemote().sendBinary(ByteBuffer.wrap(data, offset, length),
83                      new WriteResult(resource(), b.array(), semaphore));
84          } catch (NullPointerException e) {
85              patchGlassFish(e);
86              semaphore.release();
87          } catch (IllegalStateException e) {
88              semaphore.release();
89              throw e;
90          }
91          return this;
92      }
93  
94      void patchGlassFish(NullPointerException e) {
95          // https://java.net/jira/browse/TYRUS-175
96          logger.trace("", e);
97          WebSocketProcessorFactory.getDefault().getWebSocketProcessor(config().framework()).close(this, 1002);
98      }
99  
100     @Override
101     public void close() {
102         logger.trace("WebSocket.close() for AtmosphereResource {}", resource() != null ? resource().uuid() : "null");
103         try {
104             session.close();
105             // Tomcat may throw  https://gist.github.com/jfarcand/6702738
106         } catch (Exception e) {
107             logger.trace("", e);
108         }
109     }
110 
111     private final class WriteResult implements SendHandler {
112 
113         private final AtmosphereResource r;
114         private final Object message;
115         private final Semaphore semaphore;
116 
117         private WriteResult(AtmosphereResource r, Object message, Semaphore semaphore) {
118             this.r = r;
119             this.message = message;
120             this.semaphore = semaphore;
121         }
122 
123         @Override
124         public void onResult(SendResult result) {
125             semaphore.release();
126             if (!result.isOK() || result.getException() != null) {
127                 logger.trace("WebSocket {} failed to write {}", r, message);
128                 Broadcaster b = r.getBroadcaster();
129                 b.getBroadcasterConfig().getBroadcasterCache().addToCache(b.getID(), r, new BroadcastMessage(message));
130             }
131         }
132     }
133 }

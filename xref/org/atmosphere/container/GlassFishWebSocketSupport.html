16  package org.atmosphere.container;
17  
18  import com.sun.grizzly.websockets.WebSocketEngine;
19  import org.atmosphere.cpr.Action;
20  import org.atmosphere.cpr.AtmosphereConfig;
21  import org.atmosphere.cpr.AtmosphereRequest;
22  import org.atmosphere.cpr.AtmosphereResponse;
23  import org.atmosphere.util.Utils;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import javax.servlet.ServletConfig;
28  import javax.servlet.ServletException;
29  import java.io.IOException;
30  
31  /**
32   * Websocket Portable Runtime implementation on top of GlassFish 3.0.1 and up.
33   *
34   * @author Jeanfrancois Arcand
35   */
36  public class GlassFishWebSocketSupport extends GrizzlyCometSupport {
37  
38      private static final Logger logger = LoggerFactory.getLogger(GlassFishWebSocketSupport.class);
39      private GlassFishWebSocketHandler glassfishWebSocketHandler;
40  
41      public GlassFishWebSocketSupport(AtmosphereConfig config) {
42          super(config);
43      }
44  
45      @Override
46      public void init(ServletConfig sc) throws ServletException {
47          super.init(sc);
48          glassfishWebSocketHandler = new GlassFishWebSocketHandler(config);
49          WebSocketEngine.getEngine().register(glassfishWebSocketHandler);
50      }
51  
52      @Override
53      public void shutdown() {
54          WebSocketEngine.getEngine().unregister(glassfishWebSocketHandler);
55          super.shutdown();
56      }
57  
58      /**
59       * {@inheritDoc}
60       */
61      @Override
62      public Action service(AtmosphereRequest request, AtmosphereResponse response)
63              throws IOException, ServletException {
64  
65          if (!Utils.webSocketEnabled(request)) {
66              return super.service(request, response);
67          } else {
68              return suspended(request, response);
69          }
70      }
71  
72      /**
73       * Return the container's name.
74       */
75      public String getContainerName() {
76          return config.getServletConfig().getServletContext().getServerInfo() + " with WebSocket enabled.";
77      }
78  
79      @Override
80      public boolean supportWebSocket() {
81          return true;
82      }
83  }

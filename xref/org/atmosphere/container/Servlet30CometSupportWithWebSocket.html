16  package org.atmosphere.container;
17  
18  import org.atmosphere.cpr.AtmosphereServlet.Action;
19  import org.atmosphere.cpr.AtmosphereServlet.AtmosphereConfig;
20  import org.eclipse.jetty.websocket.WebSocketFactory;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import javax.servlet.ServletException;
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpServletResponse;
27  import java.io.IOException;
28  
29  /**
30   * This class is the same as {@link JettyCometSupportWithWebSocket} implementation and add Websocket support
31   * to Servlet 3.0.
32   *
33   * @author Jeanfrancois Arcand
34   */
35  public class Servlet30CometSupportWithWebSocket extends Servlet30CometSupport {
36  
37      private static final Logger logger = LoggerFactory.getLogger(Servlet30CometSupportWithWebSocket.class);
38      private final WebSocketFactory webSocketFactory;
39  
40      public Servlet30CometSupportWithWebSocket(final AtmosphereConfig config) {
41          super(config);
42  
43          boolean isJetty = config.getServletContext().getServerInfo().toLowerCase().startsWith("jetty");
44          if (isJetty) {
45              webSocketFactory = JettyWebSocketUtil.getFactory(config);
46          }  else {
47              webSocketFactory = null;
48          }
49          //TODO: Add Grizzly support here as well.
50  
51      }
52  
53      /**
54       * {@inheritDoc}
55       */
56      @Override
57      public Action service(HttpServletRequest req, HttpServletResponse res)
58              throws IOException, ServletException {
59          Action action = JettyWebSocketUtil.doService(this,req,res,webSocketFactory);
60          return action == null? super.service(req,res) : action;
61      }
62  
63      /**
64       * Return the container's name.
65       */
66      public String getContainerName() {
67          return config.getServletConfig().getServletContext().getServerInfo() + " with WebSocket enabled.";
68      }
69  
70      @Override
71      public boolean supportWebSocket() {
72          return true;
73      }
74  }

38  /**
39   * This class is used when the {@link org.atmosphere.cpr.AtmosphereFramework} detect the container
40   * detect Servlet 3.0 Asynch API.
41   *
42   * @author Jeanfrancois Arcand
43   */
44  public class Servlet30CometSupport extends AsynchronousProcessor {
45  
46      private static final Logger logger = LoggerFactory.getLogger(Servlet30CometSupport.class);
47  
48      public Servlet30CometSupport(AtmosphereConfig config) {
49          super(config);
50      }
51  
52      /**
53       * Return "javax.servlet".
54       *
55       * @return "javax.servlet"
56       */
57      @Override
58      public String getContainerName() {
59          return super.getContainerName() + " using javax.servlet/3.0";
60      }
61  
62      @Override
63      public Action service(AtmosphereRequest request, AtmosphereResponse response)
64              throws IOException, ServletException {
65  
66          Action action = suspended(request, response);
67          if (action.type() == Action.TYPE.SUSPEND) {
68              suspend(action, request, response);
69          } else if (action.type() == Action.TYPE.RESUME) {
70  
71              Action nextAction = resumed(request, response);
72              if (nextAction.type() == Action.TYPE.SUSPEND) {
73                  suspend(action, request, response);
74              }
75          }
76  
77          return action;
78      }
79  
80      /**
81       * Suspend the connection by invoking {@link AtmosphereRequest#startAsync()}
82       *
83       * @param action The {@link org.atmosphere.cpr.Action}
84       * @param req    the {@link AtmosphereRequest}
85       * @param res    the {@link AtmosphereResponse}
86       * @throws java.io.IOException
87       * @throws javax.servlet.ServletException
88       */
89      private void suspend(Action action, AtmosphereRequest req, AtmosphereResponse res)
90              throws IOException, ServletException {
91  
92          if (!req.isAsyncStarted() && !Utils.webSocketEnabled(req)) {
93              AsyncContext asyncContext = req.startAsync(req, res);
94              asyncContext.addListener(new CometListener(this, res.uuid()));
95              // Do nothing except setting the times out
96              if (action.timeout() != -1) {
97                  asyncContext.setTimeout(action.timeout());
98              } else {
99                  // Jetty 8 does something really weird if you set it to
100                 // Long.MAX_VALUE, which is to resume automatically.
101                 asyncContext.setTimeout(Integer.MAX_VALUE);
102             }
103             req.setAttribute(FrameworkConfig.ASYNC_CONTEXT, asyncContext);
104         }
105     }
106 
107     @Override
108     public void action(AtmosphereResourceImpl r) {
109         super.action(r);
110         if (r.action().type() == Action.TYPE.RESUME && r.isInScope()) {
111             endAsyncContext(r.getRequest(false));
112         }
113     }
114 
115     @Override
116     public AsyncSupport complete(AtmosphereResourceImpl r) {
117         endAsyncContext(r.getRequest(false));
118         return null;
119     }
120 
121     public void endAsyncContext(AtmosphereRequest request){
122         AsyncContext asyncContext = (AsyncContext) request.getAttribute(FrameworkConfig.ASYNC_CONTEXT);
123         if (asyncContext != null) {
124             try {
125                 asyncContext.complete();
126             } catch (IllegalStateException ex) {
127                 // Already completed. Jetty throw an exception on shutdown with log
128                 try {
129                     logger.trace("Already resumed!", ex);
130                 } catch (Exception ex2){};
131             }
132         }
133     }
134 
135     @Override
136     public Action cancelled(AtmosphereRequest req, AtmosphereResponse res)
137             throws IOException, ServletException {
138 
139         Action action = super.cancelled(req, res);
140         if (req.getAttribute(MAX_INACTIVE) != null && Long.class.cast(req.getAttribute(MAX_INACTIVE)) == -1) {
141             endAsyncContext(req);
142         }
143         return action;
144     }

146     /**
147      * Servlet 3.0 async listener support.
148      */
149     private final static class CometListener implements AsyncListener {
150 
151         private final AsynchronousProcessor p;
152         private final String uuid;
153 
154         // For JBoss 7 https://github.com/Atmosphere/atmosphere/issues/240
155         public CometListener() {
156             this.uuid = "-1";
157             p = null;
158         }
159 
160         public CometListener(AsynchronousProcessor processor, String uuid) {
161             this.p = processor;
162             this.uuid = uuid;
163         }
164 
165         @Override
166         public void onComplete(AsyncEvent event) throws IOException {
167             logger.trace("Resumed (completed): event: {}", uuid);
168         }
169 
170         @Override
171         public void onTimeout(AsyncEvent event) throws IOException {
172             logger.trace("onTimeout(): event: {}", event.getAsyncContext().getRequest());
173 
174             if (p == null) {
175                 logger.error("Invalid state - CometListener");
176                 return;
177             }
178 
179             try {
180                 p.timedout((AtmosphereRequest) event.getAsyncContext().getRequest(),
181                         (AtmosphereResponse) event.getAsyncContext().getResponse());
182             } catch (ServletException ex) {
183                 logger.warn("onTimeout(): failed timing out comet response: " + event.getAsyncContext().getResponse(), ex);
184             } finally {
185                 event.getAsyncContext().complete();
186             }
187         }
188 
189         @Override
190         public void onError(AsyncEvent event) {
191             logger.trace("onError(): event: {}", event.getAsyncContext().getResponse());
192 
193             if (p == null) {
194                 logger.error("Invalid state - CometListener");
195                 return;
196             }
197 
198             try {
199                 p.cancelled((AtmosphereRequest) event.getAsyncContext().getRequest(),
200                         (AtmosphereResponse) event.getAsyncContext().getResponse());
201             } catch (Throwable ex) {
202                 logger.warn("failed cancelling comet response: " + event.getAsyncContext().getResponse(), ex);
203             }
204         }
205 
206         @Override
207         public void onStartAsync(AsyncEvent event) {
208             logger.trace("onStartAsync(): event: {}", event.getAsyncContext().getResponse());
209         }
210     }

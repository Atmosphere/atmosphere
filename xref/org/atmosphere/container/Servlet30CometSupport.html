40  /**
41   * This class is used when the {@link org.atmosphere.cpr.AtmosphereFramework} detect the container
42   * detect Servlet 3.0 Asynch API.
43   *
44   * @author Jeanfrancois Arcand
45   */
46  public class Servlet30CometSupport extends AsynchronousProcessor {
47  
48      private static final Logger logger = LoggerFactory.getLogger(Servlet30CometSupport.class);
49  
50      public Servlet30CometSupport(AtmosphereConfig config) {
51          super(config);
52      }
53  
54      /**
55       * Return "javax.servlet".
56       *
57       * @return "javax.servlet"
58       */
59      @Override
60      public String getContainerName() {
61          return super.getContainerName() + " using javax.servlet/3.0";
62      }
63  
64      @Override
65      public Action service(AtmosphereRequest request, AtmosphereResponse response)
66              throws IOException, ServletException {
67  
68          Action action = suspended(request, response);
69          if (action.type() == Action.TYPE.SUSPEND) {
70              suspend(action, request, response);
71          } else if (action.type() == Action.TYPE.RESUME) {
72  
73              Action nextAction = resumed(request, response);
74              if (nextAction.type() == Action.TYPE.SUSPEND) {
75                  suspend(action, request, response);
76              }
77          }
78  
79          return action;
80      }
81  
82      /**
83       * Suspend the connection by invoking {@link AtmosphereRequestImpl#startAsync()}
84       *
85       * @param action The {@link org.atmosphere.cpr.Action}
86       * @param req    the {@link AtmosphereRequest}
87       * @param res    the {@link AtmosphereResponse}
88       * @throws java.io.IOException
89       * @throws javax.servlet.ServletException
90       */
91      private void suspend(Action action, AtmosphereRequest req, AtmosphereResponse res)
92              throws IOException, ServletException {
93  
94          if (!req.isAsyncStarted() && !Utils.webSocketEnabled(req)) {
95              AsyncContext asyncContext = req.startAsync(req, res);
96              asyncContext.addListener(new CometListener(this, res.uuid()));
97              // Do nothing except setting the times out
98              if (action.timeout() != -1) {
99                  asyncContext.setTimeout(action.timeout());
100             } else {
101                 // Jetty 8 does something really weird if you set it to
102                 // Long.MAX_VALUE, which is to resume automatically.
103                 asyncContext.setTimeout(Integer.MAX_VALUE);
104             }
105             req.setAttribute(FrameworkConfig.ASYNC_CONTEXT, asyncContext);
106         }
107     }
108 
109     @Override
110     public void action(AtmosphereResourceImpl r) {
111         super.action(r);
112         if (r.action().type() == Action.TYPE.RESUME && r.isInScope()) {
113             endAsyncContext(r.getRequest(false));
114         }
115     }
116 
117     @Override
118     public AsyncSupport complete(AtmosphereResourceImpl r) {
119         endAsyncContext(r.getRequest(false));
120         return null;
121     }
122 
123     public void endAsyncContext(AtmosphereRequest request){
124         final Object attribute = request.getAttribute(FrameworkConfig.ASYNC_CONTEXT);
125         if (attribute instanceof AsyncContext) {
126             AsyncContext asyncContext = (AsyncContext) attribute;
127             if (asyncContext != null) {
128                 try {
129                     asyncContext.complete();
130                 } catch (IllegalStateException ex) {
131                     // Already completed. Jetty throw an exception on shutdown with log
132                     try {
133                         logger.trace("Already resumed!", ex);
134                     } catch (Exception ex2){};
135                 } finally {
136                     request.removeAttribute(FrameworkConfig.ASYNC_CONTEXT);
137                 }
138             }
139         }
140     }
141 
142     @Override
143     public Action cancelled(AtmosphereRequest req, AtmosphereResponse res)
144             throws IOException, ServletException {
145 
146         Action action = super.cancelled(req, res);
147         if (req.getAttribute(MAX_INACTIVE) != null && Long.class.cast(req.getAttribute(MAX_INACTIVE)) == -1) {
148             endAsyncContext(req);
149         }
150         return action;
151     }
152 
153     /**
154      * Servlet 3.0 async listener support.
155      */
156     private final static class CometListener implements AsyncListener {
157 
158         private final AsynchronousProcessor p;
159         private final String uuid;
160 
161         // For JBoss 7 https://github.com/Atmosphere/atmosphere/issues/240
162         public CometListener() {
163             this.uuid = "-1";
164             p = null;
165         }
166 
167         public CometListener(AsynchronousProcessor processor, String uuid) {
168             this.p = processor;
169             this.uuid = uuid;
170         }
171 
172         @Override
173         public void onComplete(AsyncEvent event) throws IOException {
174             logger.trace("Resumed (completed): event: {}", uuid);
175         }
176 
177         @Override
178         public void onTimeout(AsyncEvent event) throws IOException {
179             logger.trace("onTimeout(): event: {}", event.getAsyncContext().getRequest());
180 
181             if (p == null) {
182                 logger.error("Invalid state - CometListener");
183                 return;
184             }
185 
186             final AsyncContext asyncContext = event.getAsyncContext();
187             try {
188                 p.timedout((AtmosphereRequest) asyncContext.getRequest(),
189                         (AtmosphereResponse) asyncContext.getResponse());
190             } catch (ServletException ex) {
191                 logger.warn("onTimeout(): failed timing out comet response: " + event.getAsyncContext().getResponse(), ex);
192             } finally {
193                 try {
194                     asyncContext.complete();
195                 } catch (IllegalStateException ex) {
196                     // The complete method has been already called.
197                     // https://tomcat.apache.org/tomcat-7.0-doc/api/org/apache/coyote/AsyncStateMachine.html
198                     logger.trace("", ex);
199                 }
200             }
201         }
202 
203         @Override
204         public void onError(AsyncEvent event) {
205             logger.trace("onError(): event: {}", event.getAsyncContext().getResponse());
206 
207             if (p == null) {
208                 logger.error("Invalid state - CometListener");
209                 return;
210             }
211 
212             try {
213                 p.cancelled((AtmosphereRequest) event.getAsyncContext().getRequest(),
214                         (AtmosphereResponse) event.getAsyncContext().getResponse());
215             } catch (Throwable ex) {
216                 logger.warn("failed cancelling comet response: " + event.getAsyncContext().getResponse(), ex);
217             }
218         }
219 
220         @Override
221         public void onStartAsync(AsyncEvent event) {
222             logger.trace("onStartAsync(): event: {}", event.getAsyncContext().getResponse());
223         }
224     }

60  /**
61   * Comet Portable Runtime implementation on top of Grizzly API included
62   * with GlassFish v2.
63   *
64   * @author Jeanfrancois Arcand
65   */
66  public class GlassFishv2CometSupport extends AsynchronousProcessor {
67  
68      private static final Logger logger = LoggerFactory.getLogger(GlassFishv2CometSupport.class);
69  
70      private static final String ATMOSPHERE = "/atmosphere";
71  
72      private String atmosphereCtx = "";
73  
74      public GlassFishv2CometSupport(AtmosphereConfig config) {
75          super(config);
76      }
77  
78      /**
79       * Init Grizzly's {@link CometContext} that will be used to suspend and
80       * resume the response.
81       *
82       * @param sc the {@link ServletContext}
83       * @throws javax.servlet.ServletException
84       */
85      @Override
86      public void init(ServletConfig sc) throws ServletException {
87          super.init(sc);
88          atmosphereCtx = sc.getServletContext().getContextPath() + ATMOSPHERE;
89  
90          CometEngine cometEngine = CometEngine.getEngine();
91          CometContext context = cometEngine.register(atmosphereCtx);
92          context.setExpirationDelay(-1);
93          logger.debug("Created CometContext for atmosphere context: {}", atmosphereCtx);
94      }
95  
96      /**
97       * {@inheritDoc}
98       */
99      public Action service(HttpServletRequest req, HttpServletResponse res)
100             throws IOException, ServletException {
101         CometContext ctx = CometEngine.getEngine().getCometContext(atmosphereCtx);
102         Action action = suspended(req, res);
103         if (action.type == Action.TYPE.SUSPEND) {
104             logger.debug("Suspending response: {}", res);
105             suspend(ctx, action, req, res);
106         } else if (action.type == Action.TYPE.RESUME) {
107             logger.debug("Resuming response: {}", res);
108             resume(req, ctx);
109         }
110         return action;
111     }
112 
113     /**
114      * Suspend the response
115      *
116      * @param ctx
117      * @param action
118      * @param req
119      * @param res
120      */
121     private void suspend(CometContext ctx, Action action, HttpServletRequest req,
122                          HttpServletResponse res) {
123         VoidCometHandler c = new VoidCometHandler(req, res);
124         ctx.setExpirationDelay(action.timeout);
125         ctx.addCometHandler(c);
126         ctx.addAttribute("Time", System.currentTimeMillis());
127         req.setAttribute(ATMOSPHERE, c.hashCode());
128 
129         if (supportSession()) {
130             // Store as well in the session in case the resume operation
131             // happens outside the AtmosphereHandler.onStateChange scope.
132             req.getSession().setAttribute(ATMOSPHERE, c.hashCode());
133         }
134     }
135 
136     /**
137      * Resume the underlying response,
138      *
139      * @param req an {@link HttpServletRequest}
140      * @param ctx a {@link CometContext}
141      */
142     private void resume(HttpServletRequest req, CometContext ctx) {
143 
144         if (req.getAttribute(ATMOSPHERE) == null) {
145             return;
146         }
147 
148         CometHandler handler = ctx.getCometHandler((Integer) req.getAttribute(ATMOSPHERE));
149         req.removeAttribute(ATMOSPHERE);
150 
151         if (handler == null && supportSession() && req.getSession(false) != null) {
152             handler = ctx.getCometHandler((Integer) req.getSession(false).getAttribute(ATMOSPHERE));
153             req.getSession().removeAttribute(ATMOSPHERE);
154         }
155 
156         if (handler != null && (config.getInitParameter(ApplicationConfig.RESUME_AND_KEEPALIVE) == null
157                 || config.getInitParameter(ApplicationConfig.RESUME_AND_KEEPALIVE).equalsIgnoreCase("false"))) {
158             ctx.resumeCometHandler(handler);
159         }
160     }
161 
162     /**
163      * {@inheritDoc}
164      */
165     @Override
166     public void action(AtmosphereResourceImpl actionEvent) {
167         super.action(actionEvent);
168         if (actionEvent.action().type == Action.TYPE.RESUME && actionEvent.isInScope()) {
169             CometContext ctx = CometEngine.getEngine().getCometContext(atmosphereCtx);
170             resume(actionEvent.getRequest(), ctx);
171         }
172     }
173 
174     @Override
175     public Action cancelled(HttpServletRequest req, HttpServletResponse res)
176             throws IOException, ServletException {
177 
178         Action action = super.cancelled(req, res);
179         if (req.getAttribute(MAX_INACTIVE) != null && Long.class.cast(req.getAttribute(MAX_INACTIVE)) == -1) {
180             resume(req, CometEngine.getEngine().getCometContext(atmosphereCtx));
181         }
182         return action;
183     }

185     /**
186      * Void {@link CometHandler}, which delegate the processing of the
187      * {@link HttpServletResponse} to an {@link AtmosphereResource}.
188      */
189     private class VoidCometHandler implements CometHandler {
190 
191         private final HttpServletRequest req;
192         private final HttpServletResponse res;
193 
194         private VoidCometHandler(HttpServletRequest req, HttpServletResponse res) {
195             this.req = req;
196             this.res = res;
197         }
198 
199         /**
200          * {@inheritDoc}
201          */
202         public void attach(Object o) {
203         }
204 
205         /**
206          * {@inheritDoc}
207          */
208         public void onEvent(CometEvent ce) throws IOException {
209         }
210 
211         /**
212          * {@inheritDoc}
213          */
214         public void onInitialize(CometEvent ce) throws IOException {
215         }
216 
217         /**
218          * {@inheritDoc}
219          */
220         public void onTerminate(CometEvent ce) throws IOException {
221         }
222 
223         /**
224          * {@inheritDoc}
225          */
226         public synchronized void onInterrupt(CometEvent ce) throws IOException {
227             long timeStamp = (Long) ce.getCometContext().getAttribute("Time");
228             try {
229                 if (ce.getCometContext().getExpirationDelay() > 0
230                         && (System.currentTimeMillis() - timeStamp) >= ce.getCometContext().getExpirationDelay()) {
231                     timedout(req, res);
232                 } else {
233                     cancelled(req, res);
234                 }
235             } catch (ServletException ex) {
236                 logger.warn("onInterrupt(): encountered exception", ex);
237             }
238         }
239     }

32  package org.atmosphere.container;
33  
34  import org.atmosphere.cpr.Action;
35  import org.atmosphere.cpr.AtmosphereConfig;
36  import org.atmosphere.cpr.AtmosphereRequest;
37  import org.atmosphere.cpr.AtmosphereResponse;
38  import org.atmosphere.cpr.WebSocketProcessorFactory;
39  import org.atmosphere.websocket.WebSocketProcessor;
40  import org.slf4j.Logger;
41  import org.slf4j.LoggerFactory;
42  
43  import javax.servlet.ServletException;
44  import java.io.IOException;
45  
46  /**
47   * Tomcat's WebSocket support. This code has been adapted from {@link org.apache.catalina.websocket.WebSocketServlet}
48   */
49  public class Tomcat7Servlet30SupportWithWebSocket extends Servlet30CometSupport implements TomcatWebSocketUtil.Delegate {
50  
51      private static final Logger logger = LoggerFactory.getLogger(Tomcat7Servlet30SupportWithWebSocket.class);
52      private static final long serialVersionUID = 1L;
53      private final WebSocketProcessor webSocketProcessor;
54  
55      public Tomcat7Servlet30SupportWithWebSocket(AtmosphereConfig config) {
56          super(config);
57          webSocketProcessor = WebSocketProcessorFactory.getDefault().getWebSocketProcessor(config.framework());
58      }
59  
60      @Override
61      public Action service(AtmosphereRequest req, AtmosphereResponse res) throws IOException, ServletException {
62          return TomcatWebSocketUtil.doService(req, res, this, config, webSocketProcessor);
63      }
64  
65      @Override
66      public Action doService(AtmosphereRequest req, AtmosphereResponse res) throws IOException, ServletException {
67          return super.service(req, res);
68      }
69  
70      @Override
71      public boolean supportWebSocket() {
72          return true;
73      }
74  }

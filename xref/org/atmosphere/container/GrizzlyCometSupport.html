27  import org.atmosphere.cpr.AtmosphereRequestImpl;
28  import org.atmosphere.cpr.AtmosphereResourceImpl;
29  import org.atmosphere.cpr.AtmosphereResponse;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import javax.servlet.ServletConfig;
34  import javax.servlet.ServletContext;
35  import javax.servlet.ServletException;
36  import java.io.IOException;
37  
38  import static org.atmosphere.cpr.ApplicationConfig.MAX_INACTIVE;
39  
40  /**
41   * Comet Portable Runtime implementation on top of Grizzly 1.5 and up.
42   *
43   * @author Jeanfrancois Arcand
44   */
45  public class GrizzlyCometSupport extends AsynchronousProcessor {
46  
47      private static final Logger logger = LoggerFactory.getLogger(GrizzlyCometSupport.class);
48  
49      private static final String ATMOSPHERE = "/atmosphere";
50  
51      private String atmosphereCtx = "";
52  
53      public GrizzlyCometSupport(AtmosphereConfig config) {
54          super(config);
55      }
56  
57      /**
58       * Init Grizzly's {@link CometContext} that will be used to suspend and resume the response.
59       *
60       * @param sc the {@link ServletContext}
61       * @throws javax.servlet.ServletException
62       */
63      @Override
64      public void init(ServletConfig sc) throws ServletException {
65          super.init(sc);
66  
67          atmosphereCtx = sc.getServletContext().getContextPath() + ATMOSPHERE;
68  
69          CometEngine cometEngine = CometEngine.getEngine();
70          CometContext context = cometEngine.register(atmosphereCtx);
71          context.setExpirationDelay(-1);
72          logger.debug("Created CometContext for atmosphere context: {}", atmosphereCtx);
73      }
74  
75      @Override
76      public Action service(AtmosphereRequest req, AtmosphereResponse res)
77              throws IOException, ServletException {
78  
79          CometContext ctx = CometEngine.getEngine().getCometContext(atmosphereCtx);
80          Action action = suspended(req, res);
81          if (action.type() == Action.TYPE.SUSPEND) {
82              suspend(ctx, action, req, res);
83          } else if (action.type() == Action.TYPE.RESUME) {
84              resume(req, ctx);
85          }
86          return action;
87      }
88  
89      /**
90       * Suspend the response
91       *
92       * @param ctx
93       * @param action
94       * @param req
95       * @param res
96       */
97      private void suspend(CometContext ctx, Action action, AtmosphereRequest req, AtmosphereResponse res) {
98          VoidCometHandler c = new VoidCometHandler(req, res);
99          ctx.setExpirationDelay(action.timeout());
100         ctx.addCometHandler(c);
101         req.setAttribute(ATMOSPHERE, c.hashCode());
102         ctx.addAttribute("Time", System.currentTimeMillis());
103     }
104 
105     /**
106      * Resume the underlying response.
107      *
108      * @param req an {@link AtmosphereRequest}
109      * @param ctx a {@link CometContext}
110      */
111     private void resume(AtmosphereRequest req, CometContext ctx) {
112 
113         if (req.getAttribute(ATMOSPHERE) == null) {
114             return;
115         }
116 
117         CometHandler handler = ctx.getCometHandler((Integer) req.getAttribute(ATMOSPHERE));
118         req.removeAttribute(ATMOSPHERE);
119 
120         if (req.resource() != null) {
121             try {
122                 AtmosphereResourceImpl.class.cast(req.resource()).cancel();
123             } catch (IOException e) {
124                 logger.trace("", e);
125             }
126         }
127 
128         if (handler != null) {
129             ctx.resumeCometHandler(handler);
130         }
131     }
132 
133     @Override
134     public void action(AtmosphereResourceImpl r) {
135         super.action(r);
136         if (r.action().type() == Action.TYPE.RESUME && r.isInScope()) {
137             complete(r);
138         }
139     }
140 
141     @Override
142     public AsyncSupport complete(AtmosphereResourceImpl r) {
143         CometContext ctx = CometEngine.getEngine().getCometContext(atmosphereCtx);
144         resume(r.getRequest(false), ctx);
145         return this;
146     }
147 
148     @Override
149     public Action cancelled(AtmosphereRequest req, AtmosphereResponse res)
150             throws IOException, ServletException {
151 
152         Action action = super.cancelled(req, res);
153         if (req.getAttribute(MAX_INACTIVE) != null && Long.class.cast(req.getAttribute(MAX_INACTIVE)) == -1) {
154             resume(req, CometEngine.getEngine().getCometContext(atmosphereCtx));
155         }
156         return action;
157     }
158 
159     /**
160      * Void {@link CometHandler}, which delegate the processing of the
161      * {@link AtmosphereRequest} to an {@link org.atmosphere.cpr.AtmosphereResourceImpl}.
162      */
163     private class VoidCometHandler implements CometHandler {
164 
165         AtmosphereRequest req;
166         AtmosphereResponse res;
167 
168         public VoidCometHandler(AtmosphereRequest req, AtmosphereResponse res) {
169             this.req = req;
170             this.res = res;
171         }
172 
173         @Override
174         public void attach(Object o) {
175         }
176 
177         @Override
178         public void onEvent(CometEvent ce) throws IOException {
179         }
180 
181         @Override
182         public void onInitialize(CometEvent ce) throws IOException {
183         }
184 
185         @Override
186         public void onTerminate(CometEvent ce) throws IOException {
187         }
188 
189         @Override
190         public synchronized void onInterrupt(CometEvent ce) throws IOException {
191             long timeStamp = (Long) ce.getCometContext().getAttribute("Time");
192             try {
193                 if (ce.getCometContext().getExpirationDelay() > 0
194                         && (System.currentTimeMillis() - timeStamp) >= ce.getCometContext().getExpirationDelay()) {
195                     timedout(req, res);
196                 } else {
197                     cancelled(req, res);
198                 }
199             } catch (ServletException ex) {
200                 logger.warn("onInterrupt() encountered exception", ex);
201             }
202         }
203     }
204 }

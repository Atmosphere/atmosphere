35  /**
36   * Jetty 7 & 8 WebSocket support.
37   */
38  public class JettyWebSocketHandler implements org.eclipse.jetty.websocket.WebSocket,
39          org.eclipse.jetty.websocket.WebSocket.OnFrame,
40          org.eclipse.jetty.websocket.WebSocket.OnBinaryMessage,
41          org.eclipse.jetty.websocket.WebSocket.OnTextMessage,
42          org.eclipse.jetty.websocket.WebSocket.OnControl {
43  
44      private static final Logger logger = LoggerFactory.getLogger(JettyWebSocketHandler.class);
45  
46      private WebSocketProcessor webSocketProcessor;
47      private final AtmosphereRequest request;
48      private final AtmosphereFramework framework;
49      private final WebSocketProtocol webSocketProtocol;
50  
51      public JettyWebSocketHandler(AtmosphereRequest request, AtmosphereFramework framework, WebSocketProtocol webSocketProtocol) {
52          this.request = request;
53          this.framework = framework;
54          this.webSocketProtocol = webSocketProtocol;
55      }
56  
57      @Override
58      public void onMessage(byte[] data, int offset, int length) {
59          logger.trace("WebSocket.onMessage (bytes)");
60          webSocketProcessor.invokeWebSocketProtocol(data, offset, length);
61          try {
62              webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent(new String(data, offset, length, "UTF-8"), MESSAGE, webSocketProcessor.webSocket()));
63          } catch (UnsupportedEncodingException e) {
64              logger.warn("UnsupportedEncodingException", e);
65  
66          }
67      }
68  
69      @Override
70      public boolean onControl(byte controlCode, byte[] data, int offset, int length) {
71          logger.trace("WebSocket.onControl.");
72          try {
73              webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent(new String(data, offset, length, "UTF-8"), CONTROL, webSocketProcessor.webSocket()));
74          } catch (UnsupportedEncodingException e) {
75              logger.warn("UnsupportedEncodingException", e);
76  
77          }
78          return false;
79      }
80  
81      @Override
82      public boolean onFrame(byte flags, byte opcode, byte[] data, int offset, int length) {
83          logger.trace("WebSocket.onFrame.");
84          // TODO: onMessage is always invoked after that method gets called, so no need to enable for now.
85          //       webSocketProcessor.broadcast(data, offset, length);
86          /* try {
87              webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent(new String(data, offset, length, "UTF-8"), MESSAGE, webSocketProcessor.webSocket()));
88          } catch (UnsupportedEncodingException e) {
89              logger.warn("UnsupportedEncodingException", e);
90  
91          }*/
92          return false;
93      }
94  
95      @Override
96      public void onHandshake(org.eclipse.jetty.websocket.WebSocket.FrameConnection connection) {
97          logger.trace("WebSocket.onHandshake");
98          try {
99              webSocketProcessor = new WebSocketProcessor(framework, new Jetty8WebSocket(connection, framework.getAtmosphereConfig()), webSocketProtocol);
100         } catch (Exception e) {
101             logger.warn("failed to connect to web socket", e);
102         }
103 
104         webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent("", HANDSHAKE, webSocketProcessor.webSocket()));
105     }
106 
107     @Override
108     public void onMessage(String data) {
109         logger.trace("WebSocket.onMessage");
110         webSocketProcessor.invokeWebSocketProtocol(data);
111         webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent(data, MESSAGE, webSocketProcessor.webSocket()));
112     }
113 
114     @Override
115     public void onOpen(org.eclipse.jetty.websocket.WebSocket.Connection connection) {
116         logger.trace("WebSocket.onOpen.");
117         try {
118             webSocketProcessor = new WebSocketProcessor(framework, new Jetty8WebSocket(connection, framework.getAtmosphereConfig()), webSocketProtocol);
119             webSocketProcessor.dispatch(request);
120             webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent("", CONNECT, webSocketProcessor.webSocket()));
121         } catch (Exception e) {
122             logger.warn("failed to connect to web socket", e);
123         }
124     }
125 
126     @Override
127     public void onClose(int closeCode, String message) {
128         request.destroy();
129         if (webSocketProcessor == null) return;
130 
131         webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent("", CLOSE, webSocketProcessor.webSocket()));
132         webSocketProcessor.close(closeCode);
133 
134     }

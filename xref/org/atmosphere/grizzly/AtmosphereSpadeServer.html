50  /**
51   * Simple Atmosphere REST based Server builder:
52   * <p>
53   * AtmosphereSpadeServer.build("http://localhost:8080/","org.foo.bar").start()
54   * </p>
55   * which listen on localhost:8080 for resources starting with package org.foo.bar.
56   *
57   * @author Jean-francois Arcand
58   */
59  public final class AtmosphereSpadeServer {
60  
61      private static final Logger logger = LoggerFactory.getLogger(AtmosphereSpadeServer.class);
62  
63      private final GrizzlyWebServer gws;
64      private final AtmosphereAdapter aa = new AtmosphereAdapter();
65      private boolean defaultToJersey = true;
66  
67      private AtmosphereSpadeServer(int port, boolean secure) {
68          gws = new GrizzlyWebServer(port, ".", secure);
69          gws.getSelectorThread().setDisplayConfiguration(false);
70          gws.addAsyncFilter(new CometAsyncFilter());
71      }

74       * Create a {@link AtmosphereSpadeServer} which listen based using the URI u.

78       */

80          return AtmosphereSpadeServer.build(u, "");
81      }

107     /**
108      * Create a {@link AtmosphereSpadeServer} which listen based on the 'u' for requests for
109      * resources defined under the resources package.
110      *
111      * @param u                The URI the server listen for requests
112      * @param resourcesPackage The resources package name.
113      * @param sslConfig        The {@link SSLConfig}
114      * @return
115      */
116     public static AtmosphereSpadeServer build(String u, String resourcesPackage, SSLConfig sslConfig) {
117         if (u == null) {
118             throw new IllegalArgumentException("The URI must not be null");
119         }
120 
121         URI uri = URI.create(u);
122         final String scheme = uri.getScheme();
123         if (!scheme.startsWith("http"))
124             throw new IllegalArgumentException("The URI scheme, of the URI " + u +
125                     ", must be equal (ignoring case) to 'http'");
126 
127         final int port = (uri.getPort() == -1) ? 80 : uri.getPort();
128         String path = uri.getPath();
129         if (path == null) {
130             throw new IllegalArgumentException("The URI path, of the URI " + uri +
131                     ", must be non-null");
132         } else if (path.length() == 0) {
133             throw new IllegalArgumentException("The URI path, of the URI " + uri +
134                     ", must be present");
135         } else if (path.charAt(0) != '/') {
136             throw new IllegalArgumentException("The URI path, of the URI " + uri +
137                     ". must start with a '/'");
138         }
139 
140         boolean secure = false;
141         if (scheme.equalsIgnoreCase("https")) {
142             secure = true;
143         }
144 
145         AtmosphereSpadeServer a = new AtmosphereSpadeServer(port, secure);
146         if (path.length() > 1) {
147             if (path.endsWith("/")) {
148                 path = path.substring(0, path.length() - 1);
149             }
150             a.aa.setContextPath(path);
151         }
152 
153         if (resourcesPackage != null) {
154             a.aa.setResourcePackage(resourcesPackage);
155         }
156 
157         if (sslConfig != null) {
158             a.setSSLConfig(sslConfig);
159         }
160 
161         return a;
162     }
163 
164     private void setSSLConfig(SSLConfig sslConfig) {
165         gws.setSSLConfig(sslConfig);
166     }
167 
168     /**
169      * Add an {@link AtmosphereHandler}
170      */
171     public AtmosphereSpadeServer addAtmosphereHandler(String mapping, AtmosphereHandler h) {
172         defaultToJersey = false;
173         AtmosphereAdapter a = new AtmosphereAdapter();
174         a.setServletInstance(aa.getServletInstance());
175         a.addAtmosphereHandler(mapping, h);
176         a.setServletPath(mapping);
177         a.setHandleStaticResources(true);
178         gws.addGrizzlyAdapter(a, new String[]{mapping});
179         return this;
180     }
181 
182     /**
183      * Set the resource package name.
184      */
185     public void setResourcePackage(String resourcePackage) {
186         aa.setResourcePackage(resourcePackage);
187     }

189     /**
190      * Start the {@link AtmosphereSpadeServer}
191      *
192      * @return
193      * @throws IOException
194      */
195     public AtmosphereSpadeServer start() throws IOException {
196         logger.info("AtmosphereSpade Server Started on port: {}", gws.getSelectorThread().getPort());
197 
198         if (defaultToJersey) {
199             aa.setHandleStaticResources(true);
200             gws.addGrizzlyAdapter(aa, new String[]{"*"});
201         }
202         gws.start();
203         return this;
204     }

206     /**
207      * Stop the {@link AtmosphereSpadeServer}
208      *
209      * @return
210      * @throws IOException
211      */
212     public AtmosphereSpadeServer stop() throws IOException {
213         gws.stop();
214         return this;
215     }

16  package org.atmosphere.annotation;
17  
18  import org.atmosphere.cpr.AtmosphereResourceEventListener;
19  import org.atmosphere.cpr.BroadcastFilter;
20  import org.atmosphere.cpr.HeaderConfig;
21  
22  import java.lang.annotation.Documented;
23  import java.lang.annotation.ElementType;
24  import java.lang.annotation.Retention;
25  import java.lang.annotation.RetentionPolicy;
26  import java.lang.annotation.Target;
27  
28  /**
29   * Suspend the response and use the Broadcaster associated with {@link org.atmosphere.cpr.HeaderConfig#X_ATMOSPHERE_TRACKING_ID}
30   * to publish the result.
31   *
32   * @author Jeanfrancois Arcand
33   */
34  @Target({ElementType.METHOD})
35  @Retention(RetentionPolicy.RUNTIME)
36  @Documented
37  public @interface Asynchronous {
38  
39      /**
40       * Wait for {@link org.atmosphere.cpr.AtmosphereResource} before executing the {@link org.atmosphere.cpr.Broadcaster#awaitAndBroadcast(Object, long, java.util.concurrent.TimeUnit)}.
41       * @return true if the broadcast operation must wait before executing.
42       */
43      boolean waitForResource() default true;
44  
45      /**
46       * Add {@link org.atmosphere.cpr.BroadcastFilter} to the broadcast operation.
47       */
48      public Class<? extends BroadcastFilter>[] broadcastFilter() default {};
49  
50      /**
51       * The header value used to create a {@link org.atmosphere.cpr.Broadcaster} that will be used to broadcast the
52       * asynchronous execution.
53       * @return The header value used to create a {@link org.atmosphere.cpr.Broadcaster}
54       */
55      String header() default HeaderConfig.X_ATMOSPHERE_TRACKING_ID;
56  
57      /**
58       * The time a connection will be suspended if no broadcast happens. Same as {@link org.atmosphere.annotation.Suspend#period()}
59       * @return The time a connection will be suspended if no broadcast happens.
60       */
61      int period() default 5 * 60 * 1000;
62  
63      /**
64       * Add {@link AtmosphereResourceEventListener} to the broadcast operation.
65       */
66      public Class<? extends AtmosphereResourceEventListener>[] eventListeners() default {};
67  
68      /**
69       * Write the returned entity back to the calling connection. Default is false.
70       * @return true if the entity needs to be written back to the calling connection.
71       */
72      public boolean writeEntity() default true;
73      /**
74       * If the @Produces annotation is missing, this value will be used instead.
75       * @return the default content-type used if the @Produces annotation is missing.
76       */
77      public String contentType() default "";
78  }

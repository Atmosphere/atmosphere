18  package org.atmosphere.annotation;
19  
20  import org.atmosphere.cpr.AtmosphereResourceEventListener;
21  
22  import java.lang.annotation.Documented;
23  import java.lang.annotation.ElementType;
24  import java.lang.annotation.Retention;
25  import java.lang.annotation.RetentionPolicy;
26  import java.lang.annotation.Target;
27  import java.util.concurrent.TimeUnit;
28  
29  /**
30   * Suspend the underlying response. Once suspended, a response might be allowed
31   * to consume {@link Broadcast} events, depending on the scope ([@link Suspend#SCOPE}).
32   * By default, a suspended response is suspended able to consume
33   * any broadcasted events executed inside the same application (SCOPE.APPLICATION).
34   * The period can also be per suspended response (SCOPE.REQUEST) or available to other
35   * application (SCOPE.VM).
36   *
37   * @author Jeanfrancois Arcand
38   */
39  @Target({ElementType.METHOD})
40  @Retention(RetentionPolicy.RUNTIME)
41  @Documented
42  public @interface Suspend {
43  
44      /**
45       * How long a response stay suspended in {@link #timeUnit}, default is -1
46       *
47       * @return
48       */
49      int period() default -1;
50  
51      /**
52       * @return
53       */
54      TimeUnit timeUnit() default TimeUnit.MILLISECONDS;
55  
56      enum SCOPE {
57          REQUEST, APPLICATION, VM
58      }
59  
60      /**
61       * The Scope of the {@link org.atmosphere.cpr.Broadcaster} that will be created once the
62       * response gets suspended. One final word on Broadcaster: by default,
63       * a Broadcaster will broadcast using
64       * all resources/classes on which the response has been suspended.
65       * This behavior is configurable and you can configure it setting the appropriate scope<ul>
66       * <li>REQUEST: broadcast events only to the suspended response associated with the current request.</li>
67       * <li>APPLICATION: broadcast events to all suspended responses associated with the current application.</li>
68       * <li>VM: broadcast events to all suspended responses created inside the current virtual machine.</li>
69       * </ul>
70       *
71       * @return The Scope of the {@link org.atmosphere.cpr.Broadcaster} that will be created once the
72       *         response gets suspended.
73       */
74      SCOPE scope() default SCOPE.APPLICATION;
75  
76      /**
77       * By default, output some comments when suspending the connection.
78       * Deprecated. No longer required
79       */
80      @Deprecated
81      boolean outputComments() default true;
82  
83      /**
84       * Resume all suspended response on the first broadcast operation.
85       */
86      public boolean resumeOnBroadcast() default false;
87  
88      /**
89       * Add {@link AtmosphereResourceEventListener} to track internal events.
90       */
91      public Class<? extends AtmosphereResourceEventListener>[] listeners() default {};
92  
93      /**
94       * Write the returned entity back to the calling connection. Default is false.
95       * @return true if the entity needs to be written back to the calling connection.
96       */
97      public boolean writeEntity() default true;
98  
99      /**
100      * If the @Produces annotation is missing, this value will be used instead.
101      * @return the default content-type used if the @Produces annotation is missing.
102      */
103     public String contentType() default "";
104 }

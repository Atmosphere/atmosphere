17  package org.atmosphere.annotation;
18  
19  import org.atmosphere.cpr.BroadcastFilter;
20  
21  import java.lang.annotation.ElementType;
22  import java.lang.annotation.Retention;
23  import java.lang.annotation.RetentionPolicy;
24  import java.lang.annotation.Target;
25  
26  /**
27   * Broadcast to all suspended response the value of the method annotated with this annotation.
28   *
29   * @author Jeanfrancois Arcand
30   */
31  @Target({ElementType.METHOD})
32  @Retention(RetentionPolicy.RUNTIME)
33  public @interface Broadcast {
34  
35      public String value() default "/*";
36  
37      /**
38       * Add {@link BroadcastFilter}s to the broadcast operation.
39       */
40      public Class<? extends BroadcastFilter>[] filters() default {};
41  
42      /**
43       * Resume all suspended response on the first broadcast operation.
44       */
45      public boolean resumeOnBroadcast() default false;
46  
47      /**
48       * Should the broadcast be delayed? A value of 0 means
49       * the broadcast be delayed until a normal broadcast operation
50       * happens. Any other value will be evaluated as seconds
51       */
52      public int delay() default -1;
53  
54      /**
55       * Write the returned entity back to the calling connection. Default is false.
56       * @return true if the entity needs to be written back to the calling connection.
57       */
58      public boolean writeEntity() default true;
59  }

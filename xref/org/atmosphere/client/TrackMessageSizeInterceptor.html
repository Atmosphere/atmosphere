45  import static org.atmosphere.cpr.ApplicationConfig.MESSAGE_DELIMITER;

47  /**
48   * An {@link org.atmosphere.cpr.AtmosphereInterceptor} that add a message size and delimiter.
49   * <p/>
50   * The special String is configurable using {@link org.atmosphere.cpr.ApplicationConfig#MESSAGE_DELIMITER} and
51   * you can configure this class to exclude some response's content-type by using the {@link ApplicationConfig#EXCLUDED_CONTENT_TYPES}
52   *
53   * @author Jeanfrancois Arcand
54   */
55  public class TrackMessageSizeInterceptor extends AtmosphereInterceptorAdapter {
56  
57      private static final Logger logger = LoggerFactory.getLogger(TrackMessageSizeInterceptor.class);
58      private final static byte[] END = "|".getBytes();
59      private final static String IN_ENCODING = "UTF-8";
60      private final static String OUT_ENCODING = "UTF-8";
61      public final static String SKIP_INTERCEPTOR = TrackMessageSizeInterceptor.class.getName() + ".skip";
62  
63      private byte[] end = END;
64      private String endString = "|";
65      private final Charset inCharset = Charset.forName(IN_ENCODING);
66      private final Charset outCharset = Charset.forName(OUT_ENCODING);
67      private final HashSet<String> excludedContentTypes = new HashSet<String>();
68  
69      private final Interceptor interceptor = new Interceptor();
70  
71      @Override
72      public void configure(AtmosphereConfig config) {
73          String s = config.getInitParameter(MESSAGE_DELIMITER);
74          if (s != null) {
75              messageDelimiter(s);
76          }
77          s = config.getInitParameter(EXCLUDED_CONTENT_TYPES);
78          if (s != null) {
79              excludedContentTypes.addAll(Arrays.asList(s.split(",")));
80          }
81      }
82  
83      /**
84       * Set the character delimiter used by this class to separate message.
85       * @param endString
86       * @return this
87       */
88      public TrackMessageSizeInterceptor messageDelimiter(String endString) {
89          this.endString = endString;
90          end = endString.getBytes();
91          return this;
92      }
93  
94      /**
95       * Exclude response's content-type from being processed by this class.
96       * @param excludedContentType the value of {@link AtmosphereResponseImpl#getContentType()}
97       * @return this
98       */
99      public TrackMessageSizeInterceptor excludedContentType(String excludedContentType) {
100         excludedContentTypes.add(excludedContentType.toLowerCase());
101         return this;
102     }
103 
104     public HashSet<String> excludedContentTypes(){
105         return excludedContentTypes;
106     }
107 
108     @Override
109     public Action inspect(final AtmosphereResource r) {
110         if (AtmosphereResource.TRANSPORT.UNDEFINED == r.transport() || Utils.webSocketMessage(r))
111             return Action.CONTINUE;
112 
113         final AtmosphereResponse response = r.getResponse();
114 
115         super.inspect(r);
116 
117         AsyncIOWriter writer = response.getAsyncIOWriter();
118         if (AtmosphereInterceptorWriter.class.isAssignableFrom(writer.getClass())) {
119             AtmosphereInterceptorWriter.class.cast(writer).interceptor(interceptor);
120         } else {
121             logger.warn("Unable to apply {}. Your AsyncIOWriter must implement {}", getClass().getName(), AtmosphereInterceptorWriter.class.getName());
122         }
123         return Action.CONTINUE;
124     }
125 
126     @Override
127     public String toString() {
128         return " Track Message Size Interceptor using " + endString;
129     }
130 
131     private final class Interceptor extends AsyncIOInterceptorAdapter {
132         @Override
133         public byte[] transformPayload(AtmosphereResponse response, byte[] responseDraft, byte[] data) throws IOException {
134 
135             boolean writeAsBytes = IOUtils.isBodyBinary(response.request());
136             if (writeAsBytes) {
137                 logger.warn("Cannot use TrackMessageSizeInterceptor with binary write. Writing the message as it is.");
138                 return responseDraft;
139             }
140 
141             if (response.request().getAttribute(SKIP_INTERCEPTOR) == null
142                     && Boolean.valueOf(response.request().getHeader(HeaderConfig.X_ATMOSPHERE_TRACKMESSAGESIZE))
143                     && (response.getContentType() == null
144                     || !excludedContentTypes.contains(response.getContentType().toLowerCase()))) {
145                 try {
146                     response.setCharacterEncoding(OUT_ENCODING);
147 
148                     CharBuffer cb = inCharset.newDecoder().decode(ByteBuffer.wrap(responseDraft, 0, responseDraft.length));
149                     String s = cb.toString();
150 
151                     if (s.trim().length() == 0) {
152                         return responseDraft;
153                     }
154 
155                     AtmosphereResource r = response.resource();
156                     if (r == null) {
157                         throw new IOException("Connection Closed by the Client " + response.uuid());
158                     }
159 
160                     int size = cb.length();
161 
162                     String csq = Integer.toString(size) + endString;
163                     ByteBuffer bb = ByteBuffer.allocate(csq.getBytes().length + responseDraft.length);
164                     CharBuffer cb2 = CharBuffer.wrap(csq);
165                     CharsetEncoder encoder = outCharset.newEncoder();
166                     encoder.encode(cb2, bb, false);
167                     encoder.encode(cb, bb, false);
168                     bb.flip();
169                     byte[] b = new byte[bb.limit()];
170                     bb.get(b);
171                     return b;
172                 } catch (MalformedInputException ex) {
173                     // https://github.com/Atmosphere/atmosphere/issues/1803
174                     logger.warn("Cannot decode the response bytes for {}. Writing the message as it is", response.uuid(), ex);
175                     return responseDraft;
176                 }
177             } else {
178                 return responseDraft;
179             }
180         }
181     }
182 
183     @Override
184     public PRIORITY priority() {
185         return InvokationOrder.BEFORE_DEFAULT;
186     }
187 }

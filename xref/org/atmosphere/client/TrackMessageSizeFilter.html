16  package org.atmosphere.client;
17  
18  import org.atmosphere.cpr.AtmosphereRequest;
19  import org.atmosphere.cpr.AtmosphereResource;
20  import org.atmosphere.cpr.PerRequestBroadcastFilter;
21  
22  import static org.atmosphere.cpr.HeaderConfig.X_ATMOSPHERE_TRACKMESSAGESIZE;
23  
24  /**
25   * A {@link PerRequestBroadcastFilter} implementation that add the expected length of the message. This is
26   * useful when used with the atmosphere.js library as the library will read the expected size and wait for the
27   * entire messages to be received before invoking its associated callback.
28   *
29   * If aren't using atmosphere.js, you need to add the {@link org.atmosphere.cpr.HeaderConfig#X_ATMOSPHERE_TRACKMESSAGESIZE} header in order to
30   * enable that Filter. The delimiter char used is '|'.
31   *
32   * For example, broadcasting String 'helloword' will be received by the client as '9 | helloword' but delivered as 'helloword'
33   * to the Javascript function/callback.
34   */
35  public class TrackMessageSizeFilter implements PerRequestBroadcastFilter {
36  
37      @Override
38      public BroadcastAction filter(AtmosphereResource r, Object message, Object originalMessage) {
39  
40          AtmosphereRequest request = r.getRequest();
41          if ("true".equalsIgnoreCase(request.getHeader(X_ATMOSPHERE_TRACKMESSAGESIZE))
42                  && message != null && String.class.isAssignableFrom(message.getClass())) {
43  
44              String msg = message.toString();
45              msg = msg.length() + "|" + msg;
46              return new BroadcastAction(BroadcastAction.ACTION.CONTINUE, msg);
47  
48          }
49          return new BroadcastAction(BroadcastAction.ACTION.CONTINUE, message);
50      }
51  
52      @Override
53      public BroadcastAction filter(Object originalMessage, Object message) {
54          return new BroadcastAction(message);
55      }
56  }

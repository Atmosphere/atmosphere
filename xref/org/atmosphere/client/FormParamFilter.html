16  package org.atmosphere.client;
17  
18  import org.atmosphere.cpr.BroadcastFilter;
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  /**
23   * Simple Form param filter that remove the first part of the request body.
24   *
25   * @author Jeanfrancois Arcand
26   */
27  public class FormParamFilter implements BroadcastFilter {
28  
29      private static final Logger logger = LoggerFactory.getLogger(FormParamFilter.class);
30  
31      @Override
32      public BroadcastAction filter(Object originalMessage, Object message) {
33  
34          if ((message instanceof String) && ((String) message).contains("=")) {
35              try {
36                  message = message.toString().split("=")[1];
37              } catch (ArrayIndexOutOfBoundsException ex) {
38                  // Don't fail, just log it.
39                  logger.warn("failed to split form param: " + message, ex);
40              }
41          }
42          return new BroadcastAction(BroadcastAction.ACTION.CONTINUE, message);
43      }
44  
45  }

71  package org.apache.catalina;

73  import javax.servlet.ServletException;
74  import javax.servlet.http.HttpServletRequest;
75  import javax.servlet.http.HttpServletResponse;
76  import java.io.IOException;
77  
78  /**
79   * The CometEvent interface.
80   *
81   * @author Filip Hanik
82   * @author Remy Maucherat
83   */
84  public interface CometEvent {
85  
86      /**
87       * Enumeration describing the major events that the container can invoke
88       * the CometProcessors event() method with
89       * BEGIN - will be called at the beginning
90       * of the processing of the connection. It can be used to initialize any relevant
91       * fields using the request and response objects. Between the end of the processing
92       * of this event, and the beginning of the processing of the end or error events,
93       * it is possible to use the response object to write data on the open connection.
94       * Note that the response object and depedent OutputStream and Writer are still
95       * not synchronized, so when they are accessed by multiple threads,
96       * synchronization is mandatory. After processing the initial event, the request
97       * is considered to be committed.
98       * READ - This indicates that input data is available, and that one read can be made
99       * without blocking. The available and ready methods of the InputStream or
100      * Reader may be used to determine if there is a risk of blocking: the servlet
101      * should read while data is reported available. When encountering a read error,
102      * the servlet should report it by propagating the exception properly. Throwing
103      * an exception will cause the error event to be invoked, and the connection
104      * will be closed.
105      * Alternately, it is also possible to catch any exception, perform clean up
106      * on any data structure the servlet may be using, and using the close method
107      * of the event. It is not allowed to attempt reading data from the request
108      * object outside of the execution of this method.
109      * END - End may be called to end the processing of the request. Fields that have
110      * been initialized in the begin method should be reset. After this event has
111      * been processed, the request and response objects, as well as all their dependent
112      * objects will be recycled and used to process other requests. End will also be
113      * called when data is available and the end of file is reached on the request input
114      * (this usually indicates the client has pipelined a request).
115      * ERROR - Error will be called by the container in the case where an IO exception
116      * or a similar unrecoverable error occurs on the connection. Fields that have
117      * been initialized in the begin method should be reset. After this event has
118      * been processed, the request and response objects, as well as all their dependent
119      * objects will be recycled and used to process other requests.
120      */
121     public enum EventType {
122         BEGIN, READ, END, ERROR
123     }
124 
125 
126     /**
127      * Event details
128      * TIMEOUT - the connection timed out (sub type of ERROR); note that this ERROR type is not fatal, and
129      * the connection will not be closed unless the servlet uses the close method of the event
130      * CLIENT_DISCONNECT - the client connection was closed (sub type of ERROR)
131      * IOEXCEPTION - an IO exception occurred, such as invalid content, for example, an invalid chunk block (sub type of ERROR)
132      * WEBAPP_RELOAD - the webapplication is being reloaded (sub type of END)
133      * SERVER_SHUTDOWN - the server is shutting down (sub type of END)
134      * SESSION_END - the servlet ended the session (sub type of END)
135      */
136     public enum EventSubType {
137         TIMEOUT, CLIENT_DISCONNECT, IOEXCEPTION, WEBAPP_RELOAD, SERVER_SHUTDOWN, SESSION_END
138     }
139 
140 
141     /**
142      * Returns the HttpServletRequest.
143      *
144      * @return HttpServletRequest
145      */
146     public HttpServletRequest getHttpServletRequest();
147 
148     /**
149      * Returns the HttpServletResponse.
150      *
151      * @return HttpServletResponse
152      */
153     public HttpServletResponse getHttpServletResponse();
154 
155     /**
156      * Returns the event type.
157      *
158      * @return EventType
159      */
160     public EventType getEventType();
161 
162     /**
163      * Returns the sub type of this event.
164      *
165      * @return EventSubType
166      */
167     public EventSubType getEventSubType();
168 
169     /**
170      * Ends the Comet session. This signals to the container that
171      * the container wants to end the comet session. This will send back to the
172      * client a notice that the server has no more data to send as part of this
173      * request. The servlet should perform any needed cleanup as if it had recieved
174      * an END or ERROR event.
175      *
176      * @throws IOException if an IO exception occurs
177      */
178     public void close() throws IOException;
179 
180     /**
181      * Sets the timeout for this Comet connection. Please NOTE, that the implementation
182      * of a per connection timeout is OPTIONAL and MAY NOT be implemented.<br/>
183      * This method sets the timeout in milliseconds of idle time on the connection.
184      * The timeout is reset every time data is received from the connection or data is flushed
185      * using <code>response.flushBuffer()</code>. If a timeout occurs, the
186      * <code>error(HttpServletRequest, HttpServletResponse)</code> method is invoked. The
187      * web application SHOULD NOT attempt to reuse the request and response objects after a timeout
188      * as the <code>error(HttpServletRequest, HttpServletResponse)</code> method indicates.<br/>
189      * This method should not be called asynchronously, as that will have no effect.
190      *
191      * @param timeout The timeout in milliseconds for this connection, must be a positive value, larger than 0
192      * @throws IOException                   An IOException may be thrown to indicate an IO error,
193      *                                       or that the EOF has been reached on the connection
194      * @throws ServletException              An exception has occurred, as specified by the root
195      *                                       cause
196      * @throws UnsupportedOperationException if per connection timeout is not supported, either at all or at this phase
197      *                                       of the invocation.
198      */
199     public void setTimeout(int timeout)
200             throws IOException, ServletException, UnsupportedOperationException;
